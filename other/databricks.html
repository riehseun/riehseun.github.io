Databricks CLI

# Install CLI.
pip3 install databricks-cli
databricks --version

# Enter Databricks URL and user token respectively.
databricks configure --token

# Add a line "insecure = True" to "~/.databrickscfg".
# List dbfs.
databricks fs ls

# List clusters. (That you have permission to view)
databricks clusters list

# Inspect a cluster.
databricks clusters get --cluster-id <cluster_id>
databricks clusters get --cluster-name <cluster_name>

# List all groups.
databricks groups list

# List all members of a group.
databricks groups list-members --group-name <group_name>

# List all groups of a user.
databricks groups list-parents --user-name <user_email>

# Add a member to a group.
databricks groups add-member --parent-name <group_name> --user-name <user_email>

# Remove a member from a group.
databricks groups remove-member --parent-name <group_name> --user-name <user_email>

# List secret scopes.
databricks secrets list-scopes

# List secrets in a scope.
databricks secrets list --scope <scope_name>

# List ACLs in a scope.
databricks secrets list-acls --scope <scope_name>
