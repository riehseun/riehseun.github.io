Next Permutation (M)

class Solution:
    def nextPermutation(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Iterate from right
        #   Find first decreasing number
        #   Find first number from right > above number
        #   Swap those two numbers
        #   Reverse right side of array
        # If there is no decreasing number, reverse entire array

        reverse = True

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        for i in range(len(nums)-1, 0, -1):
            if nums[i-1] < nums[i]:
                
                j = len(nums) - 1
                while i - 1 < j:
                    if nums[i-1] < nums[j]:
                        nums[j], nums[i-1] = nums[i-1], nums[j]
                        break
                    j -= 1

                l = len(nums) - 1
                while k < l:
                    nums[l], nums[k] = nums[k], nums[l]
                    k += 1
                    l -= 1

                reverse = False
                break

        if reverse:
            nums.reverse()

Longest Consecutive Sequence (M)

class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:

        # Assume duplicates are not counted
        # Convert nums to set (to reduce search time to O(1))
        # For each element, check
        #   If element-1 is not in the set, then this is start of sequence
        #   Then, recursively check if element+1 is in the set

        # Time O(n) 
        #   n = len(nums)
        # Space O(n)
        nums = set(nums)
        max_length = 0

        for i, num in enumerate(nums):
            if num-1 not in nums:
                current_length = 1

                while num+1 in nums:
                    current_length += 1
                    num += 1
                    i += 1

                max_length = max(max_length, current_length)

        return max_length

Longest Increasing Subsequence (M)

class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:

        # Greedy
        #   Build subsequence while scanning the array
        #   Pick the first number
        #
        #   If second number < first number
        #     Replace first number with second number
        #   Elif second number > first number
        #     Add second number to subsequence
        #
        #   If third number <= a number in subsequence
        #     Replace third number with such smallest number
        #     Equal sign is to replace numbers with same values in sub
        #   Elif third number > all numbers in subseuqence
        #     Add third number to subsequence

        sub = []
        sub.append(nums[0])

        def binary_search(nums, k):

            left = 0
            right = len(nums) - 1
            
            while left < right:
                mid = (left+right) // 2 
                if nums[mid] < k:
                    left = mid + 1
                else:
                    right = mid

            return right

        # Time O(nlogn)
        #   n = len(nums)
        # Space O(1)
        #   Ignoring space for output
        for i in range(1, len(nums)):

            if nums[i] < sub[-1]:
                # Find first number in sub that is > nums[i]
                j = binary_search(sub, nums[i])
                sub[j] = nums[i]

            if nums[i] > sub[-1]:
                sub.append(nums[i])
            
        return len(sub)

Russian Doll Envelopes (H)

class Solution:
    def maxEnvelopes(self, envelopes: List[List[int]]) -> int:

        # Sort x[0] ascending
        # Sort x[1] decending
        # Extra x[1]'s, then compute longest increasing subsequence

        def lis(nums):

            sub = []
            sub.append(nums[0])

            def binary_search(nums, k):

                left = 0
                right = len(nums) - 1
                
                while left < right:
                    mid = (left+right) // 2 
                    if nums[mid] < k:
                        left = mid + 1
                    else:
                        right = mid

                return right

            # Time O(nlogn)
            #   n = len(nums)
            # Space O(1)
            #   Ignoring space for output
            for i in range(1, len(nums)):

                if nums[i] < sub[-1]:
                    # Find first number in sub that is > nums[i]
                    j = binary_search(sub, nums[i])
                    sub[j] = nums[i]

                if nums[i] > sub[-1]:
                    sub.append(nums[i])
                
            return len(sub)       

        envelopes.sort(key=lambda x: (x[0], -x[1]))
        sequence = [x for _, x in envelopes]
        return lis(sequence)
