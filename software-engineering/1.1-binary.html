Add binary (E) (G)

class Solution:
    def addBinary(self, a: str, b: str) -> str:

        a = list(a)
        b = list(b)
        carry = 0
        binary_sum = 0
        c = ""

        # Example [1,1], [1] => 100
        # sum=2, carry=1, c=0
        # sum=1, carry=1, c=00

        # Time O(max(n,m)) n = len(a), m = len(b)
        # Space O(1)
        while a or b:
            if a:
                binary_sum += int(a.pop())
            if b:
                binary_sum += int(b.pop())

            if binary_sum == 2:
                if carry == 1:
                    carry = 1
                    c = "1" + c
                else:
                    carry = 1
                    c = "0" + c
            elif binary_sum == 1:
                if carry == 1:
                    carry = 1
                    c = "0" + c
                else:
                    carry = 0
                    c = "1" + c
            else:
                if carry == 1:
                    carry = 0
                    c = "1" + c
                else:
                    carry = 0
                    c = "0" + c

            binary_sum = 0

        if carry == 1:
            c = "1" + c

        return c

Subset (M) (F/A)

class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:

        # Time: O(n * 2**n)
        # Space: O(2**n)

        # If bit is 1, include the number. If bit is 0, exclude the number.

        result = []
        result.append([])

        num_subset = 2 ** len(nums)

        for i in range(1, num_subset):
            binary = bin(i)
            binary = binary.replace("0b","")
            padding = "0" * (len(nums)-len(binary))
            binary = padding + binary
            subset = []

            for j, bit in enumerate(binary):
                if bit == "1":
                    subset.append(nums[j])
            result.append(subset)

        return result

Single Number (E) (G)

class Solution:
    def singleNumber(self, nums: List[int]) -> int:

        # XOR is commutative and associative, so can rearrange
        # 1 xor 2 xor 3 xor 1 xor 2 xor 3 xor 4 =
        # (1 xor 1) xor (2 xor 2) xor (3 xor 3) xor 4 =
        # 0 xor 0 xor 0 xor 4 =
        # 4

        # Time O(n)
        # Space O(1)
        xor = 0

        for num in nums:
            xor ^= num

        return xor
