Zigzag Conversion (M)

class Solution:
    def convert(self, s: str, numRows: int) -> str:

        # Simulate 
        # Create matrix
        #   If direction is down, increment rows until reaching numRows
        #   If direction is up, decrement rows (also increment column)
        #   until reaching 0 
        #   If rows == numRows, change direction to up
        #   If rows == 0, change direction to down

        i, j, d = 0, 0, 1
        n, m = numRows, len(s)
        matrix = [[""] * m for _ in range(max(n,m))]

        # Time O(n)
        #   n = numRows
        # Space O(nm)
        #   m = len(s)
        for c in s:

            matrix[i][j] = c

            if d == 1:
                i += 1
            else:
                i -= 1
                j += 1

            if i == 0:
                d = 1
            if i == n - 1:
                d = -1
            
        result = ""

        for row in matrix:
            result += "".join(row)

        return result

Reorganize String (M)

class Solution:
    def reorganizeString(self, s: str) -> str:

        # Count frequency of each char in s
        # If number of most frequent char > len(s) // 2, return ""
        # Place most frequent char in every even indices
        #   If there are remaiding even indices, place any char
        #   Once even indices are all used, place chars in odd indices

        frequencies = defaultdict(int)
        max_count, max_char = 0, ""

        # Time O(n)
        #   n = len(s)
        # Space O(1)
        #   Ignoring space for result
        for c in s:
            frequencies[c] += 1
            if frequencies[c] > max_count:
                max_count = frequencies[c]
                max_char = c

        if len(s) % 2 == 0 and max_count > len(s) // 2:
            return ""
        if len(s) % 2 != 0 and max_count > len(s) // 2 + 1:
            return ""

        result = [""] * len(s)
        i = 0

        while frequencies[max_char] > 0:
            result[i] = max_char
            frequencies[max_char] -= 1
            i += 2

        for key, val in frequencies.items():
            while frequencies[key] > 0:
                if i >= len(s):
                    i = 1
                result[i] = key
                frequencies[key] -= 1
                i += 2

        return "".join(result)
