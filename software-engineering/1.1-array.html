Two sum (E) (G/A) !

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:

        # Loop through the number
        #   Check if (target-num) is in the array
        #     Store the number and its index in dictionary for fast search

        # Time O(n) n = len(nums)
        # Space O(n)
        my_dict = {}
        for i, num in enumerate(nums):
            my_dict[num] = i

        for i, num in enumerate(nums):
            if target-num in my_dict and i != my_dict[target-num]:
                return [i, my_dict[target-num]]

        return []

First missing positive (H)

class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:

        n = len(nums)

        # Set numbers outside [1,n] to 0
        for i in range(n):
            if nums[i] < 1 or nums[i] > n:
                nums[i] = 0

        for i in range(n):
            if nums[i] != 0 and nums[i] % (n+1) <= n:
                nums[(nums[i]%(n+1))-1] += (n+1)

        for i in range(n):
            if nums[i] <= n:
                return i + 1

        return n + 1

Longest consecutive sequence (M) (G/F/A)

class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:

        # Assume duplicates are not counted
        # Convert nums to set (to reduce search time to O(1))
        # For each element, check
        #   If element-1 is not in the set, then this is start of sequence
        #     Then, recursively check if element+1 is in the set

        # Time O(n) n = len(nums)
        # Space O(1)
        nums = set(nums)
        max_length = 0

        for num in nums:
            # This prevents duplicate work
            if num-1 not in nums:
                current_length = 1

                while num+1 in nums:
                    current_length += 1
                    num += 1

                max_length = max(max_length, current_length)

        return max_length

Maximum product subarray (M) (A)

class Solution:
    def maxProduct(self, nums: List[int]) -> int:

        # Reset the product when seeing 0 in the array
        # Otherwise, compute the product and remember the largest

        # Time O(n) n = len(nums)
        # Space O(1)
        def helper(nums):
            largest = -math.inf
            product = 1
            is_default_product = True

            for num in nums:
                if num == 0:
                    product = 1
                else:
                    product *= num
                    # For example, 1 is result of (-1)(-1)
                    if product == 1:
                        is_default_product = False
                largest = max(largest, product)

            if largest == 1 and is_default_product:
                return 0

            return largest

        a = helper(nums)
        nums.reverse()
        b = helper(nums)
        return max(a, b)

class Solution:
    def majorityElement(self, nums: List[int]) -> int:

        count = defaultdict(int)

        # Time O(n) n = len(nums)
        for num in nums:
            count[num] += 1

        for k,v in count.items():
            if v > len(nums)/2:
                return k

Rotate array (M) (G/A)

class Solution:
    def rotate(self, nums: List[int], k: int) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # New index is (i+k) & len(nums)

        # Time O(n) n = len(nums)
        # Space O(n)
        nums_copy = nums[:]
        for i, num in enumerate(nums):
            new_index = (i+k) % len(nums)
            nums[new_index] = nums_copy[i]

        return nums

Product of array except self (M) (F/A)

class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:

        # Compute the running products from both sides

        # Time O(n) n = len(nums)
        # Space O(n)
        products_from_left = defaultdict(int)
        products_from_left[-1] = 1

        for i, num in enumerate(nums):
            products_from_left[i] = products_from_left[i-1] * num

        products_from_right = defaultdict(int)
        products_from_right[len(nums)] = 1

        for i in range(len(nums)-1, -1, -1):
            products_from_right[i] = products_from_right[i+1] * nums[i]

        result = []
        for i, num in enumerate(nums):
            result.append(products_from_left[i-1]*products_from_right[i+1])

        return result

Missing number (E) (A)

class Solution:
    def missingNumber(self, nums: List[int]) -> int:

        n = len(nums)
        nums = set(nums)

        # Time O(n) n = len(nums)
        # Space O(1)
        for i in range(n+1):
            if i not in nums:
                return i

        return -1

Continuous Subarray Sum (M) (F)

class Solution:
    def checkSubarraySum(self, nums: List[int], k: int) -> bool:

        # Compute cummulative sum
        # Compute mod of those sums by k
        # If two mods have the same value,
        # sum betweem them should be divisible by k

        # Ex.
        # [23, 2, 4, 6, 7]
        # [23, 25, 29, 35, 42]
        # [5, 1, 5, 5, 0]

        mods = defaultdict(list)

        cummulative_sum = 0
        cummulative_sums = []

        # Time O(n) n = len(nums)
        # Space O(n)
        for num in nums:
            cummulative_sum += num
            cummulative_sums.append(cummulative_sum)

        for i, num in enumerate(cummulative_sums):
            mods[num%k].append(i)

        for key, val in mods.items():
            if len(val) > 2:
                return True
            if len(val) == 2:
                if abs(val[0]-val[1]) > 1:
                    return True

        if 0 in mods and mods[0] != [0]:
            return True

        return False

Subarray Sum Equals K (M) (G/F/A)

class Solution:
    def subarraySum(self, nums: List[int], k: int) -> int:

        # Compute cummulative sum
        # Create hash table
        #   key = a cumulative sum
        #   value = number of times a particular cumulative sum occurs
        # If cummulative_sum_a - k = cummulative_sum_b,
        # Keep the number of times cummulative_sum_b occurs

        cummulative_sums = defaultdict(int)
        cummulative_sum = 0
        count = 0
        cummulative_sums[0] = 1

        # Time O(n) n = len(nums)
        # Space O(n)
        for num in nums:
            cummulative_sum += num
            count += cummulative_sums[cummulative_sum-k]
            cummulative_sums[cummulative_sum] += 1

        return count

Unique Number of Occurrences (E) (G)

class Solution:
    def uniqueOccurrences(self, arr: List[int]) -> bool:

        # Hash table to store count

        # Time O(n) n = len(arr)
        # Space O(n)
        counts = defaultdict(int)
        for num in arr:
            counts[num] += 1

        seen = set()
        for key, val in counts.items():
            if val in seen:
                return False
            seen.add(val)

        return True

Minimum Number of Increments on Subarrays to Form a Target Array (H)

class Solution:
    def minNumberOperations(self, target: List[int]) -> int:

        # Consider the first number
        # It requires target[0] operations at the miminum
        # If target[0] > target[1]
        #   No extra operation is needed
        # If target[0] < target[1]
        #   We need extra target[1]-target[0] operations

        total_operations = target[0]
        current_operation = target[0]

        # Time O(n) n = len(target)
        # Space O(1)
        for i in range(1, len(target)):
            if target[i] < current_operation:
                current_operation = target[i]
            else:
                total_operations += (target[i]-current_operation)
                current_operation = target[i]

        return total_operations

Three Consecutive Odds (E)

class Solution:
    def threeConsecutiveOdds(self, arr: List[int]) -> bool:

        # Slide 3 numbers at once

        if len(arr) < 3:
            return False

        # Time O(n) n = len(arr)
        # Space O(!)
        for i in range(len(arr)-2):
            is_consecutive_odd = True
            for num in arr[i:i+3]:
                if num % 2 == 0:
                    is_consecutive_odd = False

            if is_consecutive_odd:
                return True

        return False

Crawler Log Folder (E)

class Solution:
    def minOperations(self, logs: List[str]) -> int:

        # Use counter to keep track of level

        count = 0

        # Time O(n) n = len(logs)
        # Space O(1)
        for log in logs:

            if log == "../":
                if count > 0:
                    count -= 1
            elif log == "./":
                continue
            else:
                count += 1

        return count

Buildings With an Ocean View (M) (F)

class Solution:
    def findBuildings(self, heights: List[int]) -> List[int]:

        # Find numbers that has decreasing sequence only
        # Compute running maximum from the right and compare

        max_num = -1
        result = []

        # Time O(n) n = len(heights)
        # Space O(n)
        for i in range(len(heights)-1, -1, -1):
            if heights[i] > max_num:
                result.append(i)
            max_num = max(max_num, heights[i])

        result.reverse()
        return result
