Two sum (E) (Meta)

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:

        # Iterate nums while storing (target-num) in dictionary
        # If num is in dictionary, return

        dic = defaultdict(int)

        # Time O(n)
        #   n = len(nums)
        # Space O(n)
        for i, num in enumerate(nums):
            if num in dic:
                return (i, dic[num])
            dic[target-num] = i
            
        return None

Next permutation (M)

class Solution:
    def nextPermutation(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Find a first decreasing number from the right
        # (Find the largest i-1 such that nums[i-1] < nums[i])
        #   Find a first number from the right, that is greater 
        #   than the first decreasing number
        #   (Find the largest index j > i-1 such that 
        #   nums[i-1] < nums[j])
        #   Swap those two numbers (Swap nums[j], nums[i-1]) 
        #   Then reverse the right side of the array, which is nums[i:]
        # When such numbers do not exist
        #   Reverse the entire array

        is_swapped = False
        
        # Time O(n) n = len(nums)
        # Space O(1)
        for i in range(len(nums)-1, 0, -1):

            # First decreasing number from right
            if nums[i-1] < nums[i]:

                # Find first number from right that is 
                # greater than the first decreasing number
                # Then, swap those numbers
                for j in range(len(nums)-1, i-1, -1):
                    if nums[i-1] < nums[j]:
                        nums[i-1], nums[j] = nums[j], nums[i-1]
                        break
                
                # Reverse right side of array
                start = 0
                end = len(nums[i:]) - 1
                while start < end:
                    nums[i+start], nums[i+end] = nums[i+end], nums[i+start]
                    start += 1
                    end -= 1

                is_swapped = True
                break
            
        if not is_swapped:
            nums.reverse()

First missing positive (H)

class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:

        n = len(nums)

        # Set numbers outside [1,n] to 0
        for i in range(n):
            if nums[i] < 1 or nums[i] > n:
                nums[i] = 0

        for i in range(n):
            if nums[i] != 0 and nums[i] % (n+1) <= n:
                nums[(nums[i]%(n+1))-1] += (n+1)

        for i in range(n):
            if nums[i] <= n:
                return i + 1

        return n + 1

Longest consecutive sequence (M)

class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:

        # Assume duplicates are not counted
        # Convert nums to set (to reduce search time to O(1))
        # For each element, check
        #   If element-1 is not in the set, then this is start of sequence
        #     Then, recursively check if element+1 is in the set

        # Time O(n) n = len(nums)
        # Space O(1)
        nums = set(nums)
        max_length = 0

        for num in nums:
            # This prevents duplicate work
            if num-1 not in nums:
                current_length = 1

                while num+1 in nums:
                    current_length += 1
                    num += 1

                max_length = max(max_length, current_length)

        return max_length

Maximum product subarray (M)

class Solution:
    def maxProduct(self, nums: List[int]) -> int:

        # Reset the product when seeing 0 in the array
        # Otherwise, compute the product and remember the largest

        # Time O(n) n = len(nums)
        # Space O(1)
        def helper(nums):
            largest = -math.inf
            product = 1
            is_default_product = True

            for num in nums:
                if num == 0:
                    product = 1
                else:
                    product *= num
                    # For example, 1 is result of (-1)(-1)
                    if product == 1:
                        is_default_product = False
                largest = max(largest, product)

            if largest == 1 and is_default_product:
                return 0

            return largest

        a = helper(nums)
        nums.reverse()
        b = helper(nums)
        return max(a, b)

Missing ranges (E)

class Solution:
    def findMissingRanges(self, nums: List[int], lower: int, upper: int) -> List[List[int]]:

        # Linear scan looping through nums
        # low = lower
        # If num == low:
        #   low += 1 
        # Else
        #   high = num - 1
        #   Insert (low, high) to result
        #   low = num + 1
        # high = upper
        # Insert (low, high) to result

        # If empty list, return the entire range
        if not nums:
            return [[lower, upper]]
        # If there is no range, return empty list
        if lower == upper:
            return []

        result = []
        low = lower

        # Time O(n) 
        #   n = len(nums)
        # Space O(1) 
        #   Ignoring space for output
        for num in nums:
            if num == low:
                low += 1
            else:
                high = num - 1
                result.append([low, high])
                low = num + 1

        # Insert range between the last num and upper
        if nums[-1] != upper:
            result.append([low, upper])

        return result

Rotate array (M)

class Solution:
    def rotate(self, nums: List[int], k: int) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # New index is (i+k) & len(nums)

        # Time O(n) n = len(nums)
        # Space O(n)
        nums_copy = nums[:]
        for i, num in enumerate(nums):
            new_index = (i+k) % len(nums)
            nums[new_index] = nums_copy[i]

        return nums

class Solution:
    def containsNearbyDuplicate(self, nums: List[int], k: int) -> bool:

        # Sliding window
        # Move window of k and check if 
        # there are two same numbers within k 
        # Use set for the window

        # Time O(n)
        #   n = len(nums)
        # Space O(k)
        nums_set = set()
        for i, c in enumerate(nums):
            if c in nums_set:
                return True
            nums_set.add(c)
            if len(nums_set) > k:
                nums_set.remove(nums[i-k])

        return False

Product of array except self (M)

class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:

        # If there are more than one 0 in the arry,
        # result is always 0
        # if there is one 0 in the array,
        # result is product of all array except that 0 divied by each num
        # If tehre is no 0 in the array, 
        # result is product of all array divided by each num

        num_zeros = 0

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        #   Ignoring space for result
        for num in nums:
            if num == 0:
                num_zeros += 1

        result = []
        product = 1
        for num in nums:
            if num != 0:
                product *= num

        if num_zeros > 1:
            return [0] * len(nums)
        elif num_zeros == 1: 
            for num in nums:
                if num == 0:
                    result.append(int(product))
                else:
                    result.append(0)
        else:
            for num in nums:
                result.append(int(product/num))

        return result

Strobogrammatic Number (E)

class Solution:
    def isStrobogrammatic(self, num: str) -> bool:

        # Identify valid pairs
        # flip 6 => 9
        # flip 8 => 8
        # flip 1 => 1
        # flip 0 => 0
        # Rotate numbers

        rotated_num = ""

        # Time O(n)
        #   n = len(num)
        # Space O(n)
        for c in num:
            if c in ("0", "1", "8"):
                rotated_num += c
            if c == "6":
                rotated_num += "9"
            if c == "9":
                rotated_num += "6"

        rotated_num = list(rotated_num)
        rotated_num.reverse()
        rotated_num = "".join(rotated_num)
        
        return num == rotated_num

Longest Increasing Subsequence (M) (Samsung)

class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:

        # Greedy
        #   Build subsequence while scanning the array
        #   Pick the first number
        #
        #   If second number < first number
        #     Replace first number with second number
        #   Elif second number > first number
        #     Add second number to subsequence
        #
        #   If third number <= a number in subsequence
        #     Replace third number with such smallest number
        #     Equal sign is to replace numbers with same values in sub
        #   Elif third number > all numbers in subseuqence
        #     Add third number to subsequence

        sub = []
        sub.append(nums[0])

        def binary_search(nums, k):

            left = 0
            right = len(nums) - 1
            
            while left < right:
                mid = (left+right) // 2 
                if nums[mid] < k:
                    left = mid + 1
                else:
                    right = mid

            return right

        # Time O(nlogn)
        #   n = len(nums)
        # Space O(1)
        #   Ignoring space for output
        for i in range(1, len(nums)):

            if nums[i] < sub[-1]:
                # Find first number in sub that is > nums[i]
                j = binary_search(sub, nums[i])
                sub[j] = nums[i]

            if nums[i] > sub[-1]:
                sub.append(nums[i])
            
        return len(sub)

class Logger:

    # Use dictionary to remember the last timestamp for each message

    def __init__(self):
        self.messages = defaultdict(int)

    # Time O(1)
    # Space O(n) n = number of unique messages
    def shouldPrintMessage(self, timestamp: int, message: str) -> bool:

        if message in self.messages \
            and timestamp - self.messages[message] < 10:
                return False
        else:
            self.messages[message] = timestamp
            return True

Insert Delete GetRandom (M)

class RandomizedSet:

    # Use list to store numbers
    # Use dictionary to store indice of numbers
    # Insert
    #   Append to list O(1)
    #   dic[element] = index O(1)
    # Remove
    #   Swap index of value to remove with last element O(1)
    #   Pop from list O(1)
    #   Remove element from dic O(1)  

    # Time O(1)
    # Space O(n)
    #   n = length of input numbers
    def __init__(self):
        self.nums = []
        self.indices = defaultdict(int)

    def insert(self, val: int) -> bool:
        if val not in self.indices:
            self.nums.append(val)
            self.indices[val] = len(self.nums) - 1
            return True
        return False
        
    def remove(self, val: int) -> bool:
        if val in self.indices:
            val_index = self.indices[val]
            self.nums[val_index] = self.nums[-1]
            self.indices[self.nums[-1]] = val_index
            self.nums.pop()
            del self.indices[val]
            return True
        return False

    def getRandom(self) -> int:
        return self.nums[random.randint(0, len(self.nums)-1)]


# Your RandomizedSet object will be instantiated and called as such:
# obj = RandomizedSet()
# param_1 = obj.insert(val)
# param_2 = obj.remove(val)
# param_3 = obj.getRandom()

Next Greater Element III (M)

class Solution:
    def nextGreaterElement(self, n: int) -> int:

        # Look from right side
        # If nums[i] > nums[i-1]
        #   Find first number from right that is greater than nums[i-1]
        #   Swap them
        #   Then, reverse right side
        # Else, continue to i-2
        # If reached beginning, return -1
        # Ex. 31347
        # Ex. 31342 => 31243

        nums = [x for x in str(n)]

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        for i in range(len(nums)-1, 0, -1):
            if int(nums[i]) > int(nums[i-1]):
                # nums[i], nums[i-1] = nums[i-1], nums[i]
                for j in range(len(nums)-1, i-1, -1):
                    if nums[i-1] < nums[j]:
                        nums[i-1], nums[j] = nums[j], nums[i-1]
                        break
                nums[i:] = nums[i:][::-1]
                result = "".join(nums)
                if int(result) > 2**31 - 1:
                    return -1
                return int(result)

        return - 1

Can Place Flowers (E)

class Solution:
    def canPlaceFlowers(self, flowerbed: List[int], n: int) -> bool:
        
        # Linear scan
        #   If nums[i] == 0 and nums[i-1] != 1 and nums[i+1] != 1
        #     place 1
        #   Else
        #     i += 1

        for i in range(len(flowerbed)):
            if flowerbed[i] == 0 \
                and (i == 0 or flowerbed[i-1] == 0) \
                and (i == len(flowerbed)-1 or flowerbed[i+1] == 0):
                flowerbed[i] = 1
                n -= 1

        return n <= 0

class FileSystem:

    # Use a hash table where 
    #   key = path
    #   value = integer associated to the path
    # __init__
    #   initialize a hash table
    # createPath
    #   If path exists in hash table, return False
    #   If parent path does not exist in hash table, return False
    #   Otherwise, hash_table[path] = value
    # get
    #   return hash_table[path]

    def __init__(self):
        self.paths = defaultdict(int)

    def createPath(self, path: str, value: int) -> bool:

        # Check if parent exists
        # Time O(n)
        #   n = len(path)
        # Space O(m)
        #   m = number of unique path
        array = path.split("/")
        directory = array[-1]
        parent = "/".join(array[:-1])

        if path in self.paths:
            return False
        
        if parent not in self.paths and parent != "":
            return False
    
        self.paths[path] = value
        return True
        
    def get(self, path: str) -> int:
        return self.paths.get(path, -1)

Unique Number of Occurrences (E)

class Solution:
    def uniqueOccurrences(self, arr: List[int]) -> bool:

        # Hash table to store count

        # Time O(n) n = len(arr)
        # Space O(n)
        counts = defaultdict(int)
        for num in arr:
            counts[num] += 1

        seen = set()
        for key, val in counts.items():
            if val in seen:
                return False
            seen.add(val)

        return True

Minimum Number of Increments on Subarrays to Form a Target Array (H)

class Solution:
    def minNumberOperations(self, target: List[int]) -> int:

        # Consider the first number
        # It requires target[0] operations at the miminum
        # If target[0] > target[1]
        #   No extra operation is needed
        # If target[0] < target[1]
        #   We need extra target[1]-target[0] operations

        total_operations = target[0]
        current_operation = target[0]

        # Time O(n) n = len(target)
        # Space O(1)
        for i in range(1, len(target)):
            if target[i] < current_operation:
                current_operation = target[i]
            else:
                total_operations += (target[i]-current_operation)
                current_operation = target[i]

        return total_operations

Three Consecutive Odds (E)

class Solution:
    def threeConsecutiveOdds(self, arr: List[int]) -> bool:

        # Slide 3 numbers at once

        if len(arr) < 3:
            return False

        # Time O(n) n = len(arr)
        # Space O(!)
        for i in range(len(arr)-2):
            is_consecutive_odd = True
            for num in arr[i:i+3]:
                if num % 2 == 0:
                    is_consecutive_odd = False

            if is_consecutive_odd:
                return True

        return False

        # Space O(1)

            elif log == "./":

Buildings With an Ocean View (M) (Meta)

class Solution:
    def findBuildings(self, heights: List[int]) -> List[int]:

        # Iterate from right
        #   Remeber the max number seen
        # If a num > max_seen, then it has ocean view

        result = []
        max_seen = 0
        
        # Time O(n)
        #   n = len(heights)
        # Space O(1)
        for i in range(len(heights)-1, -1, -1):
            if heights[i] > max_seen:
                result.append(i)
            max_seen = max(max_seen, heights[i])

        result.reverse()
        return result
