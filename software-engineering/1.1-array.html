Two sum (E) (Google)

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:

        # Iterate nums while storing (target-num) in dictionary
        # If num is in dictionary, return

        dic = defaultdict(int)

        # Time O(n)
        #   n = len(nums)
        # Space O(n)
        for i, num in enumerate(nums):
            if num in dic:
                return (i, dic[num])
            dic[target-num] = i
            
        return None

First missing positive (H) (Google)

class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:

        n = len(nums)

        # Set numbers outside [1,n] to 0
        for i in range(n):
            if nums[i] < 1 or nums[i] > n:
                nums[i] = 0

        for i in range(n):
            if nums[i] != 0 and nums[i] % (n+1) <= n:
                nums[(nums[i]%(n+1))-1] += (n+1)

        for i in range(n):
            if nums[i] <= n:
                return i + 1

        return n + 1

Longest consecutive sequence (M) (Google)

class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:

        # Assume duplicates are not counted
        # Convert nums to set (to reduce search time to O(1))
        # For each element, check
        #   If element-1 is not in the set, then this is start of sequence
        #     Then, recursively check if element+1 is in the set

        # Time O(n) n = len(nums)
        # Space O(1)
        nums = set(nums)
        max_length = 0

        for num in nums:
            # This prevents duplicate work
            if num-1 not in nums:
                current_length = 1

                while num+1 in nums:
                    current_length += 1
                    num += 1

                max_length = max(max_length, current_length)

        return max_length

Majority element (E) (Google)

class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        
        # Use hash table to store counts
        count = defaultdict(int)
        
        # Time O(n) 
        #   n = len(nums)
        # Space O(n)
        for num in nums:
            count[num] += 1
            if count[num] > len(nums) / 2:
                return num

Rotate array (M) (Google)

class Solution:
    def rotate(self, nums: List[int], k: int) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # New index is (i+k) & len(nums)

        # Time O(n) n = len(nums)
        # Space O(n)
        nums_copy = nums[:]
        for i, num in enumerate(nums):
            new_index = (i+k) % len(nums)
            nums[new_index] = nums_copy[i]

        return nums

Contains duplicate (E) (Bloomberg)

class Solution:
    def containsDuplicate(self, nums: List[int]) -> bool:
        
        # Use set to track items

        nums_set = set()

        # Time O(n)
        #   n = len(nums)
        # Space O(n)
        for num in nums:
            if num in nums_set:
                return True
            nums_set.add(num)

        return False

Product of array except self (M) (Amazon)

class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:

        # If there are more than one 0 in the arry,
        # result is always 0
        # if there is one 0 in the array,
        # result is product of all array except that 0 divied by each num
        # If tehre is no 0 in the array, 
        # result is product of all array divided by each num

        num_zeros = 0

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        #   Ignoring space for result
        for num in nums:
            if num == 0:
                num_zeros += 1

        result = []
        product = 1
        for num in nums:
            if num != 0:
                product *= num

        if num_zeros > 1:
            return [0] * len(nums)
        elif num_zeros == 1: 
            for num in nums:
                if num == 0:
                    result.append(int(product))
                else:
                    result.append(0)
        else:
            for num in nums:
                result.append(int(product/num))

        return result

Logger rate limiter (E) (Google)

class Logger:

    # Use dictionary to remember the last timestamp for each message

    def __init__(self):
        self.messages = defaultdict(int)

    # Time O(1)
    # Space O(n) n = number of unique messages
    def shouldPrintMessage(self, timestamp: int, message: str) -> bool:

        if message in self.messages \
            and timestamp - self.messages[message] < 10:
                return False
        else:
            self.messages[message] = timestamp
            return True

Insert Delete GetRandom (M) (Amazon)

class RandomizedSet:

    # Use list to store numbers
    # Use dictionary to store indice of numbers
    # Insert
    #   Append to list O(1)
    #   dic[element] = index O(1)
    # Remove
    #   Swap index of value to remove with last element O(1)
    #   Pop from list O(1)
    #   Remove element from dic O(1)  

    # Time O(1)
    # Space O(n)
    #   n = length of input numbers
    def __init__(self):
        self.nums = []
        self.indices = defaultdict(int)

    def insert(self, val: int) -> bool:
        if val not in self.indices:
            self.nums.append(val)
            self.indices[val] = len(self.nums) - 1
            return True
        return False
        
    def remove(self, val: int) -> bool:
        if val in self.indices:
            val_index = self.indices[val]
            self.nums[val_index] = self.nums[-1]
            self.indices[self.nums[-1]] = val_index
            self.nums.pop()
            del self.indices[val]
            return True
        return False

    def getRandom(self) -> int:
        return self.nums[random.randint(0, len(self.nums)-1)]


# Your RandomizedSet object will be instantiated and called as such:
# obj = RandomizedSet()
# param_1 = obj.insert(val)
# param_2 = obj.remove(val)
# param_3 = obj.getRandom()

Design File System (M) (Doordash)

class FileSystem:

    # Use a hash table where 
    #   key = path
    #   value = integer associated to the path
    # __init__
    #   initialize a hash table
    # createPath
    #   If path exists in hash table, return False
    #   If parent path does not exist in hash table, return False
    #   Otherwise, hash_table[path] = value
    # get
    #   return hash_table[path]

    def __init__(self):
        self.paths = defaultdict(int)

    def createPath(self, path: str, value: int) -> bool:

        # Check if parent exists
        # Time O(n)
        #   n = len(path)
        # Space O(m)
        #   m = number of unique path
        array = path.split("/")
        directory = array[-1]
        parent = "/".join(array[:-1])

        if path in self.paths:
            return False
        
        if parent not in self.paths and parent != "":
            return False
    
        self.paths[path] = value
        return True
        
    def get(self, path: str) -> int:
        return self.paths.get(path, -1)

Invalid Transactions (M) (Bloomberg)

class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:

        # Use dictionary where 
        #   key = name 
        #   value = another dictionary where
        #     key = time
        #     value = city
        #   Scan trans between -60 and +60 from current time
        #     See if trans happend in different city
        # If amount > 1000, put transaction into result

        result = []
        trans = {}

        for item in transactions:
            it = item.split(",")
            name, time, amount, city = it[0], int(it[1]), int(it[2]), it[3]
            if name not in trans:
                trans[name] = {}
            if time not in trans[name]:
                trans[name][time] = []
            trans[name][time].append(city)

        # Time O(n)
        #   n = len(transactions)
        # Space O(n)
        for item in transactions:
            it = item.split(",")
            name, time, amount, city = it[0], int(it[1]), int(it[2]), it[3]
            
            if amount > 1000:
                result.append(item)
                continue

            # Check all transactions within 60 range
            # under the same name
            break_outter = False
            for t in range(int(time)-60, int(time)+60):
                if t in trans[name]:
                    for c in trans[name][t]:
                        if c != city:
                            result.append(item)
                            break_outter = True
                            break
                if break_outter:
                    break

        return result

Dot Product of Two Sparse Vectors (M) (Meta)

class SparseVector:

    # Only store indices of non-zero element
    # Compute multiplication only when indices match
    # between two arrays

    # Time O(n) to create array O(l) to compute dot product
    #   n = len(nums)
    #   l = number of non-zero elements
    # Space O(l)
    def __init__(self, nums: List[int]):
        self.nums = []
        for i, num in enumerate(nums):
            if num != 0:
                self.nums.append((i, num))

    def dotProduct(self, vec: 'SparseVector') -> int:

        i, j, result = 0, 0, 0

        while i < len(self.nums) and j < len(vec.nums):

            if self.nums[i][0] > vec.nums[j][0]:
                j += 1
            elif self.nums[i][0] < vec.nums[j][0]:
                i += 1
            else:
                result += self.nums[i][1] * vec.nums[j][1]
                i += 1
                j += 1
                print(result)

        return result

Buildings With an Ocean View (M) (Meta)

class Solution:
    def findBuildings(self, heights: List[int]) -> List[int]:

        # Iterate from right
        #   Remeber the max number seen
        # If a num > max_seen, then it has ocean view

        result = []
        max_seen = 0
        
        # Time O(n)
        #   n = len(heights)
        # Space O(1)
        for i in range(len(heights)-1, -1, -1):
            if heights[i] > max_seen:
                result.append(i)
            max_seen = max(max_seen, heights[i])

        result.reverse()
        return result
