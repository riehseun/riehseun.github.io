Merge two sorted lists (E) (Google)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:

        # Iterate list1 and list2 at the same time

        result = ListNode()
        dummy_head = result

        # Time O(n+m) 
        #   n = length of list1
        #   m = length of list2
        # Space O(1) 
        while list1 and list2:
            if list1.val < list2.val:
                result.next = list1
                list1 = list1.next
            else:
                result.next = list2
                list2 = list2.next
            result = result.next

        if list1:
            result.next = list1
        if list2:
            result.next = list2

        # Returning result.next will return the last node in the linked
        # list we computed. To return the first node, we use temp
        return dummy_head.next

Merge interval (M) (Meta)

class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:

        # Sort intervals by start time
        # Check two intervals at a time [a,b] and [c,d]
        #   If b >= c, there is an overlap
        #   Merged interval is [min(a,c), max(b,d)]
        # Insert fisrt item to result
        # Then, iterate from second items in interval
        # and compare with last item in the result
        
        intervals.sort(key=lambda x: x[0])

        result = []
        result.append(intervals[0])

        # Time O(nlogn)
        #   n = len(intervals)
        # Space O(logn)
        #   Ignoring space for result
        #   Recursive stack for quicksort
        for interval in intervals[1:]:
            a, b = result[-1]
            c, d = interval
            if b >= c:
                result.pop()
                result.append([min(a,c), max(b,d)])
            else:
                result.append([c,d])

        return result

Meeting rooms II (M) (Amazon)

class Solution:
    def minMeetingRooms(self, intervals: List[List[int]]) -> int:

        # Sort by start time in an array
        # Also, sort by end time in a separate array
        # Iterate start time array
        #   If start[i] > end[j], some meeting has ended already
        #   Then, we can re-use the room (Also, increment j)  
        #   Else, increament room count

        starts, ends, n, count = [], [], len(intervals), 0

        # Time O(nlogn)
        #   n = len(intervals)
        # Space O(1)
        for s, e in intervals:
            starts.append(s)
            ends.append(e)
        
        starts.sort()
        ends.sort()

        i, j = 0, 0

        while i < n and j < n:
            if starts[i] >= ends[j]:
                j += 1
            else:
                count += 1
            i += 1

        return count

Employee free time (H)

"""
# Definition for an Interval.
class Interval:
    def __init__(self, start: int = None, end: int = None):
        self.start = start
        self.end = end
"""

class Solution:
    def employeeFreeTime(self, schedule: '[[Interval]]') -> '[Interval]':

        # Sort the interval by the start, and then end by reverse
        # If any start is after the latest end seen,
        # then there is non-intersection

        result = []

        # Time O(nlogn) n = length of combined input arrays
        # Space O(n)
        def convert_input_to_list(schedules):

            flat_array = []

            for schedule in schedules:
                for array in schedule:
                    flat_array.append([array.start, array.end])

            flat_array.sort(key = lambda x: (x[0], -x[1]))

            return flat_array

        array = convert_input_to_list(schedule)

        latest_end = 0
        for interval in array:

            if interval[0] > latest_end and latest_end != 0:
                    result.append(Interval(latest_end, interval[0]))
            latest_end = max(latest_end, interval[1])

        return result

Interval List Intersections (M) (Meta)

class Solution:
    def intervalIntersection(self, firstList: List[List[int]], secondList: List[List[int]]) -> List[List[int]]:

        # Compare A[0] and B[0]
        # Then, compare either A[0] and B[1] or A[1] and B[0]
        #   If B[0]'s end > A[0]'s end, compare A[1] and B[0]
        #   If A[0]'s end > B[0]'s end, compare A[0] and B[1]
        # [a,b] and [c,d] overlap is 
        #   [max(a,c), min(b,d)]

        intersections = []
        i, j = 0, 0
        n = len(firstList)
        m = len(secondList)

        # Time O(n+m)
        #   n = len(firstList)
        #   m = len(secondList)
        # Space O(1)
        #   Ignoring space for result
        while i < n and j < m:
            a, b = firstList[i]
            c, d = secondList[j]
            if min(b,d) >= max(a,c):
                intersections.append([max(a,c), min(b,d)])
            
            if d > b:
                i += 1
            else:
                j += 1

        return intersections
