Merge Two Sorted Lists (E)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:

        # Iterate list1 and list2 at the same time

        result = ListNode()
        dummy_head = result

        # Time O(n+m) 
        #   n = length of list1
        #   m = length of list2
        # Space O(1) 
        while list1 and list2:
            if list1.val < list2.val:
                result.next = list1
                list1 = list1.next
            else:
                result.next = list2
                list2 = list2.next
            result = result.next

        if list1:
            result.next = list1
        if list2:
            result.next = list2

        # Returning result.next will return the last node in the linked
        # list we computed. To return the first node, we use temp
        return dummy_head.next

Merge k Sorted Lists (H)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:

        # Merge two list at a time
        #   Keep merged list in the original array 
        # Repeat until only one list is left

        def merge_two_lists(l1, l2):

            head = l3 = ListNode()

            while l1 and l2:
                if l1.val < l2.val:
                    l3.next = l1
                    l1 = l1.next
                else:
                    l3.next = l2
                    l2 = l2.next
                l3 = l3.next

            if l1:
                l3.next = l1
            if l2:
                l3.next = l2

            return head.next

        n = len(lists)
        interval = 1

        # Time O(nlogk)
        #   n = total number of nodes in all linked lists
        #   k = number of linked lists
        # Space O(1)
        while interval < n:
            for i in range(0, n-interval, interval*2):
                lists[i] = merge_two_lists(lists[i], lists[i+interval])
            interval *= 2

        if n <= 0:
            return None

        return lists[0]

Merge Interval (M)

class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:

        # Sort intervals by start time
        # Check two intervals at a time [a,b] and [c,d]
        #   If b >= c, there is an overlap
        #   Merged interval is [min(a,c), max(b,d)]
        # Insert fisrt item to result
        # Then, iterate from second items in interval
        # and compare with last item in the result
        
        intervals.sort(key=lambda x: x[0])

        result = []
        result.append(intervals[0])

        # Time O(nlogn)
        #   n = len(intervals)
        # Space O(logn)
        #   Ignoring space for result
        #   Recursive stack for quicksort
        for interval in intervals[1:]:
            a, b = result[-1]
            c, d = interval
            if b >= c:
                result.pop()
                result.append([min(a,c), max(b,d)])
            else:
                result.append([c,d])

        return result

Meeting Rooms II (M)

class Solution:
    def minMeetingRooms(self, intervals: List[List[int]]) -> int:

        # Sort by start time in an array
        # Also, sort by end time in a separate array
        # Iterate start time array
        #   If start[i] > end[j], some meeting has ended already
        #   Then, we can re-use the room (Also, increment j)  
        #   Else, increament room count

        starts, ends, n, count = [], [], len(intervals), 0

        # Time O(nlogn)
        #   n = len(intervals)
        # Space O(1)
        for s, e in intervals:
            starts.append(s)
            ends.append(e)
        
        starts.sort()
        ends.sort()

        i, j = 0, 0

        while i < n and j < n:
            if starts[i] >= ends[j]:
                j += 1
            else:
                count += 1
            i += 1

        return count

Interval List Intersections (M)

class Solution:
    def intervalIntersection(self, firstList: List[List[int]], secondList: List[List[int]]) -> List[List[int]]:

        # Compare A[0] and B[0]
        # Then, compare either A[0] and B[1] or A[1] and B[0]
        #   If B[0]'s end > A[0]'s end, compare A[1] and B[0]
        #   If A[0]'s end > B[0]'s end, compare A[0] and B[1]
        # [a,b] and [c,d] overlap is 
        #   [max(a,c), min(b,d)]

        intersections = []
        i, j = 0, 0
        n = len(firstList)
        m = len(secondList)

        # Time O(n+m)
        #   n = len(firstList)
        #   m = len(secondList)
        # Space O(1)
        #   Ignoring space for result
        while i < n and j < m:
            a, b = firstList[i]
            c, d = secondList[j]
            if min(b,d) >= max(a,c):
                intersections.append([max(a,c), min(b,d)])
            
            if d > b:
                i += 1
            else:
                j += 1

        return intersections
