Clone graph (M) (F) #

"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:

        # BFS
        # Construct "explored" as a dictionary
        #   key is node object of original graph
        #   value is node object of deep copy graph

        if not node:
            return node
        
        queue = deque()
        queue.append(node)

        explored = defaultdict(Node)
        explored[node] = Node(node.val)

        # Time O(n+m) 
        #   n = number of nodes
        #   m = number of edges
        # Space O(n)
        while queue:

            n = queue.popleft()

            for next_node in n.neighbors:
                if next_node not in explored:
                    explored[next_node] = Node(next_node.val)
                    queue.append(next_node)
                
                explored[n].neighbors.append(explored[next_node])
                        
        return explored[node]

#        :rtype List[NestedInteger]
#        """

class Solution:
    def depthSum(self, nestedList: List[NestedInteger]) -> int:

        # DFS (recursive implementation)
        # Each time getList() is called, increment depth by 1
        
        # Time O(n) 
        #   n = number of nested lists and nested integers combined
        # Space O(n) 
        #   Recursion stack
        def dfs(nested_list, depth):
            
            result = 0
                
            for token in nested_list:
                if token.isInteger():
                    result += (token.getInteger()*depth)
                else:
                    result += dfs(token.getList(), depth+1)

            return result

        return dfs(nestedList, 1)

Friends Of Appropriate Ages (M) (F) !

class Solution:
    def numFriendRequests(self, ages: List[int]) -> int:

        # Construct a graph
        # Count the number of edges in graph

        count = 0

        # Time O(n**2)
        #   n = len(ages)
        # Space O(1)
        for i in range(len(ages)):
            for j in range(len(ages)):
                if i != j:
                    if ages[j] <= 0.5 * ages[i] + 7:
                        continue
                    if ages[j] > ages[i]:
                        continue
                    if ages[j] > 100 and ages[i] < 100:
                        continue
                    count += 1
        
        return count

Regions cut by slashes (M) (G) !

class Solution:
    def regionsBySlashes(self, grid: List[str]) -> int:

        # Convert each "/" and "\" into 3 by 3 grid
        # Construct bidirection graph where there is edge between connected grid

        new_grid = []
        new_grid_dimension = 3*len(grid)
        for i in range(new_grid_dimension):
            new_grid.append(new_grid_dimension*[0])

        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == "/":
                    new_grid[3*i][3*j] = 0
                    new_grid[3*i][3*j+1] = 0
                    new_grid[3*i][3*j+2] = 1
                    new_grid[3*i+1][3*j] = 0
                    new_grid[3*i+1][3*j+1] = 1
                    new_grid[3*i+1][3*j+2] = 0
                    new_grid[3*i+2][3*j] = 1
                    new_grid[3*i+2][3*j+1] = 0
                    new_grid[3*i+2][3*j+2] = 0
                elif grid[i][j] == "\\":
                    new_grid[3*i][3*j] = 1
                    new_grid[3*i][3*j+1] = 0
                    new_grid[3*i][3*j+2] = 0
                    new_grid[3*i+1][3*j] = 0
                    new_grid[3*i+1][3*j+1] = 1
                    new_grid[3*i+1][3*j+2] = 0
                    new_grid[3*i+2][3*j] = 0
                    new_grid[3*i+2][3*j+1] = 0
                    new_grid[3*i+2][3*j+2] = 1

        graph = defaultdict(set)
        nodes = set()

        # Time O(nm)
        # Space O(nm) worst case is that every number is 0
        n = len(new_grid)
        m = len(new_grid[0])
        for i in range(n):
            for j, num in enumerate(new_grid[i]):
                if num == 0:
                    if (i,j) not in nodes:
                        nodes.add((i,j))
                    # Check right
                    if j < m-1 and new_grid[i][j+1] == 0:
                        graph[(i,j)].add((i, j+1))
                        graph[(i,j+1)].add((i,j))
                    # Check left
                    if j > 0 and new_grid[i][j-1] == 0:
                        graph[(i,j)].add((i, j-1))
                        graph[(i,j-1)].add((i,j))
                    # Check up
                    if i > 0 and new_grid[i-1][j] == 0:
                        graph[(i,j)].add((i-1, j))
                        graph[(i-1,j)].add((i,j))
                    # Check down
                    if i < n-1 and new_grid[i+1][j] == 0:
                        graph[(i,j)].add((i+1, j))
                        graph[(i+1,j)].add((i,j))

        explored = set()
        count = 0

        def bfs(graph, explored, start):
            queue = deque()
            queue.append(start)

            while queue:
                node = queue.popleft()
                for child_node in graph[node]:
                    if child_node not in explored:
                        explored.add(child_node)
                        queue.append(child_node)

            return 0

        # Time O(n+m) worst case is that every number is 1
        for (i,j) in nodes:
            if (i,j) not in explored:
                bfs(graph, explored, (i,j))
                count += 1

        return count
