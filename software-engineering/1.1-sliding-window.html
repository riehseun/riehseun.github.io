Longest substring without repeating characters (M) (G/A/200) !!!

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:

        # Move the end pointer until seeing repeating chars
        # Move the start pointer until substring has no repeating chars
        # Keep the max diff between end and start pointer
        # Scan finished when both start and end pointers reach the end of string

        start = 0
        max_length = 0
        char_map = {}  # Store chracter and its index

        # Time O(n) = len(s)
        # Space O(n)
        for i,c in enumerate(s):

            # If duplicate is found, compute the length
            # using start and the i pointer
            # The second condition ensures that we are only looking
            # at the substring after the start index
            if c in char_map and char_map[c] >= start:
                max_length = max(max_length, i-start)
                # Move the start pointer to plus 1
                # of the index where the duplicate is
                start = char_map[c] + 1
            char_map[c] = i

        max_length = max(max_length, len(s)-start)

        return max_length

Mimimum window substring (H) (F) !

class Solution:
    def minWindow(self, s: str, t: str) -> str:

        # Use start and end pointer to iterate s
        # Move end pointer until all chars in t are in the substring
        # If all chars in t are in the substring
        #   Move start pointer while substring is still valid and remember the substring
        #   If substring is not valid, move the endpoint

        # How to keep track whether all chars are in substring
        #   Use a hash table store the count of each char in t
        #   The substring has to have the exact count for each char in t

        start = 0
        end = 0
        min_length = len(s)
        min_window_substring = ""

        char_count_t = defaultdict(int)
        for c in t:
            char_count_t[c] += 1
        char_count_s = defaultdict(int)
        end_pointer_moved = True

        # Time O(n*m) n = len(s) m = len(t)
        # Space O(n) n = len(s)
        while start <= end and end < len(s):

            substring = s[start:end+1]
            if end_pointer_moved:
                char_count_s[s[end]] += 1
            else:
                char_count_s[s[start-1]] -= 1

            valid_substring = True
            for key,val in char_count_t.items():
                if key not in char_count_s or val > char_count_s[key]:
                    valid_substring = False

            if valid_substring:
                if len(substring) <= min_length:
                    min_window_substring = substring
                    min_length = len(substring)
                end_pointer_moved = False
                start += 1
            else:
                end_pointer_moved = True
                end += 1

        return min_window_substring

Sliding window maximum (H) (A) !

class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:

        # Move the start and end pointer by 1
        # Compute the max in the subarray and store it
        #   1. Slicing array and computing the max is O(k^2)
        #   2. Use queue to add and remove items and compute max in the queue O(k)

        # Keep the maximum number at the front of the queue (We can do this by popping out all smaller numbers than the current number)
        # First, compute the maximum number in the first window
        # Then, everytime a window moves
        #   Pop out all smaller numbers than the current number
        #   Remove previous number, which is now outside of the window, from the queue
        #   Append the first item in the queue to the result

        # Time O(n) n = len(nums)
        # Space O(k)

        result = []
        queue = deque()

        # Compute the maximum in the first window
        for i in range(k):

            # Pop out all smaller items than the current item
            # by removing last items in the queue
            # Then the first itme in the queue will always
            # contains the maximum item
            while queue and nums[queue[-1]] < nums[i]:
                queue.pop()
            queue.append(i)

        result.append(nums[queue[0]])

        # Move the window
        for i in range(k, len(nums)):

            while queue and nums[queue[-1]] < nums[i]:
                queue.pop()
            queue.append(i)

            # Remove all indexes in the queue that is before
            # and outside than the current window
            while queue and queue[0] <= (i - k):
                queue.popleft()

            result.append(nums[queue[0]])

        return result

Longest substring with at least K repeating characters (M) [divide-and-conquer]

class Solution:
    def longestSubstring(self, s: str, k: int) -> int:

        # Find characters whose occurence is less than k in s
        # Using one of the characters from above, divide the string into two substrings
        # Recurse on those two substrings

        def helper(s, k, length_of_good_substring):

            bad_chars = get_bad_chars(s, k)

            # The entire string is a good string
            if not bad_chars:
                length_of_good_substring[s] = len(s)
                return

            start = 0

            for i in range(len(s)):
                # Check previous good string when hitting a bad char
                if s[i] in bad_chars:
                    if s[start:i] not in length_of_good_substring:
                        helper(s[start:i], k, length_of_good_substring)

                    start = i + 1

            # Check previous good string at the end
            helper(s[start:len(s)], k, length_of_good_substring)

        def get_bad_chars(s, k):

            char = defaultdict(int)

            for i in s:
                char[i] += 1

            bad_chars = set()

            for key,val in char.items():
                if val < k:
                    bad_chars.add(key)

            return bad_chars

        length_of_good_substring = {}

        helper(s, k, length_of_good_substring)

        max_len = 0
        for key, val in length_of_good_substring.items():
            max_len = max(max_len, val)

        return max_len

Longest repeating character replacement (M) (G) !

class Solution:
    def characterReplacement(self, s: str, k: int) -> int:

        # Find longest substring whose chars differ by max k
        # Initialize start and end pointer
        # Move end pointer while the string is valid
        # When string is invalid, move start pointer

        # While the window is moving, remove and add char into a dictionary

        def helper(substring, k, char_count):

            max_char = ""
            max_count = 0
            for key, val in char_count.items():
                if val > max_count:
                    max_char = key
                    max_count = val

            diff_count = 0
            for key, val in char_count.items():
                if key != max_char:
                    diff_count += val

            if k >= diff_count:
                return True
            return False

        start = 0
        end = 0
        max_length = 0
        char_count = defaultdict(int)
        end_pointer_moved = True

        # Time O(nd) n = len(s) d = size of dictionary
        # Space O(d)
        while start <= end and end < len(s):

            substring = s[start:end+1]
            if end_pointer_moved:
                char_count[s[end]] += 1
            else:
                char_count[s[start-1]] -= 1

            if helper(substring, k, char_count):
                max_length = max(max_length, len(substring))
                end += 1
                end_pointer_moved = True
            else:
                start += 1
                end_pointer_moved = False

        return max_length

Max Consecutive Ones III (M) (F/A) !

class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:

        # Sliding window
        #   End pointer moves until exhausing k 0's
        #   Then, start pointer moves
        #   Keep the max distance

        start = 0
        end = 0
        num_zeros = 0
        max_dist = 0

        # Time O(n) n = len(nums)
        # Space O(1)
        while start <= end and end < len(nums):

            if nums[end] == 0:
                if num_zeros < k:
                    num_zeros += 1
                    end += 1
                else:
                    if end == start:
                        end += 1
                        start += 1
                    else:
                        if nums[start] == 0:
                            num_zeros -= 1
                        start += 1
                max_dist = max(max_dist, end-start)
            else:
                end += 1
                max_dist = max(max_dist, end-start)

        return max_dist

class Solution:
    def minSwaps(self, nums: List[int]) -> int:

        # Sliding Windows
        # Find the highest concatenation of 1s
        # Handle circular array by concatenating nums by nums
        # Initialize the first window size with the number of 1s in nums

        size = sum(nums)
        sums = sum(nums[:size])
        largest = 0
        index = 0

        nums = nums + nums
        print(nums)

        # Time O(n) n = len(nums)
        # Space O(1)
        for i in range(len(nums)-size):
            if i > 0:
                sums = sums - nums[i-1] + nums[i+size-1]
            if largest < sums:
                largest = sums
                index = i

        return size - largest
