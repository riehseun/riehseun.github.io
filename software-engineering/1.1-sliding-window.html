Longest Substring Without Repeating Characters (M)

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:

        # Sliding window
        # Increase end while there is no duplicate in substring
        # If duplicate, increase start until there is no duplciate
        # Use set to store seen numbers

        start, end = 0, 0
        seen, max_length = set(), 0

        # Time O(n)
        #   n = len(s)
        # Space O(n)
        while start <= end and end < len(s):
            
            if s[end] not in seen:
                seen.add(s[end])
                end += 1
                max_length = max(max_length, end-start)
            else:
                seen.remove(s[start])
                start += 1

        return max_length

Minimum Window Substring (H)

class Solution:
    def minWindow(self, s: str, t: str) -> str:

        # Silding window
        #   Move end pointer until substring is vaild
        #   When substring is valid, move start pointer (remember substring)
        # How to check valid substring
        #   Use dictionary to count chars that are in t and s
        #   s_chars represents deficit for each char
          
        t_chars, s_chars = defaultdict(int), defaultdict(int)
        for c in t:
            t_chars[c] += 1
        s_chars[s[0]] = 1

        start, end, min_length, min_substring = 0, 0, len(s), ""
        
        def is_vaild_substring(s_chars, t_chars):

            for key, val in t_chars.items():
                if val > s_chars[key]:
                    return False

            return True

        # Time O(nm)
        #   n = len(s)
        #   m = len(t)
        # Space O(n+m)
        while start <= end and end < len(s):

            if not is_vaild_substring(s_chars, t_chars):
                end += 1
                if end < len(s):
                    s_chars[s[end]] += 1
            else:
                length = end - start
                if length < min_length:
                    min_length = length
                    min_substring = s[start:end+1]
                    
                s_chars[s[start]] -= 1
                start += 1

        return min_substring

Max Consecutive Ones III (M)

class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:

        # Sliding window
        #   Move end pointer while num is 1 or
        #   num is 0 but num_zeros <= k
        #   If num_zeros > k, move start pointer until 
        #   subarray is valid

        start, end, num_zeros, max_dist = 0, 0, 0, 0  
        
        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while start <= end and end < len(nums):

            if nums[end] == 1:
                end += 1
            else:
                if num_zeros < k:
                    num_zeros += 1
                    end += 1
                else:
                    # If both start and end is at 0,
                    # and there is no zeros to filp,
                    # move until finding the first 1
                    if start == end:
                        end += 1
                        start += 1
                        continue

                    if nums[start] == 0:
                        num_zeros -= 1
                    start += 1
                    
            max_dist = max(max_dist, end-start)

        # max_dist = max(max_dist, end-start)
        return max_dist
