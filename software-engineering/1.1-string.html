Roman to Integer (E)

class Solution:
    def romanToInt(self, s: str) -> int:
        """
        :type s: str
        :rtype: int
        """

        # Dictionary to store symbol-value mapping
        mapping = {
            'I':1,
            'V':5,
            'X':10,
            'L':50,
            'C':100,
            'D':500,
            'M':1000, 
            "IV":4, 
            "IX":9, 
            "XL":40, 
            "XC":90, 
            "CD":400, 
            "CM":900}
        
        i = 0
        result = 0
        
        # Time O(1) 
        #   3999 is maximum for Roman numbers
        # Space O(1)
        while i < len(s):
            if i < len(s) - 1 and s[i] + s[i+1] in mapping:
                result += mapping[s[i]+s[i+1]]
                i += 2
            else:
                result += mapping[s[i]]
                i += 1

        return result

Longest Common Prefix (M)

class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:

        # Fix first string  
        #   Compare each char in first string
        #   against each char in all other strings

        i = 0
        break_other_loop = False

        # Time O(kn)
        #   k = length of first char  
        #   n = len(strs)
        # Space O(1)
        while i < len(strs[0]):

            for j in range(1, len(strs)): 
                if i >= len(strs[j]) or strs[j][i] != strs[0][i]:
                    break_other_loop = True
                    break

            if break_other_loop:
                break
            i += 1

        return strs[0][:i]

Valid Number (H)

class Solution:
    def isNumber(self, s: str) -> bool:

        # Each char must be digit, sign, exponent, dot
        # Input cannot end with a sign or an exponent
        # There must be at least one digit in the input
        # If there is a sign, it must be either
        #   First character in the input
        #   First character after an exponent
        # There cannot be more than one exponent in the input
        # If an exponent exists, a digit must have existed beforehand
        # There cannot be more than one dot in the input
        # If a dot exists, there should be no exponent beforehand
        
        digit_exist = False
        exponent_exist = False
        dot_exist = False

        # Time O(n) 
        #   n = len(s)
        # Space O(1)
        for i, c in enumerate(s):
            if c.isdigit():
                digit_exist = True
            elif c == "-" or c == "+":
                if i > 0 \
                    and (s[i-1] != "e" and s[i-1] != "E"):
                    return False
                if i == len(s) - 1:
                    return False
            elif c == "e" or c == "E":
                if exponent_exist:
                    return False
                if not digit_exist:
                    return False
                if i == len(s) - 1:
                    return False
                exponent_exist = True
            elif c == ".":
                if dot_exist:
                    return False
                if exponent_exist:
                    return False
                dot_exist = True
            else:
                return False

        if not digit_exist:
            return False
        
        return True

Text Justification (H)

class Solution:
    def fullJustify(self, words: List[str], maxWidth: int) -> List[str]:

        # Compute running sum of available spaces
        #   If length of accumulated text > available spaces left
        #   then, insert the word in the next line
        # Each word in each line must have equal space in between
        #   Number of splits = number of words - 1
        #   Number of spaces in each split = spaces // number of splits 
        #   Extra spaces = spaces % number of splits

        result = []
        text = ""
        char_count = 0
        word_array = []
        word_length = 0

        # Time O(n)
        #   n = len(words)
        # Space O(n)
        #   word_array may contain entire words in worst case
        for word in words:
            
            if len(word) + char_count <= maxWidth:
                char_count += len(word) + 1
                word_array.append(word)
                word_length += len(word)
            else:
                if len(word_array) == 1:
                    extra_spaces = maxWidth - word_length
                    text = word_array[0]
                    for _ in range(extra_spaces):
                        text += " "
                    result.append(text)
                else:
                    num_splits = len(word_array) - 1
                    num_spaces = (maxWidth-word_length) // num_splits
                    extra_spaces = (maxWidth-word_length) % num_splits

                    text = ""
                    for j, w in enumerate(word_array[:-1]):
                        text += w
                        for _ in range(num_spaces):
                            text += " "
                        if extra_spaces > 0:
                            text += " " 
                            extra_spaces -= 1
                    text += word_array[-1]
                    result.append(text)
                
                char_count = len(word) + 1
                word_array = [word]
                word_length = len(word)

        # Process remaining words in word_array
        text = ""
        for w in word_array:
            text += w
            if len(text) < maxWidth:
                text += " "
        
        for _ in range(maxWidth-len(text)):
            text += " "
        
        result.append(text)

        return result

Valid Anagram (E)

class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        
        # For s and t, store count of each char in hash table
        # Check if two hash tables are the same
        
        # Time O(max(n,m)) 
        #   n = len(s)
        #   m = len(t)
        # Space O(1)
        #   There is only 26 chars in alphabet
        count_s, count_t = defaultdict(int), defaultdict(int)
        
        for char in s:
            count_s[char] += 1
        
        for char in t:
            count_t[char] += 1        
                
        if count_s == count_t:
            return True
        
        return False

Integer to English Words (H)

class Solution:
    def numberToWords(self, num: int) -> str:

        # Sparate by ones, thousands, millions, billions

        # Corner case
        if num == 0:
            return "Zero"
        
        ones = {
            0: "Zero",
            1: "One",
            2: "Two",
            3: "Three",
            4: "Four",
            5: "Five",
            6: "Six",
            7: "Seven",
            8: "Eight",
            9: "Nine",
            10: "Ten",
            11: "Eleven",
            12: "Twelve",
            13: "Thirteen",
            14: "Fourteen",
            15: "Fifteen",
            16: "Sixteen",
            17: "Seventeen",
            18: "Eighteen",
            19: "Nineteen"
        }

        tens = {
            2: "Twenty",
            3: "Thirty",
            4: "Forty",
            5: "Fifty",
            6: "Sixty",
            7: "Seventy",
            8: "Eighty",
            9: "Ninety",
        }

        thousands = {
            0: "",
            1: "Thousand",
            2: "Million",
            3: "Billion"
        }

        total_result = ""
        count = 0

        # Time O(1)
        # Space O(1)
        while num:

            result = ""
            quotient = num % 1000
            
            flag = True
            if quotient == 0:
                flag = False
            
            remainder = num // 1000

            if quotient >= 100:
                result += ones[quotient//100] + " Hundred "
                quotient %= 100

            if 100 > quotient >= 20:
                result += tens[quotient//10] + " "
                quotient %= 10

            if 20 > quotient > 0:
                result += ones[quotient] + " "

            num = remainder
            if flag:
                result += thousands[count] + " "
            count += 1
            total_result = result + total_result

        return total_result.strip()

Valid Word Abbreviation (E)

class Solution:
    def validWordAbbreviation(self, word: str, abbr: str) -> bool:

        # Iterate word and abbr 
        #   For abbr, differentate letters and digits
        #   If numbers are found, move indices based on that number
        #   If any indices between word and abbr don't match, return False

        n = len(word)
        m = len(abbr)
        i, j = 0, 0

        # Time O(max(n,m))
        # Space O(1)
        while i < n and j < m:

            digit = ""
            while j < m and abbr[j].isdigit():
                digit += abbr[j]
                j += 1
            if digit:
                # Check if leading zeros
                if digit.startswith("0"):
                    return False
                i += int(digit)
            else:
                # Check if char matches in each index
                if word[i] != abbr[j]:
                    return False
                i += 1
                j += 1

        return i == n and j == m

Fizz Buzz (M)

class Solution:
    def fizzBuzz(self, n: int) -> List[str]:

        result = []
        for i in range(1,n+1):
            if i % 3 == 0 and i % 5 == 0:
                result.append("FizzBuzz")
            elif i % 3 == 0 and i % 5 != 0:
                result.append("Fizz")
            elif i % 3 != 0 and i % 5 == 0:
                result.append("Buzz")
            else:
                result.append(str(i))

        return result

Reorganize String (M)

class Solution:
    def reorganizeString(self, s: str) -> str:

        # Count frequency of each char in s
        # If number of most frequent char > len(s) // 2, return ""
        # Place most frequent char in every even indices
        #   If there are remaiding even indices, place any char
        #   Once even indices are all used, place chars in odd indices

        freq = defaultdict(int)
        max_count = 0
        max_char = ""

        # Time O(n)
        #   n = len(s)
        # Space O(n)
        for c in s:
            freq[c] += 1
            if freq[c] > max_count:
                max_count = freq[c]
                max_char = c

        if len(s) % 2 == 0 and max_count > len(s) // 2:
            return ""
        if len(s) % 2 != 0 and max_count > len(s) // 2 + 1:
            return ""

        result = [""] * len(s)
        i = 0

        while freq[max_char] > 0:
            result[i] = max_char
            freq[max_char] -= 1
            i += 2

        for key, val in freq.items():
            while freq[key] > 0:
                if i >= len(s):
                    i = 1
                result[i] = key
                freq[key] -= 1
                i += 2

        return "".join(result)

Custom Sort String (M)

class Solution:
    def customSortString(self, order: str, s: str) -> str:

        # For each char in s, find its ordering 
        #   Store frequencies of each char in s
        #   Go through each char in order and see if it 
        #   is in frequencies hash table

        # Time O(n) 
        #   n = len(s)
        # Space O(n)
        frequencies = defaultdict(int)
        for c in s:
            frequencies[c] += 1

        result = ""
        for c in order:
            if c in frequencies:
                result += frequencies[c] * c

        # Find char in s, that is not in order
        for key, val in frequencies.items():
            if key not in set(order):
                result += val * key 

        return result

Minimum Add to Make Parentheses Valid (M)

class Solution:
    def minAddToMakeValid(self, s: str) -> int:

        # Count number of mismatched parenthesis
        # Count number of opening and closing parenthesis separately
        # Decrease opening count when matching closing is found

        opening_count, closing_count = 0, 0

        # Time O(n)
        #   n = len(s)
        # Space O(1)
        for c in s:
            if c == "(":
                opening_count += 1
            if c == ")":
                if opening_count > 0:
                    opening_count -= 1
                else:
                    closing_count += 1

        return opening_count + closing_count

Merge Strings Alternately (E)

class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:

        # Iterate both words using two pointers

        i, j, result = 0, 0, ""
        w1, w2 = list(word1), list(word2) 

        # Time O(max(n,m))
        #   n = len(word1)
        #   m = len(word2)
        # Space O(1)
        #   Ignoring space for result

        while i < len(w1) and j < len(w2):
            result += w1[i]
            result += w2[j]
            i += 1
            j += 1

        if i < len(w1):
            result += "".join(w1[i:])

        if j < len(w2):
            result += "".join(w2[j:])

        return result
