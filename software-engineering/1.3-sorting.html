Largest Number (M)

class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        
        # Do a quicksort
        #   Pick a pivot
        #   Put elements smaller than pivot to left of pivot
        #   Put elements larger than pivot to right of pivot
        # When comparing two numbers check if "AB" > "BA"

        if sum(nums) == 0:
            return "0"

        # Time O(nlogn) 
        #   n = len(nums)
        # Space O(logn)
        #  Recursion stack
        def quicksort(nums, l, r):
            if l >= r:
                return
            pos = partition(nums, l, r)
            quicksort(nums, l, pos-1)
            quicksort(nums, pos+1, r)

        def partition(nums, l, r):

            p = random.randint(l,r)

            for i in range(l,r+1):
                if compare(nums[i],nums[p]) and i > p:
                    nums[i], nums[p] =  nums[p], nums[i]
                if not compare(nums[i],nums[p]) and i < p:
                    nums[i], nums[p] =  nums[p], nums[i]
                p = i

            return p

        def compare(x, y):
            return str(x) + str(y) > str(y) + str(x)

        quicksort(nums, 0, len(nums)-1)
        result = ""
        for num in nums:
            result += str(num)
        return result

Kth Largest Element in an Array (M)

class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:

        # Quickselect
        # Partition array such that
        #   Chooese pivot nums[p] (index p) 
        #   Nums larger than nums[p] comes left side
        #   Nums less than nums[p] comes right side
        # If k > p, search on right
        # If k < p, search on left
        # If k == p, retur nums[p]
        # k should be k - 1

        n = len(nums)

        def partition(left, right):

            p = random.randint(left, right)
            p = (left+right) // 2
            
            for i in range(left, right+1):
                if i < p and nums[i] > nums[p]:
                    nums[i], nums[p] = nums[p], nums[i]
                if i > p and nums[i] < nums[p]:
                    nums[i], nums[p] = nums[p], nums[i]
                p = i

            return p

        def quickselect(left, right):

            p = len(nums)
            while True:
                p = partition(left, right)
                
                if n - k > p:
                    left = p + 1
                elif n - k < p:
                    right = p - 1
                else:
                    return nums[p]

        return quickselect(0, n-1)

Top K Frequent Elements (M)

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:

        # Count occurance of each num
        # Quickselect based on the occurance
        # Partition array such that 
        #   Pick pivot index randomly
        #   Numbers higher than pivot value comes left
        #   Numbers lower than pivot value comes right
        # If pivot index < k, kth should be on right
        # If pivot index > k, kth should be on left 

        # Time O(n)
        #   n = len(nums)
        # Space O(n)
        #   Storing frequencies and keys
        def partition(left, right, array):
            p = random.randint(left, right)
            
            for i in range(left, right+1):
                if array[i][0] < array[p][0] and i < p:
                    array[i], array[p] = array[p], array[i] 
                if array[i][0] > array[p][0] and i > p:
                    array[i], array[p] = array[p], array[i] 
                p = i

            return p

        def quickselect(left, right, array, k):
            
            p = partition(left, right, array)

            if p < k - 1:
                return quickselect(p+1, right, array, k)
            elif p > k - 1:
                return quickselect(left, p-1, array, k)
            else:
                return array[:p+1]
        
        count = defaultdict(int)

        for num in nums:
            count[num] += 1

        array = []
        for key, val in count.items():
            array.append([val, key])

        ret_val = quickselect(0, len(array)-1, array, k)

        result = []
        for count, num in ret_val:
            result.append(num)

        return result

cclass Solution:
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:

        # Find k smallest distance
        # Find all distances
        # Partition 
        #   Pick a random pivot
        #   Distances less than pivot are on left
        #   Distances greater than pivot are on right
        # If index of pivot < k
        #   Search on right
        # If index of pivot > k
        #   Search on left

        # Time O(n)
        #   n = len(points)
        # Space O(1)
        def partition(left, right):

            p = random.randint(left, right)
            
            for i in range(left, right+1):
                dist_p = points[p][0]**2 + points[p][1]**2
                dist_i = points[i][0]**2 + points[i][1]**2
                if i < p and dist_i > dist_p:
                    points[i], points[p] = points[p], points[i]
                if i > p and dist_i < dist_p:
                    points[i], points[p] = points[p], points[i]
                p = i

            return p 

        def quickselect(left, right):

            while True:
                p = partition(left, right)

                if k - 1 > p:
                    left = p + 1
                elif k - 1 < p:
                    right = p - 1
                else:
                    return points[:p+1]

        n = len(points)
        return quickselect(0, n-1)
