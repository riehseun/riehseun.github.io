Clone Graph (M)

"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []

class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:

        # Should return the root node?
        # Does graph have cycle?
        
        # node = [] => return None
        # node = Node(1, [Node(2,[])]) => Exactly the same object

        # Brute?

        # DFS
        # Put root node into stack
        # pop from stack
        # Iterate neighbors and insert them into stack
        # Create graph of adjacency list while DFSing
        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n)
        
        def dfs(start):

            explored = set()
            explored.add(start)

            stack = []
            stack.append(start)

            graph[start] = Node(start.val)
                
            while stack:
                node = stack.pop()
                
                for next_node in node.neighbors:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append(next_node)
                        graph[next_node] = Node(next_node.val)
                    graph[node].neighbors.append(graph[next_node])

        if not node:
            return None

        graph = {}
        dfs(node)
        return graph[node]

Word Break (M)

class Solution(object):
    def wordBreak(self, s, wordDict):
        """
        :type s: str
        :type wordDict: List[str]
        :rtype: bool
        """

        # Consider indices of start and end char of string 
        # as nodes in graph
        # For example, because "leet" is in "leetcode", 0->4
        # Return True if there is path from 0 to len(s)
        
        occurences = defaultdict(set)
        
        # Time O(nml)
        #   n = len(s)
        #   m = len(wordDict)
        #   l = average length of each word in wordDict
        # Space O(m)
        for word in wordDict:
            for i in range(len(s)):
                if s[i:].startswith(word):
                    occurences[i].add(i+len(word))
            
        # Time O(m)
        # Space O(m)
        def dfs(graph, start, end):

            explored = set()
            explored.add(start)
            
            stack = []
            stack.append(start)
            
            while stack:
                node = stack.pop()
                for next_node in graph[node]:
                    if next_node == end:
                        return True
                        
                    if next_node not in explored:
                        stack.append(next_node)
                        explored.add(next_node)
                
            return False
                    
        return dfs(occurences, 0, len(s))

Alien Dictionary (H)

class Solution:
    def alienOrder(self, words: List[str]) -> str:

        # Compare two adjacent words at a time
        # Only the first difference in char tells the ordering
        #   If second word is prefix of first word, return ""
        # Construct a graph based on the ordering
        #   Letters than don't appear in the graph are not subject to
        #   ordering, thus can go anywhere in the output
        #   If there is cycle, return ""
        # Then, apply topological order
        #   Do DFS starting at each node and track finishing time
        #   Idea is that longer finishing time means earlier in the order

        # Track all chars in words
        # Time O(c) 
        #   c = all character in words
        # Space O(1) 
        #   There are only 26 chars in alphabet
        all_chars = set()
        for word in words:
            for char in word:
                all_chars.add(char)
        
        graph = defaultdict(set)

        # Time O(c)
        # Space O(c) 
        for i in range(len(words)-1):
            # If second word is prefix of first word, 
            # it is not valid ordering
            if words[i].startswith(words[i+1]) and len(words[i]) > len(words[i+1]):
                return "" 
            for c1, c2 in zip(words[i], words[i+1]):
                if c1 != c2:
                    graph[c1].add(c2)
                    break

        def dfs(graph, start):

            explored = set()
            explored.add(start)

            stack = []
            stack.append(start)
            finishing_time = 0

            while stack:
                node = stack.pop()
                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append(next_node)
                        finishing_time += 1
                    if next_node == start:
                        return -1

            return finishing_time

        ordering = []

        # Time O(n*(n+m)) = O(1)
        #   n = number of nodes = 26 (There are 26 chars in alpahbet)
        #   m = number of edges = 26**2 worse case every node is connected
        # Space O(1)
        for char in all_chars:
            finishing_time = dfs(graph, char)
            if finishing_time == -1:
                return ""
            ordering.append([char, finishing_time])

        ordering.sort(key = lambda x: (-x[1]))
        
        result = ""
        for item in ordering:
            result += item[0]
        return result
