"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []

class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:

        # Should return the root node?
        # Does graph have cycle?
        
        # node = [] => return None
        # node = Node(1, [Node(2,[])]) => Exactly the same object

        # Brute?

        # DFS
        # Put root node into stack
        # pop from stack
        # Iterate neighbors and insert them into stack
        # Create graph of adjacency list while DFSing
        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n)
        
        def dfs(start):

            explored = set()
            explored.add(start)

            stack = []
            stack.append(start)

            graph[start] = Node(start.val)
                
            while stack:
                node = stack.pop()
                
                for next_node in node.neighbors:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append(next_node)
                        graph[next_node] = Node(next_node.val)
                    graph[node].neighbors.append(graph[next_node])

        if not node:
            return None

        graph = {}
        dfs(node)
        return graph[node]

Number of Connected Components in an Undirected Graph (M)

class Solution:
    def countComponents(self, n: int, edges: List[List[int]]) -> int:
        
        # Is graph directed or undirected?

        # edges = [], n = 3  => 3
        # edges = [[0,1]], n = 3 => 2

        # DFS at each node
        #   Increment count by 1
        # Skip already explored nodes
        # Time O(n+m) n = number of nodes, m = numder of edges
        # Space O(m)

        def dfs(start):

            stack = []
            stack.append(start)

            while stack:
                node = stack.pop()

                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append(next_node)

        explored, graph = set(), defaultdict(list)
        count = 0

        for a, b in edges:
            graph[b].append(a)
            graph[a].append(b)

        for i in range(n):
            if i not in explored:
                explored.add(i)
                dfs(i)
                count += 1

        return count
