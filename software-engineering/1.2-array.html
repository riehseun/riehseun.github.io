Valid sudoku (M/A)

class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:

        n = len(board)
        seen = set()

        # Time O(1)
        # Space O(1)

        # Validate row
        for row in board:
            for num in row:
                if num.isdigit():
                    if num not in seen:
                        seen.add(num)
                    else:
                        return False
            seen = set()

        # Validate col
        for j in range(n):
            for i in range(n):
                if board[i][j].isdigit():
                    if board[i][j] not in seen:
                        seen.add(board[i][j])
                    else:
                        return False
            seen = set()

        # Validate square from each top left index
        for i in range(0, n, 3):
            for j in range(0, n, 3):
                for k in range(0,3):
                    for l in range(0,3):
                        if board[i+k][j+l].isdigit():
                            if board[i+k][j+l] not in seen:
                                seen.add(board[i+k][j+l])
                            else:
                                return False
                seen = set()

        return True

Rotate Image (M/F/A)

class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """

        # First reverse up & down
        # Second, swap index i and j

        n = len(matrix)
        start = 0
        end = n - 1

        # Time O(nm) n = number of rows, m = number of columns
        # Space O(1)
        while start < end:
            matrix[start], matrix[end] = matrix[end], matrix[start]
            start += 1
            end -= 1

        for i in range(n):
            for j in range(i+1, n):
                matrix[i][j], matrix[j][i] = matrix[j][i], matrix[i][j]

Spiral matrix (M/G/A)

class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:

        # Remove rows and columns from the matrix while moving spiral
        # The order is
        #   Remove first row
        #   Remove last col
        #   Remove last row
        #   Remove first col
        # Repeat

        result = []

        # Time O(nm) n = number of rows, m = number of columns
        # Space O(nm)

        while matrix:
            nums = matrix.pop(0)  # Remove first row
            result += nums

            if matrix and matrix[0]:
                for row in matrix:
                    num = row.pop()  # Remove last col
                    result += [num]

            if matrix:
                nums = matrix.pop()  # Remove last row
                # Ensure the order that numbers are appended to the result
                nums.reverse()
                result += nums

            if matrix and matrix[0]:
                # Ensure the order that numbers are appended to the result
                matrix_copy = matrix.copy()
                matrix_copy.reverse()
                for row in matrix_copy:
                    num = row.pop(0)  # Remove first col
                    result += [num]

        return result

Set matrix zeroes (M)

class Solution:
    def setZeroes(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """

        # First find rows and cols that needs to be set to zero

        rows = {}
        cols = {}

        # Time O(nm) where n = number of rows, m = number of cols
        # Space O(max(n,m))

        for i in range(len(matrix)):
            for j in range(len(matrix[i])):
                if matrix[i][j] == 0:
                    rows[i] = True
                    cols[j] = True

        for i in range(len(matrix)):
            for j in range(len(matrix[i])):
                if i in rows or j in cols:
                    matrix[i][j] = 0

Search a 2D Matrix (M/A)

class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:

        # Compare target with the first and last integer of middle row
        #   If target < first integer
        #     Search on upper half of rows
        #   If target > last integer
        #     Search on lower half of row
        #   If first integer <= target <= last integer
        #     Do binary search on the current row

        # Time O(log(n)*log(m)) n = number of rows, m = number of cols
        # Space O(1)
        def binary_search(sorted_array, number):

            n = len(sorted_array)
            if n == 0:
                return False
            mid = n // 2

            if number < sorted_array[mid]:
                return binary_search(sorted_array[:mid], number)
            elif number > sorted_array[mid]:
                return binary_search(sorted_array[mid+1:], number)
            else:
                return True

        n = len(matrix)

        if n == 0:
            return False

        if n == 1:
            return binary_search(matrix[0], target)

        m = len(matrix[0])
        mid = n // 2

        if target < matrix[mid][0]:
            return self.searchMatrix(matrix[:mid], target)
        elif target > matrix[mid][m-1]:
            return self.searchMatrix(matrix[mid+1:], target)
        else:
            return binary_search(matrix[mid], target)

Longest consecutive sequence (M/G/F/A)

class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:

        # Assume duplicates are not counted
        # Convert nums to set (to reduce search time to O(1))
        # For each element, check
        #   If element-1 is not in the set, then this is start of sequence
        #     Then, recursively check if element+1 is in the set

        # Time O(n) n = len(nums)
        # Space O(1)
        nums = set(nums)
        max_length = 0

        for num in nums:
            if num-1 not in nums:  # This prevents duplicate work
                current_length = 1

                while num+1 in nums:
                    current_length += 1
                    num += 1

                max_length = max(max_length, current_length)

        return max_length

Maximum product subarray (M/A)

class Solution:
    def maxProduct(self, nums: List[int]) -> int:

        # Reset the product when seeing 0 in the array
        # Otherwise, compute the product and remember the largest

        # Time O(n) n = len(nums)
        # Space O(1)
        def helper(nums):
            largest = -math.inf
            product = 1
            is_default_product = True

            for num in nums:
                if num == 0:
                    product = 1
                else:
                    product *= num
                    # For example, 1 is result of (-1)(-1)
                    if product == 1:
                        is_default_product = False
                largest = max(largest, product)

            if largest == 1 and is_default_product:
                return 0

            return largest

        a = helper(nums)
        nums.reverse()
        b = helper(nums)
        return max(a, b)

class Solution:
    def rotate(self, nums: List[int], k: int) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # New index is (i+k) & len(nums)

        # Time O(n) n = len(nums)
        # Space O(n)
        nums_copy = nums[:]
        for i, num in enumerate(nums):
            new_index = (i+k) % len(nums)
            nums[new_index] = nums_copy[i]

        return nums

Product of array except self (M/F/A)

class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:

        # Compute the running products from both sides

        # Time O(n) n = len(nums)
        # Space O(n)
        products_from_left = defaultdict(int)
        products_from_left[-1] = 1

        for i, num in enumerate(nums):
            products_from_left[i] = products_from_left[i-1] * num

        products_from_right = defaultdict(int)
        products_from_right[len(nums)] = 1

        for i in range(len(nums)-1, -1, -1):
            products_from_right[i] = products_from_right[i+1] * nums[i]

        result = []
        for i, num in enumerate(nums):
            result.append(products_from_left[i-1]*products_from_right[i+1])

        return result

        # (i, j)
        # (i+1, j) (i, j+1)
        # (i+2, j), (i+1, j+1), (i, j+2)

Continuous Subarray Sum (M/F)

class Solution:
    def checkSubarraySum(self, nums: List[int], k: int) -> bool:

        # Compute cummulative sum
        # Compute mod of those sums by k
        # If two mods have the same value,
        # sum betweem them should be divisible by k

        # Ex.
        # [23, 2, 4, 6, 7]
        # [23, 25, 29, 35, 42]
        # [5, 1, 5, 5, 0]

        mods = defaultdict(list)

        cummulative_sum = 0
        cummulative_sums = []

        # Time O(n) n = len(nums)
        # Space O(n)
        for num in nums:
            cummulative_sum += num
            cummulative_sums.append(cummulative_sum)

        for i, num in enumerate(cummulative_sums):
            mods[num%k].append(i)

        for key, val in mods.items():
            if len(val) > 2:
                return True
            if len(val) == 2:
                if abs(val[0]-val[1]) > 1:
                    return True

        if 0 in mods and mods[0] != [0]:
            return True

        return False

Subarray Sum Equals K (M/G/F/A)

class Solution:
    def subarraySum(self, nums: List[int], k: int) -> int:

        # Compute cummulative sum
        # Create hash table
        #   key = a cumulative sum
        #   value = number of times a particular cumulative sum occurs
        # If cummulative_sum_a - k = cummulative_sum_b,
        # Keep the number of times cummulative_sum_b occurs

        cummulative_sums = defaultdict(int)
        cummulative_sum = 0
        count = 0
        cummulative_sums[0] = 1

        # Time O(n) n = len(nums)
        # Space O(n)
        for num in nums:
            cummulative_sum += num
            count += cummulative_sums[cummulative_sum-k]
            cummulative_sums[cummulative_sum] += 1

        return count

Toeplitz Matrix (E/F)

class Solution:
    def isToeplitzMatrix(self, matrix: List[List[int]]) -> bool:

        # For all elements in matrix, check if its value equals to
        # its top left element (matrix[i][j] == matrix[i-1][j-1])

        # Time O(nm) n = len(rows), m = len(columns)
        # Space O(1)
        for i in range(len(matrix)):
            for j in range(len(matrix[0])):
                if i > 0 and j > 0:
                    if matrix[i][j] != matrix[i-1][j-1]:
                        return False

        return True

Rotting Oranges (M/A)

class Solution:
    def orangesRotting(self, grid: List[List[int]]) -> int:

        # Iterate over grid

        # Time O(nm) n = number of rows, m = number of columns
        # Space O(1)
        def helper(grid):

            to_be_rot = []

            for i in range(len(grid)):
                for j in range(len(grid[0])):
                    if grid[i][j] == 2:
                        if i > 0 and grid[i-1][j] == 1:
                            to_be_rot.append((i-1,j))
                        if i < len(grid)-1 and grid[i+1][j] == 1:
                            to_be_rot.append((i+1,j))
                        if j > 0 and grid[i][j-1] == 1:
                            to_be_rot.append((i,j-1))
                        if j < len(grid[0])-1 and grid[i][j+1] == 1:
                            to_be_rot.append((i,j+1))

            return to_be_rot

        count = 0
        while True:
            to_be_rot = helper(grid)
            if not to_be_rot:
                break
            for i,j in to_be_rot:
                grid[i][j] = 2
            count += 1

        # If any fresh orange is left in grid, return -1
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    return -1

        return count

Unique Number of Occurrences (E/G)

class Solution:
    def uniqueOccurrences(self, arr: List[int]) -> bool:

        # Hash table to store count

        # Time O(n) n = len(arr)
        # Space O(n)
        counts = defaultdict(int)
        for num in arr:
            counts[num] += 1

        seen = set()
        for key, val in counts.items():
            if val in seen:
                return False
            seen.add(val)

        return True

Three Consecutive Odds (E)

class Solution:
    def threeConsecutiveOdds(self, arr: List[int]) -> bool:

        # Slide 3 numbers at once

        if len(arr) < 3:
            return False

        # Time O(n) n = len(arr)
        # Space O(!)
        for i in range(len(arr)-2):
            is_consecutive_odd = True
            for num in arr[i:i+3]:
                if num % 2 == 0:
                    is_consecutive_odd = False

            if is_consecutive_odd:
                return True

        return False

Buildings With an Ocean View (M/F)

class Solution:
    def findBuildings(self, heights: List[int]) -> List[int]:

        # Find numbers that has decreasing sequence only
        # Compute running maximum from the right and compare

        max_num = -1
        result = []

        # Time O(n) n = len(heights)
        # Space O(n)
        for i in range(len(heights)-1, -1, -1):
            if heights[i] > max_num:
                result.append(i)
            max_num = max(max_num, heights[i])

        result.reverse()
        return result
