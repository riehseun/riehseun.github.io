Jump Game II (M)

class Solution:
    def jump(self, nums: List[int]) -> int:

        # Greedy
        # From each index, find farthest index it can go to
        #   Take maximun of farthest index reachable from previous indices
        #   and i + nums[i]
        # Keep index that we step on to jump (call it "curr_index")
        # If cur_index == i, take farthest jump  
        #   Also, increment jump count 
        #   Also, set cur_index to far_index

        cur_index, far_index, count = 0, 0, 0

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        for i in range(len(nums)-1):
            far_index = max(far_index, i+nums[i])
            if i == cur_index:
                count += 1
                cur_index = far_index

        return count

Jump Game (M)

class Solution:
    def canJump(self, nums: List[int]) -> bool:

        # At each index, compute the farthest index it can reach
        # Fartest index is the max of the two
        #   Fartest from previous indexes
        #   Fartest from the current index
        # If at any point, current number is 0 and farthest index of all
        # previous indices cannot reach beyoud the current number, return False

        # Time O(n)
        # Space O(1)

        if len(nums) == 1:
            return True

        farthest_index = 0
        for i in range(len(nums)-1):
            farthest_index = max(farthest_index, nums[i]+i)
            if nums[i] == 0 and farthest_index <= i:
                return False
            if farthest_index >= len(nums)-1:
                return True

        return False

Best Time to Buy and Sell Stock (E)

class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        # Compute running min
        # Find diff between price and min at each i
        # Remember the largest diff

        profit, running_min = 0, math.inf

        # Time O(n)
        #   n = len(prices)
        # Space O(1)
        for p in prices:
            running_min = min(running_min, p)
            profit = max(profit, p-running_min)

        return profit

Gas Stations (M)

class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:

        # Since there is unique solution, check from each index
        # If total cost > total gas, there is no solution
        # Find the first index where the accumulated gas is positive

        if sum(cost) > sum(gas):
            return -1

        index, accumulated_gas = 0, 0

        # Time O(n) 
        #   n = len(gas)
        # Space O(1)
        for i in range(len(gas)):
            accumulated_gas += (gas[i]-cost[i])
            if accumulated_gas < 0:
                accumulated_gas = 0
                index = i + 1

        return index

Candy (H)

class Solution:
    def candy(self, ratings: List[int]) -> int:

        # At index i,
        #   If ratings[i] > ratings[i-1], candy[i] > candy[i-1]
        #   If ratings[i] > ratings[i+1], candy[i] > candy[i+1]
        # Initialize candy array to all 1
        # Go though the candy array from left to right incrementing candies
        # Go though the candy array from right to left incrementing candies

        n = len(ratings)
        candies = [1] * n

        # Time O(n) 
        #   n = len(ratings)
        # Space O(n)
        for i in range(1, n):
            if ratings[i] > ratings[i-1]:
                candies[i] = candies[i-1] + 1

        for j in range(n-2, -1, -1):
            if ratings[j] > ratings[j+1] and candies[j] <= candies[j+1]:
                candies[j] = candies[j+1] + 1

        return sum(candies)

Maximum Swap (M)

class Solution:
    def maximumSwap(self, num: int) -> int:

        # Find largest number from right
        #   When seeing smaller number than largest,
        #   consider them as candidate for swapping 
        # Remember index of largest number
        #   When seeing a smaller number than largest,
        #   only then set the indices to swap

        nums = list(str(num))
        largest, index_1, index_2 = 0, -1, -1
        
        # Time O(n)
        #   n = len(num)
        # Space O(n)
        for i in range(len(nums)-1, -1, -1):
            n = int(nums[i])
            if n > largest:
                largest = n
                index_largest = i
            if n < largest:
                index_1 = index_largest
                index_2 = i
                
        if index_2 == -1:
            return num
        
        nums[index_2], nums[index_1] = nums[index_1], nums[index_2]
        return int("".join(nums))

Minimum Domino Rotations For Equal Row (M)

class Solution:
    def minDominoRotations(self, tops: List[int], bottoms: List[int]) -> int:

        # For each number in tops and bottoms,
        # store its indices in a hash table

        # Time O(n) 
        #   n = len(tops)
        # Space O(n)
        tops_dict, bottoms_dict = defaultdict(list), defaultdict(list)

        for i, item in enumerate(tops):
            tops_dict[item].append(i)

        for i, item in enumerate(bottoms):
            bottoms_dict[item].append(i)

        for num in range(1,7):
            if len(set(tops_dict[num]+bottoms_dict[num])) >= len(tops):
                return len(tops) - max(len(tops_dict[num]), len(bottoms_dict[num]))

        return -1
