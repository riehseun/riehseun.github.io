Reconstruct Itinerary (H)

class Solution:
    def findItinerary(self, tickets: List[List[str]]) -> List[str]:

        # DFS 
        # Sort tickets by destination 
        # Greedly traverse until no destination is left
        # Time O(nlogn) n = len(tickets)
        # Space O(n)

        graph, result = defaultdict(list), []
        tickets.sort(key=lambda x: x[1], reverse=True)
        
        for src, det in tickets:
            graph[src].append(det)

        stack = []
        stack.append("JFK")
        
        while stack:
            while graph[stack[-1]]:
                stack.append(graph[stack[-1]].pop())
            result.append(stack.pop())

        return result[::-1]

Min Cost to Connect All Points (M)

class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:

        # Points are nodes
        # Edges exists between any two points
        # Minimum spanning tree
        # Greedly explore nodes not in MST and pick lowest cost
        # Time O(n**2) n = number of nodes
        # Space O(n)

        n, m, cost = len(points), 0, 0
        min_dist = [math.inf] * n
        min_dist[0] = 0
        mst = [False] * n

        while m < n:
            min_node, min_edge = -1, math.inf

            for node in range(n):
                if not mst[node] and min_edge > min_dist[node]:
                    min_edge = min_dist[node]
                    min_node = node

            cost += min_edge
            m += 1
            mst[min_node] = True

            for next_node in range(n):
                dist = abs(points[min_node][0]-points[next_node][0]) \
                    + abs(points[min_node][1]-points[next_node][1])
                if not mst[next_node] and min_dist[next_node] > dist:
                    min_dist[next_node] = dist

        return cost
