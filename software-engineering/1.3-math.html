Reverse integer (M/100)

class Solution:
    def reverse(self, x: int) -> int:

        is_neg = False
        x = str(x)

        if x[0] == "-":
            x = x[1:]
            is_neg = True

        arr = []
        for i in x:
            arr.append(i)

        start = 0
        end = len(x)-1

        while start < end:
            temp = arr[start]
            arr[start] = arr[end]
            arr[end] = temp
            start += 1
            end -= 1

        result = "".join(arr)

        if is_neg:
            if -int(result) < -2**31:
                return 0
            else:
                return -int(result)
        else:
            if int(result) > 2**31 - 1:
                return 0
            else:
                return int(result)

Pow(x, n) (M/F1/100)

class Solution:
    def myPow(self, x: float, n: int) -> float:

        # Use recursion to compute product from 1 to n
        # Recursive relation is x^(n) = x * x^(n-1)
        # If n is negative, x^(n) = 1 / (x * x^(n-1))

        # Time O(n)
        # Space O(1)
        def pow_pos(x, n):

            # Base case
            if n == 0:
                return 1

            return x * pow(x, n-1)

        def pow_neg(x, n):

            # Base case
            if n == 0:
                return 1

            return 1 / (x*pow(x, n-1))

        if n >= 0:
            return pow_pos(x, n)
        else:
            return pow_neg(x, -n)

Add Strings (E/F1)

class Solution:
    def addStrings(self, num1: str, num2: str) -> str:

        # Interpret each digit as integer by the difference
        # in unicode representation between the digit and "0"
        # Add digit by digit while tracking carry

        result = []
        i = len(num1) - 1
        j = len(num2) - 1
        carry = 0

        # Time O(n) n = max(len(num1), len(num2))
        # Space O(n)
        while i >= 0 or j >= 0:

            if i >= 0:
                x1 = ord(num1[i]) - ord("0")
            else:
                x1 = 0
            if j >= 0:
                x2 = ord(num2[j]) - ord("0")
            else:
                x2 = 0
            value = (x1 + x2 + carry) % 10
            carry = (x1 + x2 + carry) // 10
            result.append(str(value))
            i -= 1
            j -= 1

        if carry:
            result.append(str(carry))

        return ''.join(result[::-1])

Next Closest Time (M/G1)

class Solution:
    def nextClosestTime(self, time: str) -> str:

        # Compute permutation using 4 digits
        #   Check if time is valid
        #   Then, find smallest elapsed time from the input

        input_time = [time[0], time[1], time[3], time[4]]
        times = []

        # Time O(1)
        # Space O(1)
        for h1, h2, m1,m2 in itertools.product(input_time, repeat=4):
            if int(h1+h2) > 24:
                continue
            if int(h1+h2) == 24 \
                and (m1 != 0 or m2 != 0):
                continue
            if int(m1+m2) >= 60:
                continue
            times.append(h1+h2+m1+m2)

        input_time_in_int = int(time[0]+time[1]+time[3]+time[4])
        times.sort()

        print(times)

        for t in times:
            if int(t) > input_time_in_int:
                t = str(t)
                return t[0]+t[1]+":"+t[2]+t[3]

        t = str(times[0])
        return t[0]+t[1]+":"+t[2]+t[3]
