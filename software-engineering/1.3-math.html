Reverse integer (M/G/F/A)

class Solution:
    def reverse(self, x: int) -> int:

        is_neg = False
        x = str(x)

        # Extract negative sign if negative number
        if x[0] == "-":
            x = x[1:]
            is_neg = True

        # Place each digit in array
        # Time O(n) n = len(x)
        # Space O(n)
        arr = []
        for i in x:
            arr.append(i)
        arr.reverse()
        result = "".join(arr)

        if is_neg:
            if -int(result) < -2**31:
                return 0
            else:
                return -int(result)
        else:
            if int(result) > 2**31 - 1:
                return 0
            else:
                return int(result)

Pow(x, n) (M/F/A)

class Solution:
    def myPow(self, x: float, n: int) -> float:

        # Use recursion to compute product from 1 to n
        # Recursive relation is x^(n) = x * x^(n-1)
        # If n is negative, x^(n) = 1 / (x * x^(n-1))

        # Time O(n)
        # Space O(1)
        def pow_pos(x, n):

            # Base case
            if n == 0:
                return 1

            return x * pow(x, n-1)

        def pow_neg(x, n):

            # Base case
            if n == 0:
                return 1

            return 1 / (x*pow(x, n-1))

        if n >= 0:
            return pow_pos(x, n)
        else:
            return pow_neg(x, -n)

Add Strings (E/F)

class Solution:
    def addStrings(self, num1: str, num2: str) -> str:

        # Interpret each digit as integer by the difference
        # in unicode representation between the digit and "0"
        # Add digit by digit while tracking carry

        result = []
        i = len(num1) - 1
        j = len(num2) - 1
        carry = 0

        # Time O(n) n = max(len(num1), len(num2))
        # Space O(n)
        while i >= 0 or j >= 0:

            if i >= 0:
                x1 = ord(num1[i]) - ord("0")
            else:
                x1 = 0
            if j >= 0:
                x2 = ord(num2[j]) - ord("0")
            else:
                x2 = 0
            value = (x1 + x2 + carry) % 10
            carry = (x1 + x2 + carry) // 10
            result.append(str(value))
            i -= 1
            j -= 1

        if carry:
            result.append(str(carry))

        return ''.join(result[::-1])
