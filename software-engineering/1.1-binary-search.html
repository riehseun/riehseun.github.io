Median of two sorted arrays (H) (1)

class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:

        # Binary search
        # Assume len(nums1) < len(nums2)
        # Partition nums1 and nums2 into smaller and larger halves 
        # Number of elements in combined smaller halves should
        # equal to number of elements in combined larger halves  
        # If max_left_a <= min_right_b and max_left_b <= min_right_a,
        # partition is at the median of combined nums1 and nums2
        # If max_left_a > min_right_b,
        # max_left_a should be in larger half
        # If max_left_b > min_right_a,
        # max_left_b should be in smaller half

        if len(nums1) > len(nums2):
            return self.findMedianSortedArrays(nums2, nums1)
        
        n = len(nums1)
        m = len(nums2)

        left = 0
        right = n

        # Time O(log(min(n,m)))
        #   n = len(nums1)
        #   m = len(nums2)
        # Space O(1)
        while left <= right:
            partition_a = (left+right) // 2
            partition_b = (n+m+1) // 2 - partition_a

            if partition_a == 0:
                max_left_a = -math.inf
            else:
                max_left_a = nums1[partition_a-1]

            if partition_a == n:
                min_right_a = math.inf
            else:
                min_right_a = nums1[partition_a]

            if partition_b == 0:
                max_left_b = -math.inf
            else:
                max_left_b = nums2[partition_b-1]

            if partition_b == m:
                min_right_b = math.inf
            else:
                min_right_b = nums2[partition_b]

            if max_left_a <= min_right_b and max_left_b <= min_right_a:
                if (n+m) % 2 == 0:
                    return (
                        max(max_left_a, max_left_b) 
                        + min(min_right_a, min_right_b)
                    ) / 2
                else:
                    return max(max_left_a, max_left_b)
            if max_left_a > min_right_b:
                right = partition_a - 1
            if max_left_b > min_right_a:
                left = partition_a + 1

Find First and Last Position of Element in Sorted Array (M) (F) #

class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:

        # Binary search

        left = 0
        right = len(nums) - 1

        # Time O(logn) 
        #   n = len(nums)
        # Space O(1)
        while left <= right:
            mid = (left + right) // 2
            if target > nums[mid]:
                left = mid + 1
            elif target < nums[mid]:
                right = mid - 1
            else:
                start = mid
                while nums[mid] == nums[start]:
                    start -= 1
                    if start < 0:
                        break
                start += 1

                end = mid
                while nums[mid] == nums[end]:
                    end += 1
                    if end >= len(nums):
                        break
                end -= 1

                return [start, end]

        return [-1, -1]

Sqrt(x) (E) (G) !

class Solution:
    def mySqrt(self, x: int) -> int:

        # Binary search

        if x <= 1:
            return x

        low = 1
        high = x

        # Time O(logn)
        # Space O(1)
        while low <= high:
            mid = (low + high) // 2

            if mid * mid < x:
                low = mid + 1
            elif mid * mid > x:
                high = mid - 1
            else:
                return mid

        return high

Find peak element (M) (1/Meta)

class Solution:
    def findPeakElement(self, nums: List[int]) -> int:

        # Binary search trying to find local max
        # If nums[mid] < nums[mid+1]
        #   Numbers at the right half will either decrease
        #   or hit the end which is -math.inf
        #   There must be a peak in right half
        # If nums[mid] > nums[mid+1]
        #   Numbers at the left half will either decrease
        #   or hit the end which is -math.inf
        #   There must be a peak in left half

        left = 0
        right = len(nums) - 1

        # Time O(logn)
        # Space O(1)
        while left < right:
            mid = (left + right) // 2
            if nums[mid] > nums[mid+1]:
                right = mid
            else:
                left = mid + 1

        return left

Kth smallest element in a sorted matrix (M) (F) #

class Solution:
    def kthSmallest(self, matrix: List[List[int]], k: int) -> int:

        # Binary search
        # Min number is top-left and max number is bottom-right
        # Compute mid = min + (max - min) / 2
        # Count numbers <= mid
        # Track smallest number > mid and biggest numbers <= mid
        # If count == k, return count
        # Elif count < k, search from right of smallest number
        # Elif count > k, search from left of biggest number 

        n = len(matrix)
        left = matrix[0][0]
        right = matrix[n-1][n-1]

        # Time O(nlog(max-min))
        #   n = len(matrix) = len(matrix[0])
        # Space O(1)
        while left < right:
            mid = left + (right-left) / 2

            count = 0
            biggest = matrix[0][0] 
            smallest = matrix[n-1][n-1]

            i = n - 1
            j = 0

            # Time O(n)
            # Space O(1)
            while i >= 0 and j < n:
                if matrix[i][j] <= mid:
                    count += i + 1
                    biggest = max(biggest, matrix[i][j])
                    j += 1
                else:
                    smallest = min(smallest, matrix[i][j])
                    i -= 1

            if count == k:
                return biggest
            elif count < k:
                left = smallest
            else:
                right = biggest

        return left

Find K closest elements (M) (Meta/Doordash)

class Solution:
    def findClosestElements(self, nums: List[int], k: int, num: int) -> List[int]:

        # Because nums is sorted, closest items will be in
        # consecutive array
        # Use binary search to find closest item to num
        # Then, expand to left and right till k items are collected

        # Time O(logn+k)
        #   n = len(nums)
        # Space O(1)

        def binary_search(nums, left, right, target):

            while left <= right:
                mid = (left + right) // 2
                if target > nums[mid]:
                    left = mid + 1
                elif target < nums[mid]:
                    right = mid - 1
                else:
                    return mid
                
            return left

        index = binary_search(nums, 0, len(nums)-1, num)
        
        left = index - 1
        right = index

        while right - left - 1 < k:
            # check for out of bounds
            if left == -1:
                right += 1
                continue

            if right == len(nums) \
                or abs(nums[left]-num) <= abs(nums[right]-num):
                left -= 1
            else:
                right += 1

        return nums[left+1:right]

Koko Eating Bananas (M) (1/Doordash)

class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:

        # Think about min and max range of k
        #   Min is 1
        #   Max is max(piles)
        # Binary search
        #   If can't finish eating before h, try higher
        #   Try lower

        left = 1
        right = max(piles)

        # Time O(nlogn)
        #   n = len(piles)
        # Space O(1)
        while left <= right:

            mid = (left+right) // 2
            
            # Calculate number of hours needed 
            # to finish eating with speed "mid"
            h_prime = 0
            for pile in piles:
                quotient = pile // mid
                remainer = pile % mid
                if remainer == 0:
                    h_prime += quotient
                else:
                    h_prime += quotient + 1

            print(f"{mid}, {h_prime}")

            # Eating too fast, need to eat slower
            if h > h_prime:
                right = mid - 1
            # Easting too slow, need to eat faster
            elif h < h_prime:
                left = mid + 1
            # When eating speed is good enough, try lower speed
            else:
                right = mid - 1
                
        return left

Kth Missing Positive Number (E) (F) #

class Solution:
    def findKthPositive(self, arr: List[int], k: int) -> int:

        # Binary search
        # If number of missing positive integers < k
        # search on right (arr[mid] - mid - 1 < k)
        # Else search on left  

        left = 0
        right = len(arr) - 1

        # Time O(logn)
        #   n = len(arr)
        # Space O(1)
        while left <= right:
            mid = (left + right) // 2
            if arr[mid] - mid - 1 < k:
                left = mid + 1 
            else:
                right = mid - 1

        return left + k
