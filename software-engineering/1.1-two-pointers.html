Longest palindromic substring (M)

class Solution:
    def longestPalindrome(self, s: str) -> str:

        # At each index, look to both left and right to
        # compute the length of longest parlindrome

        max_len = 0
        longest_parlindrome = ""

        # Time O(n**2) 
        #   n = len(s)
        # Space O(1)
        for i in range(len(s)):

            # j tracks the left of i
            j = i
            while j > 0 and s[j] == s[j-1]:
                j -= 1

            # k tracks the right of i
            k = i
            while k < len(s) - 1 and s[k] == s[k+1]:
                k += 1

            while j > 0 and k < len(s) - 1 and s[j-1] == s[k+1]:
                j -= 1
                k += 1

            if max_len < len(s[j:k+1]):
                max_len = len(s[j:k+1])
                longest_parlindrome = s[j:k+1]

        return longest_parlindrome

Palindrome number (E)

class Solution:
    def isPalindrome(self, x: int) -> bool:

        # Create revered right half 
        # Ex. 1221
        # 1221 % 10 = 1
        # 122 % 10 = 2
        # Create left half

        # Edge cases
        if x < 0 or (x % 10 == 0 and x != 0):
            return False

        left = x
        right = 0

        # Time O(logn)
        #   n = len(x)
        #   Divide input by 10 at every iteration
        # Space O(1)
        while left > right:
            right = right * 10 + left % 10
            left = left // 10

        if left == right or left == right // 10:
            return True
        
        return False

Container with most water (M)

class Solution:
    def maxArea(self, height: List[int]) -> int:

        # Initialize two pointers at each end
        # Move two pointers towards middle one by one
        # The bar with higher height will contain more water

        max_area = 0
        start = 0
        end = len(height) - 1

        # Time O(n) n = len(height)
        # Space O(1)
        while start < end:
            width = end - start
            length = min(height[start], height[end])
            area = width * length
            max_area = max(max_area, area)

            # Max area can only be constructed by keeping higher height
            if height[start] > height[end]:
                end -= 1
            else:
                start += 1

        return max_area

3Sum (M)

class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        
        # First sort the array
        # Initialize two pointers at each end 
        #   If sum is less than zero, increment start
        #   Else, decrement end
        # Do this for each element of the array

        result = []
        nums.sort()
        print(nums)

        # Time O(n^2) 
        #   n = len(nums)
        # Space O(logn)
        #   Assuming quicksort where recursion stack costs logn
        #   Ignoring space for output 
        for i, num in enumerate(nums):

            # If current value > 0, the sum can never be 0
            if num > 0:
                break
            # Avoid computing starting from the same number
            if i != 0 and nums[i-1] == nums[i]:
                continue

            # There should be no two same items that 
            # just have different ordering
            start = i + 1
            end = len(nums) - 1

            while start < end:
                three_sum = num + nums[start] + nums[end]
                if three_sum == 0:
                    candidate = [num, nums[start], nums[end]]
                    result.append(candidate)
                    # Can move both start and end
                    start += 1
                    end -= 1
                    # Avoid computing starting from the same number
                    while start < end and nums[start] == nums[start-1]:
                        start += 1
                elif three_sum < 0:
                    start += 1
                else:
                    end -= 1

        return result

3Sum closest (M)

class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:

        sorted_nums = sorted(nums)
        result = sum(sorted_nums[0:3])
        min_diff = 13000

        # Time O(n^2) n = len(nums)
        # Space O(?)
        for i, num in enumerate(sorted_nums):

            # Make sure there is no two items that have different
            # ordering of same numbers
            start = i + 1  # No need to check numbers before i again
            end = len(sorted_nums) - 1

            while start < end:
                three_sum = num + sorted_nums[start] + sorted_nums[end]
                diff = three_sum-target
                # In this case, three_sum must increase
                # because it is less than target
                if diff < 0:
                    if abs(diff) < min_diff:
                        min_diff = abs(diff)
                        result = three_sum
                    start += 1
                else:
                    if abs(diff) < min_diff:
                        min_diff = abs(diff)
                        result = three_sum
                    end -= 1

        return result

4Sum (M)

class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:

        # Reduce 4sum problem to 2sum problems
        #   Fix the first two numbers
        #   Then, find the combination of remaning two numbers that addes up to the target minus the first two numbers
        #     Sort the array
        #     Initialize start and end pointer at each end
        #     If sum is less than new target
        #       Increase start
        #     Else
        #       Decrement end

        new_nums = sorted(nums)
        result = []

        # Time O(n^3) n = len(nums)
        # Space O(?)
        for i in range(len(new_nums)):
            for j in range(i+1, len(new_nums)):
                new_target = target - (new_nums[i]+new_nums[j])
                start = j + 1
                end = len(new_nums) - 1

                while start < end:
                    if new_nums[start] + new_nums[end] == new_target:
                        candidate = [new_nums[i], new_nums[j], new_nums[start], new_nums[end]]
                        if candidate not in result:
                            result.append(candidate)
                        start += 1
                    elif new_nums[start] + new_nums[end] < new_target:
                        start += 1
                    else:
                        end -= 1

        return result

Trapping rain water (H)

class Solution:
    def trap(self, height: List[int]) -> int:

        # Water that bar i can contain is
        # min(
        #    max height from left side, max height from right side)
        #    - height[i]
        # )
        # Initialize start and end pointer at each end
        # Memorize maximum from left and right 
        # while coming towards the middle

        water = 0
        start = 0
        end = len(height) - 1
        max_from_left = height[start]
        max_from_right = height[end]

        # Time O(n) 
        #.  n = len(height)
        # Space O(1)
        while start < end:

            max_from_left = max(max_from_left, height[start])
            max_from_right = max(max_from_right, height[end])
            if max_from_left < max_from_right:
                if height[start] < max_from_left:
                    water += (max_from_left-height[start])
                start += 1
            else:
                if height[end] < max_from_right:
                    water += (max_from_right-height[end])
                end -= 1

        return water

Sort colors (M)

class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Initialize three pointers
        #   low pointer at 0
        #   mid pointer at 0
        #   high pointer at len(nums)-1
        # Move mid pointer such that
        #   all 0s are before low pointer
        #   all 2s are after high pointer
        #   all 1s are between low and high pointer

        low = 0
        high = len(nums) - 1
        mid = 0

        # Time O(n) 
        #   n = len(nums)
        # Space O(1)
        while mid <= high:
            # Move 0 to the beginning and move low pointer
            if nums[mid] == 0:
                nums[low], nums[mid] = nums[mid], nums[low]
                low += 1
                mid += 1
            # Move 2 to the end
            elif nums[mid] == 2:
                nums[high], nums[mid] = nums[mid], nums[high]
                high -= 1
            else:
                mid += 1

Merge sorted array (E)

class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """

        # Compare numbers from nums1 and nums2 starting from the end
        # Pick larger number and place it in nums1 from the end 

        p1 = m - 1
        p2 = n - 1

        # Time O(n+m)
        # Space O(1)
        for i in range(n+m-1, -1, -1):

            if p2 < 0:
                break

            if p1 >= 0 and nums1[p1] > nums2[p2]:
                nums1[i] = nums1[p1]
                p1 -= 1
            else:
                nums1[i] = nums2[p2]
                p2 -= 1

        return nums1

class Solution:
    def reverseWords(self, s: str) -> str:

        # Convert multiple whitespaces to a single whitespace
        s = " ".join(s.split())  # Time O(n)
        s = s.split(" ")  # Time O(n)

        start = 0
        end = len(s) - 1

        # Time O(n) n = len(s)
        # Space P(1)
        while start < end:
            s[end], s[start] = s[start], s[end]
            start += 1
            end -= 1

        return " ".join(s)

Two sum II - input array is sorted (M)

class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:

        # Initialize two pointers at each end
        # If sum is greater than target, decrease end
        # If sum is less than target, increase start

        start = 0
        end = len(numbers) - 1

        # Time O(n) n = len(numbers)
        # Space O(1)
        while start < end:
            if numbers[start] + numbers[end] == target:
                return [start+1, end+1]
            elif numbers[start] + numbers[end] > target:
                end -= 1
            else:
                start += 1

        return []

Move zeroes (E)

class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Two pointers
        # Move i, j forward one by one
        #   i stops moving when seeing 0
        # When j is at non-zero and i is at zero, swap them

        i = 0
        j = 0

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while j < len(nums):
            if nums[j] != 0 and nums[i] == 0:
                nums[i], nums[j] = nums[j], nums[i]

            if nums[i] != 0:
                i += 1
            
            j += 1

Is Subsequence (E)

class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:

        # Two pointers
        # Iterate s and t at the same time
        # If s[i] == t[j]
        #   Move both i and j
        # Elif s[i] != t[j]
        #   Move j
        # At the end, if i != len(s)-1, return False

        i = 0
        j = 0

        # Time O(n)
        #   n = len(t)
        # Space O(1)
        while i < len(s) and j < len(t):
            if s[i] == t[j]:
                i += 1
                j += 1
            else:
                j += 1

        if i != len(s):
            return False
        
        return True

Sum of Square Numbers (M)

class Solution:
    def judgeSquareSum(self, c: int) -> bool:

        # Two pointers

        start = 0
        end = int(sqrt(c))

        # Time O(sqrt(c))
        # Space O(1)
        while start <= end:

            val = start ** 2 + end ** 2
            if val < c:
                start += 1
            elif val > c:
                end -= 1
            else:
                return True

        return False

class Solution:
    def countSubstrings(self, s: str) -> int:

        # For every center, expand substring until
        # getting palindrome
        # Check odd and even length separately

        count = 0

        # Time O(n**2) 
        #   n = len(s)
        # Space O(1)
        for i in range(len(s)):
            count += 1
            j = i
            k = i
            while j > 0 and k < len(s) - 1 and s[j-1] == s[k+1]:
                count += 1
                j -= 1
                k += 1

        for i in range(len(s)-1):
            if s[i] == s[i+1]:
                count += 1
                j = i
                k = i + 1
                while j > 0 and k < len(s) - 1 and s[j-1] == s[k+1]:
                    count += 1
                    j -= 1
                    k += 1

        return count

Valid palindrome II (E) (Meta)

class Solution:
    def validPalindrome(self, s: str) -> bool:

        # Two pointers
        # Iterate from both ends until there is mismatch
        # If there is mismatch
        #   Remove either start or end, then check again

        # Time O(n)
        #   n = len(s)
        # Space O(1)
        def valid_palindrome(s):

            start = 0
            end = len(s) - 1

            while start < end:
                if s[start] != s[end]:
                    return (start, end)
                start += 1
                end -= 1
            
            return (-1, -1)

        start, end = valid_palindrome(s)
        if (start, end) == (-1, -1):
            return True
        else:
            if valid_palindrome(s[:start]+s[start+1:]) == (-1, -1):
                return True
            if valid_palindrome(s[:end]+s[end+1:]) == (-1, -1):
                return True
            return False

Long pressed name (E)

class Solution:
    def isLongPressedName(self, name: str, typed: str) -> bool:

        # Use pointer i for name
        # Use pointer j for typed
        # If name[i] == typed[j]
        #   If name[i+1] == typed[j+1]:
        #     Increment both i and j
        #   Else
        #     if typed[j] == typed[j+1]:
        #       Increment j
        #     Else
        #       Increment both i and j
        # If name[i] != typed[j], return False

        i = 0
        j = 0

        # Time O(max(n,m)) n = len(name) m = len(typed)
        # Space O(1)
        while i < len(name) - 1 and j < len(typed) - 1:

            if name[i] == typed[j]:
                if name[i+1] == typed[j+1] or typed[j] != typed[j+1]:
                    i += 1
                    j += 1
                else:
                    j += 1
            else:
                return False

        # If index i has not reached the end of name
        if i < len(name) - 1:
            return False

        # Check the last char of name
        if name[i] != typed[j]:
            return False

        # The remaining chars in typed from index j must be the same chars
        while j < len(typed) - 1:
            if typed[j] != typed[j+1]:
                return False
            j += 1

        return True

Dot Product of Two Sparse Vectors (M)

class SparseVector:

    # Don't store 0 to deal with sparsity
    # Represent sparse array as key-value pair

    # Time O(n) to create array O(l) to compute dot product
    #   n = len(nums)
    #   l = number of non-zero elements
    # Space O(l)
    def __init__(self, nums: List[int]):
        self.pairs = [] 
        for i, num in enumerate(nums):
            if num != 0:
                self.pairs.append((i,num))

    # Return the dotProduct of two sparse vectors
    def dotProduct(self, vec: 'SparseVector') -> int:

        # Multiply only when the indices are the same
        result, i, j = 0, 0, 0

        while i < len(self.pairs) and j < len(vec.pairs):

            if self.pairs[i][0] == vec.pairs[j][0]:
                result += (self.pairs[i][1]*vec.pairs[j][1])
                i += 1
                j += 1
            elif self.pairs[i][0] < vec.pairs[j][0]:
                i += 1
            else:
                j += 1

        return result
