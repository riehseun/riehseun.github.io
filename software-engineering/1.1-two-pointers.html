Longest palindromic substring (M) (Amazon)

class Solution:
    def longestPalindrome(self, s: str) -> str:

        # Iterate string s
        # At each index, expand to left and right to find
        # the longest palindrom substring
        #   If nearby char is the same, expand the middle first
        
        max_length = 0
        max_substring = ""

        # Time O(n**2)
        #   n = len(s)
        # Space O(1)
        for i in range(len(s)):

            j, k = i, i

            while k < len(s) and s[i] == s[k]:
                k += 1

            while j >= 0 and s[i] == s[j]:
                j -= 1
        
            while j >= 0 and k < len(s):

                if s[j] == s[k]:
                    j -= 1
                    k += 1
                else:
                    break 

            if max_length < len(s[j+1:k]):
                max_length = len(s[j+1:k])
                max_substring = s[j+1:k]

        return max_substring

Palindrome number (E)

class Solution:
    def isPalindrome(self, x: int) -> bool:

        # Create revered right half 
        # Ex. 1221
        # 1221 % 10 = 1
        # 122 % 10 = 2
        # Create left half

        # Edge cases
        if x < 0 or (x % 10 == 0 and x != 0):
            return False

        left = x
        right = 0

        # Time O(logn)
        #   n = len(x)
        #   Divide input by 10 at every iteration
        # Space O(1)
        while left > right:
            right = right * 10 + left % 10
            left = left // 10

        if left == right or left == right // 10:
            return True
        
        return False

Container with most water (M) (Google)

class Solution:
    def maxArea(self, height: List[int]) -> int:

        # Iterate from both ends
        # The area is length * height 
        #   height = lower height of left and right pillar
        #   length = distance between two pillars

        start, end, max_area = 0, len(height) - 1, 0

        # Time O(n)
        #   n = len(height)
        # Space O(1)
        while start < end:

            h = min(height[start], height[end])
            l = end - start
            max_area = max(max_area, h*l)

            if height[start] < height[end]:
                start += 1
            else:
                end -= 1

        return max_area

3Sum (M)

class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        
        # First sort the array
        # Initialize two pointers at each end 
        #   If sum is less than zero, increment start
        #   Else, decrement end
        # Do this for each element of the array

        result = []
        nums.sort()
        print(nums)

        # Time O(n^2) 
        #   n = len(nums)
        # Space O(logn)
        #   Assuming quicksort where recursion stack costs logn
        #   Ignoring space for output 
        for i, num in enumerate(nums):

            # If current value > 0, the sum can never be 0
            if num > 0:
                break
            # Avoid computing starting from the same number
            if i != 0 and nums[i-1] == nums[i]:
                continue

            # There should be no two same items that 
            # just have different ordering
            start = i + 1
            end = len(nums) - 1

            while start < end:
                three_sum = num + nums[start] + nums[end]
                if three_sum == 0:
                    candidate = [num, nums[start], nums[end]]
                    result.append(candidate)
                    # Can move both start and end
                    start += 1
                    end -= 1
                    # Avoid computing starting from the same number
                    while start < end and nums[start] == nums[start-1]:
                        start += 1
                elif three_sum < 0:
                    start += 1
                else:
                    end -= 1

        return result

3Sum closest (M)

class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:

        sorted_nums = sorted(nums)
        result = sum(sorted_nums[0:3])
        min_diff = 13000

        # Time O(n^2) n = len(nums)
        # Space O(?)
        for i, num in enumerate(sorted_nums):

            # Make sure there is no two items that have different
            # ordering of same numbers
            start = i + 1  # No need to check numbers before i again
            end = len(sorted_nums) - 1

            while start < end:
                three_sum = num + sorted_nums[start] + sorted_nums[end]
                diff = three_sum-target
                # In this case, three_sum must increase
                # because it is less than target
                if diff < 0:
                    if abs(diff) < min_diff:
                        min_diff = abs(diff)
                        result = three_sum
                    start += 1
                else:
                    if abs(diff) < min_diff:
                        min_diff = abs(diff)
                        result = three_sum
                    end -= 1

        return result

4Sum (M)

class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:

        # Reduce 4sum problem to 2sum problems
        #   Fix the first two numbers
        #   Then, find the combination of remaning two numbers that addes up to the target minus the first two numbers
        #     Sort the array
        #     Initialize start and end pointer at each end
        #     If sum is less than new target
        #       Increase start
        #     Else
        #       Decrement end

        new_nums = sorted(nums)
        result = []

        # Time O(n^3) n = len(nums)
        # Space O(?)
        for i in range(len(new_nums)):
            for j in range(i+1, len(new_nums)):
                new_target = target - (new_nums[i]+new_nums[j])
                start = j + 1
                end = len(new_nums) - 1

                while start < end:
                    if new_nums[start] + new_nums[end] == new_target:
                        candidate = [new_nums[i], new_nums[j], new_nums[start], new_nums[end]]
                        if candidate not in result:
                            result.append(candidate)
                        start += 1
                    elif new_nums[start] + new_nums[end] < new_target:
                        start += 1
                    else:
                        end -= 1

        return result

Trapping rain water (H) (Amazon)

class Solution:
    def trap(self, height: List[int]) -> int:

        # Consider how much water each index can hold
        #   min(max_height_left_of_index, max_height_right_of_index) 
        #   - its_height
        # Start from both end
        #   Remember the max heights while coming towards the middle
        #   If max_height_right > max_height_left, move start
        #   Else, move end
        
        i, j = 0, len(height) - 1
        water, max_left_height, max_right_height = 0, 0, 0

        # Time O(n)
        #   n = len(height)
        # Space O(1)
        while i <= j:

            if max_right_height > max_left_height:
                water += max(min(max_left_height, max_right_height)-height[i], 0)
                max_left_height = max(max_left_height, height[i])
                i += 1
            else:
                water += max(min(max_left_height, max_right_height)-height[j], 0)
                max_right_height = max(max_right_height, height[j])
                j -= 1

        return water

Sort colors (M)

class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Initialize three pointers
        #   low pointer at 0
        #   mid pointer at 0
        #   high pointer at len(nums)-1
        # Move mid pointer such that
        #   all 0s are before low pointer
        #   all 2s are after high pointer
        #   all 1s are between low and high pointer

        low = 0
        high = len(nums) - 1
        mid = 0

        # Time O(n) 
        #   n = len(nums)
        # Space O(1)
        while mid <= high:
            # Move 0 to the beginning and move low pointer
            if nums[mid] == 0:
                nums[low], nums[mid] = nums[mid], nums[low]
                low += 1
                mid += 1
            # Move 2 to the end
            elif nums[mid] == 2:
                nums[high], nums[mid] = nums[mid], nums[high]
                high -= 1
            else:
                mid += 1

                i += 1

Merge sorted array (E) (Meta)

class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """

        # Iterate nums1 and nums2 from the right
        #   Compare and put bigger number at the end of nums1

        if not nums2:
            return

        i, j, k = m - 1, n - 1, m + n - 1

        # Time O(n+m)
        # Space O(1)
        while k >= 0:

            if j < 0:
                nums1[:k+1] = nums1[:i+1]
                return

            if i < 0:
                nums1[:k+1] = nums2[:j+1]
                return

            if nums1[i] > nums2[j]:
                nums1[k] = nums1[i]
                i -= 1

            else:
                nums1[k] = nums2[j]
                j -= 1
            
            k -= 1

Valid palindrome (E) (Meta)

class Solution:
    def isPalindrome(self, s: str) -> bool:

        # Two pointers
        # Skip pointer is char is not alphanumeric

        left = 0
        right = len(s) - 1

        # Time O(n)
        #   n = len(s)
        # Space O(1)
        while left < right:
            while not s[left].isalnum() and left < right:
                left += 1
            while not s[right].isalnum() and left < right:
                right -= 1

            if s[left].lower() == s[right].lower():
                left += 1
                right -= 1
            else:
                return False

        return True

Reverse words in a string (M)

class Solution:
    def reverseWords(self, s: str) -> str:

        # Convert multiple whitespaces to a single whitespace
        s = " ".join(s.split())  # Time O(n)
        s = s.split(" ")  # Time O(n)

        start = 0
        end = len(s) - 1

        # Time O(n) n = len(s)
        # Space P(1)
        while start < end:
            s[end], s[start] = s[start], s[end]
            start += 1
            end -= 1

        return " ".join(s)

Two sum II - input array is sorted (M)

class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:

        # Initialize two pointers at each end
        # If sum is greater than target, decrease end
        # If sum is less than target, increase start

        start = 0
        end = len(numbers) - 1

        # Time O(n) n = len(numbers)
        # Space O(1)
        while start < end:
            if numbers[start] + numbers[end] == target:
                return [start+1, end+1]
            elif numbers[start] + numbers[end] > target:
                end -= 1
            else:
                start += 1

        return []

Move zeroes (E)

class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Two pointers
        # Move i, j forward one by one
        #   i stops moving when seeing 0
        # When j is at non-zero and i is at zero, swap them

        i = 0
        j = 0

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while j < len(nums):
            if nums[j] != 0 and nums[i] == 0:
                nums[i], nums[j] = nums[j], nums[i]

            if nums[i] != 0:
                i += 1
            
            j += 1

Is Subsequence (E)

class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:

        # Two pointers
        # Iterate s and t at the same time
        # If s[i] == t[j]
        #   Move both i and j
        # Elif s[i] != t[j]
        #   Move j
        # At the end, if i != len(s)-1, return False

        i = 0
        j = 0

        # Time O(n)
        #   n = len(t)
        # Space O(1)
        while i < len(s) and j < len(t):
            if s[i] == t[j]:
                i += 1
                j += 1
            else:
                j += 1

        if i != len(s):
            return False
        
        return True

Sum of Square Numbers (M)

class Solution:
    def judgeSquareSum(self, c: int) -> bool:

        # Two pointers

        start = 0
        end = int(sqrt(c))

        # Time O(sqrt(c))
        # Space O(1)
        while start <= end:

            val = start ** 2 + end ** 2
            if val < c:
                start += 1
            elif val > c:
                end -= 1
            else:
                return True

        return False

Palindromic Substring (M)

class Solution:
    def countSubstrings(self, s: str) -> int:

        # For every center, expand substring until
        # getting palindrome
        # Check odd and even length separately

        count = 0

        # Time O(n**2) 
        #   n = len(s)
        # Space O(1)
        for i in range(len(s)):
            count += 1
            j = i
            k = i
            while j > 0 and k < len(s) - 1 and s[j-1] == s[k+1]:
                count += 1
                j -= 1
                k += 1

        for i in range(len(s)-1):
            if s[i] == s[i+1]:
                count += 1
                j = i
                k = i + 1
                while j > 0 and k < len(s) - 1 and s[j-1] == s[k+1]:
                    count += 1
                    j -= 1
                    k += 1

        return count

Valid palindrome II (E) (Meta)

class Solution:
    def validPalindrome(self, s: str) -> bool:

        # Two pointers
        # Iterate from both ends until there is mismatch
        # If there is mismatch
        #   Remove either start or end, then check again

        # Time O(n)
        #   n = len(s)
        # Space O(1)
        def valid_palindrome(s):

            start = 0
            end = len(s) - 1

            while start < end:
                if s[start] != s[end]:
                    return (start, end)
                start += 1
                end -= 1
            
            return (-1, -1)

        start, end = valid_palindrome(s)
        if (start, end) == (-1, -1):
            return True
        else:
            if valid_palindrome(s[:start]+s[start+1:]) == (-1, -1):
                return True
            if valid_palindrome(s[:end]+s[end+1:]) == (-1, -1):
                return True
            return False

Long pressed name (E)

class Solution:
    def isLongPressedName(self, name: str, typed: str) -> bool:

        # Use pointer i for name
        # Use pointer j for typed
        # If name[i] == typed[j]
        #   If name[i+1] == typed[j+1]:
        #     Increment both i and j
        #   Else
        #     if typed[j] == typed[j+1]:
        #       Increment j
        #     Else
        #       Increment both i and j
        # If name[i] != typed[j], return False

        i = 0
        j = 0

        # Time O(max(n,m)) n = len(name) m = len(typed)
        # Space O(1)
        while i < len(name) - 1 and j < len(typed) - 1:

            if name[i] == typed[j]:
                if name[i+1] == typed[j+1] or typed[j] != typed[j+1]:
                    i += 1
                    j += 1
                else:
                    j += 1
            else:
                return False

        # If index i has not reached the end of name
        if i < len(name) - 1:
            return False

        # Check the last char of name
        if name[i] != typed[j]:
            return False

        # The remaining chars in typed from index j must be the same chars
        while j < len(typed) - 1:
            if typed[j] != typed[j+1]:
                return False
            j += 1

        return True
