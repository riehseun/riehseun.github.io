Jump Game II (M) (G/A)

class Solution:
    def jump(self, nums: List[int]) -> int:

        # Compute farthest jump in each interation
        # and update current jump to farthest jump when needed

        # Time O(n)
        # Space O(1)

        farthest_jump = 0  # Farthest index we can reach
        current_jump = 0  # End index of current jump
        num_jumps = 0

        for i in range(len(nums)-1):
            farthest_jump = max(farthest_jump, nums[i]+i)
            if i == current_jump:
                current_jump = farthest_jump
                num_jumps += 1

        return num_jumps

Jump Game II (M) (G/A)

class Solution:
    def canJump(self, nums: List[int]) -> bool:

        # At each index, compute the farthest index it can reach
        # Fartest index is the max of the two
        #   Fartest from previous indexes
        #   Fartest from the current index
        # If at any point, current number is 0 and farthest index of all
        # previous indices cannot reach beyoud the current number, return False

        # Time O(n)
        # Space O(1)

        if len(nums) == 1:
            return True

        farthest_index = 0
        for i in range(len(nums)-1):
            farthest_index = max(farthest_index, nums[i]+i)
            if nums[i] == 0 and farthest_index <= i:
                return False
            if farthest_index >= len(nums)-1:
                return True

        return False

Best time to buy and sell stock (E) (G/F/A)

class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        # Remember minimum price while going through the list
        # Keep max diff between minimum price and any price

        min_price = prices[0]
        max_profit = 0

        # Time O(n) n = len(prices)
        # Space O(1)
        for price in prices[1:]:
            min_price = min(min_price, price)
            profit = max(price-min_price, 0)
            max_profit = max(max_profit, profit)

        return max_profit

Best time to buy and sell stock II (M) (A)

class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        # Find all increasing sequences in the array
        # Then, find diff in each increasing sequence and sum them

        diffs = []
        diff = 0

        # Time O(n) n = len(prices)
        # Space O(n)
        i = 0
        while i < len(prices) - 1:

            if prices[i] <= prices[i+1]:
                diff += prices[i+1] - prices[i]
            else:
                diffs.append(diff)
                diff = 0
            i += 1

        diffs.append(diff)

        return sum(diffs)

Gas stations (M) (A)

class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:

        # Since there is unique solution, check from each index
        # If total cost > total gas, there is no solution
        # Find the first index where the accumulated gas is positive

        if sum(cost) > sum(gas):
            return -1

        # Time O(n) n = len(gas)
        # Space O(1)
        index = 0
        accumulated_gas = 0
        for i in range(len(gas)):
            accumulated_gas += (gas[i]-cost[i])
            if accumulated_gas < 0:
                accumulated_gas = 0
                index = i + 1

        return index

Candy (H) (A)

class Solution:
    def candy(self, ratings: List[int]) -> int:

        # At index i,
        #   If ratings[i] > ratings[i-1], candy[i] > candy[i-1]
        #   If ratings[i] > ratings[i+1], candy[i] > candy[i+1]
        # Initialize candy array to all 1
        # Go though the candy array from left to right incrementing candies
        # Go though the candy array from right to left incrementing candies

        n = len(ratings)
        candies = [1] * n

        # Time O(n) n = len(ratings)
        # Space O(n)
        for i in range(1, n):
            if ratings[i] > ratings[i-1]:
                candies[i] = candies[i-1] + 1

        for j in range(n-2, -1, -1):
            if ratings[j] > ratings[j+1] and candies[j] <= candies[j+1]:
                candies[j] = candies[j+1] + 1

        return sum(candies)

Maximum Swap (M) (F)

class Solution:
    def maximumSwap(self, num: int) -> int:

        # Scanning from left to right, if there is larger digit laster,
        # it is best to swap at the current number
        # For each digit, remember the index of last occurance
        # Then, find the largest digit to the right
        # If there are multiple largest digits, swap with the later one

        # Time O(n) n = len(num)
        # Space O(n)
        digits = list(str(num))
        last_occurance = defaultdict(int)
        for i, digit in enumerate(digits):
            last_occurance[int(digit)] = i

        for i, c in enumerate(str(num)):
            # Check if larger digit than c occurs to the right
            for digit in range(9, int(c), -1):
                if last_occurance[digit] > i:
                    digits[i], digits[last_occurance[digit]] = digits[last_occurance[digit]], digits[i]
                    return int("".join(digits))

        return num

Boats to save people (M)

class Solution:
    def numRescueBoats(self, people: List[int], limit: int) -> int:

        # Sort the array
        # If the sum of lightest and heaviest is less than or equal to limit,
        #   Increment lightest and decrement heaviest.
        # Else
        #   Only decrement heaviest.

        # Time: O(nlogn)
        # Space: O(1)
        people.sort()
        lightest = 0
        heaviest = len(people) - 1
        num_boats = 0

        while lightest <= heaviest:
            if people[lightest] + people[heaviest] <= limit:
                lightest += 1
                heaviest -= 1
                num_boats += 1
            else:
                heaviest -= 1
                num_boats += 1

        return num_boats

Find Valid Matrix Given Row and Column Sums (M)

class Solution:
    def restoreMatrix(self, rowSum: List[int], colSum: List[int]) -> List[List[int]]:
        # Greedy
        #   Assign the maximum value that can go in each spot

        # Initialize matrix
        n = len(rowSum)
        m = len(colSum)
        matrix = [m*[0] for _ in range(n)]

        # Time O(nm) n = number of rows, m = number of columns
        # Space O(max(n,m))
        running_row_sum = n * [0]
        running_col_sum = m * [0]
        for i in range(n):
            for j in range(m):
                matrix[i][j] = min(rowSum[i]-running_row_sum[i], colSum[j]-running_col_sum[j])
                running_row_sum[i] += matrix[i][j]
                running_col_sum[j] += matrix[i][j]

        return matrix

Average Waiting Time (M)

class Solution:
    def averageWaitingTime(self, customers: List[List[int]]) -> float:

        # Compute next available time

        next_available_time = customers[0][0]
        waiting_time = 0
        accumulated_waiting_time = 0

        # Time O(n) n = len(customers)
        # Space O(1)
        for customer in customers:

            next_available_time = max(
                next_available_time+customer[1],sum(customer))
            waiting_time = next_available_time - customer[0]
            accumulated_waiting_time += waiting_time

        return accumulated_waiting_time / len(customers)

Maximum Score From Removing Substrings (M)

class Solution:
    def maximumGain(self, s: str, x: int, y: int) -> int:

        # First remove substring with higher score
        # Then, remove substring with lower score
        # Stack

        # Time O(n) n = len(s)
        # Space O(1)
        def remove_substring(string, substring, point):
            points = 0
            stack = []

            for c in string:
                if not stack:
                    stack.append(c)
                    continue

                if c == substring[1] and stack[-1] == substring[0]:
                    stack.pop()
                    points += point
                else:
                    stack.append(c)

            return "".join(stack), points

        total_points = 0

        if x > y:
            s, points = remove_substring(s, "ab", x)
            total_points += points
            s, points = remove_substring(s, "ba", y)
            total_points += points
        else:
            s, points = remove_substring(s, "ba", y)
            total_points += points
            s, points = remove_substring(s, "ab", x)
            total_points += points

        return total_points
