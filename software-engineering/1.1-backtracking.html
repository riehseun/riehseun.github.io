class Solution:
    def generateParenthesis(self, n: int) -> List[str]:

        combinations = []

        # Time ?
        # Space ?
        def backtracking(n, num_open, num_close, string, combinations):

            if num_open == n and num_close == n:
                combinations.append(string)

            if num_open < n:
                # No "return" statement
                backtracking(n, num_open+1, num_close, string+"(", combinations)

            # Number of closing parenthesie cannot exceed the number of
            # opening parenthesis at any point to construct valid patterns
            if num_close < num_open:
                # No "return" statement
                backtracking(n, num_open, num_close+1, string+")", combinations)

        backtracking(n, 0, 0, "", combinations)

        return combinations

class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:

        # Time O(n*3^m) n = number of cells, m = len(word)
        # Space O(m)

        # Sort the array
        n = len(board)
        m = len(board[0])
        directions = [[0,1],[1,0],[0,-1],[-1,0]]

        def backtrack(board, word, directions, start, n, m, visited, word_index):
            if board[start[0]][start[1]] == word[word_index]:
                if word_index == len(word) - 1:
                    return True

                for x1, y1 in directions:
                    x = start[0] + x1
                    y = start[1] + y1
                    if 0 <= x < n \
                        and 0 <= y < m \
                        and (x,y) not in visited:
                        # Make the move
                        visited.add((x,y))
                        # Move on to the next character in word
                        if backtrack(board, word, directions, (x,y), n, m, visited, word_index+1):
                            return True
                        # Backtrack
                        visited.remove((x,y))

            return False

        for i in range(n):
            for j in range(m):
                visited = {(i,j)}
                if backtrack(board, word, directions, (i,j), n, m, visited, 0):
                    return True

        return False

Word Search II (H) (A) !

class TrieNode():
    def __init__(self):
        self.children = defaultdict(TrieNode)
        self.is_word = False

class Trie():
    def __init__(self):
        self.root = TrieNode()
        self.num_words = 0

    def insert(self, word):
        node = self.root
        for w in word:
            node = node.children[w]
        node.is_word = True

class Solution:
    def findWords(self, board: List[List[str]], words: List[str]) -> List[str]:

        if not board:
            return []

        n = len(board)
        m = len(board[0])
        result = []

        trie = Trie()
        trie.num_words = len(words)
        node = trie.root
        for word in words:
            trie.insert(word)

        def backtrack(board, result, trie, node, i, j, n, m, path):

            if trie.num_words == 0:
                return

            if node.is_word:
                result.append(path)
                trie.num_words -= 1
                node.is_word = False

            if 0 <= i < n and 0 <= j < m:
                temp = board[i][j]
                node = node.children.get(temp)
                if not node:
                    return

                board[i][j] = "#"
                backtrack(board, result, trie, node, i+1, j, n, m, path+temp)
                backtrack(board, result, trie, node, i-1, j, n, m, path+temp)
                backtrack(board, result, trie, node, i, j+1, n, m, path+temp)
                backtrack(board, result, trie, node, i, j-1, n, m, path+temp)
                board[i][j] = temp

            else:
                return

            return False

        for i in range(n):
            for j in range(m):
                backtrack(board, result, trie, node, i, j, n, m, "")

        return result
