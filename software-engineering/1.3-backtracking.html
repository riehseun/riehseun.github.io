Letter Combinations of a Phone Number (M)

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:

        # Backtracking
        # At each digit, there are 3 choices

        # Time O(3**n)
        #   n = len(digits)
        # Space O(n)
        #   n is from recursion call stack
        #   Ignoring space for output
        map = {
            "2": "abc",
            "3": "def",
            "4": "ghi",
            "5": "jkl",
            "6": "mno",
            "7": "pqrs",
            "8": "tuv",
            "9": "wxyz"
        }

        def backtrack(i, path):

            if i == len(digits):
                result.append("".join(path))
                return

            for c in map[digits[i]]:
               path.append(c)
               backtrack(i+1, path)
               path.pop() 

        if len(digits) == 0:
            return []
            
        result = []
        backtrack(0, [])
        return result

Generate Parentheses (M)

class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        
        # Backtracking
        # Number of ( cannot be greater than n
        # Number of ) cannot excced number of ( at any point 
        # Generate n * 2 length of string

        # Time O(2n**n)
        # Space O(n)
        def backtrack(result, brackets, n, count):

            for c in ["(", ")"]:

                if len(brackets) == n * 2:
                    result.append("".join(brackets))
                    return

                if c == ")" and count["("] > count[")"]:
                    count[")"] += 1
                    brackets.append(")")
                    backtrack(result, brackets, n, count)
                    brackets.pop()  
                    count[")"] -= 1  
                if c == "(" and count["("] < n:
                    count["("] += 1
                    brackets.append("(")
                    backtrack(result, brackets, n, count)
                    brackets.pop()
                    count["("] -= 1
            
        result, count = [], defaultdict(int)
        backtrack(result, [], n, count)
        return result

Combination Sum (M)

class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        
        # Backtracking

        def backtrack(path, j, curr_num):

            # result.append(path[:])
            # print(f"{sum(path[:])},{path[:]}")
            if curr_num < 0:
                return
            if curr_num == 0:
                result.append(path[:])
            
            for i in range(j, len(candidates)):
                path.append(candidates[i])
                backtrack(path, i, curr_num-candidates[i])
                path.pop()

        result = []
        backtrack([], 0, target)
        return result

class Solution:

        # Backtracking

        def backtrack(j, path, curr_num):

            if curr_num < 0:
                return
            if curr_num == 0:
                result.append(path[:]) 

            for i in range(j, len(candidates)):
                if i > j and candidates[i] == candidates[i-1]:
                    continue
                path.append(candidates[i])
                backtrack(i+1, path, curr_num-candidates[i])
                path.pop()

        result = []
        candidates.sort()
        backtrack(0, [], target)
        return result

Permutations (M)

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        
        # Backtracking

        def backtrack(path):

            if len(path) == len(nums):
                result.append(path[:])
                return

            for num in nums:
                if num not in path:
                    path.append(num)
                    backtrack(path)
                    path.pop()

        result = []
        backtrack([])
        return result

class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:

        # Backtracking
        # In each cell, there are 2 choices
        #   Place a queen or not
        # To place a queen, there should be no queen
        # in the same row, column, diag, antidiag

        rows, columns, diags, antidiags = set(), set(), set(), set()

        def backtrack(result, board, n, i, columns, diags, antidiags):

            if i == n:
                new_board = []
                for row in board:
                    new_board.append("".join(row))
                # if new_board not in result:
                result.append(new_board)
                print(result)

            for j in range(n):
                if j not in columns \
                    and i + j not in diags \
                    and i - j not in antidiags:
                    board[i][j] = "Q"
                    # rows.add(i)
                    columns.add(j)
                    diags.add(i+j)
                    antidiags.add(i-j)
                    backtrack(result, board, n, i+1, columns, diags, antidiags)
                    board[i][j] = "."
                    # rows.remove(i)
                    columns.remove(j)
                    diags.remove(i+j)
                    antidiags.remove(i-j)

        result = []
        board = [["."] * n for _ in range(n)]
        backtrack(result, board, n, 0, columns, diags, antidiags)
        return result

Subset (M)

class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:

        # Backtracking
        #   j is the index of first element to add to current combination

        # Time O(n*(2**n))
        #   n = len(nums)
        # Space O(n)
        def backtrack(j, path):

            result.append(path[:])

            for i in range(j, len(nums)):
                path.append(nums[i])
                backtrack(i+1, path)
                path.pop()

        result = []
        backtrack(0, [])
        return result

Palindrome Partitioning (M)

class Solution:
    def partition(self, s: str) -> List[List[str]]:

        # Backtracking
        #   For every possible substring, check palindrome

        # Time O(n*(2**n))
        #   n = len(s)
        #   O(n) to check if substring is palindrom
        #   O(2**n) possible substrings
        # Space O(n)
        #   Recursion stack
        def backtrack(j, path):

            if j == len(s):
                result.append(path[:])

            for i in range(j, len(s)):
                if s[j:i+1] == s[j:i+1][::-1]:
                    path.append(s[j:i+1])
                    # After appending substring s[i:j+1],
                    # Start scannign from index j + 1
                    backtrack(i+1, path)
                    path.pop()

        result = []
        backtrack(0, [])
        return result

class Solution:
    def cleanRoom(self, robot):
        """
        :type robot: Robot
        :rtype: None
        """

        # Backtracking
        # Always turn right
        # Consider visited cell as obstacle
        # Explore up, right, down, left in order
        #   if no obstacle
        #     Move
        #     Explore the next cell
        #     Backtrack 
        #   Else
        #     Turn right    

        # Up, down, right, left
        dirs = [(-1, 0), (0, 1), (1, 0), (0, -1)]
        visited = set()

        def move_back():
            robot.turnRight()
            robot.turnRight()
            robot.move()
            robot.turnRight()
            robot.turnRight()

        # Time O(n-m)
        #   n = number of cells
        #   m = number of obstacles
        # Space O(n-m)
        def backtrack(p, d):

            visited.add(p)
            # Clean the current cell
            robot.clean()

            # Explore up, right, down, left in order
            # Let up = 0, right = 1, down = 2, left = 3
            for i in range(4):
                new_d = (d+i) % 4
                new_p = (p[0]+dirs[new_d][0], p[1]+dirs[new_d][1])
                
                # If no obstacle, move and explore the next cell
                if new_p not in visited and robot.move():
                    backtrack(new_p, new_d)
                    move_back()
                robot.turnRight()

        # Initial direction of the robot will be facing up
        backtrack((0, 0), 0)
