Word break (M) (F/A) #!

class Solution(object):
    def wordBreak(self, s, wordDict):
        """
        :type s: str
        :type wordDict: List[str]
        :rtype: bool
        """

        # Consider indices of start and end char of string 
        # as nodes in graph
        # For example, because "leet" is in "leetcode", 0->4
        # Return True if there is path from 0 to len(s)
        
        occurences = defaultdict(set)
        
        # Time O(nml)
        #   n = len(s)
        #   m = len(wordDict)
        #   l = average length of each word in wordDict
        # Space O(m)
        for word in wordDict:
            for i in range(len(s)):
                if s[i:].startswith(word):
                    occurences[i].add(i+len(word))
            
        # Time O(m)
        # Space O(m)
        def dfs(graph, start, end):

            explored = set()
            explored.add(start)
            
            stack = []
            stack.append(start)
            
            while stack:
                node = stack.pop()
                for next_node in graph[node]:
                    if next_node == end:
                        return True
                        
                    if next_node not in explored:
                        stack.append(next_node)
                        explored.add(next_node)
                
            return False
                    
        return dfs(occurences, 0, len(s))

Course schedule (M) (G/F/A) !#!

class Solution:
    def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        
        # Do DFS
        # If cycle, return False
        
        graph = defaultdict(set)
        nodes = set()
        
        # Time O(k^2) 
        #   k = numCourses
        # Space O(k^2) 
        for pair in prerequisites:
            graph[pair[1]].add(pair[0])
            nodes.add(pair[0])
            nodes.add(pair[1])
        
        # Time O(n+m)
        #   n = number of nodes, which is k
        #   m = number of edges, which is k^2 
        # Space O(n)
        def dfs(graph, start):

            explored = set()
            explored.add(start)

            stack = []
            stack.append(start)
            
            while stack:
                node = stack.pop()
                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append(next_node)
                    else:
                        if next_node == start:
                            return True
                            
            return False

        # Time O(n*(n+m))
        # Space O(n**2)
        for node in nodes:
            if dfs(graph, node):
                return False
            
        return True

Course schedule II (M) (A) !

class Solution:
    def findOrder(self, numCourses: int, prerequisites: List[List[int]]) -> List[int]:

        # Loop through n. Do DFS for each node
        # 1. If a node is pre-requisite for a lot of courses, it will take long to
        # finsh DFS
        # 2. If a node is pre-requisite for only small number of course or even no
        # course, DFS will finish quickly
        # Remember the number of steps taken for DFS starting at each node

        graph = defaultdict(set)

        for edge in prerequisites:
            graph[edge[1]].add(edge[0])

        # Time O(n+m)
        #   n = number of nodes = numCourses
        #   m = number of edges = len(prerequisites)
        # Space O(n)
        def dfs(graph, start):

            explored = set()
            explored.add(start)

            stack = []
            stack.append(start)
            count = 1

            while stack:
                node = stack.pop()
                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append(next_node)
                        count += 1
                    else:
                        if next_node == start:
                            return -1

            return count

        ordering = []
        for i in range(numCourses):
            count = dfs(graph, i)
            # If cycle
            if count == -1:
                return []
            ordering.append([count, i])

        ordering.sort(key=lambda x:(-x[0]))

        return [item[1] for item in ordering]
