Binary tree inorder traversal (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def inorderTraversal(self, root: Optional[TreeNode]) -> List[int]:

        # Traverse tree
        # Return the order as left, root, right

        inorder = []
        self.helper(root, inorder)
        return inorder

    # Time O(n) where n is number of nodes
    # Space O(h) where h = n if worst case, h = log(n) if best case
    def helper(self, root, inorder):

        if root:
            self.helper(root.left, inorder)
            inorder.append(root.val)
            self.helper(root.right, inorder)

        return []

Recover binary search tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """

        # Do inorder traversal
        # During inorder traversal, find elements that need to be swapped

        self.previous = TreeNode(-math.inf)
        self.first = None
        self.second = None

        # Time O(n) where n is the number of nodes in tress
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if root:
                helper(root.left)

                if self.previous.val >= root.val:
                    if not self.first:
                        self.first = self.previous
                    self.second = root
                self.previous = root

                helper(root.right)

        helper(root)
        self.first.val, self.second.val = self.second.val, self.first.val

Same tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isSameTree(self, p: Optional[TreeNode], q: Optional[TreeNode]) -> bool:

        # If both root are None, return True
        # If one root is not None but the other is None, return False
        # If at any point, the root val is different, return False

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(p, q):

            if not p and not q:
                return True

            if (not p and q) or (p and not q):
                return False

            if p.val != q.val:
                return False

            left = helper(p.left, q.left)
            right = helper(p.right, q.right)

            return left and right

        return helper(p, q)

Maximum depth of binary tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxDepth(self, root: Optional[TreeNode]) -> int:

        # Same as computing the height of tree
        # If root is None, return 0
        # At each level, return max depth of left and right, and plus one

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if not root:
                return 0

            left = helper(root.left)
            right = helper(root.right)

            return max(left, right) + 1

        return helper(root)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreexNode]:

        # Preorder - root, left, right
        # Inorder - left, root, right
        # The very first number of preorder is always the root
        # Numbers left of this root from inorder forms left subtree
        # Numbers right of this root from inorder forms left subtree

        # Time O(n) computation takes place on each node of tree

        if inorder:

            root_node = preorder.pop(0)
            root_index = inorder.index(root_node)
            root = TreeNode(root_node)
            root.left = self.buildTree(preorder[:len(inorder[:root_index])], inorder[:root_index])
            root.right = self.buildTree(preorder[len(inorder[:root_index]):], inorder[root_index+1:])

            return root

#         self.left = left
#         self.right = right

    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:

        # Inorder - left, root, right

        # Numbers left of this root from inorder forms left subtree

            root.right = self.buildTree(inorder[root_index+1:], postorder[len(inorder[:root_index]):])

            return root

#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sortedArrayToBST(self, nums: List[int]) -> Optional[TreeNode]:

        # The middle element is always the root
        # Everything on the left side constitute left subtree
        # Everything on the right side constitute right subtree

        # Time O(n) where n is len(nums)

            low = 0
            high = len(nums)-1
            mid = (low+high) // 2

            root = TreeNode(nums[mid])
            root.left = self.sortedArrayToBST(nums[:mid])
            root.right = self.sortedArrayToBST(nums[mid+1:])

            return root

Convert sorted list to binary search tree (M)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:

        # The middle element is always the root
        # Everything on the left side constitute left subtree
        # Everything on the right side constitute right subtree

        nums = []

        while head:
            nums.append(head.val)
            head = head.next

        return self.helper(nums)

    def helper(self, nums):

        # Time O(n) where n is len of the linked list
        if nums:
            low = 0
            high = len(nums)-1
            mid = (low+high) // 2

            root = TreeNode(nums[mid])
            root.left = self.helper(nums[:mid])
            root.right = self.helper(nums[mid+1:])

            return root

Balanced binary tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isBalanced(self, root: Optional[TreeNode]) -> bool:

        # Return 0 at the leaf nodes
        # If height difference at any layer is greater than 1, return False
        # If height of left or right subtree return -1, return False
        # Return max height plus 1 in each layer

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if not root:
                return 0

            left = helper(root.left)
            right = helper(root.right)

            if abs(left-right) > 1:
                return -1

            if left == -1 or right == -1:
                return -1

            return max(left, right) + 1

        retval = helper(root)

        if retval == -1:
            return False
        return True

Path sum (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def hasPathSum(self, root: Optional[TreeNode], targetSum: int) -> bool:

        # Return True when the running sum equals to target

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root, running_sum, target_sum):

            if root:

                if running_sum + root.val == target_sum and not root.left and not root.right:
                    return True

                left = helper(root.left, running_sum+root.val, target_sum)
                right = helper(root.right, running_sum+root.val, target_sum)

                return left or right

        return helper(root, 0, targetSum)

Path sum II (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:

        # Save the result when the running sum equals to target

        self.paths = []

        # Time O(n) where n is number of nodes
        # Space O(n) for each node, we are constructing its path
        def helper(root, path, running_sum, target_sum):

            if root:

                if running_sum + root.val == target_sum and not root.left and not root.right:
                    self.paths.append(path+[root.val])

                helper(root.left, path+[root.val], running_sum+root.val, target_sum)
                helper(root.right, path+[root.val], running_sum+root.val, target_sum)

            return self.paths

        return helper(root, [], 0, targetSum)

Binary tree maximum path sum (H) (Doordash)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:

        # The path sum is 
        #   node.val 
        #   + path sum of left subtree 
        #   + path sum of right subtree
        # The path sum of leaf nodes are its own value
        #   When node does not exist, return 0
        # Traverse tree
        #   For each node, compute path sum
        #   Since both left and right children must be
        #   computed first, do post_order traversal
        # Each child can either pick contribution from its
        # left child or right child to have valid path

        self.max_sum = -math.inf

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   Call stack for recursion
        def post_order(node):

            if node:
                max_sum_from_left = max(post_order(node.left), 0)
                max_sum_from_right = max(post_order(node.right), 0)
                node_sum = node.val + max_sum_from_left + max_sum_from_right
                self.max_sum = max(node_sum, self.max_sum)
                # Contribution from Subtree can either be 
                # from left or right, not both 
                return max(
                    node.val+max_sum_from_left, 
                    node.val+max_sum_from_right
                )

            else:
                return 0

        post_order(root)
        return self.max_sum/code>

class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:

        # DFS pre-order

        #   At each node, multiply running sum by 10
        #   If leaf node, add running sum to total sum

        #   n = number of nodes in tree
        # Space O(h)

                    self.total_sum += running_sum
                
                pre_order(node.left, running_sum)
                pre_order(node.right, running_sum)                

        pre_order(root, 0)    

        return self.total_sum

Binary search tree iterator (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class BSTIterator:

    # During init, store only the left subtrees into stack
    #   Node at top of stack is next smallest element

    # Time O(n) for skewed tree
    #   O(1) on average
    # Space O(n)
    def left_child(self, node):

        if node:
            self.stack.append(node)
            self.left_child(node.left)
        
    def __init__(self, root: Optional[TreeNode]):

        self.stack = []        
        self.left_child(root)

    # Time O(n) for skewed tree
    #   O(1) on average    
    def next(self) -> int:
        node = self.stack.pop()
        
        if node.right:
            self.left_child(node.right)
        
        return node.val
        
    # Time O(1)
    def hasNext(self) -> bool:
        if len(self.stack) > 0:
            return True

        return False

Lowest common ancestor of a binary tree (M) (Meta)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':

        # Do post-order
        # If leaf node, return 0
        # If node is p or q, return 1
        # At any node, if node.val + left + right == 2, 
        # the node is lca 

        self.lca = None

        # Time O(h) = O(n)
        #   h = height of tree, worst case h = n
        # Space O(1)
        def post_order(node, p, q):

            if node:
                
                left = post_order(node.left, p, q)
                right = post_order(node.right, p, q)
                    
                mid = 0
                if node.val == p.val or node.val == q.val:
                    mid = 1

                if left + right + mid == 2 \
                    and not self.lca:
                    self.lca = node
                    
                return left + right + mid

            else:
                return 0

        post_order(root, p, q)
        return self.lca

Closest binary search tree value (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def closestValue(self, root: Optional[TreeNode], target: float) -> int:

        # DO pre-order
        #   If target > root.val, search on right
        #   If target < root.val, search on left

        self.min_diff = math.inf
        self.result = None

        # Time O(h) 
        #   h = height of tree
        # Space O(1)
        def pre_order(node, target):

            if node:

                if abs(node.val-target) < self.min_diff:
                    self.min_diff = abs(node.val-target)
                    self.result = node
                # If difference is the same, keep smaller node value
                elif abs(node.val-target) == self.min_diff:
                    if node.val < self.result.val:
                        self.result = node
                
                if target < node.val:
                    pre_order(node.left, target)
                else:
                    pre_order(node.right, target)
        
        pre_order(root, target)

        return self.result.val

Serialize and deserialize binary tree (H)

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Codec:

    # Time O(n) where n is number of nodes
    # Space O(h) where h = n if worst case, h = log(n) if best case

    def serialize(self, root):
        """Encodes a tree to a single string.

        :type root: TreeNode
        :rtype: str
        """

        self.data = ""

        def helper(root):

            if root:
                self.data += str(root.val)
                self.data += ","
                helper(root.left)
                helper(root.right)
            else:
                self.data += "#"
                self.data += ","

        helper(root)
        return self.data[:len(self.data)-1]

    def deserialize(self, data):
        """Decodes your encoded data to tree.

        :type data: str
        :rtype: TreeNode
        """

        data_array = data.split(",")

        def helper(data_array):
            val = next(data_array)
            if val != "#":
                root = TreeNode(int(val))
                root.left = helper(data_array)
                root.right = helper(data_array)
                return root

        return helper(iter(data_array))

        # Then, remove leaf nodes from the tree
        # Repeat until only the root remains

        # Time O(h) h = height of tree
        # Space O(n) n = number of nodes in tree

                right = post_order(root.right, array)

        while root.left or root.right:

            post_order(root, array)
            result.append(array)

Convert Binary Search Tree to Sorted Doubly Linked List (M)

"""
# Definition for a Node.
class Node:
    def __init__(self, val, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right
"""

class Solution:
    def treeToDoublyList(self, root: 'Optional[Node]') -> 'Optional[Node]':

        # DFS in-order
        # Link nodes while traversing
    
        if not root:
            return None

        self.nodes = []
        self.first = None
        self.last = None

        # Time O(n) 
        #   n = number of nodes
        # Space O(h) 
        #   recursion stack of tree height
        def in_order(node):
            
            if node:
                in_order(node.left)

                if self.last:
                    self.last.right = node
                    node.left = self.last
                else:
                    # If no nodes existed, set current node 
                    # as first node in linked list
                    self.first = node
                # Set current node as last node
                self.last = node

                in_order(node.right)

        in_order(root)
        # Connect head and tail to construct 
        # circular linked list
        self.first.left = self.last
        self.last.right = self.first
        
        return self.first

Diameter of binary tree (E) (Meta)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:

        # Diameter or longest path is sum of 
        #   root + left subtree + right subtree
        # Do post order traverse
        #   If leaf node, return 1
        #   Else, max path is
        #     longest path of left + longest path of right + 1
        #   Valid path can only have either left or right subtree 

        self.max_path = 0

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   h = n in the worst case
        def post_order(node):

            if node:
                longest_path_left = post_order(node.left)
                longest_path_right = post_order(node.right)
                self.max_path = max(self.max_path, 
                    longest_path_left+longest_path_right+1)
                return max(longest_path_left, longest_path_right) + 1
            else:
                return 0
                
        post_order(root)
        return self.max_path - 1

Boundary of Binary Tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def boundaryOfBinaryTree(self, root: Optional[TreeNode]) -> List[int]:
        
        # root 
        # + left boundry (left most children other than root and leaf)
        # + leaves from left to right 
        # + reverse of right boundry 

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(n)
        def left_boundry(node):

            if node:
                left_nodes.append(node.val)
                if node.left:
                    left_boundry(node.left)
                else:
                    left_boundry(node.right)

        def right_boundry(node):

            if node:
                right_nodes.append(node.val)
                if node.right:
                    right_boundry(node.right)
                else:
                    right_boundry(node.left)

        def pre_order(node, root):

            if node:
                # Find leaf nodes
                if not node.left \
                    and not node.right:
                    leaf_nodes.append(node.val)
                pre_order(node.left, root)
                pre_order(node.right, root)

        leaf_nodes = []
        left_nodes = []
        right_nodes = []

        if not root.left and not root.right:
            return [root.val]   

        left_boundry(root.left)
        pre_order(root, root)
        right_boundry(root.right)
        right_nodes.reverse()

        result = []   
        if left_nodes:
            leaf_nodes = leaf_nodes[1:]
        if right_nodes:
            leaf_nodes = leaf_nodes[:-1]
        result = [root.val] + left_nodes +leaf_nodes + right_nodes

        return result

Longest univalue path (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def longestUnivaluePath(self, root: Optional[TreeNode]) -> int:

        # Longest path at a node is
        #  longest path of left subtree
        #  longest path of right subtree

        self.longest_path = 0

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case

        # Find max height of a tree only when the node values are the same
        def helper(root, node_value):

            if not root or root.val != node_value:
                return 0

            left = helper(root.left, node_value)
            right = helper(root.right, node_value)

            return max(left, right) + 1

        def helper2(root):

            if root:
                path_length_from_left = helper(root.left, root.val) - 1
                path_length_from_right = helper(root.right, root.val) - 1
                path_legnth = path_length_from_left + path_length_from_right + 2
                self.longest_path = max(self.longest_path, path_legnth)

                helper2(root.left)
                helper2(root.right)

        helper2(root)

        return self.longest_path

All Nodes Distance K in Binary Tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distanceK(self, root: TreeNode, target: TreeNode, k: int) -> List[int]:

        # DFS on tree to convert to graph
        # DFS on graph

        graph = defaultdict(set)

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(n)
        def pre_order(node, parent, graph):

            if node:
                
                if parent:
                    graph[parent.val].add(node.val)
                    graph[node.val].add(parent.val)

                pre_order(node.left, node, graph)
                pre_order(node.right, node, graph)

        pre_order(root, None, graph)

        distances = defaultdict(list)

        # Time O(n+m) 
        #   n = number of nodes
        #   m = number of edges
        # Space O(n)
        def dfs(graph, start):
            
            explored = set()
            explored.add(start)

            stack = []
            stack.append((start, 0))

            while stack:
                node, depth = stack.pop()
                distances[depth].append(node)
                
                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append((next_node, depth+1))
                        
        dfs(graph, target.val)

        return distances[k]

Range sum of bst (E) (Meta)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rangeSumBST(self, root: Optional[TreeNode], low: int, high: int) -> int:

        # DFS pre-order
        #   Add to sum only if value is within the range

        self.sum = 0

        # Time O(n)
        #   n = number of nodes
        # Space O(h)
        #   Recursion stack h = n worst case
        def pre_order(node, low, high):
            
            if node:

                if node.val >= low and node.val <= high:
                    self.sum += node.val

                pre_order(node.left, low, high)
                pre_order(node.right, low, high)

        pre_order(root, low, high)

        return self.sum

Lowest Common Ancestor of a Binary Tree III (M) (Meta)

"""
# Definition for a Node.
class Node:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None
        self.parent = None
"""

class Solution:
    def lowestCommonAncestor(self, p: 'Node', q: 'Node') -> 'Node':

        # Do post order traverse
        #   If a node is p or q, return 1
        #   Else, return 0
        # If left and right returns 2, 
        # then current root is lca
        # Also, if either left or right return 1 and current root is p or q,
        # then, current root is lca

        # First, need to find root
        
        self.lca = None

        def find_root(p):
            
            if p.parent: 
                return find_root(p.parent)
            else:
                return p

        # Time O(h) = O(n)
        #   h = height of tree, worst case h = n
        # Space O(1)
        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)
                
                mid = 0
                if node.val == p.val or node.val == q.val:
                    mid = 1

                if left + right + mid == 2 \
                    and not self.lca:
                    self.lca = node
                    
                return left + right + mid

            else:
                return 0

        root = find_root(p)
        post_order(root)
        return self.lca
