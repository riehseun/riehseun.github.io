Binary tree maximum path sum (H) (Doordash)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:

        # The path sum is 
        #   node.val 
        #   + path sum of left subtree 
        #   + path sum of right subtree
        # The path sum of leaf nodes are its own value
        #   When node does not exist, return 0
        # Traverse tree
        #   For each node, compute path sum
        #   Since both left and right children must be
        #   computed first, do post_order traversal
        # Each child can either pick contribution from its
        # left child or right child to have valid path

        self.max_sum = -math.inf

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   Call stack for recursion
        def post_order(node):

            if node:
                max_sum_from_left = max(post_order(node.left), 0)
                max_sum_from_right = max(post_order(node.right), 0)
                node_sum = node.val + max_sum_from_left + max_sum_from_right
                self.max_sum = max(node_sum, self.max_sum)
                # Contribution from Subtree can either be 
                # from left or right, not both 
                return max(
                    node.val+max_sum_from_left, 
                    node.val+max_sum_from_right
                )

            else:
                return 0

        post_order(root)
        return self.max_sum/code>

Sum root to leaf numbers (M) (Meta)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:

        # DFS pre-order
        # Calculate running sum for each path
        #   At each node, multiply running sum by 10
        #   If leaf node, add running sum to total sum  

        self.total_sum = 0

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   Recursion stack h = n in worst case
        def pre_order(node, running_sum):

            if node:
                running_sum = running_sum * 10 + node.val
                
                if not node.left and not node.right:
                    self.total_sum += running_sum
                
                pre_order(node.left, running_sum)
                pre_order(node.right, running_sum)                

        pre_order(root, 0)    

        return self.total_sum

Lowest common ancestor of a binary tree (M) (Meta)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':

        # Do post-order
        # If leaf node, return 0
        # If node is p or q, return 1
        # At any node, if node.val + left + right == 2, 
        # the node is lca 

        self.lca = None

        # Time O(h) = O(n)
        #   h = height of tree, worst case h = n
        # Space O(1)
        def post_order(node, p, q):

            if node:
                
                left = post_order(node.left, p, q)
                right = post_order(node.right, p, q)
                    
                mid = 0
                if node.val == p.val or node.val == q.val:
                    mid = 1

                if left + right + mid == 2 \
                    and not self.lca:
                    self.lca = node
                    
                return left + right + mid

            else:
                return 0

        post_order(root, p, q)
        return self.lca

Serialize and deserialize binary tree (H) ()

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Codec:

    # Time O(n) where n is number of nodes
    # Space O(h) where h = n if worst case, h = log(n) if best case

    def serialize(self, root):
        """Encodes a tree to a single string.

        :type root: TreeNode
        :rtype: str
        """

        self.data = ""

        def helper(root):

            if root:
                self.data += str(root.val)
                self.data += ","
                helper(root.left)
                helper(root.right)
            else:
                self.data += "#"
                self.data += ","

        helper(root)
        return self.data[:len(self.data)-1]

    def deserialize(self, data):
        """Decodes your encoded data to tree.

        :type data: str
        :rtype: TreeNode
        """

        data_array = data.split(",")

        def helper(data_array):
            val = next(data_array)
            if val != "#":
                root = TreeNode(int(val))
                root.left = helper(data_array)
                root.right = helper(data_array)
                return root

        return helper(iter(data_array))

Find leaves of binary tree (M) ()

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def findLeaves(self, root: Optional[TreeNode]) -> List[List[int]]:

        # DFS post-order
        # Add leaf nodes to array
        # Then, remove leaf nodes from the tree
        # Repeat until only the root remains

        # Time O(h) h = height of tree
        # Space O(n) n = number of nodes in tree
        def post_order(root, array):

            if root:

                # Check if leaf node
                if not root.left and not root.right:
                    array.append(root.val)
                    return -1

                left = post_order(root.left, array)
                right = post_order(root.right, array)
                # Remove leaf node on left side
                if left == -1:
                    root.left = None
                # Remove leaf node on right side
                if right == -1:
                    root.right = None
                print(root.val)

        result = []

        while root.left or root.right:
            array = []
            post_order(root, array)
            result.append(array)
            print(array)

        result.append([root.val])
        return result

Convert Binary Search Tree to Sorted Doubly Linked List (M) ()

"""
# Definition for a Node.
class Node:
    def __init__(self, val, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right
"""

class Solution:
    def treeToDoublyList(self, root: 'Optional[Node]') -> 'Optional[Node]':

        # DFS in-order
        # Link nodes while traversing
    
        if not root:
            return None

        self.nodes = []
        self.first = None
        self.last = None

        # Time O(n) 
        #   n = number of nodes
        # Space O(h) 
        #   recursion stack of tree height
        def in_order(node):
            
            if node:
                in_order(node.left)

                if self.last:
                    self.last.right = node
                    node.left = self.last
                else:
                    # If no nodes existed, set current node 
                    # as first node in linked list
                    self.first = node
                # Set current node as last node
                self.last = node

                in_order(node.right)

        in_order(root)
        # Connect head and tail to construct 
        # circular linked list
        self.first.left = self.last
        self.last.right = self.first
        
        return self.first

Diameter of binary tree (E) (Meta)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:

        # Diameter or longest path is sum of 
        #   root + left subtree + right subtree
        # Do post order traverse
        #   If leaf node, return 1
        #   Else, max path is
        #     longest path of left + longest path of right + 1
        #   Valid path can only have either left or right subtree 

        self.max_path = 0

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   h = n in the worst case
        def post_order(node):

            if node:
                longest_path_left = post_order(node.left)
                longest_path_right = post_order(node.right)
                self.max_path = max(self.max_path, 
                    longest_path_left+longest_path_right+1)
                return max(longest_path_left, longest_path_right) + 1
            else:
                return 0
                
        post_order(root)
        return self.max_path - 1

All Nodes Distance K in Binary Tree (M) ()

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distanceK(self, root: TreeNode, target: TreeNode, k: int) -> List[int]:

        # DFS on tree to convert to graph
        # DFS on graph

        graph = defaultdict(set)

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(n)
        def pre_order(node, parent, graph):

            if node:
                
                if parent:
                    graph[parent.val].add(node.val)
                    graph[node.val].add(parent.val)

                pre_order(node.left, node, graph)
                pre_order(node.right, node, graph)

        pre_order(root, None, graph)

        distances = defaultdict(list)

        # Time O(n+m) 
        #   n = number of nodes
        #   m = number of edges
        # Space O(n)
        def dfs(graph, start):
            
            explored = set()
            explored.add(start)

            stack = []
            stack.append((start, 0))

            while stack:
                node, depth = stack.pop()
                distances[depth].append(node)
                
                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append((next_node, depth+1))
                        
        dfs(graph, target.val)

        return distances[k]

Range sum of bst (E) (Meta)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rangeSumBST(self, root: Optional[TreeNode], low: int, high: int) -> int:

        # DFS pre-order
        #   Add to sum only if value is within the range

        self.sum = 0

        # Time O(n)
        #   n = number of nodes
        # Space O(h)
        #   Recursion stack h = n worst case
        def pre_order(node, low, high):
            
            if node:

                if node.val >= low and node.val <= high:
                    self.sum += node.val

                pre_order(node.left, low, high)
                pre_order(node.right, low, high)

        pre_order(root, low, high)

        return self.sum

Lowest Common Ancestor of a Binary Tree III (M) (Meta)

"""
# Definition for a Node.
class Node:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None
        self.parent = None
"""

class Solution:
    def lowestCommonAncestor(self, p: 'Node', q: 'Node') -> 'Node':

        # Do post order traverse
        #   If a node is p or q, return 1
        #   Else, return 0
        # If left and right returns 2, 
        # then current root is lca
        # Also, if either left or right return 1 and current root is p or q,
        # then, current root is lca

        # First, need to find root
        
        self.lca = None

        def find_root(p):
            
            if p.parent: 
                return find_root(p.parent)
            else:
                return p

        # Time O(h) = O(n)
        #   h = height of tree, worst case h = n
        # Space O(1)
        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)
                
                mid = 0
                if node.val == p.val or node.val == q.val:
                    mid = 1

                if left + right + mid == 2 \
                    and not self.lca:
                    self.lca = node
                    
                return left + right + mid

            else:
                return 0

        root = find_root(p)
        post_order(root)
        return self.lca
