Binary tree maximum path sum (H) (F)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:

        # If node is None, return 0
        # Consider each node. Max sum at each node is the sum of
        #   node value itself
        #   max sum of left subtree
        #   max sum of right subtree
        # Also, in each node, the path can only take either left or right

        self.max_sums = []

        # Time O(n) n = number of nodes
        # Space O(n) for each node, we are constructing its max sum
        def helper(root):

            if not root:
                return 0

            max_sum_left = max(helper(root.left), 0)
            max_sum_right = max(helper(root.right), 0)

            max_sum = root.val + max_sum_left + max_sum_right
            self.max_sums.append(max_sum)

            max_path = root.val + max(max_sum_left, max_sum_right)
            return max_path

        helper(root)

        return max(self.max_sums)

Sum root to leaf numbers (M) (F) !

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:

        # Do pre-order to find all path from root to all leaf nodes

        self.total_num = 0

        # Time O(n) where n is number of nodes
        # Space O(h) to keep recursion stack
        def pre_order(root, num):

            if root:

                num = num * 10 + root.val

                # If leaf node
                if not root.left and not root.right:
                    self.total_num += num

                pre_order(root.left, num)
                pre_order(root.right, num)

        pre_order(root, 0)

        return self.total_num

Lowest common ancestor of a binary tree (M) (F/A) !

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':

        # If subtrees return one p and one q, the current root is LCA
        # If only one of subtrees return one of p and q, then the other one of
        # p and q must be contained under that subtree

        if root == p or root == q:
            return root

        left = None
        right = None

        # Time O(h) h = height of tree
        # Space O(1)
        if root.left:
            left = self.lowestCommonAncestor(root.left, p, q)
        if root.right:
            right = self.lowestCommonAncestor(root.right, p, q)

        if left and right:
            return root
        elif left and not right:
            return left
        elif not left and right:
            return right
        else:
            return None

Closest binary search tree value (E) (F)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def closestValue(self, root: Optional[TreeNode], target: float) -> int:

        # Perform pre-order traversal
        # Remember the difference and value

        self.closest_value = math.inf
        self.difference = math.inf

        # Time O(n) n = number of nodes in tree
        # Space O(1)
        def pre_order(root, target):

            if root:

                if abs(root.val-target) < self.difference:
                    self.difference = abs(root.val-target)
                    self.closest_value = root.val
                elif abs(root.val-target) == self.difference:
                    self.closest_value = min(self.closest_value, root.val)

                pre_order(root.left, target)
                pre_order(root.right, target)

        pre_order(root, target)

        return self.closest_value

Serialize and deserialize binary tree (H) (A)

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Codec:

    # Time O(n) where n is number of nodes
    # Space O(h) where h = n if worst case, h = log(n) if best case

    def serialize(self, root):
        """Encodes a tree to a single string.

        :type root: TreeNode
        :rtype: str
        """

        self.data = ""

        def helper(root):

            if root:
                self.data += str(root.val)
                self.data += ","
                helper(root.left)
                helper(root.right)
            else:
                self.data += "#"
                self.data += ","

        helper(root)
        return self.data[:len(self.data)-1]

    def deserialize(self, data):
        """Decodes your encoded data to tree.

        :type data: str
        :rtype: TreeNode
        """

        data_array = data.split(",")

        def helper(data_array):
            val = next(data_array)
            if val != "#":
                root = TreeNode(int(val))
                root.left = helper(data_array)
                root.right = helper(data_array)
                return root

        return helper(iter(data_array))

Convert Binary Search Tree to Sorted Doubly Linked List (M) (F) !

"""
# Definition for a Node.
class Node:
    def __init__(self, val, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right
"""

class Solution:
    def treeToDoublyList(self, root: 'Optional[Node]') -> 'Optional[Node]':

        # Do in-order traversal
        # Link nodes while traversing

        if not root:
            return None

        self.nodes = []
        self.first = None
        self.last = None

        # Time O(n) n = number of nodes
        # Space O(n) recursion stack of the size of tree height
        def in_order(root):

            if root:
                in_order(root.left)

                # Link two nodes
                if self.last:
                    self.last.right = root
                    root.left = self.last
                else:
                    self.first = root
                self.last = root

                in_order(root.right)

        in_order(root)
        self.last.right = self.first
        self.first.left = self.last

        return self.first

Diameter of binary tree (E) (F) !

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:

        # Diameter of tree at a node is the sum of
        #   Max height of right subtree
        #   Max height of left subtree

        self.max_diameter = 0

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case

        # Find max height of a tree
        def get_max_diameter(root):

            if not root:
                return 0

            left = get_max_diameter(root.left)
            right = get_max_diameter(root.right)
            self.max_diameter = max(self.max_diameter, left+right)

            return max(left, right) + 1

        get_max_diameter(root)

        return self.max_diameter

Range sum of bst (E) (F) !

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rangeSumBST(self, root: Optional[TreeNode], low: int, high: int) -> int:

        # Get all nodes first
        # Then, find numbers within the range

        self.nodes = []

        # Time O(n) where n is number of nodes
        # Space O(n)
        def pre_order(root):

            if root:
                self.nodes.append(root.val)
                pre_order(root.left)
                pre_order(root.right)

        pre_order(root)

        range_sum = 0
        for node in self.nodes:
            if node >= low and node <= high:
                range_sum += node

        return range_sum

Lowest Common Ancestor of a Binary Tree III (M) (F) !

"""
# Definition for a Node.
class Node:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None
        self.parent = None
"""

class Solution:
    def lowestCommonAncestor(self, p: 'Node', q: 'Node') -> 'Node':

        # If subtrees return one p and one q, the current root is LCA
        # If only one of subtrees return one of p and q, then the other one of
        # p and q must be contained under that subtree

        def get_root(p):

            if not p.parent:
                return p

            return get_root(p.parent)

        root = get_root(p)

        def lowest_common_ancestor(root, p, q):

            if root == p or root == q:
                return root

            left = None
            right = None

            # Time O(h) h = height of tree
            # Space O(1)
            if root.left:
                left = lowest_common_ancestor(root.left, p, q)
            if root.right:
                right = lowest_common_ancestor(root.right, p, q)

            if left and right:
                return root
            elif left and not right:
                return left
            elif not left and right:
                return right
            else:
                return None

        return lowest_common_ancestor(root, p, q)
