Clone graph (M/F1)

"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

from typing import Optional
class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:

        # Do standard BFS

        if not node:
            return node

        explored = {}
        explored[node] = Node(node.val)

        queue = deque()
        queue.append(node)

        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n)
        while queue:

            n = queue.popleft()

            for next_node in n.neighbors:
                if next_node not in explored:
                    explored[next_node] = Node(next_node.val)
                    queue.append(next_node)

                explored[n].neighbors.append(explored[next_node])

        return explored[node]

Number of islands (M/G1/F1/100)

class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:

        # Construct a graph
        # Two nodes are connected when node is 1 and adjacent is also 1
        # Count number of connnected component using BFS

        graph = defaultdict(list)
        nodes = set()

        # Time O(nm)
        # Space O(nm) worst case is that every number is 1
        n = len(grid)
        m = len(grid[0])
        for i in range(n):
            for j, num in enumerate(grid[i]):
                if num == "1":
                    if (i,j) not in nodes:
                        nodes.add((i,j))
                    # Check right
                    if j < m-1 and grid[i][j+1] == "1":
                        graph[(i,j)].append((i, j+1))
                        graph[(i,j+1)].append((i,j))
                    # Check left
                    if j > 0 and grid[i][j-1] == "1":
                        graph[(i,j)].append((i, j-1))
                        graph[(i,j-1)].append((i,j))
                    # Check up
                    if i > 0 and grid[i-1][j] == "1":
                        graph[(i,j)].append((i-1, j))
                        graph[(i-1,j)].append((i,j))
                    # Check down
                    if i < n-1 and grid[i+1][j] == "1":
                        graph[(i,j)].append((i+1, j))
                        graph[(i+1,j)].append((i,j))

        explored = set()
        count = 0

        # Time O(nm) worst case visit every node in graph
        # Space O(nm)
        for (i,j) in nodes:
            if (i,j) not in explored:
                self.bfs(graph, explored, (i,j))
                count += 1
        return count

    def bfs(self, graph, explored, start):
        queue = deque()
        queue.append(start)

        while queue:
            node = queue.popleft()
            for child_node in graph[node]:
                if child_node not in explored:
                    explored.add(child_node)
                    queue.append(child_node)

Remove invalid parentheses (H/F1)

class Solution:
    def removeInvalidParentheses(self, s: str) -> List[str]:

        # Treat "s" as root.
        # If node is a valid paranthesis expression, add to result
        # Else, construcut child nodes by removing one parenthesis at a time
        # Keep track of distances from root to child nodes

        explored = set()
        explored.add(s)

        queue = deque()
        queue.append((s,0))

        result = []
        min_dist = len(s)

        # Time O(n+m) n = len(s) ^ 2, m = len(s) ^ 2
        # Space O(n)
        while queue:
            node, dist = queue.popleft()

            if self.is_valid(node):
                # If not the shortest distance
                if dist > min_dist:
                    return result
                min_dist = min(min_dist, dist)
                result.append(node)

            else:
                for i in range(len(node)):
                    if node[i] == "(" or node[i] == ")":
                        next_node = node[:i] + node[i+1:]
                        if next_node not in explored:
                            explored.add(next_node)
                            queue.append((next_node,dist+1))

        return result

    def is_valid(self, string):

        dic = {'(': 1, ')': -1}
        count = 0
        for char in string:
            count += dic.get(char, 0)
            if count < 0:
                return False # early exit coz once count of ')' > '(', s can't be valid
        return count == 0
