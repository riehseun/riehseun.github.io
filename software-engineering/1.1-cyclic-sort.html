First missing positive (H)

class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:
        n = len(nums)

        # Set numbers outside [1,n] to 0.
        for i in range(n):
            if nums[i] < 1 or nums[i] > n:
                nums[i] = 0

        for i in range(n):
            if nums[i] != 0 and nums[i]%(n+1) <= n:
                nums[(nums[i]%(n+1))-1] += (n+1)

        for i in range(n):
            if nums[i] <= n:
                return i + 1

        return n + 1

Missing number (M/A)

class Solution:
    def missingNumber(self, nums: List[int]) -> int:

        n = len(nums)
        nums = set(nums)

        # Time O(n) n = len(nums)
        # Space O(1)
        for i in range(n+1):
            if i not in nums:
                return i

        return -1

Find the duplicate number (M)

class Solution:
    def findDuplicate(self, nums: List[int]) -> int:

        # Convert nums[num] to its negative
        for num in nums:
            num = abs(num)
            if nums[num] < 0:
                return num
            nums[num] = -nums[num]

        return len(nums)
