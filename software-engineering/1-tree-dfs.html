Maximum depth of binary tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxDepth(self, root: Optional[TreeNode]) -> int:

        # Same as computing the height of tree
        # If root is None, return 0
        # At each level, return max depth of left and right, and plus one

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if not root:
                return 0

            left = helper(root.left)
            right = helper(root.right)

            return max(left, right) + 1

        return helper(root)

Convert Sorted Array to Binary Search Tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sortedArrayToBST(self, nums: List[int]) -> Optional[TreeNode]:

        # The middle element is always the root
        # Everything on the left side constitute left subtree
        # Everything on the right side constitute right subtree

        # Time O(n) where n is len(nums)
        if nums:
            low = 0
            high = len(nums)-1
            mid = (low+high) // 2

            root = TreeNode(nums[mid])
            root.left = self.sortedArrayToBST(nums[:mid])
            root.right = self.sortedArrayToBST(nums[mid+1:])

            return root

Convert sorted list to binary search tree (M)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:

        # The middle element is always the root
        # Everything on the left side constitute left subtree
        # Everything on the right side constitute right subtree

        nums = []

        while head:
            nums.append(head.val)
            head = head.next

        return self.helper(nums)

    def helper(self, nums):

        # Time O(n) where n is len of the linked list
        if nums:
            low = 0
            high = len(nums)-1
            mid = (low+high) // 2

            root = TreeNode(nums[mid])
            root.left = self.helper(nums[:mid])
            root.right = self.helper(nums[mid+1:])

            return root

Balanced binary tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isBalanced(self, root: Optional[TreeNode]) -> bool:

        # Return 0 at the leaf nodes
        # If height difference at any layer is greater than 1, return False
        # If height of left or right subtree return -1, return False
        # Return max height plus 1 in each layer

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if not root:
                return 0

            left = helper(root.left)
            right = helper(root.right)

            if abs(left-right) > 1:
                return -1

            if left == -1 or right == -1:
                return -1

            return max(left, right) + 1

        retval = helper(root)

        if retval == -1:
            return False
        return True

Path sum (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def hasPathSum(self, root: Optional[TreeNode], targetSum: int) -> bool:

        # Return True when the running sum equals to target

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root, running_sum, target_sum):

            if root:

                if running_sum + root.val == target_sum and not root.left and not root.right:
                    return True

                left = helper(root.left, running_sum+root.val, target_sum)
                right = helper(root.right, running_sum+root.val, target_sum)

                return left or right

        return helper(root, 0, targetSum)

Path sum II (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:

        # Save the result when the running sum equals to target

        self.paths = []

        # Time O(n) where n is number of nodes
        # Space O(n) for each node, we are constructing its path
        def helper(root, path, running_sum, target_sum):

            if root:

                if running_sum + root.val == target_sum and not root.left and not root.right:
                    self.paths.append(path+[root.val])

                helper(root.left, path+[root.val], running_sum+root.val, target_sum)
                helper(root.right, path+[root.val], running_sum+root.val, target_sum)

            return self.paths

        return helper(root, [], 0, targetSum)

Binary tree maximum path sum (H)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:

        # If node is None, return 0
        # Consider each node. Max sum at each node is the sum of
        #   node value itself
        #   max sum of left subtree
        #   max sum of right subtree
        # Also, in each node, the path can only take either left or right

        self.max_sums = []

        # Time O(n) where n is number of nodes
        # Space O(n) for each node, we are constructing its max sum
        def helper(root):

            if not root:
                return 0

            max_sum_left = max(helper(root.left), 0)
            max_sum_right = max(helper(root.right), 0)

            max_sum = root.val + max_sum_left + max_sum_right
            self.max_sums.append(max_sum)

            max_path = root.val + max(max_sum_left, max_sum_right)
            return max_path

        helper(root)

        return max(self.max_sums)

class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:

        # Find all path from root to all leaf nodes

        self.paths = []

        # Time O(n) where n is number of nodes
        # Space O(n) for each node, we are constructing its path

            if root:

        final_sum = 0
        result = helper(root, [])
        for path in result:
            string = ""

Diameter of binary tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:

        # Diameter of tree at a node is the sum of
        #   Max height of right subtree
        #   Max height of left subtree

        self.max_diameter = 0

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case

        # Find max height of a tree
        def helper(root):

            if not root:
                return 0

            left = helper(root.left)
            right = helper(root.right)

            return max(left, right) + 1

        def helper2(root):

            if root:
                max_height_from_left = helper(root.left) - 1
                max_height_from_right = helper(root.right) - 1
                diameter = max_height_from_left + max_height_from_right + 2

                self.max_diameter = max(self.max_diameter, diameter)
                helper2(root.left)
                helper2(root.right)

        helper2(root)

        return self.max_diameter

Longest univalue path (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def longestUnivaluePath(self, root: Optional[TreeNode]) -> int:

        # Longest path at a node is
        #  longest path of left subtree
        #  longest path of right subtree

        self.longest_path = 0

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case

        # Find max height of a tree only when the node values are the same
        def helper(root, node_value):

            if not root or root.val != node_value:
                return 0

            left = helper(root.left, node_value)
            right = helper(root.right, node_value)

            return max(left, right) + 1

        def helper2(root):

            if root:
                path_length_from_left = helper(root.left, root.val) - 1
                path_length_from_right = helper(root.right, root.val) - 1
                path_legnth = path_length_from_left + path_length_from_right + 2
                self.longest_path = max(self.longest_path, path_legnth)

                helper2(root.left)
                helper2(root.right)

        helper2(root)

        return self.longest_path
