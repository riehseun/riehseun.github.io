Letter Combinations of a Phone Number (M/F1/100)

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        # Store the mapping between number and letter in a dictionary.
        mapping = {
            "2":"abc",
            "3":"def",
            "4":"ghi",
            "5":"jkl",
            "6":"mno",
            "7":"pqrs",
            "8":"tuv",
            "9":"wxyz"
        }

        result = []

        # Base case.
        if len(digits) == 0:
            return []
        else:
            for char in mapping[digits[0]]:
                result.append(char)

        # Recursively compute all possible combinations.

        # Key idea: add each letter mathcing the current digit to all
        #           existing items in the resulting array.

        # Example
        # a b c
        # ad bd cd ae be ce af bf cf

        for digit in digits[1:]:
            result_old = result[:]
            for char in mapping[digit]:
                for item in result_old:
                    result.append(item+char)
            for item in result_old:
                result.remove(item)

        return result

Generate parentheses (M/100)

class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        combinations = []

        self.subroutine(n, 0, 0, "", combinations)

        return combinations

    def subroutine(self, n, num_open, num_close, string, combinations):
        if num_open == n and num_close == n:
            combinations.append(string)

        if num_open < n:
            # No "return" statement.
            self.subroutine(n, num_open+1, num_close, string+"(", combinations)

        # Number of closing parenthesie cannot exceed the number of
        # opening parenthesis at any point to construct valid patterns.
        if num_close < num_open:
            # No "return" statement.
            self.subroutine(n, num_open, num_close+1, string+")", combinations)

Permutation (M)

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        result = []
        self.permute_helper(result, nums, 0, len(nums)-1)
        return result

    def permute_helper(self, result, nums, start, end):
        if start == end:
            result.append(nums)
            return
        else:
            for i in range(start, end+1):
                nums_copy = nums.copy()
                nums_copy[i], nums_copy[start] = nums_copy[start], nums_copy[i]
                self.permute_helper(result, nums_copy, start+1, end)

Subset (M)

class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:

        # Time: O(n * 2**n)
        # Space: O(2**n)

        # If bit is 1, include the number. If bit is 0, exclude the number.

        result = []
        result.append([])

        num_subset = 2 ** len(nums)

        for i in range(1, num_subset):
            binary = bin(i)
            binary = binary.replace("0b","")
            padding = "0" * (len(nums)-len(binary))
            binary = padding + binary
            subset = []

            for j, bit in enumerate(binary):
                if bit == "1":
                    subset.append(nums[j])
            result.append(subset)

        return result

Find K-sum subsets

def get_k_sum_subsets(set_of_integers, target_sum):

    # Time: O(n * 2**n)
    # Space: O(2**n)

    all_subsets = find_all_subsets(set_of_integers)

    result = []
    for subset in all_subsets:
        if sum(subset) == target_sum:
            result.append(subset)

    return result


def find_all_subsets(v):

    result = []

    num_subset = 2 ** len(v)

        binary = padding + binary
        subset = []

            if bit == "1":
                subset.append(v[j])
        result.append(subset)

    return result
