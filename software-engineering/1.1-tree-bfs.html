Binary tree right side view (M) (F) #

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:

        # Do BFS
        # For each depth, find the right-most value
        # Because BFS scans each depth left by right,
        # the right-most value is the latest value
        # explored in that layer

        self.right_side_view = {}

        # Time O(n) 
        #   n = number of nodes
        # Space O(d)
        #   d = depth of tree
        def bfs(root):

            explored = set()
            explored.add(root)

            queue = deque()
            queue.append((root,0))

            while queue:
                root, depth = queue.popleft()
                self.right_side_view[depth] = root.val
                if root.left and root.left not in explored:
                    explored.add(root.left)
                    queue.append((root.left,depth+1))
                if root.right and root.right not in explored:
                    explored.add(root.right)
                    queue.append((root.right,depth+1))

        if not root:
            return []

        result = []
        bfs(root)
        for key,val in self.right_side_view.items():
            result.append(val)

        return result

Binary tree vertical order traversal (M) (F/100) #!

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def verticalOrder(self, root: Optional[TreeNode]) -> List[List[int]]:

        # Use BFS
        # Use hash table to store nodes
        #   BFS guarantees vertical order
        #   Increment/decrement position to remember horizontal order

        ordering = defaultdict(list)
        
        queue = deque()
        queue.append((root, 0))  # Initially, position is 0 

        # Time O(max(n+m, nlogk)) 
        #   n = number of nodes
        #   m = number of edges
        #   k = breadth of tree
        # Space O(n)
        while queue:
            node, position = queue.popleft()

            if node:
                ordering[position].append(node.val)

                queue.append((node.left, position-1))
                queue.append((node.right, position+1))
        
        ordering_sorted_by_key = dict(sorted(ordering.items()))
        result = []

        for k,v in ordering_sorted_by_key.items():
            result.append(v)
            
        return result

Vertical Order Traversal of a Binary Tree (H) (F) !

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def verticalTraversal(self, root: Optional[TreeNode]) -> List[List[int]]:

        # Use BFS
        # Use hash table to store nodes
        #   Storing nodes as traversing guarantees vertical order
        #   Use key in hash table to remember horizontal order

        # Time O(nlogn) n = number of nodes
        # Space O(n)

        nodes = []

        queue = deque()
        queue.append((root, 0, 0))

        while queue:
            node, row, column = queue.popleft()

            if node:
                nodes.append((node.val, row, column))

                queue.append((node.left, row+1, column-1))
                queue.append((node.right, row+1, column+1))

        nodes.sort(key = lambda x: (x[2], x[1], x[0]))

        result = []
        last_column = -math.inf
        group = []
        for node in nodes:
            if node[2] != last_column:
                group = [node[0]]
                last_column = node[2]
                result.append(group)
            else:
                group.append(node[0])

        return result

Delete Nodes And Return Forest (M) (100) !

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def delNodes(self, root: Optional[TreeNode], to_delete: List[int]) -> List[TreeNode]:

        # BFS

        if not root:
            return []

        result = []

        # Time O(n) n = number of nodes
        # Space O(n)
        def bfs(root):

            explored = set()
            explored.add(root)

            queue = deque()
            queue.append(root)

            while queue:
                node = queue.popleft()

                if node.left and node.left not in explored:
                    explored.add(node.left)
                    queue.append(node.left)
                    # Disconnect left sub tree
                    if node.left.val in to_delete:
                        node.left = None
                if node.right and node.right not in explored:
                    explored.add(node.right)
                    queue.append(node.right)
                    # Disconnect right sub tree
                    if node.right.val in to_delete:
                        node.right = None

                # If current node is deleted, add its children to result
                if node.val in to_delete:
                    if node.left:
                        result.append(node.left)
                    if node.right:
                        result.append(node.right)

        bfs(root)

        if root.val not in to_delete:
            result.append(root)

        return result
