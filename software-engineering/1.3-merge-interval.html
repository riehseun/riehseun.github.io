Merge interval (M/Ga/F1/100)

class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:

        # Save the first interval into the result
        # While iterating the rest of intervals
        #   If the end of the last interval is greater than the current,
        #   then there is an overlap
        #   Then, merge these two intervals
        #   Else, just append the current interval to the result array

        if len(intervals) <= 1:
            return intervals

        intervals.sort(key = lambda x: x[0])
        result = []
        result.append(intervals[0])

        # Time O(n) n = len(intervals)
        # Space O(1) ignoring space for output
        for interval in intervals[1:]:
            # If overlap is found
            if result[-1][1] >= interval[0] and result[-1][0] <= interval[1]:
                last_interval = result.pop()
                merged_interval = [
                    min(last_interval[0], interval[0]),
                    max(last_interval[1], interval[1])
                ]
                result.append(merged_interval)
            else:
                result.append(interval)

        return result

Meeting rooms II (M/G1/F1/100)

class Solution:
    def minMeetingRooms(self, intervals: List[List[int]]) -> int:

        # Sort all the meetings by start time
        # Insert the end time of the first meeting into min heap
        #
        # Loop through intervals
        #   If earliest end time in heap <= start time of current meeting
        #     No additional room is required
        #     Pop from heap because the meeting at the root of heap is now
        #     scheduled back to back with the current meeting
        #   Else
        #     Increment room count
        #   Insert the current meeting into heap to update the earlist end time

        heap = []
        intervals.sort(key = lambda x: (x[0]))
        heappush(heap, intervals[0][1])
        num_rooms = 1

        # Time O(nlogn) n = number of intervals, heap operation is logn
        # Space O(n) worst case every interval need separate room
        for interval in intervals[1:]:
            if heap[0] <= interval[0]:
                heappop(heap)
            else:
                num_rooms += 1

            heappush(heap, interval[1])

        return num_rooms

Employee free time (H/G1)

"""
# Definition for an Interval.
class Interval:
    def __init__(self, start: int = None, end: int = None):
        self.start = start
        self.end = end
"""

class Solution:
    def employeeFreeTime(self, schedule: '[[Interval]]') -> '[Interval]':

        # Sort the interval by the start, and then end by reverse
        # If any start is after the latest end seen,
        # then there is non-intersection

        result = []

        # Time O(nlogn) n = length of combined input arrays
        # Space O(n)
        def convert_input_to_list(schedules):

            flat_array = []

            for schedule in schedules:
                for array in schedule:
                    flat_array.append([array.start, array.end])

            flat_array.sort(key = lambda x: (x[0], -x[1]))

            return flat_array

        array = convert_input_to_list(schedule)

        latest_end = 0
        for interval in array:

            if interval[0] > latest_end and latest_end != 0:
                    result.append(Interval(latest_end, interval[0]))
            latest_end = max(latest_end, interval[1])

        return result

Interval intersection (M/F1/100)

class Solution:
    def intervalIntersection(self, firstList: List[List[int]], secondList: List[List[int]]) -> List[List[int]]:

        # Time: O(n+m)
        # Space: O(min(n,m))
        # n, m = the lengths of input arrays.

        if not firstList or not secondList:
            return []

        intersection = self.find_intersection(firstList[0], secondList[0])

        result = []
        i = 0
        j = 0
        while i < len(firstList) \
            and j < len(secondList):

            # Check for intersection
            intersection = self.find_intersection(firstList[i], secondList[j])

            if intersection:
                result.append(intersection)

            if firstList[i][1] <= secondList[j][1]:
                i += 1
            else:
                j += 1

        return result

    def find_intersection(self, a, b):

        if a[0] <= b[1] and b[0] <= a[1]:
            intersection = [max(a[0], b[0]), min(a[1], b[1])]
        else:
            intersection = []

        return intersection
