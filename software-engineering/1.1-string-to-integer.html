Group Anagrams (M)

class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:

        # Anagrams have the same char count
        # Use a tuple (of size 26) to store alphabet count
        #   Use tuple as key in dictionary
        #   Put anagram as value in dictionary

        anagrams = defaultdict(list)

        # Time O(n)
        #   n = total number of chars in strs
        # Space O(n)
        for s in strs:
            alphabet = [0] * 26
            for c in s:
                index = ord("a") - ord(c)
                alphabet[index] += 1
            anagrams[tuple(alphabet)].append(s)

        result = []
        for key, val in anagrams.items():
            result.append(val)

        return result
