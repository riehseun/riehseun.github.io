Reverse integer (M)

class Solution:
    def reverse(self, x: int) -> int:

        is_neg = False
        x = str(x)

        # Extract negative sign if negative number
        if x[0] == "-":
            x = x[1:]
            is_neg = True

        # Place each digit in array
        # Time O(n) n = len(x)
        # Space O(n)
        arr = []
        for i in x:
            arr.append(i)
        arr.reverse()
        result = "".join(arr)

        if is_neg:
            if -int(result) < -2**31:
                return 0
            else:
                return -int(result)
        else:
            if int(result) > 2**31 - 1:
                return 0
            else:
                return int(result)

Multiply Strings (M)

class Solution:
    def multiply(self, num1: str, num2: str) -> str:

        # Ex.
        # 23
        # 56
        #    18
        #   12
        #   15
        #  10

        # Multiply digit by digt, 
        # adding zeros to output based on location of digits 
        # Sum all outputs

        n = len(num1)
        m = len(num2)

        result = 0 

        # Time O(nm)
        #   n = len(num1)
        #   m = len(num2)
        # Space O(1)
        for i in range(n-1, -1, -1):
            for j in range(m-1, -1, -1):
                out = int(num1[i]) * int(num2[j])
                out *= 10 ** (n-i-1+m-j-1)
                result += out 
        
        return str(result)

Pow(x, n) (M) (Meta)

class Solution:
    def myPow(self, x: float, n: int) -> float:

        # Ex. 2 ** 10
        #     = (2*2) ** 5
        #.    = 4 ** 5
        #.    = 4 * (4 ** 4)
        #.    = 4 * (4*4) ** 2
        #.    = 4 * (16 ** 2)
        #     = 4 * (16*16) ** 1  
        # If n is even, convert to (x*x) ** (n/2)
        # If n is odd, convert to x * (x ** (n-1)) and multiply x to result

        result = 1

        negative = False
        if n < 0:
            negative = True
            n = -n

        # Time O(n)
        #   n is divided by half
        # Space O(1)
        while n > 0:

            if n % 2 == 0:
                x *= x
                n /= 2
            else:    
                result *= x
                n -= 1

        if negative:
            return 1/result
        return result

Plus One (E)

class Solution:
    def plusOne(self, digits: List[int]) -> List[int]:

        # Add 1 to last digit remembering the carry

        n = len(digits) - 1

        # Time O(n)
        #   n = len(digits)
        # Space O(1)
        while n >= 0:
            digits[n] += 1
            if digits[n] % 10 == 0:
                digits[n] = 0
                n -= 1
            else:
                break

        # Carry overflew to before the first digit
        if n == -1:
            digits = [1] + digits
        return digits

Random Pick Index (M)

class Solution:

    # Use hash table to map num to indexes
    # Use random function to generate an integer 
    # with a range that equals to size of indexes array  

    # Time O(n)
    #   n = len(nums)
    # Space O(n)
    def __init__(self, nums: List[int]):
        
        self.indexes = defaultdict(list)
        for i, num in enumerate(nums):
            self.indexes[num].append(i)
        
    # Time O(1)
    # Space O(n)
    def pick(self, target: int) -> int:

        size = len(self.indexes[target])
        random_number = randint(0, size-1)
        return self.indexes[target][random_number]

Add Strings (E)

class Solution:
    def addStrings(self, num1: str, num2: str) -> str:

        # Interpret each digit as integer by the difference
        # in unicode representation between the digit and "0"
        # Add digit by digit while tracking carry

        result = []
        i = len(num1) - 1
        j = len(num2) - 1
        carry = 0

        # Time O(n) 
        #   n = max(len(num1), len(num2))
        # Space O(1)
        #   Ignoring space for output
        while i >= 0 or j >= 0:
            value = carry

            if i >= 0:
                value += (ord(num1[i])-ord("0"))
            if j >= 0:
                value += (ord(num2[j])-ord("0"))

            if value >= 10:
                carry = 1
            else:
                carry = 0
            result.append(str(value%10))
            i -= 1
            j -= 1

        if carry:
            result.append(str(carry))

        return ''.join(result[::-1])

Fibonacci Number (E)

class Solution:
    def fib(self, n: int) -> int:

        if n == 0:
            return 0
        if n == 1:
            return 1

        return self.fib(n-1) + self.fib(n-2)
