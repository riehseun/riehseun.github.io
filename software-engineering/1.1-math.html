Reverse integer (M) (Google)

class Solution:
    def reverse(self, x: int) -> int:

        r, sign = 0, "+"
        if x < 0:
            sign = "-"
            x = -x
            
        # Time O(logn)
        #   Dividing input by 10 in every iteration
        # Space O(1)
        while x > 0:
            r = r * 10 + x % 10
            x = x // 10
        
        if sign == "-":
            r = -r

        if -2**31 <= r <= 2**31 - 1:
            return r

        return 0

Palindrome number (E) (Google)

class Solution:
    def isPalindrome(self, x: int) -> bool:

        # Ex. 12321
        # Find reverse of right half and check it matches the left half
        #   left half - continue x // 10
        #   reverse of right half - collect results from x % 10   

        if x < 0 or (x % 10 == 0 and x != 0):
            return False

        left, right = x, 0

        # Time O(logn)
        #   Dividing input by 10 in every iteration
        # Space O(1)
        while left > right:
            right = right * 10 + left % 10
            left = left // 10
            
        if left == right or left == right // 10:
            return True

        return False

Pow(x, n) (M) (Meta)

class Solution:
    def myPow(self, x: float, n: int) -> float:

        # Ex. 2 ** 10
        #     = (2*2) ** 5
        #.    = 4 ** 5
        #.    = 4 * (4 ** 4)
        #.    = 4 * (4*4) ** 2
        #.    = 4 * (16 ** 2)
        #     = 4 * (16*16) ** 1  
        # If n is even, convert to (x*x) ** (n/2)
        # If n is odd, convert to x * (x ** (n-1)) and multiply x to result

        result = 1

        negative = False
        if n < 0:
            negative = True
            n = -n

        # Time O(n)
        #   n is divided by half
        # Space O(1)
        while n > 0:

            if n % 2 == 0:
                x *= x
                n /= 2
            else:    
                result *= x
                n -= 1

        if negative:
            return 1/result
        return result

Plus One (E) (Google)

class Solution:
    def plusOne(self, digits: List[int]) -> List[int]:

        # Add 1 to the last element of array remembering the carry

        carry = 1

        # Time O(n)
        #  n = len(digits)
        # Space O(1)
        for i in range(len(digits)-1, -1, -1):
            if digits[i] + carry > 9:
                carry = 1
                digits[i] = 0
            else:
                carry = 0
                digits[i] += 1
                break

        if carry == 1:
            digits = [1] + digits
            
        return digits

Add Strings (E) (Meta)

class Solution:
    def addStrings(self, num1: str, num2: str) -> str:

        # Interpret each digit as integer by the difference
        # in unicode representation between the digit and "0"
        # Add digit by digit while tracking carry

        result = []
        i = len(num1) - 1
        j = len(num2) - 1
        carry = 0

        # Time O(n) 
        #   n = max(len(num1), len(num2))
        # Space O(1)
        #   Ignoring space for output
        while i >= 0 or j >= 0:
            value = carry

            if i >= 0:
                value += (ord(num1[i])-ord("0"))
            if j >= 0:
                value += (ord(num2[j])-ord("0"))

            if value >= 10:
                carry = 1
            else:
                carry = 0
            result.append(str(value%10))
            i -= 1
            j -= 1

        if carry:
            result.append(str(carry))

        return ''.join(result[::-1])
