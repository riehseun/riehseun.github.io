Reverse integer (M) (G/A/100) !!!

class Solution:
    def reverse(self, x: int) -> int:

        is_neg = False
        x = str(x)

        # Extract negative sign if negative number
        if x[0] == "-":
            x = x[1:]
            is_neg = True

        # Place each digit in array
        # Time O(n) n = len(x)
        # Space O(n)
        arr = []
        for i in x:
            arr.append(i)
        arr.reverse()
        result = "".join(arr)

        if is_neg:
            if -int(result) < -2**31:
                return 0
            else:
                return -int(result)
        else:
            if int(result) > 2**31 - 1:
                return 0
            else:
                return int(result)

Pow(x, n) (M) (G/F) !#

class Solution:
    def myPow(self, x: float, n: int) -> float:
        
        # In while loop, multiply result, by x, n times
        # If n < 0, invert x
        # Optimize whule loop
        #   If n is even, reduce n by half, and square x by x
        
        # Time O(logn)
        #   n is reduced by half in each iteration
        # Space O(1)

        if n < 0:
            n = -n
            x = 1 / x

        result = 1
        while n != 0:

            if n % 2 == 1:
                result *= x
                n -= 1
            else:
                x *= x
                n /= 2

        return result

cclass Solution:
    def calculate(self, s: str) -> int:

        # If token is a digit
        #   Set current number to the digit
        #   If two consecutive tokens are digits, 
        #   multiply first digit by 10
        #   If three consecutive tokens are digits,
        #   multiply first digit by 100
        #   And so on
        # Check if last operation is "+" or "-", 
        #   The expression should be evaludated later
        #   Add/subtract the expression to/from result
        #   Set last_num to curr_num
        # Check if last operation is "*" or "/"
        #   The expression should be evaludated immediately
        #   Multiply/divide last_num by curr_num
        
        # Remove all whitespaces from the input string
        s = s.replace(" ", "")
        # This allow last operator to be evaluated
        s = s + "+"
        # Assume operator of first digit is positive
        last_op = "+"

        curr_num, last_num, result = 0, 0, 0
    
        # Time O(n) 
        #   n = len(s)
        # Space O(1)
        for token in s:

            if token.isdigit():
                curr_num = curr_num * 10 + int(token)
                continue    
            
            if last_op == "+":
                result += last_num
                last_num = curr_num
            elif last_op == "-":
                result += last_num
                last_num = -curr_num
            elif last_op == "*":
                last_num *= curr_num
            elif last_op == "/":
                last_num = int(last_num/curr_num)
            curr_num = 0
            last_op = token
                
        return result + last_num

class Solution:
    def numberToWords(self, num: int) -> str:

        # Sparate by ones, thousands, millions, billions

        # Corner case
        if num == 0:
            return "Zero"

        ones = {
            0: "Zero",
            1: "One",
            2: "Two",
            3: "Three",
            4: "Four",
            5: "Five",
            6: "Six",
            7: "Seven",
            8: "Eight",
            9: "Nine",
            10: "Ten",
            11: "Eleven",
            12: "Twelve",
            13: "Thirteen",
            14: "Fourteen",
            15: "Fifteen",
            16: "Sixteen",
            17: "Seventeen",
            18: "Eighteen",
            19: "Nineteen"
        }

        tens = {
            2: "Twenty",
            3: "Thirty",
            4: "Forty",
            5: "Fifty",
            6: "Sixty",
            7: "Seventy",
            8: "Eighty",
            9: "Ninety",
        }

        thousands = {
            0: "",
            1: "Thousand",
            2: "Million",
            3: "Billion"
        }

        total_result = ""
        count = 0

        # Time O(1)
        # Space O(1)
        while num:

            result = ""
            quotient = num % 1000

            flag = True
            if quotient == 0:
                flag = False

            remainder = num // 1000

            if quotient >= 100:
                result += ones[quotient//100] + " Hundred "
                quotient %= 100

            if 100 > quotient >= 20:
                result += tens[quotient//10] + " "
                quotient %= 10

            if 20 > quotient > 0:
                result += ones[quotient] + " "

            num = remainder
            if flag:
                result += thousands[count] + " "
            count += 1
            total_result = result + total_result

        return total_result.strip()

Random Pick Index (M) (F) !

class Solution:

    # Use hash table to map number to indexes
    # Use random function to generate an integer
    # with a range that equals to the size of indexes array

    # Time O(n) n = len(nums)
    # Space O(n)
    def __init__(self, nums: List[int]):

        self.indexes = defaultdict(list)
        for i, num in enumerate(nums):
            self.indexes[num].append(i)

    def pick(self, target: int) -> int:

        size = len(self.indexes[target])
        random_number = randint(0, size-1)
        return self.indexes[target][random_number]

Add Strings (E) (F) !

class Solution:
    def addStrings(self, num1: str, num2: str) -> str:

        # Interpret each digit as integer by the difference
        # in unicode representation between the digit and "0"
        # Add digit by digit while tracking carry

        result = []
        i = len(num1) - 1
        j = len(num2) - 1
        carry = 0

        # Time O(n) n = max(len(num1), len(num2))
        # Space O(n)
        while i >= 0 or j >= 0:

            if i >= 0:
                x1 = ord(num1[i]) - ord("0")
            else:
                x1 = 0
            if j >= 0:
                x2 = ord(num2[j]) - ord("0")
            else:
                x2 = 0
            value = (x1 + x2 + carry) % 10
            carry = (x1 + x2 + carry) // 10
            result.append(str(value))
            i -= 1
            j -= 1

        if carry:
            result.append(str(carry))

        return ''.join(result[::-1])
