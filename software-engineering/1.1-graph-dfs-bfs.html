Longest increasing path in a matrix (H/Ga)

class Solution:
    def longestIncreasingPath(self, matrix: List[List[int]]) -> int:

        n = len(matrix)
        m = len(matrix[0])

        cache = [[0 for _ in range(m)] for _ in range(n)]
        max_path = 1
        for i in range(n):
            for j in range(m):
                curr_path = self.dfs(matrix, i, j, n, m, cache)
                max_path = max(max_path, curr_path)

        return max_path

    def dfs(self, matrix, i, j, n, m, cache):
        dirs = [(-1,0), (0,1), (1,0), (0,-1)]
        if cache[i][j] != 0:
            return cache[i][j]

        max_path = 1

        for d in dirs:
            new_i = i + d[0]
            new_j = j + d[1]
            # if new i, j is less than or equal to the current, skip
            if new_i < 0 or new_i >= n or new_j < 0 or new_j >= m or matrix[new_i][new_j] <= matrix[i][j]:
                continue

            curr_path = 1 + self.dfs(matrix, new_i, new_j, n, m, cache)
            max_path = max(max_path, curr_path)

        cache[i][j] = max_path
        return max_path

Evaluate division (M/Ga)

class Solution:
    def calcEquation(self, equations: List[List[str]], values: List[float], queries: List[List[str]]) -> List[float]:

        # Construct a graph such that
        #   If a / b = 2.0, a->b has weight 2.0 and b->a has weight 0.5

        graph = defaultdict(list)
        nodes = set()

        for equation, value in zip(equations, values):
            graph[equation[0]].append((equation[1],value))
            graph[equation[1]].append((equation[0],1/value))
            nodes.add(equation[0])
            nodes.add(equation[1])

        result = []
        for query in queries:
            if query[0] == query[1] and query[0] in nodes:
                result.append(1)
            else:
                result.append(self.dfs(graph, query[0], query[1]))

        return result

    # Number of nodes n = size of input array (equations)
    # Number of edges m = size of input array (equations)
    # Time O(nm)
    # Space O(n)
    def dfs(self, graph, start, end):

        explored = set()
        explored.add(start)

        stack = []
        stack.append((start, 1))

        while stack:
            node, ret_val = stack.pop()
            for next_node, value in graph[node]:
                if next_node not in explored:
                    explored.add(next_node)
                    stack.append((next_node, ret_val*value))

                    if next_node == end:
                        return ret_val*value

        return -1

Accounts Merge (M/F1)

class Solution:
    def accountsMerge(self, accounts: List[List[str]]) -> List[List[str]]:

        # Construct a graph
        #   Node is email
        #   Edge between nodes (emails) is when they belong to the same person
        # Compute connected components

        graph = defaultdict(list)
        email_to_person = defaultdict(str)

        for account in accounts:
            # There is only one email
            if len(account) == 2:
                graph[account[1]]
                email_to_person[account[1]] = account[0]
            for i in range(1, len(account)):
                for j in range(1, len(account)):
                    if i != j:
                        graph[account[i]].append(account[j])
                        email_to_person[account[i]] = account[0]

        # Time O(n+m)
        #   n = number of emails
        #   m = number of connnected emails
        # Space O(n)
        def bfs(graph, explored, start):

            queue = deque()
            queue.append(start)

            nodes = []

            while queue:
                node = queue.popleft()
                if node not in nodes:
                    nodes.append(node)

                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        queue.append(next_node)

            return nodes

        result = []

        explored = set()
        for node in email_to_person.keys():
            if node not in explored:
                nodes = bfs(graph, explored, node)
                nodes.sort()
                person = email_to_person[nodes[0]]
                result.append([person]+nodes)

        return result

All Nodes Distance K in Binary Tree (M/F1)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distanceK(self, root: TreeNode, target: TreeNode, k: int) -> List[int]:

        # Convert tree to graph
        # DFS

        graph = defaultdict(list)

        # Time O(n) n = number of nodes in tree
        # Space O(n)
        def build_graph(graph, node, parent):
            if node and parent:
                graph[node.val].append(parent.val)
                graph[parent.val].append(node.val)
            if node.left:
                build_graph(graph, node.left, node)
            if node.right:
                build_graph(graph, node.right, node)

        build_graph(graph, root, None)

        print(graph)

        distances = defaultdict(list)

        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n)
        def dfs(graph, start, k):

            explored = set()
            explored.add(start)

            stack = []
            stack.append((start, 0))

            while stack:
                node, depth = stack.pop()
                distances[depth].append(node)

                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append((next_node, depth+1))

        dfs(graph, target.val, k)

        return distances[k]
