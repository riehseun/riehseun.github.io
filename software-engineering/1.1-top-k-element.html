Kth largest element in an array (M) (G/F/A) !!!

class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:

        # Push number into a min heap
        # if exceed k, pop from heap

        min_heap = []

        # Time O(nlogk) n = len(nums)
        # Space O(k)
        for num in nums:
            heappush(min_heap, num)
            if len(min_heap) > k:
                heappop(min_heap)

        return min_heap[0]

Kth largest element in a stream (E) (A)

class KthLargest:

    # Initialize min_heap with just K elements from input array
    # As items are added, maintain K elements in min_heap

    def __init__(self, k: int, nums: List[int]):
        self.k = k
        self.min_heap = nums
        heapify(self.min_heap)
        while len(self.min_heap) > k:
            heappop(self.min_heap)

    # Time O(nlogn)
    # Space O(n)
    def add(self, val: int) -> int:
        heappush(self.min_heap, val)
        if len(self.min_heap) > self.k:
            heappop(self.min_heap)
        return self.min_heap[0]
