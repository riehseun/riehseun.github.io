Serialize and deserialize binary tree (H/Fa)

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Codec:

    # Time O(n) where n is number of nodes
    # Space O(h) where h = n if worst case, h = log(n) if best case

    def serialize(self, root):
        """Encodes a tree to a single string.

        :type root: TreeNode
        :rtype: str
        """

        self.data = ""

        def helper(root):

            if root:
                self.data += str(root.val)
                self.data += ","
                helper(root.left)
                helper(root.right)
            else:
                self.data += "#"
                self.data += ","

        helper(root)
        return self.data[:len(self.data)-1]

    def deserialize(self, data):
        """Decodes your encoded data to tree.

        :type data: str
        :rtype: TreeNode
        """

        data_array = data.split(",")

        def helper(data_array):
            val = next(data_array)
            if val != "#":
                root = TreeNode(int(val))
                root.left = helper(data_array)
                root.right = helper(data_array)
                return root

        return helper(iter(data_array))

Range sum of bst (E/F)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rangeSumBST(self, root: Optional[TreeNode], low: int, high: int) -> int:

        # Get all nodes first
        # Then, find numbers within the range

        self.nodes = []

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def pre_order(root):

            if root:
                self.nodes.append(root.val)
                pre_order(root.left)
                pre_order(root.right)

        pre_order(root)

        range_sum = 0
        for node in self.nodes:
            if node >= low and node <= high:
                range_sum +=  node

        return range_sum
