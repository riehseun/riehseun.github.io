Rotate Image (M)

class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """

        # First, reverse up and down
        # Second, swap i and j

        i, j = 0, len(matrix) - 1
        
        # Time O(nm)
        #   n = number of rows
        #   m = number of cols
        # Space O(1)
        while i < j:
            matrix[i], matrix[j] = matrix[j], matrix[i]
            i += 1
            j -= 1

        for i in range(len(matrix)):
            for j in range(i+1, len(matrix[0])):
                matrix[i][j], matrix[j][i] = matrix[j][i], matrix[i][j]

Spiral Matrix (M)

class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:

        # Remove rows and columns from the matrix while moving spiral
        # The order is
        #   Remove first row
        #   Remove last col
        #   Remove last row
        #   Remove first col
        # Repeat

        result = []

        # Time O(nm) 
        #   n = number of rows
        #   m = number of columns
        # Space O(nm)

        while matrix:
            nums = matrix.pop(0)  # Remove first row
            result += nums

            if matrix and matrix[0]:
                for row in matrix:
                    num = row.pop()  # Remove last col
                    result += [num]

            if matrix:
                nums = matrix.pop()  # Remove last row
                # Ensure the order that numbers are appended to the result
                nums.reverse()
                result += nums

            if matrix and matrix[0]:
                # Ensure the order that numbers are appended to the result
                matrix_copy = matrix.copy()
                matrix_copy.reverse()
                for row in matrix_copy:
                    num = row.pop(0)  # Remove first col
                    result += [num]

        return result

Set Matrix Zeroes (M)

class Solution:
    def setZeroes(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """
        
        # Use a dictionary to make rows and cols of 0s

        rows, cols = defaultdict(bool), defaultdict(bool)
        
        # Time O(nm)
        #   n = number of rows
        #   m = number of columns
        # Space O(n+m)
        for i in range(len(matrix)):
            for j in range(len(matrix[i])):
                if matrix[i][j] == 0:
                    rows[i] = True
                    cols[j] = True

        for i in range(len(matrix)):
            for j in range(len(matrix[i])):
                if i in rows or j in cols:
                    matrix[i][j] = 0

Design Tic-Tac-Toe (M)

class TicTacToe:

    # After each move, check
    #   Diagonal win
    #   Anti-diagonal win
    #   Vertical win
    #   Horizontal win
    # Keep an array for each player to track the number of times
    # each player marked in each row, col, diagonal, anti-diagonal

    def __init__(self, n: int):
        self.rows = defaultdict(int)
        self.cols = defaultdict(int)
        self.diag = 0
        self.anti_diag = 0
        self.n = n

    # Time O(1)
    # Space O(n)
    def move(self, row: int, col: int, player: int) -> int:
        
        if player == 1:
            self.rows[row] += 1
            self.cols[col] += 1
            if row == col:
                self.diag += 1
            if col == self.n - row - 1:
                self.anti_diag += 1
        elif player == 2: 
            self.rows[row] -= 1
            self.cols[col] -= 1
            if row == col:
                self.diag -= 1
            if col == self.n - row - 1:
                self.anti_diag -= 1

        if abs(self.rows[row]) == self.n \
            or abs(self.cols[col]) == self.n \
            or abs(self.diag) == self.n \
            or abs(self.anti_diag) == self.n:
            return player
                
        return 0
