Largest number (M) (G/100) !!

class Solution:
    def largestNumber(self, nums: List[int]) -> str:

        # Do a quicksort
        #   Pick a pivot
        #   Put elements smaller than pivot to left of pivot
        #   Put elements larger than pivot to right of pivot
        # When comparing two numbers check if "AB" > "BA"

        if sum(nums) == 0:
            return "0"

        # Time O(nlogn) n = len(nums)
        # Space O(1)
        def quicksort(nums, l, r):
            if l >= r:
                return
            pos = partition(nums, l, r)
            quicksort(nums, l, pos-1)
            quicksort(nums, pos+1, r)

        def partition(nums, l, r):

            p = random.randint(l,r)

            for i in range(l,r+1):
                if compare(nums[i],nums[p]) and i > p:
                    nums[i], nums[p] =  nums[p], nums[i]
                if not compare(nums[i],nums[p]) and i < p:
                    nums[i], nums[p] =  nums[p], nums[i]
                p = i

            return p

        def compare(x, y):
            return str(x) + str(y) > str(y) + str(x)

        quicksort(nums, 0, len(nums)-1)
        result = ""
        for num in nums:
            result += str(num)
        return result

class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:

        # Quickselect
        # Partition
        #   Numbers greater than pivot should be on left side
        #   Numbers less than pivot should be on right side
        # If k < p, search on left side
        # If k > p, search on right side
        # If k == p, return value at pivot

        # Time O(n) on average 
        #   n = len(nums)
        # Space O(1)
        def quickselect(nums, l, r):

            p = partition(nums, l, r)
            if k-1 < p:
                return quickselect(nums, l, p-1)
            elif k-1 > p:
                return quickselect(nums, p+1, r)
            else:
                return nums[p]

        def partition(nums, l, r):

            p = random.randint(l, r)
            
            for i in range(l, r+1):
                if nums[i] > nums[p] and i > p:
                    nums[i], nums[p] = nums[p], nums[i]
                if nums[i] < nums[p] and i < p:
                    nums[i], nums[p] = nums[p], nums[i]
                p = i

            return p
                
        return quickselect(nums, 0, len(nums)-1)

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:

        # Quickselect
        # Partition
        #   Frequencies greater than pivot should be on left side
        #   Frequencies less than pivot should be on right side
        # If k < p, search on left side
        # If k > p, search on right side
        # If k == p, return value at pivot

        # Time O(n)
        #   n = len(nums)
        # Space O(n)
        nums_dict = defaultdict(int)
        for num in nums:
            nums_dict[num] += 1

        nums_arr = list(nums_dict.keys())
        
        # Time O(n) on average
        # Space O(1)
        def quickselect(nums_dict, nums_arr, l, r):

            p = partition(nums_dict, nums_arr, l, r)
            if k-1 < p:
                return quickselect(nums_dict, nums_arr, l, p-1)
            elif k-1 > p:
                return quickselect(nums_dict, nums_arr, p+1, r)
            else:
                return nums_arr[:p+1]

        def partition(nums_dict, nums_arr, l, r):

            p = random.randint(l, r)
            
            for i in range(l, r+1):
                if nums_dict[nums_arr[i]] > nums_dict[nums_arr[p]] and i > p:
                    nums_arr[i], nums_arr[p] = nums_arr[p], nums_arr[i]
                if nums_dict[nums_arr[i]] < nums_dict[nums_arr[p]] and i < p:
                    nums_arr[i], nums_arr[p] = nums_arr[p], nums_arr[i]
                p = i

            return p
                
        return quickselect(nums_dict, nums_arr, 0, len(nums_arr)-1)

Minimum Time Difference (M) (100) !

class Solution:
    def findMinDifference(self, timePoints: List[str]) -> int:

        # Convert time format to integer representing number of minutes
        # Sort (bucket sort since range of values is knwon)
        times = []

        for timepoint in timePoints:
            hour = timepoint[0:2]
            minute = timepoint[3:5]
            times.append(int(hour)*60+int(minute))

        # Time O(n) n = len(timePoints)
        # Space O(1) size is limited to 24 * 60
        buckets = defaultdict(int)
        for time in times:
            if time in buckets:
                return 0
            buckets[time] = True

        result = math.inf
        prev = math.inf
        first = math.inf
        last = math.inf

        # Time O(1)
        for i in range(24*60):
            if buckets[i]:
                if prev != math.inf:
                    result = min(result, i-prev)
                prev = i
                if first == math.inf:
                    first = i
                last = i

        return min(result, 24*60-last+first)

class Solution:
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:

        # Quickselect
        # Partition
        #   Distances less than pivot should be on left side
        #   Distances greater than pivot should be on right side
        # If k < p, search on left side
        # If k > p, search on right side
        # If k == p, return value at pivot

        # Time O(n) on average 
        #   n = len(points)
        # Space O(1)
        def quickselect(points, l, r):

            p = partition(points, l, r)
            if k-1 < p:
                return quickselect(points, l, p-1)
            elif k-1 > p:
                return quickselect(points, p+1, r)
            else:
                return points[:p+1]

        def partition(points, l, r):

            p = random.randint(l, r)
            
            for i in range(l, r+1):
                if dist(points[i]) < dist(points[p]) and i > p:
                    points[i], points[p] = points[p], points[i]
                if dist(points[i]) > dist(points[p]) and i < p:
                    points[i], points[p] = points[p], points[i]
                p = i

            return p

        def dist(point):
            return point[0]**2+point[1]**2
                
        return quickselect(points, 0, len(points)-1)

Minimum Difference Between Largest and Smallest Value in Three Moves (M) (100) !

class Solution:
    def minDifference(self, nums: List[int]) -> int:

        # Potential three numbers to change
        #   3 largest
        #   2 largest and 1 smallest
        #   1 largest and 2 smallest
        #   3 smallest
        # Sort

        if len(nums) <= 3:
            return 0

        # Time O(nlogn) n = len(nums)
        # Space O(1)
        nums.sort()

        return min(
            nums[len(nums)-4]-nums[0],
            nums[len(nums)-3]-nums[1],
            nums[len(nums)-2]-nums[2],
            nums[len(nums)-1]-nums[3])
