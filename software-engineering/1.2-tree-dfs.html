Same Tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isSameTree(self, p: Optional[TreeNode], q: Optional[TreeNode]) -> bool:
        
        # Pre-order traversal

        self.is_same = True

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   Recursion stack
        def pre_order(n1, n2):

            if n1 and n2:
                if n1.val != n2.val:
                    self.is_same = False
                pre_order(n1.left, n2.left)
                pre_order(n1.right, n2.right)
            else:
                if n1 or n2:
                    self.is_same = False

        pre_order(p, q)
        return self.is_same

Balanced Binary Tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isBalanced(self, root: Optional[TreeNode]) -> bool:

        # Post-order traverse
        # If height diff at any layer > 1, return False

        self.result = True

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   Recursion stack
        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)
                if abs(left-right) > 1:
                    self.result = False
                return max(left, right) + 1
            else:
                return 0
        
        post_order(root)
        return self.result

Binary Tree Maximum Path Sum (H)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:

        # The path sum is 
        #   node.val 
        #   + path sum of left subtree 
        #   + path sum of right subtree
        # The path sum of leaf nodes are its own value
        #   When node does not exist, return 0
        # Traverse tree
        #   For each node, compute path sum
        #   Since both left and right children must be
        #   computed first, do post_order traversal
        # Each child can either pick contribution from its
        # left child or right child to have valid path

        self.max_sum = -math.inf

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   Call stack for recursion
        def post_order(node):

            if node:
                max_sum_from_left = max(post_order(node.left), 0)
                max_sum_from_right = max(post_order(node.right), 0)
                node_sum = node.val + max_sum_from_left + max_sum_from_right
                self.max_sum = max(node_sum, self.max_sum)
                # Contribution from Subtree can either be 
                # from left or right, not both 
                return max(
                    node.val+max_sum_from_left, 
                    node.val+max_sum_from_right
                )

            else:
                return 0

        post_order(root)
        return self.max_sum

Sum Root to Leaf Numbers (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:

        # DFS pre-order
        # Calculate running sum for each path
        #   At each node, multiply running sum by 10
        #   If leaf node, add running sum to total sum  

        self.total_sum = 0

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   Recursion stack h = n in worst case
        def pre_order(node, running_sum):

            if node:
                running_sum = running_sum * 10 + node.val
                
                if not node.left and not node.right:
                    self.total_sum += running_sum
                
                pre_order(node.left, running_sum)
                pre_order(node.right, running_sum)                

        pre_order(root, 0)    

        return self.total_sum

Invert Binary Tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def invertTree(self, root: Optional[TreeNode]) -> Optional[TreeNode]:
        
        # Pre-order traverse
        # Swap left and right children

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(logh)
        #   h = height of tree
        #   Recursion stack
        def pre_order(node):

            if node:
                node.left, node.right = node.right, node.left

                pre_order(node.left)
                pre_order(node.right)

        pre_order(root)
        return root

Lowest Common Ancestor of a Binary Tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':

        # Post-order traverse
        # If leaf node, return 0
        # If p or q is found, return 1
        # If node.val + left child + right child == 2
        #   Current node is lca    

        self.lca = None

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   Recursion stack
        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)

                parent = 0
                if node.val == p.val or node.val == q.val:
                    parent = 1

                if parent + left + right == 2 \
                    and not self.lca:
                    self.lca = node
                    
                return parent + left + right
            else:
                return 0
            
        post_order(root)
        return self.lca

Serialize and Deserialize Binary Tree (H)

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Codec:

    # Time O(n)
    #   n = number of nodes
    # Space O(h)
    #   h = n if worst case
    #   h = log(n) if best case

    def serialize(self, root):
        """Encodes a tree to a single string.

        :type root: TreeNode
        :rtype: str
        """

        self.data = ""

        def pre_order(node):

            if node:
                self.data += str(node.val)
                self.data += ","
                pre_order(node.left)
                pre_order(node.right)
            else:
                self.data += "#"
                self.data += ","

        pre_order(root)
        return self.data[:len(self.data)-1]

    def deserialize(self, data):
        """Decodes your encoded data to tree.

        :type data: str
        :rtype: TreeNode
        """

        data_array = data.split(",")

        def pre_order(data_array):
            val = next(data_array)
            if val != "#":
                node = TreeNode(int(val))
                node.left = pre_order(data_array)
                node.right = pre_order(data_array)
                return node

        return pre_order(iter(data_array))

Find Leaves of Binary Tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def findLeaves(self, root: Optional[TreeNode]) -> List[List[int]]:

        # DFS post-order
        # Add leaf nodes to array
        # Then, remove leaf nodes from the tree
        # Repeat until only the root remains

        # Time O(h) 
        #   h = height of tree
        # Space O(n)
        #   n = number of nodes in tree 
        def post_order(node, array):

            if node:

                # Check if leaf node
                if not node.left and not node.right:
                    array.append(node.val)
                    return -1
                
                left = post_order(node.left, array)
                right = post_order(node.right, array)
                # Remove leaf node on left side
                if left == -1:
                    node.left = None
                # Remove leaf node on right side
                if right == -1:
                    node.right = None
                
        result = []

        while root.left or root.right:
            array = []
            post_order(root, array)
            result.append(array)
            
        result.append([root.val])
        return result

Diameter of Binary Tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:

        # Diameter or longest path is sum of 
        #   root + left subtree + right subtree
        # Do post order traverse
        #   If leaf node, return 1
        #   Else, max path is
        #     longest path of left + longest path of right + 1
        #   Valid path can only have either left or right subtree 

        self.max_path = 0

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   Recursion stack
        self.max_path = 0

        def post_order(node):

            if node:
                max_right = post_order(node.right)
                max_left = post_order(node.left)
                self.max_path = max(self.max_path, max_right+max_left+1)
                return max(max_right, max_left) + 1
            else:
                return 0

        post_order(root)
        return self.max_path - 1

Subtree of Another Tree(E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isSubtree(self, root: Optional[TreeNode], subRoot: Optional[TreeNode]) -> bool:

        # Pre-order traversal with null node

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(n)
        def pre_order(node):
            
            if node:
                self.nodes.append("*")
                self.nodes.append(str(node.val))
                pre_order(node.left)
                pre_order(node.right)
            else:
                self.nodes.append("#")
        
        self.nodes = []
        pre_order(root)
        string = "".join(self.nodes)
        
        self.nodes = []
        pre_order(subRoot)
        substring = "".join(self.nodes)
        
        if substring in string:
            return True
        return False

All Nodes Distance K in Binary Tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distanceK(self, root: TreeNode, target: TreeNode, k: int) -> List[int]:

        # DFS on tree to convert to graph
        # DFS on graph

        graph = defaultdict(set)

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(n)
        def pre_order(node, parent, graph):

            if node:
                
                if parent:
                    graph[parent.val].add(node.val)
                    graph[node.val].add(parent.val)

                pre_order(node.left, node, graph)
                pre_order(node.right, node, graph)

        pre_order(root, None, graph)

        distances = defaultdict(list)

        # Time O(n+m) 
        #   n = number of nodes
        #   m = number of edges
        # Space O(n)
        def dfs(graph, start):
            
            explored = set()
            explored.add(start)

            stack = []
            stack.append((start, 0))

            while stack:
                node, depth = stack.pop()
                distances[depth].append(node)
                
                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append((next_node, depth+1))
                        
        dfs(graph, target.val)

        return distances[k]

Range Sum of BST (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rangeSumBST(self, root: Optional[TreeNode], low: int, high: int) -> int:

        # DFS pre-order
        #   Add to sum only if value is within the range

        self.sum = 0

        # Time O(n)
        #   n = number of nodes
        # Space O(h)
        #   Recursion stack h = n worst case
        def pre_order(node, low, high):
            
            if node:

                if node.val >= low and node.val <= high:
                    self.sum += node.val

                pre_order(node.left, low, high)
                pre_order(node.right, low, high)

        pre_order(root, low, high)

        return self.sum

class Solution:
    def goodNodes(self, root: TreeNode) -> int:
        
        # Pre-order traverse
        # Remember max value 
        #   If node val is greater than max value, its good

        self.count = 0

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   Recursion stack
        def pre_order(node, max_value):

            if node:
                if node.val >= max_value:
                    self.count += 1
                    print(node.val)
                max_value = max(max_value, node.val)

                pre_order(node.left, max_value)
                pre_order(node.right, max_value)

        pre_order(root, -math.inf)
        return self.count

Lowest Common Ancestor of a Binary Tree III (M)

"""
# Definition for a Node.
class Node:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None
        self.parent = None
"""

class Solution:
    def lowestCommonAncestor(self, p: 'Node', q: 'Node') -> 'Node':

        # Do post order traverse
        #   If a node is p or q, return 1
        #   Else, return 0
        # If left and right returns 2, 
        # then current root is lca
        # Also, if either left or right return 1 and current root is p or q,
        # then, current root is lca

        # First, need to find root
        
        self.lca = None

        def find_root(p):
            
            if p.parent: 
                return find_root(p.parent)
            else:
                return p

        # Time O(h) = O(n)
        #   h = height of tree, worst case h = n
        # Space O(1)
        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)
                
                mid = 0
                if node.val == p.val or node.val == q.val:
                    mid = 1

                if left + right + mid == 2 \
                    and not self.lca:
                    self.lca = node
                    
                return left + right + mid

            else:
                return 0

        root = find_root(p)
        post_order(root)
        return self.lca
