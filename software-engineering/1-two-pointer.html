Container with most water (M) (Interview/Liked)

class Solution:
    def maxArea(self, height: List[int]) -> int:

        # Initialize two pointers at each end
        # Move two pointers towards middle one by one
        # The bar with higher height will contain more water

        max_area = 0
        start = 0
        end = len(height) - 1

        # Time O(n) n = length(height)
        # Space O(1)
        while start < end:
            width = end - start
            length = min(height[start], height[end])
            area = width * length
            max_area = max(max_area, area)

            if height[start] > height[end]:
                end -= 1
            else:
                start += 1

        return max_area

3Sum (M) (Interview/Liked)

class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:

        # For each element in the array
        #   Prepare sorted version of the array
        #   Initialize two pointers at each end
        #     If sum is less than zero, increment start
        #     Else, decrement end

        result = []
        sorted_nums = sorted(nums)

        # Time O(n^2) n = len(nums)
        # Space O(?)
        for i, num in enumerate(sorted_nums):

            # Make sure there is no two items that have different
            # ordering of same numbers
            start = i + 1  # No need to check numbers before i again
            end = len(sorted_nums) - 1

            while start < end:
                three_sum = num + sorted_nums[start] + sorted_nums[end]
                if three_sum == 0:
                    candidate = [num, sorted_nums[start], sorted_nums[end]]
                    if candidate not in result:
                        result.append(candidate)
                    # Can move either start or end
                    start += 1
                elif three_sum < 0:
                    start += 1
                else:
                    end -= 1

        return result

3Sum closest (M)

class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:

        sorted_nums = sorted(nums)
        result = sum(sorted_nums[0:3])
        min_diff = 13000

        # Time O(n^2) n = len(nums)
        # Space O(?)
        for i, num in enumerate(sorted_nums):

            # Make sure there is no two items that have different
            # ordering of same numbers
            start = i + 1  # No need to check numbers before i again
            end = len(sorted_nums) - 1

            while start < end:
                three_sum = num + sorted_nums[start] + sorted_nums[end]
                diff = three_sum-target
                # In this case, three_sum must increase
                # because it is less than target
                if diff < 0:
                    if abs(diff) < min_diff:
                        min_diff = abs(diff)
                        result = three_sum
                    start += 1
                else:
                    if abs(diff) < min_diff:
                        min_diff = abs(diff)
                        result = three_sum
                    end -= 1

        return result

4Sum (M)

class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:

        # Reduce 4sum problem to 2sum problems
        #   Fix the first two numbers
        #   Then, find the combination of remaning two numbers that addes up to the target minus the first two numbers
        #     Sort the array
        #     Initialize start and end pointer at each end
        #     If sum is less than new target
        #       Increase start
        #     Else
        #       Decrement end

        new_nums = sorted(nums)
        result = []

        # Time O(n^3) n = len(nums)
        # Space O(?)
        for i in range(len(new_nums)):
            for j in range(i+1, len(new_nums)):
                new_target = target - (new_nums[i]+new_nums[j])
                start = j + 1
                end = len(new_nums) - 1

                while start < end:
                    if new_nums[start] + new_nums[end] == new_target:
                        candidate = [new_nums[i], new_nums[j], new_nums[start], new_nums[end]]
                        if candidate not in result:
                            result.append(candidate)
                        start += 1
                    elif new_nums[start] + new_nums[end] < new_target:
                        start += 1
                    else:
                        end -= 1

        return result

Next permutation (M)

class Solution:
    def nextPermutation(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Find the largest i-1 such that nums[i-1] < nums[i]
        #   Find the largest index j > i-1 such that nums[i-1] < nums[j]
        #   Swap that nums[j], nums[i-1]
        #   Then reverse the right side of the array, which is nums[i:]
        # When such numbers do not exist
        #   Reverse the entire array

        is_swapped = False

        # Time O(n) n = len(nums)
        # Space O(1)
        for i in range(len(nums)-1, 0, -1):
            if nums[i-1] < nums[i]:

                for j in range(len(nums)-1, i-1, -1):
                    if nums[i-1] < nums[j]:
                        nums[i-1], nums[j] = nums[j], nums[i-1]
                        break

                start = 0
                end = len(nums[i:]) - 1
                while start < end:
                    nums[i+start], nums[i+end] = nums[i+end], nums[i+start]
                    start += 1
                    end -= 1

                is_swapped = True
                break

        if not is_swapped:
            nums.sort()

Trapping rain water (H) (Interview/Liked)

class Solution:
    def trap(self, height: List[int]) -> int:

        # Water that bar i can contain is
        # min(max height from left side, max height from right side) - height[i]
        # Initialize start and end pointer at each end
        # Memorize maximum from left and right while coming towards the middle

        water = 0
        start = 0
        end = len(height) - 1
        max_from_left = height[start]
        max_from_right = height[end]

        # Time O(n) n = len(height)
        # Space O(1)
        while start < end:

            max_from_left = max(max_from_left, height[start])
            max_from_right = max(max_from_right, height[end])
            if max_from_left < max_from_right:
                if height[start] < max_from_left:
                    water += (max_from_left-height[start])
                start += 1
            else:
                if height[end] < max_from_right:
                    water += (max_from_right-height[end])
                end -= 1

        return water

Sort colors (M) [Three pointer]

class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Initialize three pointers
        #   low pointer at 0
        #   mid pointer at 0
        #   high pointer at len(nums)-1
        # Move mid pointer such that
        #   all 0s are before low pointer
        #   all 2s are after high pointer
        #   all is are between low and high pointer

        low = 0
        high = len(nums) - 1
        mid = 0

        # Time O(n) n = len(nums)
        # Space O(1)
        while mid <= high:
            # Move 0 to the beginning and move low pointer
            if nums[mid] == 0:
                nums[low], nums[mid] = nums[mid], nums[low]
                low += 1
                mid += 1
            # Move 2 to the end
            elif nums[mid] == 2:
                nums[high], nums[mid] = nums[mid], nums[high]
                high -= 1
            else:
                mid += 1

Remove duplicates from sorted array II (M) [Not two pointers]

class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:

        # Increment i only when numbers are not duplicated
        # more than two, in other words nums[i] > nums[i-2]

        i = 0

        # Time O(n) n = len(nums)
        # Space O(1)
        for num in nums:
            if i < 2 or num > nums[i-2]:
                nums[i] = num
                i += 1

        for _ in range(len(nums)-i):
            nums.pop()

Reverse words in a string (M)

class Solution:
    def reverseWords(self, s: str) -> str:

        # Convert multiple whitespaces to a single whitespace
        s = " ".join(s.split())  # Time O(n)
        s = s.split(" ")  # Time O(n)

        start = 0
        end = len(s) - 1

        # Time O(n) n = len(s)
        # Space P(1)
        while start < end:
            s[end], s[start] = s[start], s[end]
            start += 1
            end -= 1

        return " ".join(s)

Two sum II - input array is sorted (M)

class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:

        # Initialize two pointers at each end
        # If sum is greater than target, decrease end
        # If sum is less than target, increase start

        start = 0
        end = len(numbers) - 1

        # Time O(n) n = len(numbers)
        # Space O(1)
        while start < end:
            if numbers[start] + numbers[end] == target:
                return [start+1, end+1]
            elif numbers[start] + numbers[end] > target:
                end -= 1
            else:
                start += 1

        return []

class Solution:
    def validPalindrome(self, s: str) -> bool:

        # When mismatch is found, remove mismatched char and check if palindrome

        def helper(s):

            start = 0
            end = len(s) - 1

            while start < end:
                if s[start] != s[end]:
                    return False
                start += 1
                end -= 1
            return True

        start = 0
        end = len(s) - 1
        num_mismatch = 0

        # Time O(n^2) where n = len(s)
        # Space O(1)
        while start < end:
            if s[start] == s[end]:
                start += 1
                end -= 1
            else:
                s_excluding_start_pointer = s[:start] + s[start+1:]
                s_excluding_end_pointer = s[:end] + s[end+1:]
                return helper(s_excluding_start_pointer) or helper(s_excluding_end_pointer)

        return True
