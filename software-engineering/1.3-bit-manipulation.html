Single Number (E)

class Solution:
    def singleNumber(self, nums: List[int]) -> int:

        # nums = [1,1,2] => 2
        # nums = [2,3,4,3,4] => 2
        # XOR two same nums results in 0
        # XOR a num and zero result in the num
        # Time O(n) n = len(nums)
        # Space O(1)

        result = 0

        for num in nums:
            result ^= num

        return result

Number of 1 Bits (E)

class Solution:
    def hammingWeight(self, n: int) -> int:

        # n = 11 => 1011 (3)
        # n = 1 => 1 (1)

        # Represent 1 in binary 
        # 0000 0000 0000 0000 0000 0000 0000 0001 
        # Perform AND operation between n and n - 1
        # Time O(1)
        # Space O(1)

        count = 0

        while n != 0:
            count += 1
            n = n & (n-1)

        return count
