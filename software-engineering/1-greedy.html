Jump Game II (M) (Liked)

class Solution:
    def jump(self, nums: List[int]) -> int:

        # At each index, compute the farthest index it can reach
        # Fartest index is the max of the two
        #   Fartest from previous indexes
        #   Fartest from the current index

        # Time O(n)
        # Space O(1)
        farthest_index = 0
        current_index = 0
        num_jumps = 0

        for i in range(len(nums)-1):
            farthest_index = max(farthest_index, nums[i]+i)
            if i == current_index:
                current_index = farthest_index
                num_jumps += 1

        return num_jumps

Jump game (M) (Interview/Liked)

class Solution:
    def canJump(self, nums: List[int]) -> bool:

        # At each index, compute the farthest index it can reach
        # Fartest index is the max of the two
        #   Fartest from previous indexes
        #   Fartest from the current index
        # If at any point, current number is 0 and farthest index of all
        # previous indices cannot reach beyoud the current number, return False

        # Time O(n)
        # Space O(1)

        if len(nums) == 1:
            return True

        farthest_index = 0
        for i in range(len(nums)-1):
            farthest_index = max(farthest_index, nums[i]+i)
            if nums[i] == 0 and farthest_index <= i:
                return False
            if farthest_index >= len(nums)-1:
                return True

        return False

Best time to buy and sell stock (E)

class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        # Remember minimum price while going through the list
        # Keep max diff between minimum price and any price

        min_price = prices[0]
        max_profit = 0

        # Time O(n) n = len(prices)
        # Space O(1)
        for price in prices[1:]:
            min_price = min(min_price, price)
            profit = max(price-min_price, 0)
            max_profit = max(max_profit, profit)

        return max_profit

Best time to buy and sell stock II (M)

class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        # Find all increasing sequences in the array
        # Then, find diff in each increasing sequence and sum them

        diffs = []
        diff = 0

        # Time O(n) n = len(prices)
        # Space O(n)
        i = 0
        while i < len(prices) - 1:

            if prices[i] <= prices[i+1]:
                diff += prices[i+1] - prices[i]
            else:
                diffs.append(diff)
                diff = 0
            i += 1

        diffs.append(diff)

        return sum(diffs)

Gas stations (M) (Interview)

class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:

        # Since there is unique solution, check from each index
        # If total cost > total gas, there is no solution
        # Find the first index where the accumulated gas is positive

        if sum(cost) > sum(gas):
            return -1

        # Time O(n) n = len(gas)
        # Space O(1)
        index = 0
        accumulated_gas = 0
        for i in range(len(gas)):
            accumulated_gas += (gas[i]-cost[i])
            if accumulated_gas < 0:
                accumulated_gas = 0
                index = i + 1

        return index

Candy (H)

class Solution:
    def candy(self, ratings: List[int]) -> int:

        # At index i,
        #   If ratings[i] > ratings[i-1], candy[i] > candy[i-1]
        #   If ratings[i] > ratings[i+1], candy[i] > candy[i+1]
        # Initialize candy array to all 1
        # Go though the candy array from left to right incrementing candies
        # Go though the candy array from right to left incrementing candies

        n = len(ratings)
        candies = [1] * n

        # Time O(n) n = len(ratings)
        # Space O(n)
        for i in range(1, n):
            if ratings[i] > ratings[i-1]:
                candies[i] = candies[i-1] + 1

        for j in range(n-2, -1, -1):
            if ratings[j] > ratings[j+1] and candies[j] <= candies[j+1]:
                candies[j] = candies[j+1] + 1

        return sum(candies)

Boats to save people (M)

class Solution:
    def numRescueBoats(self, people: List[int], limit: int) -> int:

        # Sort the array
        # If the sum of lightest and heaviest is less than or equal to limit,
        #   Increment lightest and decrement heaviest.
        # Else
        #   Only decrement heaviest.

        # Time: O(nlogn)
        # Space: O(1)
        people.sort()
        lightest = 0
        heaviest = len(people) - 1
        num_boats = 0

        while lightest <= heaviest:
            if people[lightest] + people[heaviest] <= limit:
                lightest += 1
                heaviest -= 1
                num_boats += 1
            else:
                heaviest -= 1
                num_boats += 1

        return num_boats

Minimum domino rotations for equal row (M)

class Solution:
    def minDominoRotations(self, tops: List[int], bottoms: List[int]) -> int:

        # For each number in tops and bottoms,
        # store its indices in a hash table

        # Time O(n) n = len(tops)
        # Space O(n)
        tops_dict = defaultdict(list)
        bottoms_dict = defaultdict(list)

        for i, item in enumerate(tops):
            tops_dict[item].append(i)

        for i, item in enumerate(bottoms):
            bottoms_dict[item].append(i)

        for num in range(1,7):
            if len(set(tops_dict[num]+bottoms_dict[num])) >= len(tops):
                return len(tops) - max(len(tops_dict[num]), len(bottoms_dict[num]))

        return -1

Two city scheduling (M)

class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:

        # Sort based on difference between cost A and cost B
        # Big difference would mean its cheaper to send to A than B
        # Send the first half of sorted array to A and rest to B

        # Time O(nlogn) n = len(costs)
        # Space O(n)

        cost_array = []
        for cost_a, cost_b in costs:
            cost_array.append([cost_a-cost_b, cost_a, cost_b])

        cost_array.sort(key = lambda x: x[0])

        h = len(costs) // 2
        total_cost = 0

        for i in range(h):
            total_cost += cost_array[i][1]
        for i in range(h, 2*h):
            total_cost += cost_array[i][2]

        return total_cost
