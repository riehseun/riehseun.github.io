Jump Game II (M) (Liked)

class Solution:
    def jump(self, nums: List[int]) -> int:

        # At each index, compute the farthest index it can reach
        # Fartest index is the max of the two
        #   Fartest from previous indexes
        #   Fartest from the current index

        # Time O(n)
        # Space O(1)
        farthest_index = 0
        current_index = 0
        num_jumps = 0

        for i in range(len(nums)-1):
            farthest_index = max(farthest_index, nums[i]+i)
            if i == current_index:
                current_index = farthest_index
                num_jumps += 1

        return num_jumps

Jump game (M) (Interview/Liked)

class Solution:
    def canJump(self, nums: List[int]) -> bool:

        # At each index, compute the farthest index it can reach
        # Fartest index is the max of the two
        #   Fartest from previous indexes
        #   Fartest from the current index
        # If at any point, current number is 0 and farthest index of all
        # previous indices cannot reach beyoud the current number, return False

        # Time O(n)
        # Space O(1)

        if len(nums) == 1:
            return True

        farthest_index = 0
        for i in range(len(nums)-1):
            farthest_index = max(farthest_index, nums[i]+i)
            if nums[i] == 0 and farthest_index <= i:
                return False
            if farthest_index >= len(nums)-1:
                return True

        return False

Best time to buy and sell stock (E)

class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        # Remember minimum price while going through the list
        # Keep max diff between minimum price and any price

        min_price = prices[0]
        max_profit = 0

        # Time O(n) n = len(prices)
        # Space O(1)
        for price in prices[1:]:
            min_price = min(min_price, price)
            profit = max(price-min_price, 0)
            max_profit = max(max_profit, profit)

        return max_profit

Best time to buy and sell stock II (M)

class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        # Find all increasing sequences in the array
        # Then, find diff in each increasing sequence and sum them

        diffs = []
        diff = 0

        # Time O(n) n = len(prices)
        # Space O(n)
        i = 0
        while i < len(prices) - 1:

            if prices[i] <= prices[i+1]:
                diff += prices[i+1] - prices[i]
            else:
                diffs.append(diff)
                diff = 0
            i += 1

        diffs.append(diff)

        return sum(diffs)

Gas stations (M) (Interview)

class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:

        # Time: O(n)
        # Space: O(1)

        # If sum(gas) >= sum(cost), there is enough gas to make a trip from at least one index.
        # Solution is guaranteed to be unique from the problem.

        # If this condition is not met, there is at least one index
        # where the trip can be started with enough gas.

        if sum(gas) < sum(cost):
            return -1

        n = len(gas)
        accumulated_gas = 0
        starting_index = 0

        for i in range(n):
            accumulated_gas += (gas[i]-cost[i])
            if accumulated_gas < 0:
                accumulated_gas = 0
                starting_index = i + 1

        return starting_index

Boats to save people (M)

class Solution:
    def numRescueBoats(self, people: List[int], limit: int) -> int:

        # Time: O(nlogn)
        # Space: O(1)

        # Sort the array.
        # If the sum of lightest and heaviest is less than or equal to limit, increment lightest and decrement heaviest.
        # If not, only decrement heaviest.

        people.sort()
        lightest = 0
        heaviest = len(people) - 1
        num_boats = 0

        while lightest <= heaviest:
            if people[lightest] + people[heaviest] <= limit:
                lightest += 1
                heaviest -= 1
                num_boats += 1
            else:
                heaviest -= 1
                num_boats += 1

        return num_boats

Two city scheduling (M)

class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:

        # Time O(nlogn)
        # Space O(n)

        # First half of sorted array will be the ones that would save the most cost by travelling to A rather than B.

        array = []

        for cost in costs:
            array.append([cost[0]-cost[1], cost[0], cost[1]])

        array.sort(key = lambda x: x[0])

        n = int(len(costs)/2)

        total_cost = 0
        for i in range(n):
            total_cost += array[i][1]
        for i in range(n,2*n):
            total_cost += array[i][2]

        return total_cost
