Implement Trie (Prefix Tree) (M)

class Trie:

    # Use nested dictionary for each char
    
    def __init__(self):
        self.trie = {}

    # Time O(n)
    #   n = len(word)
    # Space O(n)
    def insert(self, word: str) -> None:
        node = self.trie
        for c in word:
            if c not in node:
                node[c] = {}
            node = node[c] 
        node["word"] = True

    # Time O(n)
    # Space O(1)
    def search(self, word: str) -> bool:
        node = self.trie
        for c in word:
            if c not in node:
                return False
            node = node[c]  
        return "word" in node

    # Time O(n)
    # Space O(1)
    def startsWith(self, prefix: str) -> bool:
        node = self.trie
        for c in prefix:
            if c not in node:
                return False
            node = node[c] 
        return True

Design Add and Search Words Data Structure (M)

class WordDictionary:

    def __init__(self):
        self.trie = {}

    # Time O(n)
    #   n = len(word)
    # Space O(n)
    def addWord(self, word: str) -> None:
        node = self.trie
        for c in word:
            if c not in node:
                node[c] = {}
            node = node[c] 
        node["word"] = True

    # Time O(26**n)
    #   n = len(word)
    # Space O(n)
    #   Recursion stack
    def search(self, word: str) -> bool:

        def recursive_search(word, node):
            for i, c in enumerate(word):
                if c not in node:
                    if c == ".":
                        for child in node:
                            if child != "word" \
                                and recursive_search(word[i+1:], node[child]):
                                return True
                    return False
                else:
                    node = node[c]

            return "word" in node

        return recursive_search(word, self.trie)

class Solution:
    def findWords(self, board: List[List[str]], words: List[str]) -> List[str]: 

        # Use Trie
        # Insert all words in words into Trie

        n, m = len(board), len(board[0])
        result = []
        
        trie = {}
        for word in words:
            node = trie
            for c in word:
                if c not in node:
                    node[c] = {}
                node = node[c]
            node["word"] = word
        
        def backtrack(board, result, node, i, j, n, m):
    
            c = board[i][j]
            node = node[c]
            if "word" in node:
                if node["word"] not in result:
                    result.append(node["word"])

            dirs = [(-1, 0), (0, 1), (1, 0), (0, -1)]
            board[i][j] = "#"
            for x, y in dirs:
                if 0 <= i + x < n \
                    and 0 <= j + y < m \
                    and board[i+x][j+y] in node:
                    backtrack(board, result, node, i+x, j+y, n, m)
            board[i][j] = c

        for i in range(n):
            for j in range(m):
                if board[i][j] in trie:
                    backtrack(board, result, trie, i, j, n, m)
                          
        return result
