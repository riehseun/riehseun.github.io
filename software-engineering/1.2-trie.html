Implement Trie (Prefix Tree) (M)

class Trie:

    # Use nested dictionary for each char
    
    def __init__(self):
        self.trie = {}

    # Time O(n)
    #   n = len(word)
    # Space O(n)
    def insert(self, word: str) -> None:
        node = self.trie
        for c in word:
            if c not in node:
                node[c] = {}
            node = node[c] 
        node["word"] = True

    # Time O(n)
    # Space O(1)
    def search(self, word: str) -> bool:
        node = self.trie
        for c in word:
            if c not in node:
                return False
            node = node[c]  
        return "word" in node

    # Time O(n)
    # Space O(1)
    def startsWith(self, prefix: str) -> bool:
        node = self.trie
        for c in prefix:
            if c not in node:
                return False
            node = node[c] 
        return True

Search Suggestions System (M)

class Solution:
    def suggestedProducts(self, products: List[str], searchWord: str) -> List[List[str]]:

        # Contruct a trie where
        #   root contains 1 char
        #   next children contain 2 chars
        #   and so on
        # Sort the children alphabetically
        # Run DFS using searchWord
        #   Start node should be first word, first 2 words, and so on
        #   Collect the first 3 results and return

        trie = defaultdict(set)

        # Time O(n)
        #   n = total number of chars in products
        # Space O(n)
        for product in products:
            for i in range(len(product)):
                # "m" -> "mo" -> "mob"
                trie[product[:i]].add(product[:i+1])

        # Sort trie in alphabetical order
        # Time O(26*m*klogk) = O(mklogk)
        #   m = len(searchWord)
        #   There are 26 letters in alphabet
        #   k = max length of set containing prefixes
        for key, val in trie.items():
            val = list(val)
            val.sort()
            val.reverse()
            trie[key] = val
        
        # Time O(p+q)
        #   p = number of nodes = 26*m = m
        #   q = number of edges = p*k = mk
        # Space O(p) 
        def dfs(trie, start, products):

            explored = set()
            explored.add(start)

            stack = []
            stack.append(start)

            result = []

            while stack:
                node = stack.pop()

                if node in products:
                    result.append(node)
                    if len(result) >= 3:
                        return result

                for next_node in trie[node]:
                    explored.add(next_node)
                    stack.append(next_node)

            return result

        result = []
        for i in range(len(searchWord)):
            out = dfs(trie, searchWord[:i+1], products)
            result.append(out)

        return result
