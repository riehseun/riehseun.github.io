Climbing stairs (E) (G/F) !#

class Solution(object):
    def climbStairs(self, n):
        
        # Number of ways = number of ways (n-1) + number of ways (n-2)
        # dp[i] = dp[i-1] + dp[i-2]

        if n == 1:
            return 1

        dp = [0] * (n+1)
        dp[1] = 1
        dp[2] = 2

        # Time O(n)
        # Space O(n)
        for i in range(3, n+1):
            dp[i] = dp[i-1] + dp[i-2]

        return dp[n]

House robber (M)

class Solution:
    def rob(self, nums: List[int]) -> int:

        # a) rob the current house
        # b) don't rob the current house
        # max(current + rob(i-2), rob(i-1))

        # 1 2 3 1
        # 1 2 4 4

        n = len(nums)
        dp = [[0] for _ in range(n)]

        if n == 1:
            return nums[0]
        if n == 2:
            return max(nums[0], nums[1])

        dp[0] = nums[0]
        dp[1] = max(nums[0], nums[1])

        # Time O(n) n = len(nums)
        # Space O(n)
        for i in range(2, n):
            dp[i] = max(nums[i]+dp[i-2], dp[i-1])

        return dp[n-1]
