Merge two sorted lists

def merge_lists(lst1, lst2):

    # Time: O(n+m)

    merged = []
    s1 = 0
    s2 = 0

    while s1 < len(lst1) and s2 < len(lst2):
        if lst1[s1] < lst2[s2]:
            merged.append(lst1[s1])
            s1 += 1
        else:
            merged.append(lst2[s2])
            s2 += 1

    while s1 < len(lst1):
        merged.append(lst1[s1])
        s1 += 1

    while s2 < len(lst2):
        merged.append(lst2[s2])
        s2 += 1

    return merged

Two numbers adding up to k

def find_sum(lst, k):

    # Time: O(nlogn)

    lst.sort()

    i = 0
    j = len(lst) - 1

    while i < j:
        if lst[i] + lst[j] > k:
            j -= 1
        elif lst[i] + lst[j] < k:
            i += 1
        else:
            return lst[i], lst[j]

List of products of all elements

def find_product(lst):

    # Time: O(n)

    total_product = 1
    num_zeros = 0
    for v in lst:
        if v != 0:
            total_product *= v
        else:
            num_zeros += 1

    for i in range(len(lst)):
        if num_zeros == 0:
            lst[i] = int(total_product/lst[i])
        elif num_zeros == 1:
            if lst[i] == 0:
                lst[i] = total_product
            else:
                lst[i] = 0
        else:
            lst[i] = 0

    return lst

First non-repeating integer in a list

from collections import defaultdict

def find_first_unique(lst):

    # Time: O(n)
    # Space: O(n)

    my_dict = defaultdict(set)
    for i, num in enumerate(lst):
        my_dict[num].add(i)

    for num in lst:
        if len(my_dict[num]) == 1:
            return num

    return None

Find second maximum value in a list

import math

def find_second_maximum(lst):

    # Time: O(n)
    # Space: O(1)

    max_num = max(lst)
    second_max_num = -math.inf
    for v in lst:
        if v < max_num:
            second_max_num = max(v, second_max_num)

    return second_max_num

Right rotate list

def right_rotate(lst, k):

    # Time: O(n)
    # Space: O(n)

    result = [0] * len(lst)
    for i, num in enumerate(lst):
        new_index = (i+k) % len(lst)
        result[new_index] = lst[i]

    return result

Rearrange sorted list in max/min form

  • Use two step tricks! (Work for positive numbers only!)
# Example [1, 2, 3, 4, 5, 6] => [43, 9 , 38, 18, 33, 27] => [6, 1, 5, 2, 4, 3]

def max_min(lst):

    # Time: O(n)
    # Space: O(1)

    if not lst:
        return []

    max_index = len(lst) - 1
    min_index = 0
    max_number = lst[max_index] + 1

    for i, num in enumerate(lst):
        if i % 2 == 0:
            lst[i] += (lst[max_index] % max_number) * max_number
            max_index -= 1
        else:
            lst[i] += (lst[min_index] % max_number) * max_number
            min_index += 1

    for i, num in enumerate(lst):
        lst[i] = lst[i] // max_number

    return lst

Maximum sum sublist

  • Compute maximum so far ending at each time.
def find_max_sum_sublist(lst):

    # Time: O(n)
    # Space: O(1)

    current_max = -1000000
    global_max = -1000000
    for num in lst:
        # If maximum computed so far is less than 0, discard them.
        if current_max < 0:
            current_max = num
        # Otherwise, continue adding to the maximum so far.
        else:
            current_max += num
        global_max = max(global_max, current_max)

    return global_max
