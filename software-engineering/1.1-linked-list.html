# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:

        # Add digits by digits while maintaining the carry

        prev = None
        carry = 0

        # Time O(max(n,m)) n = len of linked list1, m = len of linked list2
        while l1 and l2:
            val = (l1.val + l2.val + carry) % 10

            if l1.val + l2.val + carry >= 10:
                carry = 1
            else:
                carry = 0

            l1 = l1.next
            l2 = l2.next

            if not prev:
                prev = ListNode(val, None)
                head = prev
            else:
                prev.next = ListNode(val, None)
                prev = prev.next

        while l1:
            val = (l1.val + carry) % 10

            if l1.val + carry >= 10:
                carry = 1
            else:
                carry = 0

            l1 = l1.next
            prev.next = ListNode(val, None)
            prev = prev.next

        while l2:
            val = (l2.val + carry) % 10

            if l2.val + carry >= 10:
                carry = 1
            else:
                carry = 0

            l2 = l2.next
            prev.next = ListNode(val, None)
            prev = prev.next

        if carry == 1:
            prev.next = ListNode(1, None)
            prev = prev.next

        return head

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:

        result = ListNode()
        temp = result

        # Time O(n) n = combined length of two linked lists
        # Space O(1)
        while list1 and list2:
            if list1.val < list2.val:
                result.next = list1
                list1 = list1.next
            else:
                result.next = list2
                list2 = list2.next
            result = result.next

        if list1:
            result.next = list1
        if list2:
            result.next = list2

        # Returning result.next will return the last node in the linked
        # list we computed. To return the first node, we use temp
        return temp.next

Merge k Sorted Lists (H) (G/F/A/100) !!!!

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:

        # Merge 2 linked lists at a time

        def merge_two_lists(list1, list2):

            result = ListNode()
            temp = result

            while list1 and list2:
                if list1.val < list2.val:
                    result.next = list1
                    list1 = list1.next
                else:
                    result.next = list2
                    list2 = list2.next
                result = result.next

            if list1:
                result.next = list1
            if list2:
                result.next = list2

            # Returning result.next will return the last node in the linked
            # list we computed. To return the first node, we use temp
            return temp.next

        n = len(lists)
        interval = 1

        # Time O(nlogk)
        #   n = numbers of items in all linked lists combined
        #   k = number of linked list
        # Space O(1)
        while interval < n:
            for i in range(0, n-interval, interval*2):
                lists[i] = merge_two_lists(lists[i], lists[i+interval])
            interval *= 2

        if n > 0:
            return lists[0]
        else:
            return None

Copy list with random pointer (M) (F/A) #!

"""
# Definition for a Node.
class Node:
    def __init__(self, x: int, next: 'Node' = None, random: 'Node' = None):
        self.val = int(x)
        self.next = next
        self.random = random
"""

class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':

        # First, create duplicates for each node in the linked list
        # Then, random pointers of duplicated nodes should point to
        # next of random pointers of original nodes
        # Finally, remove link to the original nodes

        # Time O(n) 
        #   n = number of nodes in linked list
        # Space O(1)

        if not head:
            return head

        original_head = head
        
        # Duplicate each node in the linked list
        while head:
            temp = head.next
            head.next = Node(head.val)
            if head.next:
                head.next.next = temp

            head = head.next.next

        # Set random pointers for duplicated nodes
        head = original_head
        while head:
            if head.random:
                head.next.random = head.random.next
            head = head.next.next

        # Remove links to original nodes
        head = original_head
        original_new_head = None
        while head:
            if head.next:
                temp = head.next
                head.next = head.next.next
                head = temp
                if not original_new_head:
                    original_new_head = head
            else:
                break

        return original_new_head

LRU cache (M) (G/F/A/100) !#!!

class ListNode:

    def __init__(self, key, val):
        self.prev = None
        self.next = None
        self.key = key
        self.val = val


class LRUCache:

    # Get
    #   Use a dictionary
    #   Key is number
    #   Value is linked list node
    #   Keep most recently used at the head
    #     If existing element is get, append to head - O(1)
    # Put
    #   Use doubly linked list 
    #   Keep least recently used at the tail
    #     When the capacity exceeds, remove from tail - O(1)
    #   Keep most recently used at the head
    #     If new element is added, append to head - O(1)
    #     If existing element is updated, move to head - O(1)
    
    def __init__(self, capacity: int):

        self.capacity = capacity

        # Initialize dictionary
        self.dictionary = defaultdict(int)

        # Initialize linked list
        self.head = ListNode(-1, -1)
        self.tail = ListNode(-1, -1)
        self.head.next = self.tail
        self.tail.prev = self.head

    def get(self, key: int) -> int:
        if key in self.dictionary:
            node = self.dictionary[key]
            # Move the existing node to head
            node.next.prev = node.prev
            node.prev.next = node.next
            node.prev = self.head
            node.next = self.head.next
            self.head.next.prev = node
            self.head.next = node
            return node.val
        return -1

    def put(self, key: int, value: int) -> None:
        if key in self.dictionary:
            node = self.dictionary[key]
            # Update the existing node's key and val
            node.key = key
            node.val = value
            # Move the existing node to head
            node.next.prev = node.prev
            node.prev.next = node.next
            node.prev = self.head
            node.next = self.head.next
            self.head.next.prev = node
            self.head.next = node
        else:
            # Create a new node
            node = ListNode(key, value)
            # Move the new node to head
            node.next = self.head.next
            node.prev = self.head
            self.head.next.prev = node
            self.head.next = node
            # Create a new item in dictionary
            self.dictionary[key] = node

            if len(self.dictionary) > self.capacity:
                del self.dictionary[self.tail.prev.key]
                # Remove a node from tail
                temp = self.tail.prev.prev
                self.tail.prev.prev = None
                self.tail.prev.next = None
                self.tail.prev = temp
                temp.next = self.tail

# param_1 = obj.get(key)
# obj.put(key,value)

Insert into a Sorted Circular Linked List (M) (F) !

"""
# Definition for a Node.
class Node:
    def __init__(self, val=None, next=None):
        self.val = val
        self.next = next
"""

class Solution:
    def insert(self, head: 'Optional[Node]', insertVal: int) -> 'Node':

        # Traverse linked list
        # If any value <= insertVal and its next value >= insertVal
        # place insertVal after that value

        if not head:
            head = Node(insertVal)
            head.next = head
            return head

        node = head

        # Time O(n) n = number of node in linked list
        # Space O(1)
        while node.next != head:

           # If increasing
            if node.val <= node.next.val:
                if node.val <= insertVal <= node.next.val:
                    break
            # If decreasing
            else:
                if node.val <= insertVal or insertVal <= node.next.val:
                    break

            node = node.next

        next_node = node.next
        node.next = Node(insertVal)
        node.next.next = next_node

        return head
