Add Two Numbers (M)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        
        # Go through two lists at a time
        # Remember the carry and apply in the next operation

        carry = 0
        head, prev = None, None

        # Time O(n)
        #   n = max length of l1 and l2
        # Space O(1)
        while l1 or l2:

            val = carry

            if l1:
                val += l1.val
                l1 = l1.next
            if l2:
                val += l2.val
                l2 = l2.next

            if val > 9:
                carry = 1
            else:
                carry = 0
            
            node = ListNode(val%10)
            if prev:
                prev.next = node
                prev = node
            else:
                prev = node
                head = prev
            
        # Handle carry from the last digits
        if carry == 1:
            node = ListNode(1)
            prev.next = node

        return head

Copy List with Random Pointer (M)

"""
# Definition for a Node.
class Node:
    def __init__(self, x: int, next: 'Node' = None, random: 'Node' = None):
        self.val = int(x)
        self.next = next
        self.random = random
"""

class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':

        # For each node in original LL, 
        #   Create a new node right next to each node
        #   Set new_node.next = node.next
        #   Set node.next = new_node 
        #   
        # Then, iterate newly created LL,
        #   Set node.next.random = node.random.next
        #
        # Lastly, from copied LL, remove links to original LL

        # Time O(n)
        #   n = number of nodes in LL
        # Space O(1)
        #   Ignoring space for resulting LL
        node = head
        while node:
            new_node = Node(node.val)
            new_node.next = node.next
            node.next = new_node
            node = node.next.next

        node = head
        while node and node.next:
            if node.random:
                node.next.random = node.random.next
            node = node.next.next

        head_next = None
        
        # Separate original and copied LL
        node = head
        while node and node.next:
            if not head_next:
                head_next = node.next
            next_node = node.next
            node.next = node.next.next
            node = next_node
    
        return head_next

Reverse Linked List (E)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:

        # Iterate LL
        # At each node
        #   Remember node.next (next = node.next)
        #   Reverse link (node.next = prev)
        #   Update prev (prev = node)
        #   Move to next node (node = next)    
        # Return prev

        prev, temp = None, None

        # Time O(n)
        #   n = number of nodes in LL
        # Space O(1)
        while head:

            # If head node
            if not prev:
                prev = head
                head = head.next
                prev.next = None
            else:
                next = head.next
                head.next = prev
                prev = head
                if head.next:
                    head = next
                
        return prev

Insert into a Sorted Circular Linked List (M)

"""
# Definition for a Node.
class Node:
    def __init__(self, val=None, next=None):
        self.val = val
        self.next = next
"""

class Solution:
    def insert(self, head: 'Optional[Node]', insertVal: int) -> 'Node':

        # Traverse linked list and find insertion point
        # If linked list is increasing in value
        #   node.val <= insertVal <= node.next.val
        # If linked list is decreasing in value, 
        # meaning node is at the end of linked list
        #   node.val <= insertVal or node.next.val >= insertVal

        # When there is no linked list, 
        # just create node with insertVal and return
        if not head:
            head = Node(insertVal)
            head.next = head
            return head

        node = head

        # Time O(n) 
        #   n = number of node in linked list
        # Space O(1)
        while node.next != head:
            
            # If increasing
            if node.val <= node.next.val:
                if node.val <= insertVal <= node.next.val:
                    break
            # If decreasing
            else:
                if node.val <= insertVal or insertVal <= node.next.val:
                    break

            node = node.next

        next_node = node.next
        node.next = Node(insertVal)
        node.next.next = next_node

        return head
