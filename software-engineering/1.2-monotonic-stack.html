Largest Rectangle in Histogram (H)

class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        
        # For each i, find the largest rectangle
        # Expand to left and right while its height >= heights[i]
        # How to find right boundry
        #   Push each element into stack
        #   When decreasing in sequence, 
        #   current element is the right boundry of 
        #   previous element (which is in stack)
        # How to find left boundry
        #   previous previous is left boundry
        #   as long as stack only keeps increasing sequence

        stack, max_area = [], 0
        stack.append(-1)  # Left boundry of heights[0]

        # Time O(n)
        #   n = len(heights)
        # Space O(n)
        for i, h in enumerate(heights):

            while stack[-1] != -1 and heights[stack[-1]] > h:
                height = heights[stack.pop()]  # current height
                right = i
                left = stack[-1]
                max_area = max(max_area, height*(right-left-1))
                print(max_area)
            stack.append(i)

        while stack[-1] != -1:
            height = heights[stack.pop()]
            right = len(heights)
            left = stack[-1]
            max_area = max(max_area, height*(right-left-1))

        return max_area

Daily Temperatures (M)

class Solution:
    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:
        
        # Want to know location of next greater number
        # Append index-value pair into stack
        #   If stack.top() < current_val, compute diff between indexes

        stack = []
        result = [0] * len(temperatures)

        # Time O(n)
        #   n = len(temperatures)
        # Space O(n)
        for i, t in enumerate(temperatures):

            while stack and stack[-1][1] < t:
                index, value = stack.pop()
                result[index] = i - index
            stack.append((i,t))

        return result
