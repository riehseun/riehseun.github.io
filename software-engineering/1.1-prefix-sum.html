Maximum Subarray (M)

class Solution:
    def maxSubArray(self, nums: List[int]) -> int:

        # Prefix sum
        #   If prefix sum so far < 0, discard that prefix sum (set to 0)
        #   Remember max sum

        prefix_sum, max_sum = 0, -math.inf

        # Time O(n)
        #   n = len(s)
        # Space O(1)
        for num in nums:
            prefix_sum += num
            max_sum = max(max_sum, prefix_sum)
            if prefix_sum < 0:
                prefix_sum = 0
            
        return max_sum

Random Pick with Weight (M)

class Solution:

    # Compute prefix sum
    # Pick random integer between smallest and largest
    # Return index of first number that is >= random integer
    # Ex. [1,3,6] => [1,4,10]
    #   If 1, pick index 0
    #   If 2,3,4 pick index 1
    #   If 5,6,7,8,9,10 pick index 2
        
    # Time O(n)
    #   n = len(w)
    # Space O(n)
    def __init__(self, w: List[int]):
        self.prefix_sums = []
        sum = 0
        for num in w:
            sum += num
            self.prefix_sums.append(sum)

    # Time O(logb)
    # Space O(1)
    def pickIndex(self) -> int:
        num = self.prefix_sums[-1] * random.random()
        left = 0
        right = len(self.prefix_sums) - 1
        while left < right:
            mid = (left+right) // 2
            if self.prefix_sums[mid] < num:
                left = mid + 1
            else:
                right = mid
        return right

Subarray Sum Equals K (M)

class Solution:
    def subarraySum(self, nums: List[int], k: int) -> int:
        
        # Ex. [1,1,1] k = 2
        #     [1,2,3]
        #     [-1, 0, 1]
        # Ex. [1,2,3] k = 3
        #     [1,3,6]
        #     [-2, 0, 3]
        # Ex. [1] k = 0
        #     [1]
        #     [1]
        # Compute prefix sums
        # Compute prefix sums - k
        # Count += number of occurance of (prefix sum - k)
        
        prefix_sums = defaultdict(int)
        prefix_sums[0] = 1
        sum = 0
        count = 0

        # Time O(n)
        #   n = len(nums)
        # Space O(n)
        for num in nums:
            sum += num
            count += prefix_sums[sum-k]
            prefix_sums[sum] += 1

        return count

Maximum Frequency After Subarray Operation (M)

class Solution:
    def maxFrequency(self, nums: List[int], k: int) -> int:

        # Try adding a number to entire array 
        #   If any element becomes k, put 1 
        #   If existing element is k, put -1
        #   Otherwise, put 0
        # Run maximum subarray sum on transformed array
        #   Use prefix sum
        #   Result is maximum sum + number of k

        num_k, max_freq = 0, 0

        def max_subarray_sum(nums):

            prefix_sum = 0
            largest_sum = nums[0]

            for i in range(len(nums)):
                prefix_sum += nums[i]
                prefix_sum = max(prefix_sum, nums[i])
                largest_sum = max(largest_sum, prefix_sum)
                
            return largest_sum

        for num in nums:
            if num == k:
                num_k += 1

        # Time O(n)
        #   n = len(nums)
        # Space O(n)
        for i in range(-50, 50):
            nums_new = []
            for j in nums:
                if j == k:
                    nums_new.append(-1)
                    continue
                if i + j == k:
                    nums_new.append(1) 
                else:
                    nums_new.append(0)
                
            max_freq = max(max_freq, max_subarray_sum(nums_new))

        return max_freq + num_k
