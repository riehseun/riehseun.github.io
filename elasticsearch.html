Endpoint

# Check license.
curl -k -u elastic:elastic123 -X GET https://<endpoint>:9200/_license

# Create an API key.
curl -k -u elastic:elastic123 -X POST https://<endpoint>:9200/_security/api_key --header "Content-Type: application/json" --date '{}'

# Create an index.
apiKey=<encoded_api_key_from_above>
curl -k -H "Authorization: ApiKey $apiKey" -X PUT https://<endpoint>:9200/collections

# List all indices.
curl -k -u elastic:elastic123 -X GET https://<endpoint>:9200/_cat/indices

# Get an index.
curl -k -u elastic:elastic123 -X GET https://<endpoint>:9200/<index_name>

# Check types.
curl -k -u elastic:elastic123 -X GET https://<endpoint>:9200/<index_name>/_mapping

# Create a document.
apiKey=<encoded_api_key_from_above>
curl -k -H "Authorization: ApiKey $apiKey" -H "Content-Type: application/json" -X PUT https://<endpoint>:9200/collections/_doc/1 -d '<JSON>'

# Check a document.
curl -k -u elastic:elastic123 -X GET https://<endpoint>:9200/<index_name>/_doc/<document_id>
