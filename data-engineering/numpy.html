Numpy

import numpy as np

Array

  • Python list with added features.
  • dtype keyword manually casts the array to the specified type.
  • If array has mix of int and float, all integers will be casted to float.
  • astype keyword allows casting.
  • np.nan keyword can be used as a placeholder value in the array.
  • Postive and negative infinity are represented by np.inf and -np.inf.
A = np.array([[2, 2],[2, 2]])
A = np.array([-1, 2, 5], dtype=np.float32)
A = np.array([0, 0.1, 2])  # array([0., 0.1, 2.])
A = A.astype(np.float32)
A = np.array([np.nan, 1, 2])
A = np.array([np.inf, -np.inf, 5])

Ranged data

  • np.arange is used where you can define start, end, step size.
  • np.linspace is used where you can define start, end, number of elements (rather than step size).
arr = np.arange(5)  # array([0, 1, 2, 3, 4])
arr = np.arange(-1, 4)  # array([-1,  0,  1,  2,  3])
arr = np.arange(-1.5, 4, 2)  # array([-1.5,  0.5,  2.5])
arr = np.linspace(5, 11, num=4)  # array([ 5.,  7.,  9., 11.])
arr = np.linspace(5, 11, num=4, endpoint=False)  # array([5. , 6.5, 8. , 9.5])
arr = np.linspace(5, 11, num=4, dtype=np.int32)  # array([ 5,  7,  9, 11], dtype=int32)

Reshape and flatten data

A = np.arange(8)
reshaped_A = np.reshape(A, (2, 4))  # array([[0, 1, 2, 3], [4, 5, 6, 7]])
reshaped_A = np.reshape(A, (-1, 2, 2))  # array([ [[0, 1], [2, 3]], [[4, 5], [6, 7]] ])
flattened_A = reshaped_A.flatten()  # array([0, 1, 2, 3, 4, 5, 6, 7])

Transpose

A.T

A = np.arange(8)
reshaped_A = np.reshape(A, (2, 4))
transposed_A = np.transpose(reshaped_A)  # array([[0, 4], [1, 5], [2, 6], [3, 7]])

A = np.arange(24)
reshaped_A = np.reshape(A, (3, 4, 2))  # Shape is (3, 4, 2)
transposed_A = np.transpose(reshaped_A, axes=(1, 2, 0))  # Shape is (4, 2, 3)

Zero and one

A = np.zeros(4)  # array([0., 0., 0., 0.])
A = np.ones((2, 3))  # array([[1., 1., 1.], [1., 1., 1.]])

Arithmetic

A = np.array([[1, 2], [3, 4]])
A + 1  # array([[2, 3], [4, 5]])
A - 1.2  # array([[-0.2, 0.8], [1.8, 2.8]])
A * 2  # array([[2, 4], [6, 8]])
A / 2  # array([[0.5, 1. ], [1.5, 2. ]])
A // 2  # array([[0, 1], [1, 2]])
A ** 2  # array([[1, 4], [9, 16]])
A ** 0.5  # array([[1., 1.41421356], [1.73205081, 2.]])

Non-linear functions

A = np.array([[1, 2], [3, 4]])
np.exp(A)  # array([[2.71828183, 7.3890561], [20.08553692, 54.59815003]])
np.exp2(A)  # array([[2., 4.], [8., 16.]])

A = np.array([[1, 10], [np.e, np.pi]])
np.log(A)  # array([[0., 2.30258509], [1., 1.14472989]])
np.log10(A)  # array([[0., 1.], [0.43429448, 0.49714987]])

Multiplication

A1 = np.array([1, 2, 3])
A2 = np.array([-3, 0, 10])
np.matmul(A1, A2)  # array([[5, 4, -7], [9, 8, -13], [13, 12, -19]])

A3 = np.array([[1, 2], [3, 4], [5, 6]])
A4 = np.array([[-1, 0, 1], [3, 2, -4]])
np.matmul(A3, A4)  # array([[4, 4],[-11, -10]])

Random integers

A = np.random.randint(5)  # Lower bound is 0 and upper boud is 5.
A = np.random.randint(5, high=6)  # Lower bound is 5 and upper boud is 6.
A = np.random.randint(-3, high=14, size=(2, 2))  # Size of output array is 2 by 2.

Utility functions

  • np.random.seed controls output of random functions.
  • np.random.shuffle randomly shuffles array.
np.random.seed(1)
A = np.array([1, 2, 3, 4, 5])
np.random.shuffle(A)

Distribution

  • np.random.uniform draws from uniform distribution.
  • np.random.normal draws from normal distribution.
np.random.uniform()
np.random.uniform(low=-1.5, high=2.2)
np.random.uniform(size=3)
np.random.uniform(low=-3.4, high=5.9, size=(2, 2))

np.random.normal()
np.random.normal(loc=1.5, scale=3.5)  # loc is mean and scale is standard deviation.
np.random.normal(loc=-2.4, scale=4.0, size=(2, 2))

Argmin and argmax

  • Returns indexes.
  • axis=0 means each column.
  • axis=1 means each row.
  • axis=-1 means the last dimension.
A = np.array([[-2, -1, -3],
                [4, 5, -6],
                [-3, 9, 1]])
np.argmin(A[0])  # 2
np.argmin(A[2])  # 1
np.argmin(A)  # 5 (index of element -6 is index 5 in the flattened version of A)

np.argmin(A, axis=0)  # array([2, 0, 1])
np.argmin(A, axis=1)  # array([2, 2, 0])
np.argmin(A, axis=-1)  # array([2, 2, 0])

Min and max

  • Returns values.
A = np.array([[0, 72, 3],
              [1, 3, -60],
              [-3, -2, 4]])
A.min()  # -60
A.max()  # 72
A.min(axis=0)  # array([ -3,  -2, -60])
A.max(axis=1)  # array([72,  3,  4])

Statistic

A = np.array([[0, 72, 3],
              [1, 3, -60],
              [-3, -2, 4]])
np.mean(A)  # 2.0
np.var(A)  # 977.3333333333334
np.median(A)  # 1.0
np.median(A, axis=-1)  # array([ 3.,  1., -2.])

Filter

  • np.isnan determine which locations of the array contain np.nan values.
A = np.array([[0, 2, 3],
              [1, 3, -6],
              [-3, -2, 1]])
A > 0  # array([[False, False,  True],
       #        [False,  True, False],
       #        [False, False, False]])

A = np.array([[0, 2, np.nan],
              [1, np.nan, -6],
              [np.nan, -2, 1]])
np.isnan(A)  # array([[False, False,  True],
             #        [False,  True, False],
             #        [ True, False, False]])

A = np.array([[0, 2, 3],
              [1, 0, 0],
              [-3, 0, 0]])
x_ind, y_ind = np.where(A != 0)
x_ind  # array([0, 0, 1, 2])
y_ind  # array([1, 2, 0, 0])

np_filter = np.array([[True, False], [False, True]])
positives = np.array([[1, 2], [3, 4]])
negatives = np.array([[-2, -5], [-1, -8]])
np.where(np_filter, positives, negatives)  # array([[ 1, -5],
                                           #        [-1,  4]])

A = np.array([[-2, -1, -3],
              [4, 5, -6],
              [3, 9, 1]])
np.any(A > 0)  # True
np.all(A > 0)  # False
np.any(A > 0, axis=1)  # array([False,  True,  True])
np.all(A > 0, axis=1)  # array([False, False,  True])
