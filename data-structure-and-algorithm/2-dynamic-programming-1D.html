Climbing Stairs (75/150/250)

class Solution:
    def climbStairs(self, n: int) -> int:
        
        # DP
        # Init: dp = [0] * n, dp[1] = 1, dp[2] = 2
        # Recurse: dp[i] = dp[i-1] + dp[i-2]
        # Time O(n)
        # Space O(n)

        dp = [0] * (n+1)
        if n <= 2:
            return n
        dp[1] = 1
        dp[2] = 2

        for i in range(3, n+1):
            dp[i] = dp[i-1] + dp[i-2]

        return dp[n]

Decode Ways (75/150/250)

class Solution:
    def numDecodings(self, s: str) -> int:
        
        # DP
        # dp[i] - number of decode ways of s[:i]
        # Init
        #   dp = [0] * (n+1)
        #   dp[0] = 1
        #   dp[1] = 1 if s[0] != 0
        # Recurse
        #   dp[i] += dp[i-1] if s[i-1] != 0
        #   dp[i] += dp[i-2] if 10 <= s[i-2:i] <= 26
        # Time O(n) n = len(s)
        # Space O(n)

        n = len(s)
        dp = [0] * (n+1)
        dp[0] = 1
        if int(s[0]) != 0:
            dp[1] = 1

        for i in range(2, n+1):
            if int(s[i-1]) != 0:
                dp[i] += dp[i-1]
            if 10 <= int(s[i-2:i]) <= 26:
                dp[i] += dp[i-2]
                
        return dp[n]

Word Break (75/150/250)

class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        
        # DP
        # dp[i] - if s[:i] can be constructed by words in wordDick
        # Init
        #   dp = [False] * (n+1)
        #   dp[0] = True
        # Recurse
        #   dp[i] = True if s[j:i] is in wordDict for j in range(i)
        # Time O(kn**2) n = len(s), k = length of word
        # Space O(n)

        n = len(s)
        word_set = set(wordDict)
        dp = [False] * (n+1)         
        dp[0] = True

        for i in range(1, n+1):
            for j in range(i):
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break

        return dp[n]

Word Break II (H)

class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> List[str]:

        # Let dp[i] = all possible outputs for s[:i+1]
        # At dp[i], scan "s" from right to left to check if 
        # any substring is in "wordDict"
        
        word_set = set(wordDict)
        n = len(s)

        dp = [[""] for _ in range(n+1)]

        # Time O(n**2)
        # Space O(n**2)
        for i in range(n):
            for j in range(i, -1, -1):
                if s[j:i+1] in word_set:
                    for sol in dp[j-1]:
                        dp[i].append(sol+" "+s[j:i+1])

        result = []
        for string in dp[n-1]:
            if len(string.strip().replace(" ", "")) == n:
                result.append(string.strip())

        return result

Maximum Product Subarray (75/150/250)

class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        
        # DP
        # max_prod - max product seen so far
        # min_prod - min product seen so far
        # If seen negative, swap dp_max and dp_min
        # Time O(n) n = len(nums)
        # Space O(1)

        max_prod, min_prod, result = nums[0], nums[0], nums[0]

        for i in range(1, len(nums)):
            if nums[i] < 0:
                max_prod, min_prod = min_prod, max_prod

            max_prod = max(nums[i], max_prod*nums[i])
            min_prod = min(nums[i], min_prod*nums[i])
            result = max(result, max_prod)

        return result

House Robber (75/150/250)

class Solution:
    def rob(self, nums: List[int]) -> int:
        
        # DP
        # dp[i] - max amount robs till index i
        # Init: dp = [0] * (n+1)
        # Recurse: dp[i] = max(dp[i-1], dp[i-2]+nums[i])
        # Time O(n) n = len(nums)
        # Space O(n)

        n = len(nums)
        dp = [0] * (n+1)

        for i in range(1, n+1):
            dp[i] = max(dp[i-1], dp[i-2]+nums[i-1])
        
        return dp[n]

House Robber II (75/150/250)

class Solution:
    def rob(self, nums: List[int]) -> int:
        
        # DP
        # dp[i] = maximum amount robbed till index i
        # Either rob first or not (try both and return better result)
        # Init: dp = [0] * (n+1) 
        # Recurse: dp[i] = max(dp[i-1], dp[i-2]+nums[i])
        # Time O(n) n = len(nums)
        # Space O(n)

        def dp_helper(nums):
            n = len(nums)
            dp = [0] * (n+1)
            for i in range(1, n+1):
                dp[i] = max(dp[i-1], dp[i-2]+nums[i-1])

            return dp[n]

        if len(nums) == 1:
            return nums[0]
        return max(dp_helper(nums[1:]), dp_helper(nums[:-1]))

Best Time to Buy and Sell Stock with Cooldown (150/250)

class Solution:
    def maxProfit(self, prices: List[int]) -> int:

        # DP
        # s[i] = profit at i after selling
        # b[i] = profit at i after buying
        # s[i] = max(s[i-1], prices[i]+b[i-1])
        # b[i] = max(b[i-1], s[i-2]-prices[i])
        # Time O(n) n = len(prices)
        # Space O(n)

        n = len(prices)
        s = [0 for _ in range(n+1)]
        b = [-math.inf for _ in range(n+1)]

        for i in range(1, n+1):
            s[i] = max(s[i-1], prices[i-1]+b[i-1])
            b[i] = max(b[i-1], s[i-2]-prices[i-1])

        return s[n]

Coin Change (75/150/250)

class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        
        # DP
        # dp[i] = min number of ways to achieve amount i
        # Init
        #   dp = [0] * (amount+1)
        # Recurse
        #   dp[i] = dp[i-j] + 1 for all j in coins
        #   It must be i - j >= 0
        # Time O(amount*n) n = len(coins)
        # Space O(amount)

        dp = [math.inf] * (amount+1)
        dp[0] = 0

        for i in range(1, amount+1):
            for j in coins:
                if i - j >= 0:
                    dp[i] = min(dp[i], dp[i-j]+1)

        if dp[amount] == math.inf:
            return -1
        return dp[amount]

Partition Equal Subset Sum (150/250)

class Solution:
    def canPartition(self, nums: List[int]) -> bool:

        # Compute half sum by computing total sum / 2
        # Check if half sum is achievable 
        # dp[i] = whether sum up to i can be created
        # Time O(nm) n = len(nums), m = half_sum
        # Space O(m)

        n = len(nums)
        if sum(nums) % 2 != 0:
            return False
        half_sum = sum(nums) // 2
        dp = [False for _ in range(half_sum+1)]
        dp[0] = True

        for num in nums:
            for i in range(half_sum, num-1, -1):
                dp[i] = dp[i] or dp[i-num]
        
        return dp[half_sum]

Maximum Profit in Job Scheduling (H)

class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:

        # Sort jobs based on start time
        # At each interval, there are two choices
        #   Skip the current job and move to next job
        #   Execute current job and move to next non-conflicting job
        # Let dp[i] = profit from jobs from index i to n
        #   If skip job i, dp[i] = dp[i+1]
        #   If execute job i, dp[i] = profit at i + d[next_i]
        # Use binary search to find next_i since start time is sorted

        # Time O(nlogn)
        intervals = []
        for x, y, z in zip(startTime, endTime, profit):
            intervals.append([x,y,z])

        intervals.sort(key=lambda x: x[0])
        
        n = len(intervals)
        dp = [0] * (n+1)

        def binary_search(i, intervals):

            end_t = intervals[i][1]
            left = 0
            right = len(intervals) - 1
            next_i = len(intervals)
            
            while left <= right:
                mid = (left+right) // 2
                if intervals[mid][0] >= end_t:
                    right = mid - 1
                    next_i = mid
                else:
                    left = mid + 1
                
            return next_i

        # Time O(nlogn)
        # Space O(n)
        for i in range(n-1, -1, -1):
            next_i = binary_search(i, intervals)
            print(f"{i}, {next_i}")
            if next_i == -1:
                dp[i] = max(dp[i+1], intervals[i][2])
            else:
                dp[i] = max(dp[i+1], intervals[i][2]+dp[next_i])

        return dp[0]
