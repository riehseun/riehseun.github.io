Single Number (150)

class Solution:
    def singleNumber(self, nums: List[int]) -> int:

        # nums = [1,1,2] => 2
        # nums = [2,3,4,3,4] => 2
        # XOR two same nums results in 0
        # XOR a num and zero result in the num
        # Time O(n) n = len(nums)
        # Space O(1)

        result = 0

        for num in nums:
            result ^= num

        return result

Reverse Bits (75/150)

class Solution:
    def reverseBits(self, n: int) -> int:

        print(1>>1)

        # 0011 ? 1100 

        # Consider n >> 1
        # 0101 -> 0010 -> 0001 -> 0000
        # While shifting, add last bit to front of result
        # 1 -> 10 -> 101 -> 1010
        # Time O(1) integer is represented by 32 bits
        # Space O(1)

        result, p = 0, 31

        while n:
            result += (n&1) << p
            n = n >> 1
            p -= 1
            
        return result

Number of 1 Bits (75/150)

class Solution:
    def hammingWeight(self, n: int) -> int:

        # 11011 ? 4 

        # Consider n & (n-1)
        # In each iteration, 1 bit is removed one by one
        # 1011 (11) 1010 (10) 1000 (8) 
        # 1010 (10) 1001 (9)  0111 (7) 
        # 1010 (10) 1000 (8)  0000 (0) 
        # 110 (6) 100 (4)
        # 101 (5) 011 (3)
        # 100 (4) 000 (0)
        # Time O(1) integer is represented by 32 bits by default
        # Space O(1)

        cnt = 0

        while n != 0:
            n = n & (n-1)
            cnt += 1

        return cnt

Missing Number (75/150)

class Solution:
    def missingNumber(self, nums: List[int]) -> int:

        # [0,1,3] ? 2
        # [3,0,1] ? 2

        # Consider index and num
        # [0,1,2] ^ [3,0,1] 
        # Find the number that only appears once and not len(nums)
        # XOR len(nums) with every index and num 
        # Time O(n) n = len(nums)
        # Space O(1)

        result = len(nums)

        for i, num in enumerate(nums):
            result = result ^ i ^ num

        return result

Counting Bits (75/150)

class Solution:
    def countBits(self, n: int) -> List[int]:
        
        # n = 4 ? [0,1,1,2,1] ([000,001,010,011,100])

        # Consider n & (n-1)
        # 001 (1) 010 (2) 011 (3) 100 (4)  
        # 000 (0) 001 (1) 010 (2) 011 (3)
        # 000 (0) 000 (0) 010 (1) 000 (0)
        # dp[i] = dp[i&(i-1)] + 1
        # Time (n)
        # Space O(1) ignoring space for result

        dp = [0 for _ in range(n+1)]

        for i in range(1,n+1):
            dp[i] = dp[i&(i-1)] + 1

        return dp

Sum of Two Integers (75/150)

class Solution:
    def getSum(self, a: int, b: int) -> int:
        
        # a = 2, b = 3 => 10, 11 => 101
        # Time O(1)
        # Space O(1)

        digit, mask = 0, 0xFFFFFFFF
        
        while b != 0:
            digit = (a^b) & mask 
            carry = ((a&b)<<1) & mask
            a = digit
            b = carry
        
        if a < 0x7FFFFFFF:
            return a
        else:
            return ~(a^mask)
