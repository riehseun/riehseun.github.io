Single Number (150)

class Solution:
    def singleNumber(self, nums: List[int]) -> int:

        # nums = [1,1,2] => 2
        # nums = [2,3,4,3,4] => 2
        # XOR two same nums results in 0
        # XOR a num and zero result in the num
        # Time O(n) n = len(nums)
        # Space O(1)

        result = 0

        for num in nums:
            result ^= num

        return result

Number of 1 Bits (75/150)

class Solution:
    def hammingWeight(self, n: int) -> int:

        # n = 11 => 1011 (3)
        # n = 1 => 1 (1)

        # Represent 1 in binary 
        # 0000 0000 0000 0000 0000 0000 0000 0001 
        # Perform AND operation between n and n - 1
        # Time O(1)
        # Space O(1)

        count = 0

        while n != 0:
            count += 1
            n = n & (n-1)

        return count

Missing Number (75/150)

class Solution:
    def missingNumber(self, nums: List[int]) -> int:

        # Use XOR
        # XOR with itself => 0
        # n replaces the missing number
        # Time O(n) n = len(nums)
        # Space O(1)

        missing = len(nums)

        for i, num in enumerate(nums):
            missing ^= (i^num)

        return missing

# n = 11 => [0, 1, 10, 11, 100, 101, ...] => [0, 1, 1, 2, 1, 2, ...]
        # n = 11, 1011, 1010, 1001, 1000, 0111, 0110, 0101, 0100, 0011, 0010, 0001
        # Perform AND between n and n - 1 where n is rightmost 1
        
        # Ex. 0 1 2 3 4 5 6 7
        #     0 1 1 2 1 2 2 3
        # 1 & 0 = 0 
        # 2 & 1 = 0
        # 3 & 2 = 2
        # 4 & 3 = 0
        # 5 & 4 = 4
        # dp[i] = dp[i&(i-1)] + 1
        # Time O(n)
        # Space O(n)
        
        dp = [0 for _ in range(n+1)]

        for i in range(1,n+1):
            dp[i] = dp[i&(i-1)] + 1

        return dp

Sum of Two Integers (75/150)

class Solution:
    def getSum(self, a: int, b: int) -> int:
        
        # a = 2, b = 3 => 10, 11 => 101
        # Time O(1)
        # Space O(1)

        digit, mask = 0, 0xFFFFFFFF
        
        while b != 0:
            digit = (a^b) & mask 
            carry = ((a&b)<<1) & mask
            a = digit
            b = carry
        
        if a < 0x7FFFFFFF:
            return a
        else:
            return ~(a^mask)
