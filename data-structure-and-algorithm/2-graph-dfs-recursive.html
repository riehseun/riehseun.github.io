class Solution:
    def depthSum(self, nestedList: List[NestedInteger]) -> int:

        # DFS recursively

        self.sum = 0

        # Time O(n)
        #   Number of recursive calls = number of nested lists
        # Space O(n)
        #   Recursion stack 
        def dfs(nestedList, depth):

            for item in nestedList:
                if item.isInteger():
                    self.sum += (item.getInteger()*depth)
                else:
                    dfs(item.getList(), depth+1)

        dfs(nestedList, 1) 
        return self.sum
