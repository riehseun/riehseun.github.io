Course Schedule (75/150/250)

class Solution:
    def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        
        # Topological sort
        # Create a directed graph
        # Find indegree of each node (number of incoming nodes)
        # Start with nodes with indegree of 0
        # Remove all outgoing edges of those node and traverse to next nodes
        # Time O(n+m) n = number of nodes = numCourses, m = number of edges = len(prerequisites)
        # Space O(n)

        graph, indegree = defaultdict(set), defaultdict(int)
        for a, b in prerequisites:
            graph[b].add(a)
            indegree[a] += 1

        queue = deque()
        for i in range(numCourses):
            if indegree[i] == 0:
                queue.append(i)

        while queue:
            node = queue.popleft()
            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 0:
                    queue.append(next_node)

        for key, val in indegree.items():
            if val > 0:
                return False
        return True

Course Schedule II (150/250)

class Solution:
    def findOrder(self, numCourses: int, prerequisites: List[List[int]]) -> List[int]:
        
        # Topological sort
        # Find indegree of each node
        # Remove nodes with indegree == 0
        # Update indegree of impacted nodes
        # Repeat until no nodes left
        # If nodes left, cycle
        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n)

        graph, indegree = defaultdict(set), defaultdict(int)

        for a, b in prerequisites:
            graph[b].add(a)
            indegree[a] += 1

        queue = deque()
        for i in range(numCourses):
            if indegree[i] == 0:
                queue.append(i)

        order = []
        while queue:
            node = queue.popleft()
            order.append(node)
            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 0:
                    queue.append(next_node)

        if len(order) != numCourses:
            return []
        return order

Graph Valid Tree (75/150/250)

class Solution:
    def validTree(self, n: int, edges: List[List[int]]) -> bool:

        # 0->(1,2,3->4) ? True
        # [] ? True if n == 1 False o/w
        # Cannot have cycles
        # All nodes are connected

        # Topological sort to detect cycle
        # Check if number of explored nodes = n

        graph, indegree = defaultdict(set), defaultdict(int)
        visited, queue, stack = set(), deque(), [0]
        
        for a, b in edges:
            graph[b].add(a)
            graph[a].add(b)
            indegree[a] += 1
            indegree[b] += 1

        for i in range(n):
            if indegree[i] == 1:
                queue.append(i)

        while queue:
            node = queue.popleft()
            
            for next_node in graph[node]:
                indegree[next_node] -= 1
                indegree[node] -= 1
                if indegree[next_node] == 1:
                    queue.append(next_node)

        while stack:
            node = stack.pop()

            for next_node in graph[node]:
                if next_node not in visited:
                    visited.add(next_node)
                    stack.append(next_node)

        if not edges:
            if n == 1:
                return True
            return False
        for a, b in edges:
            if indegree[a] >= 2 and indegree[b] >= 2:
                return False
        return len(visited) == n

Alien Dictionary (75/150/250)

class Solution:
    def foreignDictionary(self, words: List[str]) -> str:

        # [a, b] ? ab
        # [ab, ac] ? bc

        # Construct graph based on ordering
        # Time O(c) c = total number of chars in words 
        # Topological sort  
        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n) = O(1) 26 chars in alphabet

        graph, indegree, nodes = defaultdict(set), defaultdict(int), set()
        
        for i in range(len(words)-1):
            # If next word is prefix of current word, False
            if words[i].startswith(words[i+1]) \
                and len(words[i]) > len(words[i+1]):
                return ""
            j = 0
            while j < len(words[i]) and j < len(words[i+1]):
                if words[i][j] == words[i+1][j]:
                    j += 1
                    continue
                graph[words[i][j]].add(words[i+1][j])
                break

        for key, val in graph.items():
            for c in val:
                indegree[c] += 1

        for i in range(len(words)):
            for c in words[i]:
                nodes.add(c)

        queue = deque()
        for c in nodes:
            if indegree[c] == 0:
                queue.append(c)
        
        result = []
        while queue:
            node = queue.popleft()
            result.append(node)
            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 0:
                    queue.append(next_node)
        
        # If cycle, False
        for key, val in indegree.items():
            if val > 0:
                return ""
        return "".join(result)

Longest Increasing Path in a Matrix (150/250)

class Solution:
    def longestIncreasingPath(self, matrix: List[List[int]]) -> int:
        
        # Create directed graph 
        # Topological sort to find deepest depth
        # Time O(nm) n = number of rows, m = number of cols
        # Space O(nm)

        graph, indegree = defaultdict(list), defaultdict(int)
        n, m = len(matrix), len(matrix[0])
        dirs = [(-1,0),(1,0),(0,-1),(0,1)] 
        for i in range(n):
            for j in range(m):
                for x, y in dirs:
                    if 0 <= i + x < n \
                        and 0 <= j + y < m \
                        and matrix[i][j] < matrix[i+x][j+y]:
                        graph[(i,j)].append((i+x,j+y))
                        indegree[(i+x,j+y)] += 1

        queue = deque()
        for i in range(n):
            for j in range(m):
                if indegree[(i,j)] == 0:
                    queue.append(((i,j),0))

        while queue:
            node, depth = queue.popleft()
            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 0:
                    queue.append((next_node, depth+1))

        return depth + 1

Redundant Connection (150/250)

class Solution:
    def findRedundantConnection(self, edges: List[List[int]]) -> List[int]:
        
        # Topological sort
        # Essential edges are removed and redundant edges remain
        # Time O(n) n = number of nodes
        # Space O(n)

        graph, indegree = defaultdict(set), defaultdict(int)

        for a, b in edges:
            graph[a].add(b)
            graph[b].add(a)
            indegree[a] += 1
            indegree[b] += 1

        queue = deque()
        for i in range(1, len(edges)+1):
            if indegree[i] == 1:
                queue.append(i)

        while queue:
            node = queue.popleft()
            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 1:
                    queue.append(next_node)

        edges.reverse()
        for a, b in edges:
            if indegree[a] == 2 and indegree[b] == 2:
                return [a, b]
        return []
