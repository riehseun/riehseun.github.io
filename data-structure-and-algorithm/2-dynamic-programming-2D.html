Regular Expression Matching (150/250)

class Solution:
    def isMatch(self, s: str, p: str) -> bool:
        
        # DP
        # dp[i][j] if s[i] can be matched by p[j]
        # dp[0][0] = True
        # If p[j] == "*", dp[0][j] = dp[0][j-2]
        # If p[j] == s[i], dp[i][j] = dp[i-1][j-1]
        # If p[j] == ".", dp[i][j] = dp[i-1][j-1]
        # If p[j] == "*", dp[i][j] = dp[i][j-2]
        #   If p[j-1] == ".", dp[i][j] = dp[i][j] or dp[i-1][j]
        #   If p[j-1] == s[i], dp[i][j] = dp[i][j] or dp[i-1][j]
        # Time O(nm) n = len(s), m = len(p)
        # Space O(nm)

        n, m = len(s), len(p)
        dp = [[False]*(m+1) for _ in range(n+1)]
        dp[0][0] = True
        for j in range(2, m+1):
            dp[0][j] = dp[0][j-2]

        for i in range(1, n+1):
            for j in range(1, m+1):
                if p[j-1] == s[i-1]:
                    dp[i][j] = dp[i-1][j-1]
                elif p[j-1] == ".": 
                    dp[i][j] = dp[i-1][j-1]
                elif p[j-1] == "*": 
                    dp[i][j] = dp[i][j-2]
                    if p[j-2] == ".":
                        dp[i][j] = dp[i][j] or dp[i-1][j]
                    if p[j-2] == s[i-1]:
                        dp[i][j] = dp[i][j] or dp[i-1][j]

        return dp[n][m]

Unique Paths (75/150/250)

class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        
        # DP
        # dp[i][j] - number of paths at (i,j)
        # Init
        #   dp = [[1]*(m+1) for _ in range(n+1)]
        # Recurse
        #   dp[i][j] = dp[i-1][j] + dp[i][j-1]  
        # Time O(nm)
        # Space O(nm)
        
        dp = [[1]*(m+1) for _ in range(n+1)]        
        for i in range(1, n):
            for j in range(1, m):
                dp[i][j] = dp[i-1][j] + dp[i][j-1]  

        return dp[n-1][m-1]

Edit Distance (150/250)

class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        
        # DP
        # dp[i][j] = min ops to make word1 to word2
        #   n e a t c d e e
        # n 0 1 2 3 4 5 6 7
        # e 1 0
        # e 2
        # t 3
        # c 4
        # o 5
        # d 6
        # e 7  
        # If word1[i] == word2[j], dp[i][j] = dp[i-1][j-1]
        # Else, dp[i][j] = min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]) + 1
        # Time O(nm) n = len(word1), m = len(word2)
        # Space O(nm)

        n, m = len(word1), len(word2)
        if n == 0: return m
        if m == 0: return n
        dp = [[0]*(m+1) for _ in range(n+1)]
        for i in range(1, n):
            dp[i][0] = dp[i-1][0] + 1
        for j in range(1, m):
            dp[0][j] = dp[0][j-1] + 1

        for i in range(1, n+1):
            for j in range(1, m+1):
                if word1[i-1] == word2[j-1]:
                    dp[i][j] = dp[i-1][j-1]
                else:
                    dp[i][j] = min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]) + 1

        return dp[n][m]

Interleaving String (150/250)

class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:

        # "aaaa", "bbbb", "aabbbbaa" ? True

        # DP
        #   0 b b b b
        # 0 T F F F F
        # a T F F F F
        # a T T T T T
        # a F F F F T
        # a F F F F T
        # dp[i][j] = True if s3 up to i + j can be formed by s1[i], s2[j]
        # If i == 0, dp[i][j] = dp[i][j-1] and s2[j-1] == s3[i+j-1] (1)
        # If j == 0, dp[i][j] = dp[i-1][j] and s1[i-1] == s3[i-1+j] (2)
        # If i != 0 and j != 0, dp[i][j] = (1) or (2)
        # Time O(nm) n = len(s1), m = len(s2)
        # Space O(nm) 

        n, m = len(s1), len(s2)
        if n + m != len(s3):
            return False

        dp = [[False]*(m+1) for _ in range(n+1)]
        dp[0][0] = True

        for i in range(n+1):
            for j in range(m+1):
                if i == 0 and j != 0:
                    dp[i][j] = dp[i][j-1] and s2[j-1] == s3[i+j-1]
                if i != 0 and j == 0:
                    dp[i][j] = dp[i-1][j] and s1[i-1] == s3[i-1+j]
                if i != 0 and j != 0:
                    dp[i][j] = (dp[i][j-1] and s2[j-1] == s3[i+j-1]
                        or dp[i-1][j] and s1[i-1] == s3[i-1+j])
        
        return dp[n][m]Distinct Subsequences

Distinct Subsequences (150/250)

class Solution:
    def numDistinct(self, s: str, t: str) -> int:
        
        # DP
        #   0 c a a a t
        # 0 1 1 1 1 1 1
        # c 0 1 1 1 1 1
        # a 0 0 1 2 3 3
        # t 0 0 0 0 0 3
        # dp[i][j] = number of way to generate t[j] from s[i]
        # If s[i-1] == t[j-1], dp[i][j] = dp[i-1][j-1] + dp[i-1][j]
        # Else, dp[i][j] = dp[i-1][j]
        # Time O(nm) n = len(s), m = len(t)
        # Space O(nm)

        n, m = len(s), len(t)
        dp = [[0]*(m+1) for _ in range(n+1)]
        for i in range(n+1):
            dp[i][0] = 1

        for i in range(1, n+1):
            for j in range(1, m+1):
                if s[i-1] == t[j-1]:
                    dp[i][j] = dp[i-1][j-1] + dp[i-1][j]
                else:
                    dp[i][j] = dp[i-1][j]

        return dp[n][m]

Burst Balloons (150/250)

class Solution:
    def maxCoins(self, nums: List[int]) -> int:
        
        # DP
        # dp[i][j] = gain by bursting all balloons in (i, j)
        # dp[i][j] = max(dp[i][k]+dp[k][j]+nums[i]*nums[k]*nums[j])
        # for all k in (i, j)
        # Time O(n**3)
        # Space O(n**2)

        # Add 1 on each end to handle edge case
        nums = [1] + nums + [1]
        n = len(nums)
        dp = [[0]*n for _ in range(n)]

        # Exclude first and last ballon
        for length in range(2, n):
            for i in range(n-length):
                j = i + length
                for k in range(i+1, j):
                    dp[i][j] = max(dp[i][j],
                        dp[i][k]+dp[k][j]+nums[i]*nums[k]*nums[j])
                
        return dp[0][n-1]

Target Sum (150/250)

class Solution:
    def findTargetSumWays(self, nums: List[int], target: int) -> int:
        
        # We want to find positive number and negative numbers so that
        # sum(pos) - sum(neg) = target
        # We know sum(pos) + sum(neg) = sum(nums)
        # Then, sum(pos) = (target+sum(nums)) / 2
        # dp[i][j] = number of ways to have sum i using numbers up to j
        # If i < nums[j], number j cannot be used, dp[i][j] = dp[i][j-1]
        # Else, number j can either be used or not used
        # dp[i][j] = dp[i][j-1] + dp[i-nums[j]][j]
        # dp[0][0] = 1 sum 0 can be achieved with no number

        temp = target + sum(nums)
        if temp % 2 != 0:
            return 0
        
        n = temp // 2
        m = len(nums)
        dp = [[0]*(m+1) for _ in range(n+1)]
        dp[0][0] = 1
        
        for i in range(n+1):
            for j in range(1, m+1):
                if i < nums[j-1]:
                    dp[i][j] = dp[i][j-1]
                else:
                    dp[i][j] = dp[i][j-1] + dp[i-nums[j-1]][j-1]

        return dp[n][m]

Coin Change II (150/250)

class Solution:
    def change(self, amount: int, coins: List[int]) -> int:
        
        # DP
        # dp[i][j] = number of ways to make amount i using coins from j
        # dp[0][j] = 1 (amount 0 can be made by not selecting any coins)
        # If i < coins[j], coin j cannot be used
        # dp[i][j] = dp[i][j-1]
        # Else, coin j can either be used or not used
        # dp[i][j] = dp[i][j-1] + dp[i-coins[j]][j]
        # Time O(nm) n = len(coins), m = amount
        # Space O(nm)

        n, m = amount, len(coins)
        dp = [[0]*(m+1) for _ in range(n+1)]
        for j in range(m+1):
            dp[0][j] = 1

        for i in range(1, n+1):
            for j in range(1, m+1):
                if i < coins[j-1]:
                    dp[i][j] = dp[i][j-1]
                else:
                    dp[i][j] = dp[i][j-1] + dp[i-coins[j-1]][j]

        return dp[n][m]

Longest Common Subsequence (75/150/250)

class Solution:
    def longestCommonSubsequence(self, text1: str, text2: str) -> int:

        # apple, ape ? 3

        # DP
        # dp[i][j] = number of common at i (text1) and j (text2)
        # dp[i][j] = dp[i-1][j-1] + 1 if text1[i] == text2[j]
        # dp[i][j] = max(dp[i-1][j], dp[i][j-1]) if text1[i] != text2[j]
        # Time O(nm) n = len(text1), m = len(text2)
        # Space O(nm)

        n, m = len(text1), len(text2)
        dp = [[0]*(m+1) for _ in range(n+1)]

        for i in range(1, n+1):
            for j in range(1, m+1):
                if text1[i-1] == text2[j-1]:
                    dp[i][j] = dp[i-1][j-1] + 1
                else:
                    dp[i][j] = max(dp[i-1][j], dp[i][j-1])

        return dp[n][m]
