Median of Two Sorted Arrays (150/250)

class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        
        # Time O(log(n1+n2))
        # Space O(1)

        def find_kth_element(k):
            l1 = l2 = 0
            n1, n2 = len(nums1), len(nums2)
            while True:
                if l1 == n1:
                    return nums2[l2+k]
                if l2 == n2:
                    return nums1[l1+k]
                if k == 0:
                    return min(nums1[l1], nums2[l2])
                step = (k-1) // 2
                i = min(l1+step, n1-1)
                j = min(l2+step, n2-1)
                if nums1[i] <= nums2[j]:
                    removed = i - l1 + 1
                    l1 = i + 1
                    k -= removed
                else:
                    removed = j - l2 + 1
                    l2 = j + 1
                    k -= removed

        n = len(nums1) + len(nums2)
        if n % 2 == 1:
            return find_kth_element(n//2)
        return (find_kth_element(n//2-1) + find_kth_element(n//2)) / 2

Search in Rotated Sorted Array (75/150/250)

class Solution:
    def search(self, nums: List[int], target: int) -> int:
        
        # [1,2,3,4,5,6] 4 ? 3
        # [3,4,5,6,1,2] 4 ? 2

        # Consider mid index 
        # If nums[mid] < nums[n], right side is sorted
        #   If mid < target <= right, target is on right side, left = mid + 1
        #   Else, target is on left side, right = mid - 1
        # Else, left side is sorted
        #   If mid > target >= left, target is on left side, right = mid - 1
        #   Else, target is on right side, left = mid + 1
        # If nums[mid] == target, return mid
        # Time O(logn) n = len(nums)
        # Space O(1)

        left, right = 0, len(nums) - 1

        while left <= right:
            mid = (left+right) // 2
            if nums[mid] == target:
                return mid
            if nums[mid] < nums[right]:
                if nums[mid] < target <= nums[right]:
                    left = mid + 1
                else:
                    right = mid - 1
            else:
                if nums[mid] > target >= nums[left]:
                    right = mid - 1
                else:
                    left = mid + 1

        return -1

Find First and Last Position of Element in Sorted Array (M)

class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:

        # Binary search

        left = 0
        right = len(nums) - 1

        # Time O(logn) 
        #   n = len(nums)
        # Space O(1)
        while left <= right:
            mid = (left+right) // 2
            if target > nums[mid]:
                left = mid + 1
            elif target < nums[mid]:
                right = mid - 1
            else:
                start = mid
                while nums[mid] == nums[start]:
                    start -= 1
                    if start < 0:
                        break
                start += 1

                end = mid
                while nums[mid] == nums[end]:
                    end += 1
                    if end >= len(nums):
                        break
                end -= 1

                return [start, end]

        return [-1, -1]

Search a 2D Matrix (150/250)

class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        
        # Find a correct row, then execute binary search
        # Find mid row
        # If nums[mid][0] > target, right = mid - 1
        # If nums[mid][m-1] < target, left = mid + 1
        # Else, row = mid
        # Time O(logm+logn)
        # Space O(1)

        n, m = len(matrix), len(matrix[0])
        left, right = 0, n - 1
        row = -1
        while left <= right:
            mid = (left+right) // 2
            if matrix[mid][0] > target:
                right = mid - 1
            elif matrix[mid][m-1] < target:
                left = mid + 1
            else:
                row = mid
                break

        if row == -1:
            return False

        left, right = 0, m - 1
        while left <= right:
            mid = (left+right) // 2
            if matrix[row][mid] > target:
                right = mid - 1
            elif matrix[row][mid] < target:
                left = mid + 1
            else:
                return True

        return False

Find Minimum in Rotated Sorted Array (75/150/250)

class Solution:
    def findMin(self, nums: List[int]) -> int:

        # [3,4,5,6,1,2] ? 1
        # [5,6,1,2,3,4] ? 1
        # [1,2,3,4,5,6] ? 1
        # [4,5,6,1,2,3] ? 1

        # Consider mid point
        # If nums[mid] > nums[n], left side is sorted
        #   Min is on right side, set left = mid + 1
        # Else, right side is sorted
        #   Min is on left side, set right = mid - 1
        # If nums[mid] > nums[mid+1], mid + 1 is the min!
        # Similarly, nums[mid-1] > nums[mid], mid is the min
        # Time O(logn)
        # Space O(1)

        left, right = 0, len(nums) - 1

        while left < right:
            mid = (left+right) // 2
            if mid < len(nums) -1 and nums[mid] > nums[mid+1]:
                return nums[mid+1]
            if mid > 0 and nums[mid-1] > nums[mid]:
                return nums[mid]
            if nums[mid] > nums[right]:
                left = mid + 1
            else:
                right = mid - 1

        return nums[left]

Find Peak Element (M)

class Solution:
    def findPeakElement(self, nums: List[int]) -> int:

        # Binary search
        # If nums[mid] > nums[mid+1], peak must exist on left
        # If nums[mid] < nums[mid+1], peak must exist on right

        # Time O(logn)
        #   n = len(nums)
        # Space O(1)
        left = 0
        right = len(nums) - 1

        while left < right:
            mid = (left+right) // 2

            if nums[mid] > nums[mid+1]:
                right = mid
            else:
                left = mid + 1

        return left

Find K Closest Elements (M)

class Solution:
    def findClosestElements(self, nums: List[int], k: int, num: int) -> List[int]:

        # Because nums is sorted, k closest integers will be in
        # consecutive array
        # Find a number that is closest to x
        #   Then, expand left and right from that number
        #   to gather k closest numbers
        # Run binary seach to find closest number of x
        #   If mid > x, search on left side
        #   If mid < x, search on right side
        #   If mid == x, return mid
        #   Remember the last left and right pointer, 
        #   and return it when right >= left
        # Expand
        #   Initialize k' = 1
        #   Compare left and right numbers
        #   If k' + 2 <= k
        #     Then, include both numbers
        #   Else
        #     Compare left and right number and
        #     pick number that is closer to the 
        #     cloest number   

        # Time O(logn+k)
        #   n = len(nums)
        # Space O(1)

        def binary_search(nums, left, right, target):

            while left <= right:
                mid = (left + right) // 2
                if target > nums[mid]:
                    left = mid + 1
                elif target < nums[mid]:
                    right = mid - 1
                else:
                    return mid
                
            return left

        index = binary_search(nums, 0, len(nums)-1, num)
        
        left = index - 1
        right = index

        while right - left - 1 < k:
            # check for out of bounds
            if left == -1:
                right += 1
                continue

            if right == len(nums) \
                or abs(nums[left]-num) <= abs(nums[right]-num):
                left -= 1
            else:
                right += 1

        return nums[left+1:right]

Koko Eating Bananas (150/250)

class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        
        # Try eating rate between 1 and max(piles)
        # Calculate h', hours to take to finish eating
        # If h' < h, eating too fast, reduce eating rate
        # If h' > h, eating too slow, increase eating rate
        # If h' == h, try lower eating rate
        # Time O(nlogn) n = len(piles)
        # Space O(1)

        left, right = 1, max(piles)

        while left <= right:
            mid = (left+right) // 2
            
            h_ = 0
            for p in piles: 
                if p % mid == 0:
                    h_ += p // mid
                else:
                    h_ += p // mid + 1
            
            if h_ <= h:
                right = mid - 1
            else:
                left = mid + 1

        return left

Time Based Key-Value Store (150/250)

class TimeMap:

    # Use dictionary of tuple (timestamp, value)
    # If key does not exist, return ""
    # If key exist
    #   Find last timestamp <= timestamp 
    # Time O(1) for set and O(logn) for get
    # Space O(n) n = data in stream

    def __init__(self):
        self.data = defaultdict(list)

    def set(self, key: str, value: str, timestamp: int) -> None:
        self.data[key].append((timestamp, value))

    def get(self, key: str, timestamp: int) -> str:
        if key not in self.data:
            return ""
        if timestamp < self.data[key][0][0]:
            return ""
        
        left, right = 0, len(self.data[key])
        while left < right:
            mid = (left+right) // 2
            if self.data[key][mid][0] <= timestamp:
                left = mid + 1
            else:
                right = mid

        if left == 0:
            return ""
        return self.data[key][left-1][1]

House Robber IV (M)

class Solution:
    def minCapability(self, nums: List[int], k: int) -> int:
        
        # Binary search
        # Find max steal and min steal from any one house
        # Compute mid value
        # Check if at least k houses can be robbed 
        # (but no house whose value is greater than mid value can be robbed) 
        
        left, right, n = min(nums), max(nums), len(nums)

        # Time O(nlogm)
        #   n = len(nums)
        #   m = difference between min and max number in nums
        # Space O(1)
        while left < right:
            mid = (left+right) // 2

            i, num_robbed = 0, 0
            while i < n:
                if nums[i] <= mid:
                    num_robbed += 1
                    i += 2
                else:
                    i += 1

            # If feasible, decrease next value to check
            if num_robbed >= k:
                right = mid
            # If not feasible, increase next value to check
            else:
                left = mid + 1

        return left
