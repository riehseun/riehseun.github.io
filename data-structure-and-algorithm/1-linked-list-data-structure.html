LRU Cache (150/250)

class Node:

    def __init__(self, key=0, val=0, prev=None, next=None):
        self.key = key
        self.val = val
        self.prev = prev
        self.next = next


class LRUCache:

    # Use hash table
    # Get
    #   Return the value or -1, done in O(1)
    #   Mark the key as most recently used?
    # Put
    #   Add/update the value in dictionary - O(1)
    #   Mark the key as most recently used?
    #   Evict least recently used if necessary?
    # Use LL
    #   Head points to mostly recently used
    #   Tail points to least recently used
    #   Nodes in the middle are ordered 

    def __init__(self, capacity: int):
        self.dictionary = defaultdict(Node)
        self.capacity = capacity
        self.size = 0
        self.head = Node()
        self.tail = Node()
        self.head.next = self.tail
        self.tail.prev = self.head

    def get(self, key: int) -> int:
        if key in self.dictionary:
            self.mark(self.head, self.dictionary[key])
            return self.dictionary[key].val
        else:
            return -1
        
    def put(self, key: int, value: int) -> None:
        if key in self.dictionary:
            self.mark(self.head, self.dictionary[key])
            self.dictionary[key].val = value
        else:
            node = Node(key,value)
            self.add(self.head, node)
            self.dictionary[key] = node
            self.size += 1
        
        if self.size > self.capacity:
            key = self.evict(self.tail)
            del self.dictionary[key]
            self.size -= 1

    def add(self, head: Node, node: Node) -> None:
        node.next = head.next
        head.next.prev = node
        head.next = node
        node.prev = head
        
    def mark(self, head: Node, node: Node) -> None:
        # Remove
        node.prev.next = node.next
        node.next.prev = node.prev
        # Add
        node.next = head.next
        head.next.prev = node
        head.next = node
        node.prev = head
        
    def evict(self, tail: Node) -> None:
        key = tail.prev.key
        tail.prev.prev.next = tail
        tail.prev = tail.prev.prev
        return key
                
    
# Your LRUCache object will be instantiated and called as such:
# obj = LRUCache(capacity)
# param_1 = obj.get(key)
# obj.put(key,value)

All O`one Data Structure (H)

class ListNode:
    def __init__(self, val, keys=None, next=None, prev=None):
        self.val = val 
        self.keys = keys
        self.next = next
        self.prev = prev


class AllOne:

    # Use dictionary to store string and node mapping
    #   Look up is faster than iterating linked list
    # Use linked list where head contains min and tail contains max
    #   Each node stores frequency and keys with that frequency
    # Increament
    #   If key exists in dictionary
    #     Consider node and node.next
    #     If node.next.val != node.val + 1, create new node
    #     Else, key should be moved from current node to next node   
    #       If not node.keys, remove current node
    #   If key does not exists in dictionary
    #     If there is node with value 1, add key to that node
    #     Else, create a new node with value 1 at head
    # Decrement
    #   If key exists in a dictionary and value is > 1
    #     Consider node and node.prev
    #     if node.prev.val != node.val - 1, create new node
    #     Else, key should be moved from current node to previous node
    #     If not node.keys, remove current node
    #   If key exists in a dictionary and value is == 1
    #     Remove key from node.keys
    #     If not node.keys, remove current node
    # Max
    #   Return tail
    # Min
    #   Return head
   
    def __init__(self):
        self.nodes = {}
        self.head = ListNode(0)
        self.tail = ListNode(0)
        self.head.next = self.tail
        self.tail.prev = self.head

    def create_node_forward(self, node):
        new_node = ListNode(node.val+1, set())
        new_node.prev = node
        new_node.next = node.next
        node.next.prev = new_node
        node.next = new_node
        return new_node

    def create_node_backward(self, node):
        new_node = ListNode(node.val-1, set())
        new_node.prev = node.prev
        new_node.next = node
        node.prev.next = new_node
        node.prev = new_node
        return new_node  
        
    def remove_node(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev

    def inc(self, key: str) -> None:

        if key in self.nodes:
            node = self.nodes[key]
            if node.val + 1 == node.next.val:
                # Move key
                node.next.keys.add(key)
                node.keys.remove(key)
                # Remove current node and update dic
                if not node.keys:
                    self.remove_node(node)
                self.nodes[key] = node.next    
            else:
                # Create new node (node.val+1)
                new_node = self.create_node_forward(node)
                new_node.keys.add(key)
                node.keys.remove(key)
                # Remove current node and update dic
                if not node.keys:
                    self.remove_node(node)    
                self.nodes[key] = new_node    
        else:
            if self.head.next.val == 1:
                self.head.next.keys.add(key)
                self.nodes[key] = self.head.next
            else:
                # Create new node next to head
                new_node = self.create_node_forward(self.head)
                new_node.keys.add(key)
                self.nodes[key] = new_node

    def dec(self, key: str) -> None:
        
        node = self.nodes[key]
        if node.val > 1:
            if node.val - 1 == node.prev.val:
                # Move key
                node.prev.keys.add(key)
                node.keys.remove(key)
                # Remove current node and update dic
                if not node.keys:
                    self.remove_node(node)
                self.nodes[key] = node.prev
            else:
                # Create new node (node.val-1)
                new_node = self.create_node_backward(node)
                new_node.keys.add(key)
                node.keys.remove(key)
                # Remove current node and update dic
                if not node.keys:
                    self.remove_node(node)
                    del self.nodes[key]
                self.nodes[key] = new_node
        else:
            node.keys.remove(key)
            # Remove current node and update dic
            if not node.keys:
                self.remove_node(node)
            del self.nodes[key]
    
    def getMaxKey(self) -> str:
        
        if not self.tail.prev.keys:
            return ""
        else:
            return next(iter(self.tail.prev.keys))

    def getMinKey(self) -> str:
       
        if not self.head.next.keys:
            return ""
        else:
            return next(iter(self.head.next.keys))
