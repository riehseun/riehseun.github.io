LRU Cache (150/250)

class Node:
    def __init__(self, key=0, val=0, next=None, prev=None):
        self.key = key
        self.val = val
        self.next = next
        self.prev = prev

class LRUCache:

    # Create a dictionary to keep tract of key-value pair
    # If cache reaches capacity, remove one before putting new
    # Use LL where head is most recently used and tail is least
    # When get is called
    #   If key does exist, return -1
    #   Update LL to move "get" item to head
    #   Return value from dictionary
    # When put is called
    #   If value does not exist in dic, create node at head
    #   Else, find node and move it to head   
    #   Insert new key-value in dictionary 

    def __init__(self, capacity: int):
        self.dic = defaultdict(int)
        self.head = Node()
        self.tail = Node()
        self.head.next = self.tail
        self.tail.prev = self.head
        self.capacity = capacity

    def get(self, key: int) -> int:
        if key not in self.dic:
            return -1
        else:
            node = self.dic[key]
            self.move_to_head(node)
            return node.val
        
    def put(self, key: int, value: int) -> None:
        if key in self.dic:
            node = self.dic[key]
            node.val = value
            self.move_to_head(node)
        else:
            node = Node(key, value)
            self.dic[key] = node
            self.add_to_head(node)
            if len(self.dic) > self.capacity:
                removed_node = self.remove_from_tail()
                del self.dic[removed_node.key]
            
    def move_to_head(self, node):
        node_next = node.next
        node_prev = node.prev
        node_next.prev = node_prev
        node_prev.next = node_next

        head_next = self.head.next
        self.head.next = node
        head_next.prev = node
        node.next = head_next
        node.prev = self.head

    def add_to_head(self, node):
        head_next = self.head.next
        self.head.next = node
        head_next.prev = node
        node.next = head_next
        node.prev = self.head

    def remove_from_tail(self):
        node = self.tail.prev
        node_prev = self.tail.prev.prev
        self.tail.prev = node_prev
        node_prev.next = self.tail
        return node

All O`one Data Structure (H)

class ListNode:
    def __init__(self, val, keys=None, next=None, prev=None):
        self.val = val 
        self.keys = keys
        self.next = next
        self.prev = prev


class AllOne:

    # Use dictionary to store string and node mapping
    #   Look up is faster than iterating linked list
    # Use linked list where head contains min and tail contains max
    #   Each node stores frequency and keys with that frequency
    # Increament
    #   If key exists in dictionary
    #     Consider node and node.next
    #     If node.next.val != node.val + 1, create new node
    #     Else, key should be moved from current node to next node   
    #       If not node.keys, remove current node
    #   If key does not exists in dictionary
    #     If there is node with value 1, add key to that node
    #     Else, create a new node with value 1 at head
    # Decrement
    #   If key exists in a dictionary and value is > 1
    #     Consider node and node.prev
    #     if node.prev.val != node.val - 1, create new node
    #     Else, key should be moved from current node to previous node
    #     If not node.keys, remove current node
    #   If key exists in a dictionary and value is == 1
    #     Remove key from node.keys
    #     If not node.keys, remove current node
    # Max
    #   Return tail
    # Min
    #   Return head
   
    def __init__(self):
        self.nodes = {}
        self.head = ListNode(0)
        self.tail = ListNode(0)
        self.head.next = self.tail
        self.tail.prev = self.head

    def create_node_forward(self, node):
        new_node = ListNode(node.val+1, set())
        new_node.prev = node
        new_node.next = node.next
        node.next.prev = new_node
        node.next = new_node
        return new_node

    def create_node_backward(self, node):
        new_node = ListNode(node.val-1, set())
        new_node.prev = node.prev
        new_node.next = node
        node.prev.next = new_node
        node.prev = new_node
        return new_node  
        
    def remove_node(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev

    def inc(self, key: str) -> None:

        if key in self.nodes:
            node = self.nodes[key]
            if node.val + 1 == node.next.val:
                # Move key
                node.next.keys.add(key)
                node.keys.remove(key)
                # Remove current node and update dic
                if not node.keys:
                    self.remove_node(node)
                self.nodes[key] = node.next    
            else:
                # Create new node (node.val+1)
                new_node = self.create_node_forward(node)
                new_node.keys.add(key)
                node.keys.remove(key)
                # Remove current node and update dic
                if not node.keys:
                    self.remove_node(node)    
                self.nodes[key] = new_node    
        else:
            if self.head.next.val == 1:
                self.head.next.keys.add(key)
                self.nodes[key] = self.head.next
            else:
                # Create new node next to head
                new_node = self.create_node_forward(self.head)
                new_node.keys.add(key)
                self.nodes[key] = new_node

    def dec(self, key: str) -> None:
        
        node = self.nodes[key]
        if node.val > 1:
            if node.val - 1 == node.prev.val:
                # Move key
                node.prev.keys.add(key)
                node.keys.remove(key)
                # Remove current node and update dic
                if not node.keys:
                    self.remove_node(node)
                self.nodes[key] = node.prev
            else:
                # Create new node (node.val-1)
                new_node = self.create_node_backward(node)
                new_node.keys.add(key)
                node.keys.remove(key)
                # Remove current node and update dic
                if not node.keys:
                    self.remove_node(node)
                    del self.nodes[key]
                self.nodes[key] = new_node
        else:
            node.keys.remove(key)
            # Remove current node and update dic
            if not node.keys:
                self.remove_node(node)
            del self.nodes[key]
    
    def getMaxKey(self) -> str:
        
        if not self.tail.prev.keys:
            return ""
        else:
            return next(iter(self.tail.prev.keys))

    def getMinKey(self) -> str:
       
        if not self.head.next.keys:
            return ""
        else:
            return next(iter(self.head.next.keys))
