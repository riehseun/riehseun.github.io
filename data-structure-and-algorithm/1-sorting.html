Largest Number (M)

class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        
        # Do a quicksort
        #   Pick a pivot
        #   Put elements smaller than pivot to left of pivot
        #   Put elements larger than pivot to right of pivot
        # When comparing two numbers check if "AB" > "BA"

        if sum(nums) == 0:
            return "0"

        # Time O(nlogn) 
        #   n = len(nums)
        # Space O(logn)
        #  Recursion stack
        def quicksort(nums, l, r):
            if l >= r:
                return
            pos = partition(nums, l, r)
            quicksort(nums, l, pos-1)
            quicksort(nums, pos+1, r)

        def partition(nums, l, r):

            p = random.randint(l,r)

            for i in range(l,r+1):
                if compare(nums[i],nums[p]) and i > p:
                    nums[i], nums[p] =  nums[p], nums[i]
                if not compare(nums[i],nums[p]) and i < p:
                    nums[i], nums[p] =  nums[p], nums[i]
                p = i

            return p

        def compare(x, y):
            return str(x) + str(y) > str(y) + str(x)

        quicksort(nums, 0, len(nums)-1)
        result = ""
        for num in nums:
            result += str(num)
        return result

Kth Largest Element in an Array (150)

from heapq import heappush, heappop

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:

        # nums = [1,2,3,2,1,1] k = 2 ? [1,2]
        # nums = [1,2,3,2,1,1] k = 1 ? [1]
        # nums = [1,2,3,2,1,1] k = 3 ? [1,2,3]

        # Compute frequency of each num
        # Insert frequency into max heap of size k
        # Return elements from the heap
        # Time O(n+nlogk) n = len(nums)
        # Space O(n+k)

        # Quickselect
        # Store mapping between frequency and number
        # Retrieve top k frequency by 
        #   Parition freq array by random pivot
        #   Nums greater than pivot comes left
        #   Nums less than pivot go to right
        # If k < p, search on right
        # If k > p, search on left
        # If k == p, return
        # Time O(n) 
        # Space O(n) for calculating frequency, O(1) for quickselect

        min_heap, freq = [], defaultdict(int)

        for num in nums:
            freq[num] += 1

        for key, val in freq.items():
            heappush(min_heap, (val,key))
            if len(min_heap) > k:
                heappop(min_heap)
            
        result = []
        while min_heap:
            result.append(heappop(min_heap)[1])

        return result

K Closest Points to Origin (150)

cclass Solution:
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:

        # Find k smallest distance
        # Find all distances
        # Partition 
        #   Pick a random pivot
        #   Distances less than pivot are on left
        #   Distances greater than pivot are on right
        # If index of pivot < k
        #   Search on right
        # If index of pivot > k
        #   Search on left

        # Time O(n)
        #   n = len(points)
        # Space O(1)
        def partition(left, right):

            p = random.randint(left, right)
            
            for i in range(left, right+1):
                dist_p = points[p][0]**2 + points[p][1]**2
                dist_i = points[i][0]**2 + points[i][1]**2
                if i < p and dist_i > dist_p:
                    points[i], points[p] = points[p], points[i]
                if i > p and dist_i < dist_p:
                    points[i], points[p] = points[p], points[i]
                p = i

            return p 

        def quickselect(left, right):

            while True:
                p = partition(left, right)

                if k - 1 > p:
                    left = p + 1
                elif k - 1 < p:
                    right = p - 1
                else:
                    return points[:p+1]

        n = len(points)
        return quickselect(0, n-1)
