Graph Valid Tree (75/150/250)

class Solution:
    def validTree(self, n: int, edges: List[List[int]]) -> bool:
        
        # In valid tree, all nodes are connected
        # Also, there is no cycle
        # Union-find
        # Time O(n+m) m = number of edges = len(edges)
        # Space O(n)

        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]

        def union(a, b):
            pa, pb = find(a), find(b)
            if pa == pb:  # Cycle
                return False
            parent[pa] = pb
            return True

        if len(edges) != n - 1:
            return False

        parent = []
        for i in range(n):
            parent.append(i)

        for a, b in edges:
            if not union(a, b):
                return False
        return True

Single Circle Group (Google)

class Solution:
    def isSingleCircleGroup(circles):

        # Consider two circles
        # If distance between their centroid <= sum of their radius, they overlap or touch
        # Consider each circle as node in graph
        # Union circles that overlap
        # Time O(n**2) n = number of circles 
        # Space O(n)

        if not circles:
            return True

        n = len(circles)
        parent = list(range(n))

        def find(i):
            if parent[i] == i:
                return i
            parent[i] = find(parent[i])
            return parent[i]

        def union(i, j):
            root_i = find(i)
            root_j = find(j)
            if root_i != root_j:
                parent[root_i] = root_j

        # Check intersections
        for i in range(n):
            for j in range(i+1, n):
                x1, y1, r1 = circles[i]
                x2, y2, r2 = circles[j]
                
                dist_sq = (x2-x1) ** 2 + (y2-y1) ** 2
                radii_sum_sq = (r1+r2) ** 2
                
                if dist_sq <= radii_sum_sq:
                    union(i, j)

        # Find the root of the first circle
        first_root = find(0)
        
        # Check if all circles belong to the same group
        for i in range(1, n):
            if find(i) != first_root:
                return False
                
        return True
