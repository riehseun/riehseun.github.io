class Solution:
    def validTree(self, n: int, edges: List[List[int]]) -> bool:
        
        # In valid tree, all nodes are connected
        # Also, there is no cycle
        # Union-find
        # Time O(n+m) m = number of edges = len(edges)
        # Space O(n)

        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]

        def union(a, b):
            pa, pb = find(a), find(b)
            if pa == pb:  # Cycle
                return False
            parent[pa] = pb
            return True

        if len(edges) != n - 1:
            return False

        parent = []
        for i in range(n):
            parent.append(i)

        for a, b in edges:
            if not union(a, b):
                return False
        return True
