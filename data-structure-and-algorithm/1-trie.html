Implement Trie (Prefix Tree) (75/150)

class PrefixTree:

    # dog ? d -> o -> g
    # dooog? d -> o -> o -> o -> g

    # Use adjacency list
    # Insert - break into chars, each dict must know if char is ending
    # Search - break into chars and last char must not be an ending
    # StartsWith - break into chars and last char must be an ending 
    # Time O(n) n = length of string
    # Space O(n) 

    def __init__(self):
        self.trie = {}

    def insert(self, word: str) -> None:
        node = self.trie
        for c in word:
            if c not in node:
                node[c] = {}
            node = node[c]
        node["end"] = True
            
    def search(self, word: str) -> bool:
        node = self.trie
        for c in word:
            if c not in node:
                return False
            node = node[c]
        return "end" in node

    def startsWith(self, prefix: str) -> bool:
        node = self.trie
        for c in prefix:
            if c not in node:
                return False
            node = node[c]
        return True

Design Add and Search Words Data Structure (75/150)

class WordDictionary:

    # Insert "abc" ? a -> b -> c
    # Search "a.c" ? True

    # Construct trie
    # Use adjacency list
    # If dot, search all chars at next level via recursive search
    # Insert Time O(n) n = length of string
    # Insert Space O(n)
    # Search Time O(26**n) worst case "........"
    # Search Space O(1)

    def __init__(self):
        self.trie = {}

    def addWord(self, word: str) -> None:
        node = self.trie
        for c in word:
            if c not in node:
                node[c] = {}
            node = node[c]
        node["end"] = True

    def search(self, word: str) -> bool:

        def recursive_search(word, node):
            for i, c in enumerate(word):
                if c not in node:
                    if c == ".":
                        for next_node in node:
                            if next_node != "end" \
                                and recursive_search(word[i+1:], node[next_node]):
                                return True
                    return False
                else:
                    node = node[c]

            return "end" in node

        return recursive_search(word, self.trie)

Word Search II (75/150)

class Solution:
    def findWords(self, board: List[List[str]], words: List[str]) -> List[str]:
        
        # [[a,b],[c,d]] words = [abc, abd, cd] ? [abd, cd]

        # Construct a trie and backtrack
        # Insert all words into trie
        # Move to next char if it can be searched in trie
        # Backtrack
        # Time O(k) + O(nm(4**l)) 
        #   k = all chars in words
        #   l = number of chars in word
        #   n = number of rows
        #   m = number of cols
        # Space O(k) = O(1) Up to 26 chars in alphabet

        trie = {}
        for word in words:
            node = trie
            for c in word:
                if c not in node:
                    node[c] = {}
                node = node[c]
            node["word"] = word
            
        def backtrack(i, j, node):

            node = node[board[i][j]]
            if "word" in node:
                result.add(node["word"])
                
            c = board[i][j]
            board[i][j] = "#"
            for x, y in dirs:
                if 0 <= i + x < n \
                    and 0 <= j + y < m \
                    and board[i+x][j+y] in node:
                    backtrack(i+x, j+y, node)
            board[i][j] = c

        result = set()
        dirs = [(-1,0), (1,0), (0,-1), (0,1)]
        n, m = len(board), len(board[0])

        for i in range(n):
            for j in range(m):
                if board[i][j] in trie:
                    backtrack(i, j, trie)

        return list(result)
