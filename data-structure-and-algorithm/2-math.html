Reverse Integer (150)

class Solution:
    def reverse(self, x: int) -> int:

        # Divide input by 10 in every iteration
        # Time O(logn) 
        # Space O(1)

        r, sign = 0, "+"
        if x < 0:
            sign = "-"
            x = -x
            
        while x > 0:
            r = r * 10 + x % 10
            x = x // 10
        
        if sign == "-":
            r = -r

        if -2**31 <= r <= 2**31 - 1:
            return r

        return 0

Palindrome Number (E)

class Solution:
    def isPalindrome(self, x: int) -> bool:

        # Ex. 12321
        # Find reverse of right half and check it matches the left half
        #   left half - continue x // 10
        #   reverse of right half - collect results from x % 10   

        if x < 0 or (x % 10 == 0 and x != 0):
            return False

        left, right = x, 0

        # Time O(logn)
        #   Dividing input by 10 in every iteration
        # Space O(1)
        while left > right:
            right = right * 10 + left % 10
            left = left // 10
            
        if left == right or left == right // 10:
            return True

        return False

Multiply Strings (150)

class Solution:
    def multiply(self, num1: str, num2: str) -> str:

        # Multiply digit by digt
        # Add zeros to output based on location of digits 
        # Time O(nm) n = len(nums1), m = len(nums2)
        # Space O(1) ignoring space for output

        # Ex.
        # 23
        # 56
        #    18
        #   12
        #   15
        #  10

        n, m = len(num1), len(num2)
        result = 0 

        for i in range(n-1, -1, -1):
            for j in range(m-1, -1, -1):
                out = int(num1[i]) * int(num2[j])
                out *= 10 ** (n-i-1+m-j-1)
                result += out 
        
        return str(result)

Pow(x, n) (150)

# Ex. 2 ** 10
        #     = (2*2) ** 5
        #     = 4 ** 5
        #     = 4 * (4 ** 4)
        #     = 4 * (4*4) ** 2
        #     = 4 * (16 ** 2)
        #     = 4 * (16*16) ** 1  
        # If n is even, convert to (x*x) ** (n/2)
        # If n is odd, convert to x * (x ** (n-1)) and multiply x to result
        # Time O(n) n is divided by half in every two iteration
        # Space O(1)

        result = 1

        negative = False
        if n < 0:
            negative = True
            n = -n

        while n > 0:

            if n % 2 == 0:
                x *= x
                n /= 2
            else:    
                result *= x
                n -= 1

        if negative:
            return 1/result
            
        return result

Plus One (150)

class Solution:
    def plusOne(self, digits: List[int]) -> List[int]:

        # Add 1 to the last element of array remembering the carry
        # Time O(n) n = len(digits)
        # Space O(n) create new array if final carry is 1

        carry = 1
        
        for i in range(len(digits)-1, -1, -1):
            if digits[i] + carry > 9:
                carry = 1
                digits[i] = 0
            else:
                carry = 0
                digits[i] += 1
                break

        if carry == 1:
            digits = [1] + digits
            
        return digits

Happy Number (150)

class Solution:
    def isHappy(self, n: int) -> bool:

        # When result exists in an array of seen numbers, terminate
        # If 1 is in seen numbers, return True
        # Else, return False
        # Time O(logn)
        # Space O(logn) recursion stack

        seen = set()

        def get_next(n):

            r = 0
            for c in str(n):
                r += int(c) ** 2

            return r

        while n not in seen:
            seen.add(n)
            n = get_next(n)
            
        if 1 in seen:
            return True

        return False

Add Strings (E)

class Solution:
    def addStrings(self, num1: str, num2: str) -> str:

        # Interpret each digit as integer by the difference
        # in unicode representation between the digit and "0"
        # Add digit by digit while tracking carry

        result = []
        i = len(num1) - 1
        j = len(num2) - 1
        carry = 0

        # Time O(n) 
        #   n = max(len(num1), len(num2))
        # Space O(1)
        #   Ignoring space for output
        while i >= 0 or j >= 0:
            value = carry

            if i >= 0:
                value += (ord(num1[i])-ord("0"))
            if j >= 0:
                value += (ord(num2[j])-ord("0"))

            if value >= 10:
                carry = 1
            else:
                carry = 0
            result.append(str(value%10))
            i -= 1
            j -= 1

        if carry:
            result.append(str(carry))

        return ''.join(result[::-1])

Car Fleet (150)

class Solution:
    def carFleet(self, target: int, position: List[int], speed: List[int]) -> int:

        # Consider arriving times for cars
        # Sort cars by position
        #   If times[i+1] < times[i], fleet at i+1 will never be caught
        #   If times[i+1] > times[i], fleet at i+1 will be caught at some point

        # Time O(nlogn)
        #   n = len(position)
        # Space O(n)
        result, cars = 0, []
        for p, s in zip(position, speed):
            cars.append([p,s])
        cars.sort(key=lambda x: x[0])

        times = [(target-p)/s for p, s in cars]

        while len(times) > 1:
            lead = times.pop()
            if lead < times[-1]:
                result += 1
            else:
                times[-1] = lead

        if times:
            result += 1

        return result
