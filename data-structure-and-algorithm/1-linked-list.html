Add Two Numbers (150/250) (Intact)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        
        # Iterate both LL, creating new node with their sum of node val
        # Maintain a carry and apply it when adding each node
        # Time O(n) n = len(max(l1, l2))
        # Space O(1) ignoring space for result

        dummy = ListNode(0)
        l3 = dummy
        carry = 0
        while l1 or l2 or carry:
            sum = carry
            if l1:
                sum += l1.val
                l1 = l1.next
            if l2:
                sum += l2.val
                l2 = l2.next
            if sum > 9:
                sum -= 10
                carry = 1
            else:
                carry = 0
            l3.next = ListNode(sum)
            l3 = l3.next
            
        return dummy.next

Remove Nth Node From End of List (75/150/250)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        
        # 1->2->3->4, n = 2? 1->2->4

        # Two pointers
        # Move fast pointer n times
        # Then, move slow and fast pointer together
        # Use dummy head to prevent slow.next being null
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        dummy_head = ListNode(0)
        dummy_head.next = head
        slow = fast = dummy_head
        
        for _ in range(n):
            fast = fast.next

        while fast.next:
            slow = slow.next
            fast = fast.next

        slow.next = slow.next.next
        return dummy_head.next

Merge Two Sorted Lists (75/150/250)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:
        
        # 1->3->6, 2->4->5 ? 1->2->3->4->5->6

        # If l1.val < l2.val, node.next = l1, l1 = l1.next
        # Else, node.next = l2, l2 = l2.next
        # If not l1, node.next = l2
        # If not l2, node.next = l1
        # Time O(n+m) n = number of nodes in list1, m = number of nodes in list2
        # Space O(1) ignoring space for result

        dummy_head = node = ListNode(0)

        while list1 or list2:
            if not list1:
                node.next = list2
                break
            if not list2:
                node.next = list1
                break
            if list1.val < list2.val:
                node.next = list1
                node = node.next
                list1 = list1.next
            else:
                node.next = list2
                node = node.next
                list2 = list2.next

        return dummy_head.next

Merge k Sorted Lists (75/150/250)

from heapq import heappush, heappop

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:    
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        
        # [1,3,5], [2,6], [4,7] ? [1,2,3,4,5,6,7]

        # Pull head of each LL into a heap
        # Pop from heap to create new node
        # Time O(mlogn) n = number of LL, m = number of nodes in LL
        # Space O(n)

        min_heap = []
        dummy_head = node = ListNode(0)

        for i, head in enumerate(lists):
            if head:
                heappush(min_heap, (head.val, i, head))

        while min_heap:
            val, i, head = heappop(min_heap)
            if head.next:
                heappush(min_heap, (head.next.val, i, head.next))
            node.next = head
            node = node.next

        return dummy_head.next

Reverse Nodes in k-Group (150/250)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def reverseKGroup(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        
        # N(1)->N(2)->N(3)->N(4), k = 2 ? N(1)<-N(2) N(3)<-N(4)

        # Reverse nodes if remaining number of nodes >= k
        # Ending node of last group goes to start node of next group
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        dummy = ListNode(0)
        dummy.next = head
        group_prev = dummy

        while True:
            kth = group_prev
            for _ in range(k):
                kth = kth.next
                if not kth:
                    return dummy.next
            group_next = kth.next

            prev = group_next
            curr = group_prev.next
            while curr != group_next:
                nxt = curr.next
                curr.next = prev
                prev = curr
                curr = nxt

            old_group_start = group_prev.next
            group_prev.next = kth
            group_prev = old_group_start

Copy List with Random Pointer (150/250)

"""
# Definition for a Node.
class Node:
    def __init__(self, x: int, next: 'Node' = None, random: 'Node' = None):
        self.val = int(x)
        self.next = next
        self.random = random
"""

class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':
        
        # N(1)->N(2)->N(3) ? N'(1)->N'(2)-N'(3)

        # For each node, create new node in front of it
        # For example, N(1)->N'(1)->N(2)->N'(2)->N(3)->N'(3)->
        #   new_node.next = node.next
        #   node.next = new_node
        # Then, configure new_node's random  
        #   node.next.random = node.random.next
        # Then, remove link to original LL
        #   node.next = node.next.next 
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        if not head:
            return None

        node = head
        while node:
            new_node = Node(node.val)
            new_node.next = node.next
            node.next = new_node
            node = node.next.next

        node = head
        while node:
            if node.random:
                node.next.random = node.random.next
            node = node.next.next

        node, copy_head = head, head.next
        while node:
            copy = node.next
            node.next = copy.next
            if copy.next:
                copy.next = copy.next.next
            node = node.next

        return copy_head

Linked List Cycle (75/150/250)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def hasCycle(self, head: Optional[ListNode]) -> bool:
        
        # 1->2->3->4->2->3->4 ? True
        # 1->2->3 ? False

        # Two pointers - slow, fast
        # If slow meets fast, cycle
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        slow = fast = head
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
            if slow == fast:
                return True

        return False

Reorder List (75/150/250)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def reorderList(self, head: Optional[ListNode]) -> None:
        
        # 1->2->3->4->5->6 ? 1->6->2->5->3->4
        # 1->2->3->4<-5<-6

        # head_next = head.next
        # head.next = tail
        # tail_next = tail.next
        # tail.next = head_next
        # head = head_next
        # tail = tail_next
        # Find mid point
        # Reverse second half
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        fast = slow = head
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next

        node = slow.next
        slow.next = None
        prev = None
        while node:
            node_next = node.next
            node.next = prev
            prev = node
            node = node_next

        tail = prev
        while tail:
            head_next = head.next
            head.next = tail
            tail_next = tail.next
            tail.next = head_next
            head = head_next
            tail = tail_next

Reverse Linked List (75/150/250)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        
        # 1->2->3 ? 1<-2<-3

        # Need to know previous
        # next_node = node.next
        # node.next = prev
        # prev = node
        # node = next_node
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        node, prev = head, None
        while node:
            next_node = node.next
            node.next = prev
            prev = node
            node = next_node

        return prev

Find the Duplicate Number (150/250)

class Solution:
    def findDuplicate(self, nums: List[int]) -> int:

        # [1,3,2,2] ? 2

        # Unique numbers have 1-1 mapping between index and value
        # Duplicate numbers have don't have 1-1 mapping
        # For example
        # 0, 1, 2, 3 
        # 1, 3, 2, 2
        # 0->1->3->2->2->2->2... (cycle)
        # Create slow and fast pointers
        # If slow == fast, pointers moved to cycle
        # Move slow to beginnig, then move both slow and fast one by one
        # Time O(n) n = len(nums)
        # Space O(1)

        slow, fast = 0, 0

        while True:
            slow = nums[slow]
            fast = nums[nums[fast]]
            if slow == fast:
                break

        slow = 0
        while slow != fast:
            slow = nums[slow]
            fast = nums[fast]
        return slow

Insert into a Sorted Circular Linked List

"""
# Definition for a Node.
class Node:
    def __init__(self, val=None, next=None):
        self.val = val
        self.next = next
"""

class Solution:
    def insert(self, head: 'Optional[Node]', insertVal: int) -> 'Node':

        # Traverse linked list and find insertion point
        # If linked list is increasing in value
        #   node.val <= insertVal <= node.next.val
        # If linked list is decreasing in value, 
        # meaning node is at the end of linked list
        #   node.val <= insertVal or node.next.val >= insertVal

        # When there is no linked list, 
        # just create node with insertVal and return
        if not head:
            head = Node(insertVal)
            head.next = head
            return head

        node = head

        # Time O(n) 
        #   n = number of node in linked list
        # Space O(1)
        while node.next != head:
            
            # If increasing
            if node.val <= node.next.val:
                if node.val <= insertVal <= node.next.val:
                    break
            # If decreasing
            else:
                if node.val <= insertVal or insertVal <= node.next.val:
                    break

            node = node.next

        next_node = node.next
        node.next = Node(insertVal)
        node.next.next = next_node

        return head
