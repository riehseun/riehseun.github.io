Add Two Numbers (150)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        
        # Add numbers while remembering the carry

        carry, prev, dummy_head = 0, None, None
        
        # Time O(n)
        #   n = max length of l1 and l2
        # Space O(1)
        while l1 or l2:

            v3 = carry
            if l1:
                v3 += l1.val
                l1 = l1.next
            if l2:
                v3 += l2.val
                l2 = l2.next
            
            carry = 1 if v3 > 9 else 0

            l3 = ListNode(v3%10)
            if prev:
                prev.next = l3
                prev = l3
            else:
                prev = l3
                dummy_head = prev
                
        if carry == 1:
            prev.next = ListNode(1)

        return dummy_head

Remove Nth Node From End of List (75/150)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        
        # N(1)->N(2)->N(3)->N(4) n = 2 ? N(1)->N(2)->N(4)

        # Remove note at length - n + 1
        # Iterate first to find length of node
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        length, node = 0, head
        while node:
            length += 1
            node = node.next

        # Use dummy_head to deal with Null cases
        dummy_head = ListNode()
        dummy_head.next = head
        node, i = dummy_head, 1
        while i < length - n + 1:
            i += 1
            node = node.next
        node.next = node.next.next

        return dummy_head.next

Merge Two Sorted Lists (75/150)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:
        
        # N(1)->N(2)->N(4), N(3)->N(5) ? N(1)->N(2)->N(3)->N(4)->N(5)

        # Iterate both LL
        # If l1.val < l2.val, create new node with l1.val and move l1
        # Else, create new node with l2.val and move l2
        # Time O(n+m) n, m = number of node in LLs (list1 and list2) 
        # Space O(1) ignoring space for output

        head = prev = ListNode(0)
        
        while list1 or list2:

            if not list1:
                prev.next = list2
                break
            if not list2:
                prev.next = list1
                break

            if list1.val < list2.val:
                node = ListNode(list1.val)
                list1 = list1.next
            else:
                node = ListNode(list2.val)
                list2 = list2.next
            
            prev.next = node
            prev = node

        return head.next

Merge k Sorted Lists (75/150)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:

        # Merge two LL at a time
        # Repeat until there is only one LL

        def merge(l1, l2):
            
            head = dummy_head = ListNode()

            while l1 and l2:
                if l1.val < l2.val:
                    head.next = l1
                    l1 = l1.next
                else:
                    head.next = l2
                    l2 = l2.next
                head = head.next

            if l1:
                head.next = l1
            if l2:
                head.next = l2

            return dummy_head.next
            
        # Time O(nlogk)
        #   n = total number of nodes in all linked lists
        #   k = number of linked lists
        # Space O(1)
        i, n = 1, len(lists)
        while i < n:
            for j in range(0, n-i, i*2):
                lists[j] = merge(lists[j], lists[j+i])
            i *= 2

        if n <= 0:
            return None

        return lists[0]

Reverse Nodes in k-Group (150)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseKGroup(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:

        # Move k nodes and reverse

        if k == 1:
            return head

        def reverse(node, current_node):

            prev = start = end = None

            while node:

                if not prev:
                    end = node

                next_node = node.next
                node.next = prev
                prev = node
                node = next_node

                if node == current_node:
                    start = prev
                    break

            return start, end

        dummy_node = node = head
        start1 = end1 = start2 = end2 = None
        count = k

        # Time O(kn)
        #   n = length of LL
        # Space O(1)
        while node:
            count -= 1
            node = node.next

            if count == 0:
                start2, end2 = reverse(dummy_node, node)
                if not start1 and not end1:
                    start1 = start2
                    end1 = end2
                else:
                    end1.next = start2
                    end1 = end2
                dummy_node = node
                count = k
        
        end2.next = dummy_node
        return start1

Copy List with Random Pointer (150)

"""
# Definition for a Node.
class Node:
    def __init__(self, x: int, next: 'Node' = None, random: 'Node' = None):
        self.val = int(x)
        self.next = next
        self.random = random
"""

class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':

        # Create new nodes after each node in original LL
        # Set
        #   new_node.next = node.next
        #   node.next = new_node
        # Then, iterate newly modified LL
        #   node.next.random = node.random.next
        # Then, remove links from original LL
        #   node.next = node.next.next
        #   node = node.next

        # Time O(n)
        #   n = number of nodes in LL
        # Space O(1)
        node = head
        while node:
            new_node = Node(node.val)
            new_node.next = node.next
            node.next = new_node
            node = node.next.next

        node = head
        while node:
            if node.random:
                node.next.random = node.random.next
            node = node.next.next

        node, node_copy = head, None
        while node and node.next:
            if not node_copy:
                node_copy = node.next
            next_node = node.next
            node.next = node.next.next
            node = next_node

        return node_copy

Linked List Cycle (75/150)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def hasCycle(self, head: Optional[ListNode]) -> bool:

        # N(1)->N(2)->N(3)->N(4)->N(2)->N(3)->N(4)...

        # Iterate one pointer by 1 and another pointer by 2
        # If two pointers ever meet, there is cycle
        # If both pointers exit before meeting, there is no cycle
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        slow, fast = head, head

        while slow and fast and fast.next:
            slow = slow.next
            fast = fast.next.next

            if slow == fast:
                return True

        return False

Reorder List (75/150)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def reorderList(self, head: Optional[ListNode]) -> None:

        # N(1)->N(2)->N(3)->N(4)->N(5) ? N(1)->N(5)->N(2)->N(4)->N(3)

        # First node points to last
        # Last node points to first.next
        # First.next points to last.prev
        # Repeat
        # Revese second half of LL
        # Ex. N(1)->N(2)->N(3)<-N(4)<-N(5)
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        # Find middle node
        slow, fast = head, head
        while slow and fast and fast.next:
            slow = slow.next
            fast = fast.next.next

        # Reverse second half
        prev, node = None, slow
        while node:
            node_next = node.next
            node.next = prev
            prev = node
            node = node_next

        # Reorder
        while prev.next:
            head_next = head.next
            head.next = prev
            head = head_next

            prev_next = prev.next
            prev.next = head_next
            prev = prev_next

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
                
        # N(1)->N(2)->N(3)->N(4) ? N(1)<-N(2)<-N(3)<-N(4)

        # Iterate while remembering prev node
        # next_node = node.next
        # node.next = prev
        # prev = node
        # node = next_node
        # At the end, return prev
        # Time O(n) n = number of nodes in LL
        # Space O(1)

        node, prev = head, None

        while node:
            next_node = node.next
            node.next = prev
            prev = node
            node = next_node

        return prev

Find the Duplicate Number (150)

class Solution:
    def findDuplicate(self, nums: List[int]) -> int:
        
        # If there is no duplicate number, there should be
        # 1-to-1 mapping between index and value
        # Because there is duplicate number, more than two indexes
        # map to same value, which creates a cycle
        # How to find entrance to cycle?
        #   When slow and fast pointers crash,
        #   Move slow to the beginning,
        #   And move slow and fast under the same speed  

        slow, fast = nums[0], nums[0]

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while True:
            slow = nums[slow]
            fast = nums[nums[fast]]
            if slow == fast:
                break

        slow = nums[0]
        while slow != fast:
            slow = nums[slow]
            fast = nums[fast]

        return slow

Insert into a Sorted Circular Linked List (M)

"""
# Definition for a Node.
class Node:
    def __init__(self, val=None, next=None):
        self.val = val
        self.next = next
"""

class Solution:
    def insert(self, head: 'Optional[Node]', insertVal: int) -> 'Node':

        # Traverse linked list and find insertion point
        # If linked list is increasing in value
        #   node.val <= insertVal <= node.next.val
        # If linked list is decreasing in value, 
        # meaning node is at the end of linked list
        #   node.val <= insertVal or node.next.val >= insertVal

        # When there is no linked list, 
        # just create node with insertVal and return
        if not head:
            head = Node(insertVal)
            head.next = head
            return head

        node = head

        # Time O(n) 
        #   n = number of node in linked list
        # Space O(1)
        while node.next != head:
            
            # If increasing
            if node.val <= node.next.val:
                if node.val <= insertVal <= node.next.val:
                    break
            # If decreasing
            else:
                if node.val <= insertVal or insertVal <= node.next.val:
                    break

            node = node.next

        next_node = node.next
        node.next = Node(insertVal)
        node.next.next = next_node

        return head
