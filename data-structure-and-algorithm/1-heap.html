Kth Largest Element in an Array (150/250)

from heapq import heappush, heappop

class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        
        # Use min_heap of size k
        # If new num > root of min_heap, replace
        # Time O(nlogk) n = len(nums)
        # Space O(k) 

        min_heap = []

        for num in nums:
            if len(min_heap) < k:
                heappush(min_heap, num)
            elif num > min_heap[0]:
                heappop(min_heap)
                heappush(min_heap, num)

        return min_heap[0]

Find Median from Data Stream (75/150/250)

from heapq import heappush, heappop

class MedianFinder:

    # Use two heaps where number of values are equal
    # Send upper half to min_heap
    # Send lower half to max_heap
    # Push to max heap
    # Adjust if max_heap[0] > min_heap[0]
    # Balance
    # Add
    #   Time O(logn) n = number of elements coming in to stream
    #   Space O(n)
    # Median
    #   Time O(1)

    def __init__(self):
        self.min_heap = []
        self.max_heap = []

    def addNum(self, num: int) -> None:
        heappush(self.max_heap, -num)
        if self.max_heap and self.min_heap and (-self.max_heap[0] > self.min_heap[0]):
            val = -heappop(self.max_heap)
            heappush(self.min_heap, val)

        if len(self.max_heap) > len(self.min_heap) + 1:
            heappush(self.min_heap, -heappop(self.max_heap))

        if len(self.min_heap) > len(self.max_heap):
            heappush(self.max_heap, -heappop(self.min_heap))

    def findMedian(self) -> float:
        if len(self.max_heap) > len(self.min_heap):
            return float(-self.max_heap[0])
        return (-self.max_heap[0] + self.min_heap[0]) / 2

Top K Frequent Elements (75/150/250)

from heapq import heappush, heappop

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:

        # Create a dictionary of char count
        # Iterate dictionary 
        #   Push (count, num) tuple into min heap of size k
        # Return root 
        # Time O(nlogk) n = len(nums)
        # Space O(k)

        freq = defaultdict(int)
        for num in nums:
            freq[num] += 1

        min_heap = []
        for num, count in freq.items():
            heappush(min_heap, (count, num))
            if len(min_heap) > k:
                heappop(min_heap)

        result = []
        for count, num in min_heap:
            result.append(num)
        return result

Design Twitter (150/250)

from heapq import heappush, heappop

class Twitter:

    def __init__(self):
        self.time = 0
        self.following = defaultdict(set)
        self.tweets = defaultdict(list)

    def postTweet(self, userId: int, tweetId: int) -> None:
        self.tweets[userId].append((self.time, tweetId))
        self.time += 1

    def getNewsFeed(self, userId: int) -> List[int]:
        result = []
        max_heap = []
        user_ids = self.following[userId].copy()
        user_ids.add(userId)

        for user_id in user_ids:
            if self.tweets[user_id]:
                index = len(self.tweets[user_id]) - 1
                timestamp, tweet_id = self.tweets[user_id][index]
                heappush(max_heap, (-timestamp, tweet_id, user_id, index-1))
        
        while max_heap and len(result) < 10:
            neg_time, tweet_id, user_id, next_index = heappop(max_heap)
            result.append(tweet_id)
            if next_index >= 0:
                timestamp, next_tweet = self.tweets[user_id][next_index]
                heappush(max_heap, (-timestamp, next_tweet, user_id, next_index-1))

        return result

    def follow(self, followerId: int, followeeId: int) -> None:
        if followerId != followeeId:
            self.following[followerId].add(followeeId)

    def unfollow(self, followerId: int, followeeId: int) -> None:
        if followeeId in self.following[followerId]:
            self.following[followerId].remove(followeeId)

Task Scheduler (150/250)

from heapq import heappush, heappop

class Solution:
    def leastInterval(self, tasks: List[str], n: int) -> int:
        
        # [A,A,B,B], n = 2 ? 5 (A -> B -> idle -> A -> B)

        # Need to handle most frequent task first
        # Use heap to keep track of most frequent task
        # Remember the last position of each task
        # Insert idle appropriately
        # Instead of reducing frequency by 1 and immediately putting it
        # back to heap, store it in temp array (this is to avoid using
        # same tasks in a cycle)
        # Time O(nlogn) n = len(tasks)
        # Space O(n)

        freq = defaultdict(int)
        for t in tasks:
            freq[t] += 1
        
        max_heap = []
        for key, val in freq.items():
            heappush(max_heap, -val)

        cooldown = deque()
        time = 0

        while max_heap or cooldown:
            time += 1
            if max_heap:
                count = heappop(max_heap) + 1
                if count != 0:
                    cooldown.append((time+n, count))

            if cooldown and cooldown[0][0] == time:
                _, count = cooldown.popleft()
                heappush(max_heap, count)

        return time

from heapq import heappush, heappop

class KthLargest:

    # Create a min heap
    # When the size of heap > k, pop root element
    # Time O(logk) 
    # Space O(k)

    def __init__(self, k: int, nums: List[int]):
        self.min_heap = []
        self.k = k
        for num in nums:
            heappush(self.min_heap, num)
            if len(self.min_heap) > k:
                heappop(self.min_heap)

    def add(self, val: int) -> int:
        heappush(self.min_heap, val)
        if len(self.min_heap) > self.k:
            heappop(self.min_heap)
        return self.min_heap[0]

Max Stack

class MaxStack:

    # Push, pop, top is handled by python list in O(1)
    # Use heap to store max value and its index in the list
    # Handle pop
    #   Remember popped elements from stack or heap
    #   Remove popped elements when top/pop/peekMax/popMax  

    # Time O(logn)
    # Space O(n)
    def __init__(self):
        self.count = 0
        self.stack = []
        self.max_heap = []
        self.deleted = set() 
        
    def push(self, x: int) -> None:
        self.stack.append((x, self.count))
        heappush(self.max_heap, (-x, -self.count))
        self.count += 1

    def pop(self) -> int:
        while self.stack and self.stack[-1][1] in self.deleted:
            self.stack.pop()
        num, index = self.stack.pop()
        self.deleted.add(index)
        return num

    def top(self) -> int:
        while self.stack and self.stack[-1][1] in self.deleted:
            self.stack.pop()
        return self.stack[-1][0]

    def peekMax(self) -> int:
        while self.max_heap and -self.max_heap[0][1] in self.deleted:
            heappop(self.max_heap)
        return -self.max_heap[0][0]

    def popMax(self) -> int:
        while self.max_heap and -self.max_heap[0][1] in self.deleted:
            heappop(self.max_heap)
        num, index = heappop(self.max_heap)
        self.deleted.add(-index)
        return -num

K Closest Points to Origin (150/250)

from heapq import heappush, heappop

class Solution:
    def kClosest(self, points: List[List[int]], k: int) -> List[List[int]]:

        # Iterate points
        #   Compute distance to origin for each point
        #   Store (distance, point) tuple into max_heap of size k
        # Return length of max heap
        # Time O(nlogk) n = len(points)
        # Space O(k)

        max_heap = []
        for point in points:
            dist = point[0] ** 2 + point[1] ** 2
            heappush(max_heap, (-dist, point))
            if len(max_heap) > k:
                heappop(max_heap)
        
        result = []
        for dist, point in max_heap:
            result.append(point)
        return result

Last Stone Weight (150/250)

from heapq import heappush, heappop

class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        
        # Use max heap
        # Time O(nlogn) n = len(stones)
        # Space O(n)

        max_heap = []

        for stone in stones:
            heappush(max_heap, -stone)

        while len(max_heap) > 1:
            x = -heappop(max_heap)
            y = -heappop(max_heap)
            if x != y:
                heappush(max_heap, -x+y)

        if max_heap:
            return -max_heap[0]
        else:
            return 0

Stock Price Fluctuation

class StockPrice:

    # We only need latest (or greatest) timestamp
    # Use max heap to keep tuple of timestamp
    # Use dictionary to keep latest price per timestamp
    # How to find min and max?
    #   Use min heap and max heap for prices
    #   Insert incoming (timestamp, price) to both heaps
    #   When getting min/max, if (price, timestamp) matches
    #   the dictionary, return it. Otherwise, pop ot
      
    def __init__(self):
        self.max_heap_timestamp = []
        self.prices = defaultdict(int)
        self.max_heap_price = []
        self.min_heap_price = []
        
    # Time O(logn)
    #   n = number of update operation
    # Space O(n)
    def update(self, timestamp: int, price: int) -> None:
        heappush(self.max_heap_timestamp, -timestamp)
        self.prices[timestamp] = price
        heappush(self.max_heap_price, (-price, timestamp))
        heappush(self.min_heap_price, (price, timestamp))
        
    # Time O(1)
    def current(self) -> int:
        timestamp = self.max_heap_timestamp[0]
        return self.prices[-timestamp]

    def maximum(self) -> int:
        price, timestamp = self.max_heap_price[0]
        # If price is not outdated
        if -price == self.prices[timestamp]:
            return -price
        else:
            heappop(self.max_heap_price)
            return self.maximum() 

    def minimum(self) -> int:
        price, timestamp = self.min_heap_price[0]
        # If price is not outdated
        if price == self.prices[timestamp]:
            return price
        else:
            heappop(self.min_heap_price)
            return self.minimum()
