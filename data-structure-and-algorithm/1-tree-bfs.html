Binary Tree Level Order Traversal (75/150)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:

        # BFS

        result, values = [], defaultdict(list)
        queue = deque()
        queue.append((root,0))

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(n)
        while queue:
            node, depth = queue.popleft()
            if node:
                values[depth].append(node.val)

            if node and node.left:
                queue.append((node.left, depth+1))
            if node and node.right:
                queue.append((node.right, depth+1))

        for key, val in values.items():
            result.append(val)

        return result

Maximum Depth of Binary Tree (75/150)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxDepth(self, root: Optional[TreeNode]) -> int:

        # BFS

        depth = 0
        queue = deque()
        if root:
            queue.append((root, depth+1))

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(n)
        while queue:
            node, depth = queue.popleft()

            if node and node.left:
                queue.append((node.left, depth+1))
            if node and node.right:
                queue.append((node.right, depth+1))

        return depth

Binary Tree Right Side View (150)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:

        # BFS
        # Only keep hte last item for each level

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(d)
        #   d = depth of tree
        values = defaultdict(int)
        queue = deque()
        queue.append((root, 0))

        while queue:
            node, depth = queue.popleft()
            if node:
                values[depth] = node.val

            if node and node.left:
                queue.append((node.left, depth+1))
            if node and node.right:
                queue.append((node.right, depth+1))

        return list(values.values())

Binary Tree Vertical Order Traversal (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def verticalOrder(self, root: Optional[TreeNode]) -> List[List[int]]:

        # Vertical ordering 
        #   Use BFS while keeping depth of tree
        # Horizontal ordering
        #   During BFS, remember column of each node
        # Group result by column, then by depth
        #   Ordering of depth is guaranteed in BFS
        #   Ordering of column can be calculated by computing min/max
        #   value of column and considering all intergers in the middle

        # Edge case
        if not root:
            return []

        nodes = defaultdict(list)
        min_column, max_column = 0, 0

        queue = deque()
        queue.append((root, 0, 0))

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   h = height of tree
        #   h = n in the worst case
        while queue:
            node, depth, column = queue.popleft()
            min_column = min(min_column, column)
            max_column = max(max_column, column)
            nodes[column].append(node.val)
            
            if node.left:
                queue.append((node.left, depth+1, column-1))
            if node.right:
                queue.append((node.right, depth+1, column+1))

        result = []
        for i in range(min_column, max_column+1):
            result.append(nodes[i])

        return result

Vertical Order Traversal of a Binary Tree (H)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def verticalTraversal(self, root: Optional[TreeNode]) -> List[List[int]]:

        # BFS
        # Storing node depth as guarantees vertical order
        # Subtract 1 from column value for left child
        # Add 1 to column value for right child

        # Time O(nlogn) 
        #   n = number of nodes
        # Space O(n)

        nodes = []
        
        queue = deque()
        queue.append((root, 0, 0))

        while queue:
            node, depth, column = queue.popleft()
            nodes.append((node.val, depth, column))

            if node.left:
                queue.append((node.left, depth+1, column-1))
            if node.right:
                queue.append((node.right, depth+1, column+1))

        nodes.sort(key = lambda x: (x[2], x[1], x[0]))
        
        dic = defaultdict(list)
        for node in nodes:
            dic[node[2]].append(node[0])

        result = []
        for key, val in dic.items():
            result.append(val)

        return result
