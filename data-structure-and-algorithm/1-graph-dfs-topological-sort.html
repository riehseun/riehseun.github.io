Course Schedule (75/150)

class Solution:
    def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        
        # [[0,1],[1,2]] ? True (2->1->0)
        # [[0,1],[1,0]] ? False

        # Detect cycle
        # Use topological ordering
        # Find indegree of each node
        # Reduce indegree of each node by 1
        # Iterate until no nodes left
        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n)

        graph, indegree = defaultdict(set), defaultdict(int)
        queue = deque()

        for a, b in prerequisites:
            graph[b].add(a)
            indegree[a] += 1

        for i in range(numCourses):
            if indegree[i] == 0:
                queue.append(i)

        while queue:
            node = queue.popleft()

            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 0:
                    queue.append(next_node)

        for key, val in indegree.items():
            if val > 0:
                return False
        return True

Course Schedule II (150)

# Is there case like [[0,1],[1,0]]

        # 3, [[0,1],[1,2]] => 2->1->0
        # 3, [[0,1]] => 1->0->2
        # 3, [[0,1],[0,2]] => 2->1->0
        
        # Create graph with adjancency list
        # At each node, check if cycle
        # Time O(n*(n+m)) n = numCourses, m = len(prerequisites)
        # Space O(n)

        # Topological sort
        # Calculate indegree (incoming nodes) for each node
        # Start from nodes with indegree = 0
        # Time O(n+m) visiting nodes only once
        # Space O(n)
    
        graph, indegree = defaultdict(list), defaultdict(int)
        queue = deque()
        
        for a, b in prerequisites:
            graph[b].append(a)
            indegree[a] += 1

        for i in range(numCourses):
            if indegree[i] == 0:
                queue.append(i)

        nodes, num_explored_nodes = [], 0
        while queue:
            node = queue.popleft()
            nodes.append(node)
            num_explored_nodes += 1

            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 0:
                    queue.append(next_node)

        if num_explored_nodes == numCourses:
            return nodes
        
        return []

Graph Valid Tree (75/150)

class Solution:
    def validTree(self, n: int, edges: List[List[int]]) -> bool:

        # 0->(1,2,3->4) ? True
        # [] ? True if n == 1 False o/w
        # Cannot have cycles
        # All nodes are connected

        # Topological sort to detect cycle
        # Check if number of explored nodes = n

        graph, indegree = defaultdict(set), defaultdict(int)
        visited, queue, stack = set(), deque(), [0]
        
        for a, b in edges:
            graph[b].add(a)
            graph[a].add(b)
            indegree[a] += 1
            indegree[b] += 1

        for i in range(n):
            if indegree[i] == 1:
                queue.append(i)

        while queue:
            node = queue.popleft()
            
            for next_node in graph[node]:
                indegree[next_node] -= 1
                indegree[node] -= 1
                if indegree[next_node] == 1:
                    queue.append(next_node)

        while stack:
            node = stack.pop()

            for next_node in graph[node]:
                if next_node not in visited:
                    visited.add(next_node)
                    stack.append(next_node)

        if not edges:
            if n == 1:
                return True
            return False
        for a, b in edges:
            if indegree[a] >= 2 and indegree[b] >= 2:
                return False
        return len(visited) == n

Alien Dictionary (75/150)

# Compare two adjacent words at a time
        # Only the first difference in char tells the ordering
        #   If second word is prefix of first word, return ""
        # Construct a graph based on the ordering
        #   Letters than don't appear in the graph are not subject to
        #   ordering, thus can go anywhere in the output
        #   If there is cycle, return ""
        # Then, topological sort
        # Time O(c), c = all character in words
        # Space O(1)

        all_chars = set()
        for word in words:
            for char in word:
                all_chars.add(char)
        
        graph, indegree = defaultdict(set), defaultdict(int)

        # Time O(c)
        # Space O(c) 
        for i in range(len(words)-1):
            # If second word is prefix of first word, 
            # it is not valid ordering
            if words[i].startswith(words[i+1]) and len(words[i]) > len(words[i+1]):
                return "" 
            for c1, c2 in zip(words[i], words[i+1]):
                if c1 != c2:
                    graph[c1].add(c2)
                    break

        for key, val in graph.items():
            for c in val:
                indegree[c] += 1

        queue, order = deque(), ""
        for c in all_chars:
            if indegree[c] == 0:
                queue.append(c)

        while queue:
            node = queue.popleft()
            order += node

            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 0:
                    queue.append(next_node)

        for key, val in indegree.items():
            if val > 0:
                return ""

        return order

Longest Increasing Path in a Matrix (150)

class Solution:
    def longestIncreasingPath(self, matrix: List[List[int]]) -> int:

        # Topological sort
        # Time O(nm) nm = number of nodes
        # Space O(nm)
        
        n, m = len(matrix), len(matrix[0])
        dirs = [(-1,0), (1,0), (0,-1), (0,1)]
        graph, indegree = defaultdict(list), defaultdict(int)

        for i in range(n):
            for j in range(m):
                for x, y in dirs:
                    if 0 <= i +x < n \
                        and 0 <= j + y < m \
                        and matrix[i+x][j+y] > matrix[i][j]:
                        graph[(i,j)].append((i+x,j+y))
                        indegree[(i+x,j+y)] += 1

        queue = deque()
        for key, val in graph.items():
            if key not in indegree:
                queue.append((key, 0))

        max_depth = 0
        
        while queue:
            node, depth = queue.popleft()
            max_depth = max(max_depth, depth)
            
            for next_node in graph[node]:
                indegree[next_node] -= 1
                if indegree[next_node] == 0:
                    queue.append((next_node, depth+1))
        
        return max_depth + 1

Redundant Connection (150)

class Solution:
    def findRedundantConnection(self, edges: List[List[int]]) -> List[int]:
        
        # Detect cycle in undirected graph?

        # edges = [] => invalid input
        # edges = [[0,1],[1,2],[2,0]], n = 3 => [2,0]

        # Cycle detection? topological sort
        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n)

        graph, indegree = defaultdict(list), defaultdict(int)
        queue = deque()
        
        for a, b in edges:
            graph[b].append(a)
            graph[a].append(b)
            indegree[a] += 1
            indegree[b] += 1

        for i in range(1, len(edges)+1):
            if indegree[i] == 1:
                queue.append(i)

        while queue:
            node = queue.popleft()
        
            for next_node in graph[node]:
                indegree[next_node] -= 1
                indegree[node] -= 1
                if indegree[next_node] == 1:
                    queue.append(next_node)

        edges.reverse()
        for a, b in edges:
            if indegree[a] >= 2 and indegree[b] >= 2:
                return [a, b]

        return []
