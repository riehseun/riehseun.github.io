Longest Palindromic Substring (75/150)

class Solution:
    def longestPalindrome(self, s: str) -> str:

        # s = "a" => a
        # s = "abcd" => "a"

        # For each substring of varying length, check parlindrome
        # Time O(n**3)

        # Iterate string
        # At each char, expand to both left and right
        # Time O(n**2) n = len(s)
        # Space O(1)

        n = len(s)
        max_length, max_string = 0, ""

        for i, c in enumerate(s):
            j, k = i, i

            while j - 1 >= 0 and s[i] == s[j-1]:
                j -= 1

            while k + 1 < n and s[k] == s[k+1]:
                k += 1

            while j - 1 >= 0 \
                and k + 1 < n \
                and s[j-1] == s[k+1]:
                j -= 1
                k += 1
            
            if k - j + 1 > max_length:
                max_length = k - j + 1
                max_substring = s[j:k+1]

        return max_substring

Container With Most Water (75/150)

class Solution:
    def maxArea(self, height: List[int]) -> int:
        
        # Water that can be stored between i and j
        # min(heights[i], heights[j]) * (j-i)
        # if heights[i] > heights[j], try moving j

        start, end, max_water = 0, len(height) - 1, 0

        # Time O(n)
        #   n = len(height)
        # Space O(1)
        while start < end:
            water = min(height[start], height[end]) * (end-start)
            max_water = max(max_water, water)
            if height[start] > height[end]:
                end -= 1
            else:
                start += 1

        return max_water

3Sum (75/150)

class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:

        # [-1,0,1,2,3] ? [[-1,0,1]]

        # Sort array
        # Fix a value
        # Check from both ends to see value + nums[i] + nums[j] == 0
        # Time O(nlogn+n**2) n = len(nums)
        # Space O(1)
        
        nums.sort()
        result = set()

        for i in range(len(nums)):
            # Skip duplicate check
            if i != 0 and nums[i] == nums[i-1]:
                continue
            
            j, k = i + 1, len(nums) - 1
            while j < k:
                three_sum = nums[j] + nums[i] + nums[k]
                if three_sum < 0:
                    j += 1
                elif three_sum > 0:
                    k -= 1
                else:
                    result.add((nums[i], nums[j], nums[k]))
                    j += 1
                    k -= 1

        return list(result)

4Sum (M)

class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:

        # Reduce 4sum problem to 3sum problem
        #   Fix the first two numbers
        #   Then, find the combination of remaning two numbers

        nums.sort()
        result = set()

        # Time O(n**3) 
        #   n = len(nums)
        # Space O(1)
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                k = j + 1
                l = len(nums) - 1

                while k < l:
                    sum = nums[i] + nums[j] + nums[k] + nums[l]
                    if sum == target:
                        quadlet = [nums[i], nums[j], nums[k], nums[l]]
                        quadlet.sort()
                        result.add(tuple(quadlet))
                        k += 1
                        l -= 1
                    elif sum > target:
                        l -= 1
                    elif sum < target:
                        k += 1

        return list(result)

Remove Duplicates from Sorted Array (E)

class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
       
        # Use i to interate nums
        # Use j to denote insert positions
        # When nums[i+1] != nums[i], nums[j] = nums[i+1] and j += 1

        j = 1

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        for i in range(len(nums)-1):
            if nums[i+1] != nums[i]:
                nums[j] = nums[i+1]
                j += 1

        return j

        #   Highest bar to the right
        #   Highest bar to the left
        # that is greater than height[i]

        # Remember the max height from left and right

        water, max_height_left, max_height_right = 0, 0, 0

        #   n = len(height)
        # Space O(1)
        while start <= end:
            if max_height_left < max_height_right:
                max_height_left = max(max_height_left, height[start])
                h = min(max_height_left, max_height_right)
                water += max(h-height[start], 0)
                start += 1
            else:
                max_height_right = max(max_height_right, height[end])
                h = min(max_height_left, max_height_right)
                water += max(h-height[end], 0)
                end -= 1
                
        return water

Sort Colors (M)

class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Initialize three pointers
        #   low pointer at 0
        #   mid pointer at 0
        #   high pointer at len(nums)-1
        # Move mid pointer such that
        #   all 0s are before low pointer
        #   all 2s are after high pointer
        #   all 1s are between low and high pointer

        low = 0
        high = len(nums) - 1
        mid = 0

        # Time O(n) 
        #   n = len(nums)
        # Space O(1)
        while mid <= high:
            # Move 0 to the beginning and move low pointer
            if nums[mid] == 0:
                nums[low], nums[mid] = nums[mid], nums[low]
                low += 1
                mid += 1
            # Move 2 to the end
            elif nums[mid] == 2:
                nums[high], nums[mid] = nums[mid], nums[high]
                high -= 1
            else:
                mid += 1

Merge Sorted Array (E)

class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """

        # Iterate nums1 and nums2 from the right
        #   Compare and put bigger number at the end of nums1

        if not nums2:
            return

        i, j, k = m - 1, n - 1, m + n - 1

        # Time O(n+m)
        # Space O(1)
        while k >= 0:

            if j < 0:
                nums1[:k+1] = nums1[:i+1]
                return

            if i < 0:
                nums1[:k+1] = nums2[:j+1]
                return

            if nums1[i] > nums2[j]:
                nums1[k] = nums1[i]
                i -= 1

            else:
                nums1[k] = nums2[j]
                j -= 1
            
            k -= 1

Valid Palindrome (75/150)

class Solution:
    def isPalindrome(self, s: str) -> bool:

        # "I and I" ? True
        # "You are him" ? False

        # Iterate two pointers are both ends
        # Skip non-alphanumeric chars
        # Convert char to lowercase before comparing
        # Time O(n) n = len(s)
        # Space O(1)

        i, j = 0, len(s) - 1

        while i < j:
            while not s[i].isalnum() and i < j:
                i += 1
            while not s[j].isalnum() and i < j:
                j -= 1
            if s[i].lower() != s[j].lower():
                return False
            i += 1
            j -= 1

        return True

Two Sum II - Input Array Is Sorted (150)

class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:
        
        # Array is sorted
        # Two pointers
        #   If sum > target, decrease end pointer
        #   If sum < target, increase start pointer

        start, end = 0, len(numbers) - 1

        # Time O(n)
        #   n = len(numbers)
        # Space O(1)
        while start < end:
            sum = numbers[start] + numbers[end]
            if sum > target:
                end -= 1
            elif sum < target:
                start += 1
            else:
                return [start+1, end+1]

        return None

Move Zeroes (E)

class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """

        # Iterate array
        #   When seeing non-zero element, swap with first zero
        #   Remember index of first zero

        i, j = 0, 0

        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while j < len(nums):

            if nums[j] != 0 and nums[i] == 0:
                nums[j], nums[i] = nums[i], nums[j]

            if nums[i] != 0:
                i += 1

            j += 1

Palindromic Substring (75/150)

class Solution:
    def countSubstrings(self, s: str) -> int:

        # For every center, expand substring until
        # getting palindrome
        # Check odd and even length separately

        count = 0

        # Time O(n**2) 
        #   n = len(s)
        # Space O(1)
        # s = [] => Invalid
        # s = ["a"] => 1
        # s = ["ab"] => 2

        # Iterate string
        # At each char, expand to both left and right
        # Check odd and even length separately
        # Time O(n**2) n = len(s)
        # Space O(1)

        count = 0
        n = len(s)

        for i in range(n):
            count += 1
            j, k = i, i
            while j - 1 >= 0 \
                and k + 1 < n \
                and s[j-1] == s[k+1]:
                count += 1
                j -= 1
                k += 1

        for i in range(n-1):
            if s[i] == s[i+1]:
                count += 1
                j, k = i, i + 1
                while j - 1 >= 0 \
                    and k + 1 < n \
                    and s[j-1] == s[k+1]:
                    count += 1
                    j -= 1
                    k += 1

        return count

Valid Palindrome II (E)

class Solution:
    def validPalindrome(self, s: str) -> bool:

        # Two pointers
        # Iterate from both ends until there is mismatch
        # If there is mismatch
        #   Remove either start or end, then check again

        # Time O(n)
        #   n = len(s)
        # Space O(1)
        def valid_palindrome(s):

            start = 0
            end = len(s) - 1

            while start < end:
                if s[start] != s[end]:
                    return (start, end)
                start += 1
                end -= 1
            
            return (-1, -1)

        start, end = valid_palindrome(s)
        if (start, end) == (-1, -1):
            return True
        else:
            if valid_palindrome(s[:start]+s[start+1:]) == (-1, -1):
                return True
            if valid_palindrome(s[:end]+s[end+1:]) == (-1, -1):
                return True
            return False
