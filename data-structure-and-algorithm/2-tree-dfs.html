Validate Binary Search Tree (75/150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:

        # [1,2,3,4,5,6,7] ? False
        # [10,5,15,3,7,12,17] ? True

        # In-order traverse - left, root, right
        # Check left < root < right
        # Previous value must be less that current value
        # Time O(n) n = number of nodes in tree
        # Space O(n) recursion stack worst case h = n

        def in_order(node):

            if node:
                left = in_order(node.left)
                if self.prev and self.prev.val >= node.val:
                    self.is_valid_bst = False
                self.prev = node
                right = in_order(node.right)
                
        self.prev, self.is_valid_bst = None, True
        in_order(root)
        return self.is_valid_bst

Same Tree (75/150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def isSameTree(self, p: Optional[TreeNode], q: Optional[TreeNode]) -> bool:
        
        # p = [1,2,3] q = [1,2,3] ? True
        # p = [1,2,3] q = [1,3,2] ? False

        # Pre-order traverse on both p and q
        # Check if values at each ndoe is the same
        # Time O(n) n = number of nodes in tree
        # Space O(n) recursion stack worst case h = n

        def pre_order(p, q, result=True):

            if p and q:
                if p.val != q.val:
                    self.result = False
                pre_order(p.left, q.left)
                pre_order(p.right, q.right)
            if (p and not q) \
                or (q and not p):
                self.result = False

        self.result = True
        pre_order(p, q)
        return self.result

Construct Binary Tree from Preorder and Inorder Traversal (75/150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        
        # preorder = [1,2,3,4] inorder = [4,2,1,3] ? [1,2,3,4,N,N,N]

        # Pre-order traverse - root, left, right
        # Root is first element in preorder
        # Left subtree is inorder[:root]
        # Right subtree is inorder[root+1:]
        # Time O(n) n = number of nodes in tree
        # Space O(n) recursion stack worst case h = n

        def pre_order(preorder, inorder):

            if inorder:
                node_val = preorder.pop(0)
                node = TreeNode(node_val)
                index = inorder.index(node_val) 
                node.left = pre_order(preorder, inorder[:index])
                node.right = pre_order(preorder, inorder[index+1:])

                return node

        return pre_order(preorder, inorder)

Balanced Binary Tree (150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def isBalanced(self, root: Optional[TreeNode]) -> bool:

        # At each node, compare height of left and right subtree
        # At leaf node, return 0
        # At each node, return max(left, right) + 1
        # Time O(n) n = number of nodes in tree
        # Space O(n) recursion stack

        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)
                if abs(left-right) > 1:
                    self.result = False
                return max(left, right) + 1
            else:
                return 0

        self.result = True
        post_order(root)
        return self.result

Binary Tree Maximum Path Sum (75/150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:

        # [1,2,3] ? 6
        # [1,-2,3] ? 3

        # Post-order traverse - left, right, root
        # At each node, take max of left and right
        # If left and right has path sum < 0, take 0
        # Time O(n) n - number of nodes in tree
        # Space O(n) recursion stack worst case h = n

        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)

                path_sum = max(left, 0) + max(right, 0) + node.val
                self.max_path_sum = max(self.max_path_sum, path_sum)
                return max(node.val, node.val+left, node.val+right)
            else:
                return 0

        self.max_path_sum = -math.inf
        post_order(root)
        return self.max_path_sum

Sum Root to Leaf Numbers (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:

        # DFS pre-order
        # Calculate running sum for each path
        #   At each node, multiply running sum by 10
        #   If leaf node, add running sum to total sum  

        self.total_sum = 0

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(h)
        #   Recursion stack h = n in worst case
        def pre_order(node, running_sum):

            if node:
                running_sum = running_sum * 10 + node.val
                
                if not node.left and not node.right:
                    self.total_sum += running_sum
                
                pre_order(node.left, running_sum)
                pre_order(node.right, running_sum)                

        pre_order(root, 0)    

        return self.total_sum

Invert Binary Tree (75/150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def invertTree(self, root: Optional[TreeNode]) -> Optional[TreeNode]:
        
        # [1,2,3,4,5,6,7] ? [1,3,2,7,6,5,4]

        # Pre-order traverse - root, left, right
        # Swap left and right node
        # Time O(n) n = number of nodes in tree
        # Space O(n) recursion stack worst case h = n 

        def pre_order(node):
            
            if node:
                node.left, node.right = node.right, node.left
                pre_order(node.left)
                pre_order(node.right)

        pre_order(root)
        return root

Kth Smallest Element in a BST (75/150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:

        # [2,1,3] k = 1 ? 1 
        # [2,1,3] k = 2 ? 2

        # In-order traverse - left, root, right
        # Return kth number
        # Time O(n) n = number of nodes in tree
        # Space O(n) recursion stack worst case h = n

        def in_order(node):

            if node:
                in_order(node.left)
                self.k -= 1
                if self.k == 0:
                    self.kth_smallest = node.val
                in_order(node.right)
        
        self.kth_smallest, self.k = math.inf, k
        in_order(root)
        return self.kth_smallest

Lowest Common Ancestor of a Binary Tree (75/150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def lowestCommonAncestor(self, root: TreeNode, p: TreeNode, q: TreeNode) -> TreeNode:
        
        # [1,2,3,4,5,6,7] p = 2 q = 3 ? 1
        # [1,2,3,4,5,6,7] p = 2 q = 4 ? 2

        # Post-order traverse - left, right, root
        # If node == p or q, return 1
        # If left child + right child + root == 2, return root
        # Time O(n) n = number of nodes in tree
        # Space O(n) recursion stack worst case h = n

        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)

                parent = 0
                if node.val == p.val or node.val == q.val:
                    parent = 1

                if left + right + parent == 2 \
                    and not self.lca:
                    self.lca = node
                    
                return left + right + parent
            else:
                return 0

        self.lca = None
        post_order(root)
        return self.lca

Serialize and Deserialize Binary Tree (75/150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Codec:

    # [1,2,3,N,N,4,5] ? *1*2*#*#*3*4*#*#*5*#*#

    # Pre-order traverse - root, left, right
    # Insert * before node val
    # Insert # for null node
    # Time O(n) n = number of nodes in tree
    # Space O(n) recursion stack worst case h = n

    # Split by *
    # Pop the first element
    # If element != #, create root node with first element
    # Recurse on left and right child with remaining array  
    
    # Encodes a tree to a single string.
    def serialize(self, root: Optional[TreeNode]) -> str:

        def pre_order(node):

            if node:
                self.array.append("*")
                self.array.append(str(node.val))
                pre_order(node.left)
                pre_order(node.right)
            else:
                self.array.append("*")
                self.array.append("#")

        self.array = []
        pre_order(root)
        return "".join(self.array)
        
    # Decodes your encoded data to tree.
    def deserialize(self, data: str) -> Optional[TreeNode]:
        
        def pre_order(array):
            
            value = array.pop(0)
            if value != "#":
                node = TreeNode(int(value))
                node.left = pre_order(array)
                node.right = pre_order(array)
                return node

        array = data.split("*")
        array.pop(0)  # Remove first *
        return pre_order(array)

Find Leaves of Binary Tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def findLeaves(self, root: Optional[TreeNode]) -> List[List[int]]:

        # DFS post-order
        # Add leaf nodes to array
        # Then, remove leaf nodes from the tree
        # Repeat until only the root remains

        # Time O(h) 
        #   h = height of tree
        # Space O(n)
        #   n = number of nodes in tree 
        def post_order(node, array):

            if node:

                # Check if leaf node
                if not node.left and not node.right:
                    array.append(node.val)
                    return -1
                
                left = post_order(node.left, array)
                right = post_order(node.right, array)
                # Remove leaf node on left side
                if left == -1:
                    node.left = None
                # Remove leaf node on right side
                if right == -1:
                    node.right = None
                
        result = []

        while root.left or root.right:
            array = []
            post_order(root, array)
            result.append(array)
            
        result.append([root.val])
        return result

Diameter of Binary Tree (150/250)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:
        
        # Diameter = root + left subtree + right subtree
        # Post-order traverse
        # At leaf node, return 0
        # At root, take 1 + max(left ,right)
        # Time O(n) n = number of nodes in tree
        # Space O(n) recursion stack

        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)
                self.max_d = max(self.max_d, 1+left+right)
                return 1 + max(left, right)
            else:
                return 0

        self.max_d = 0
        post_order(root)
        return self.max_d - 1

        # [1,2,3,4,5,6,7] [2,4,5] ? True
        # [1,2,3,4,5,6,7] [2,4] ? False

        # Pre-order traverse - root, left, right
        # Serialize tree in a unique way
        # Compare if substring is in string

        # Space O(n+m) recursion stack worst case h = n, m

        def pre_order(node, array):

            if node:
                array.append("*")
                array.append(str(node.val))
                pre_order(node.left, array)
                pre_order(node.right, array)
            else:
                array.append("#")

        tree, subtree = [], []
        pre_order(root, tree)
        pre_order(subRoot, subtree)
        tree_str, subtree_str = "".join(tree), "".join(subtree)
        return subtree_str in tree_str

All Nodes Distance K in Binary Tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distanceK(self, root: TreeNode, target: TreeNode, k: int) -> List[int]:

        # DFS on tree to convert to graph
        # DFS on graph

        graph = defaultdict(set)

        # Time O(n)
        #   n = number of nodes in tree
        # Space O(n)
        def pre_order(node, parent, graph):

            if node:
                
                if parent:
                    graph[parent.val].add(node.val)
                    graph[node.val].add(parent.val)

                pre_order(node.left, node, graph)
                pre_order(node.right, node, graph)

        pre_order(root, None, graph)

        distances = defaultdict(list)

        # Time O(n+m) 
        #   n = number of nodes
        #   m = number of edges
        # Space O(n)
        def dfs(graph, start):
            
            explored = set()
            explored.add(start)

            stack = []
            stack.append((start, 0))

            while stack:
                node, depth = stack.pop()
                distances[depth].append(node)
                
                for next_node in graph[node]:
                    if next_node not in explored:
                        explored.add(next_node)
                        stack.append((next_node, depth+1))
                        
        dfs(graph, target.val)

        return distances[k]

Range Sum of BST (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rangeSumBST(self, root: Optional[TreeNode], low: int, high: int) -> int:

        # DFS pre-order
        #   Add to sum only if value is within the range

        self.sum = 0

        # Time O(n)
        #   n = number of nodes
        # Space O(h)
        #   Recursion stack h = n worst case
        def pre_order(node, low, high):
            
            if node:

                if node.val >= low and node.val <= high:
                    self.sum += node.val

                pre_order(node.left, low, high)
                pre_order(node.right, low, high)

        pre_order(root, low, high)

        return self.sum

Lowest Common Ancestor of a Binary Tree III (M)

"""
# Definition for a Node.
class Node:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None
        self.parent = None
"""

class Solution:
    def lowestCommonAncestor(self, p: 'Node', q: 'Node') -> 'Node':

        # Do post order traverse
        #   If a node is p or q, return 1
        #   Else, return 0
        # If left and right returns 2, 
        # then current root is lca
        # Also, if either left or right return 1 and current root is p or q,
        # then, current root is lca

        # First, need to find root
        
        self.lca = None

        def find_root(p):
            
            if p.parent: 
                return find_root(p.parent)
            else:
                return p

        # Time O(h) = O(n)
        #   h = height of tree, worst case h = n
        # Space O(1)
        def post_order(node):

            if node:
                left = post_order(node.left)
                right = post_order(node.right)
                
                mid = 0
                if node.val == p.val or node.val == q.val:
                    mid = 1

                if left + right + mid == 2 \
                    and not self.lca:
                    self.lca = node
                    
                return left + right + mid

            else:
                return 0

        root = find_root(p)
        post_order(root)
        return self.lca
