Rotting Oranges (150/250)

class Solution:
    def orangesRotting(self, grid: List[List[int]]) -> int:
        
        # Multi source BFS
        # From all rotten node, expand at the same time
        # If seen fresh node, convert it to rotten
        # If seen empty cell, don't expand
        # Time O(n) n = number of nodes
        # Space O(n)

        n, m = len(grid), len(grid[0])
        dirs = [(-1,0), (1,0), (0,1), (0,-1)]
        queue = deque()
        depth = 0

        for i in range(n):
            for j in range(m):
                if grid[i][j] == 2:
                    queue.append((i,j,0))

        while queue:
            i, j, d = queue.popleft()

            for x, y in dirs:
                if 0 <= i + x < n \
                    and 0 <= j + y < m \
                    and grid[i+x][j+y] == 1:
                    queue.append((i+x, j+y, d+1))
                    grid[i+x][j+y] = 2
            depth = d

        for i in range(n):
            for j in range(m):
                if grid[i][j] == 1:
                    return -1
        return depth

Islands and Treasure (M)

class Solution:
    def islandsAndTreasure(self, grid: List[List[int]]) -> None:
        
        # From each treasure chest, find distance to land
        # Keep minimum distance in each cell
        # Multi source BFS
        # Time O(n) n = number of nodes
        # Space O(n)

        n, m = len(grid), len(grid[0])
        dirs = [(-1,0), (1,0), (0,-1), (0,1)]
        queue = deque()

        for i in range(n):
            for j in range(m):
                if grid[i][j] == 0:
                    queue.append((i,j,0))

        while queue:
            i, j, d = queue.popleft()

            for x, y in dirs:
                if 0 <= i + x < n \
                    and 0 <= j + y < m \
                    and grid[i+x][j+y] != -1 \
                    and grid[i+x][j+y] != 0 \
                    and grid[i+x][j+y] > d + 1:
                    queue.append((i+x, j+y, d+1))
                    grid[i+x][j+y] = d + 1
