Largest Rectangle in Histogram (150/250)

class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        
        # For each i, find the largest rectangle
        # Expand to left and right while its height >= heights[i]
        # How to find right boundry
        #   Push each element into stack
        #   When decreasing in sequence, 
        #   current element is the right boundry of 
        #   previous element (which is in stack)
        # How to find left boundry
        #   previous previous is left boundry
        #   as long as stack only keeps increasing sequence

        stack, max_area = [], 0
        stack.append(-1)  # Left boundry of heights[0]

        # Time O(n)
        #   n = len(heights)
        # Space O(n)
        for i, h in enumerate(heights):

            while stack[-1] != -1 and heights[stack[-1]] > h:
                height = heights[stack.pop()]  # current height
                right = i
                left = stack[-1]
                max_area = max(max_area, height*(right-left-1))
                print(max_area)
            stack.append(i)

        while stack[-1] != -1:
            height = heights[stack.pop()]
            right = len(heights)
            left = stack[-1]
            max_area = max(max_area, height*(right-left-1))

        return max_area

Min Stack (150/250)

class MinStack:

    # Push, pop, top is easy with list
    # Use another stack that is monotonically decreasing
    # Min is always at the top
    # Push 
    #   Push to stack
    #   Push to min_stack if val <= min_stack[-1]
    # Pop 
    #   Pop from stack (say "val")
    #   Pop from min_stack if "val" == min_stack[-1] 
    # Time O(1) 
    # Space O(n) n = number of items in stack

    def __init__(self):
        self.stack, self.min_stack = [], []

    def push(self, val: int) -> None:
        self.stack.append(val)
        if not self.min_stack or val <= self.min_stack[-1]:
            self.min_stack.append(val)

    def pop(self) -> None:
        val = self.stack.pop()
        if self.min_stack and val == self.min_stack[-1]:
            self.min_stack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.min_stack[-1]
