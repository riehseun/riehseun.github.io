class CountSquares:

    # Use hash table to store coordinate and its count
    # For given point (x, y) find (x+d, y+d) 
    # The difference of each axis must be the same (d)
    # Add
    # Time O(1)
    # Space O(n) n = number of points
    # Count
    # Time O(n)
    # Space O(1)

    def __init__(self):
        self.coord = defaultdict(int)

    def add(self, point: List[int]) -> None:
        self.coord[tuple(point)] += 1

    def count(self, point: List[int]) -> int:

        x, y = point
        cnt = 0
        for key, val in self.coord.items():
            px, py = key
            if abs(x-px) != abs(y-py):
                continue
            if x == px:
                continue
            a, b = 0, 0
            if (x, py) in self.coord:
                a = self.coord[(x, py)]
            if (px, y) in self.coord:
                b = self.coord[(px, y)]
            cnt += val * a * b
            
        return cnt
