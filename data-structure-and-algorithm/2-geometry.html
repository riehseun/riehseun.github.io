Detect Squares (150)

class DetectSquares:

    # Use list of list
    # For point i, j need a diagonal point x, y where
    # abs(i-x) == abs(j-y)
    # Time O(n) n = number of adds
    # Space O(n)

    def __init__(self):
        self.points = defaultdict(int)

    def add(self, point: List[int]) -> None:
        self.points[tuple(point)] += 1
        
    def count(self, point: List[int]) -> int:
        
        total_count = 0
        
        for key, val in self.points.items():
            if abs(key[0]-point[0]) != abs(key[1]-point[1]):
                continue
            if key[0] == point[0]:
                continue
            one, two = 0, 0
            if (point[0], key[1]) in self.points:
                one = self.points[(point[0], key[1])]
            if (key[0], point[1]) in self.points:
                two = self.points[(key[0], point[1])]
            total_count += val * one * two
                        
        return total_count
