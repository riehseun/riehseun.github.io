Reconstruct Itinerary (150/250)

class Solution:
    def findItinerary(self, tickets: List[List[str]]) -> List[str]:
        
        # DFS (with cycle)
        # Sort tickets by destination
        # Greedily process last item amongst destinations
        # Time O(nlogn+n+m) n = number of nodes, m = number of edges
        # Space O(n)

        graph = defaultdict(list)
        sorted_ticket = tickets[:]
        sorted_ticket.sort(key=lambda x: x[1], reverse=True)

        for src, det in sorted_ticket:
            graph[src].append(det)

        stack = []
        stack.append("JFK")
        result = []

        while stack:
            while graph[stack[-1]]:
                next_node = graph[stack[-1]].pop()
                stack.append(next_node)
            result.append(stack.pop())

        return result[::-1]

Network Delay Time (150/250)

class Solution:
    def networkDelayTime(self, times: List[List[int]], n: int, k: int) -> int:
        
        # BFS
        # Keep minimum time to reach each node
        # Only move to next node if it is cheaper to do so
        # Is there a cycle?
        # Time O(n+m) n = number of nodes, m = number of edges
        # Space O(n)

        graph, visited = defaultdict(list), defaultdict(int)

        for a, b, t in times:
            graph[a].append((b,t))

        queue = deque()
        queue.append((k,0))
        visited[k] = 0
        
        while queue:
            node, time = queue.popleft()
            
            for next_node, next_time in graph[node]:
                if next_node not in visited:
                    visited[next_node] = time + next_time
                    queue.append((next_node, time+next_time))
                else:
                    if visited[next_node] > time + next_time:
                        visited[next_node] = time + next_time
                        queue.append((next_node, time+next_time))

        if len(visited) != n:
            return -1
        
        return max(list(visited.values()))

Swim in Rising Water (150/250)

class Solution:
    def swimInWater(self, grid: List[List[int]]) -> int:

        # Starting at (0,0), add all nearby cells to min_heap
        # Always try walking on cell with lowest possible
        # Biggest number seen in the path is the answer
        # Time O((n**2)log(n**2))
        # Space O(n**2)

        min_heap = [] 
        min_heap.append([grid[0][0], 0, 0])
        dirs = [(-1,0), (1,0), (0,-1), (0,1)]
        visited = set()
        visited.add((0,0))
        n = len(grid)
        result = 0

        while min_heap:
            h, i, j = heappop(min_heap)
            result = max(result, h)
            if i == j == n - 1:
                break
            for x, y in dirs:
                if 0 <= i + x < n \
                    and 0 <= j + y < n \
                    and (i+x, j+y) not in visited:
                    visited.add((i+x, j+y))
                    heappush(min_heap, [grid[i+x][j+y], i+x, j+y])

        return result

Making A Large Island (H)

class UnionFind:

    def __init__(self):
        self.parent = defaultdict(tuple)
        self.size = defaultdict(int)

    def union(self, node_1: tuple, node_2: tuple):

        leader_1 = self.find(node_1)
        leader_2 = self.find(node_2)

        if leader_1 == leader_2:
            return

        if self.size[leader_1] > self.size[leader_2]:
            self.parent[leader_2] = leader_1
            self.size[leader_1] += self.size[leader_2]
        else:
            self.parent[leader_1] = leader_2
            self.size[leader_2] += self.size[leader_1]

    def find(self, node: tuple) -> int:

        if self.parent[node] == node:
            return node

        return self.find(self.parent[node])

class Solution:
    def largestIsland(self, grid: List[List[int]]) -> int:

        # UnionFind
        # Initialize 
        #   Parent of each node to be itself
        #   Size of each node is 1
        # For each node with value 1, 
        # find neighbors with value 1, then union them
        # Then, for each node with value 0, 
        # find neighboring islands

        unionfind = UnionFind()
        directions = [
            (-1, 0), 
            (0, -1), 
            (0, 1), 
            (1, 0),
        ]

        is_zero = False
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                unionfind.parent[(i,j)] = (i,j)
                unionfind.size[(i,j)] = 1
                if grid[i][j] == 0:
                    is_zero = True
        
        if not is_zero:
            return len(grid) * len(grid[0])

        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    for row, col in directions:
                        if 0 <= row + i < len(grid) \
                            and 0 <= col + j < len(grid) \
                            and grid[row+i][col+j] == 1:
                            unionfind.union((i,j),(row+i,col+j))

        max_size = 0
        unique_roots = set()
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 0:
                    current_size = 1
                    for row, col in directions:
                        if 0 <= row + i < len(grid) \
                            and 0 <= col + j < len(grid) \
                            and grid[row+i][col+j] == 1:
                            root = unionfind.find((row+i,col+j))
                            unique_roots.add(root)
                    for root in unique_roots:
                        current_size += unionfind.size[root]
                    unique_roots.clear()
                    max_size = max(max_size, current_size)

        print(unionfind.size)
        return max_size

Min Cost to Connect All Points (150/250)

class Solution:
    def minCostConnectPoints(self, points: List[List[int]]) -> int:
        
        # Minimum spanning tree
        # Find a node not in MST, which has lowest cost
        # Add that node to MST
        # Update all nearby nodes
        # Repeat
        # Time O(n**2) n = number of nodes
        # Space O(n)

        n = len(points)
        nodes = [math.inf] * n
        nodes[0] = 0
        mst = set()
        total_cost = 0
        
        while n > len(mst):

            min_node, min_cost = -1, math.inf
            for i in range(n):
                if i not in mst and nodes[i] < min_cost:
                    min_cost = nodes[i]
                    min_node = i

            total_cost += min_cost
            mst.add(min_node)

            for i in range(n):
                d = abs(points[min_node][0]-points[i][0]) \
                    + abs(points[min_node][1]-points[i][1])
                if i not in mst and d < nodes[i]:
                    nodes[i] = d
            
        return total_cost
