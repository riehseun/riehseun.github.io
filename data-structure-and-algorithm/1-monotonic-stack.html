Largest Rectangle in Histogram (150/250)

class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        
        # [1,2,3] ? 4

        # At each index, expand left and right till hitting lower height
        # Keep only increasing sequence to stack
        # When sequene decreases, current num is right boundry of top of stack
        # left boundry of top of stack is its previous element in stack
        # Time O(n) n = len(heights)
        # Space O(n)

        stack, max_area = [-1], 0
        for i, h in enumerate(heights):
            
            while stack[-1] != -1 and h < heights[stack[-1]]:
                height = heights[stack.pop()]
                right = i
                left = stack[-1]
                max_area = max(max_area, (right-left-1)*height)
            stack.append(i)

        while stack[-1] != -1:
            height = heights[stack.pop()]
            right = len(heights)
            left = stack[-1]
            max_area = max(max_area, (right-left-1)*height)

        return max_area

Min Stack (150/250)

class MinStack:

    # Push, pop, top is easy with list
    # Use another stack that is monotonically decreasing
    # Min is always at the top
    # Push 
    #   Push to stack
    #   Push to min_stack if val <= min_stack[-1]
    # Pop 
    #   Pop from stack (say "val")
    #   Pop from min_stack if "val" == min_stack[-1] 
    # Time O(1) 
    # Space O(n) n = number of items in stack

    def __init__(self):
        self.stack, self.min_stack = [], []

    def push(self, val: int) -> None:
        self.stack.append(val)
        if not self.min_stack or val <= self.min_stack[-1]:
            self.min_stack.append(val)

    def pop(self) -> None:
        val = self.stack.pop()
        if self.min_stack and val == self.min_stack[-1]:
            self.min_stack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.min_stack[-1]
