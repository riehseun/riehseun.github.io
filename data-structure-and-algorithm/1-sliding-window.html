Longest Substring Without Repeating Characters (75/150/250)

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:

        # Ex. "abbcftbk" ? 5 (cftbk)

        # Sliding window
        # Use start and end pointer
        # Move end pointer while substring does not have duplicate char
        # If substring has duplicate chars, then move start pointer
        # Memorize max length of valid substring
        # Use a set to keep track of chars
        # Time O(n) n = len(s)
        # Space O(n)

        start, end = 0, 0
        n = len(s)
        chars = set()
        max_len = 0

        while start <= end and end < n:
            if s[end] not in chars:
                chars.add(s[end])
                end += 1
            else:
                chars.remove(s[start])
                start += 1
            max_len = max(max_len, end-start)
        return max_len

Minimum Window Substring (75/150/250)

class Solution:
    def minWindow(self, s: str, t: str) -> str:

        # "ABCDEFG", "CDF" ? "CDEF"

        # Sliding window
        # Substring is valid when occurance of each char in substring
        # > occurance of each char in t 
        # Time O(n) n = len(s)
        # Space O(m)

        i, j, min_len, min_str = 0, 0, math.inf, ""
        freq_t, freq_s = defaultdict(int), defaultdict(int)

        for c in t:
            freq_t[c] += 1
        required = len(freq_t)
        satisfied = 0

        while i <= j and j < len(s):
            
            freq_s[s[j]] += 1
            if s[j] in freq_t and freq_s[s[j]] == freq_t[s[j]]:
                satisfied += 1
            j += 1

            while satisfied == required:
                if j - i < min_len:
                    min_len = j - i
                    min_str = s[i:j]
                freq_s[s[i]] -= 1
                if s[i] in freq_t and freq_t[s[i]] > freq_s[s[i]]:
                    satisfied -= 1
                i += 1
                
        return min_str

Sliding Window Maximum (150/250)

class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        
        # Use a queue of size k
        # Enqueue incoming number in queue
        # Deque outgoing number in queue if size is exceeded
        #   Keep maximum number in front by
        #   removing all numbers smaller than incoming number
        #   from the end
        
        # Ex. [1,2,1,-1,4,2,6]
        # [1,2,1] => [2,1]
        # [2,1,-1]  
        # [2,1,-1,4] => [4]
        # [4,2] 
        # [4,2,6] => [6]

        queue, result = deque(), []

        for i in range(k):
            while queue and nums[queue[-1]] < nums[i]:
                queue.pop()
            queue.append(i)
        result.append(nums[queue[0]])
        
        # Time O(n)
        #   n = len(nums)
        # Space O(k)
        for i in range(k, len(nums)):
            while queue and nums[queue[-1]] < nums[i]:
                queue.pop()
            queue.append(i)
            
            while queue and queue[0] <= i - k:
                queue.popleft()
            result.append(nums[queue[0]])

        return result

Longest Repeating Character Replacement (75/150/250)

class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        
        # AABBAA, 2 ? 6
        # CCAAABB, 2 ? 5

        # Sliding window
        # Substring is valid when counts of total chars minus
        # counts of most frequent char don't exceed k
        # Move end pointer while substring is valid
        # When substring is invalid, move start pointer
        # Time O(n) n = len(s)
        # Space O(n) up to 26 chars 

        most_freq_count = 0
        i, j, max_len = 0, 0, 0
        freq = defaultdict(int)

        while i <= j and j < len(s):

            freq[s[j]] += 1
            most_freq_count = max(most_freq_count, freq[s[j]])
            j += 1

            while i <= j and j - i - most_freq_count > k:
                freq[s[i]] -= 1
                i += 1

            max_len = max(max_len, j-i)
            
        return max_len

Permutation in String (150/250)

class Solution:
    def checkInclusion(self, s1: str, s2: str) -> bool:

        # "abc", "kabmc" ? False
        # "abc", "kbacm" ? True

        # Create dictionary to frequency for s1
        # Move window of size s1 in s2
        # Check is char count match
        # Time O(nm) n = len(s2)
        # Space O(m) m = len(s1)

        cnts_1, cnts_2 = defaultdict(int), defaultdict(int)
        n, m = len(s2), len(s1)

        for c in s1:
            cnts_1[c] += 1

        for c in s2[:m]:
            if c in cnts_1:
                cnts_2[c] += 1
        
        if cnts_2 == cnts_1:
            return True

        for i in range(n-m):
            old_c = s2[i]
            new_c = s2[i+m]
            if old_c in cnts_1:
                cnts_2[old_c] -= 1
            if cnts_2[old_c] == 0:
                del cnts_2[old_c]
            if new_c in cnts_1:
                cnts_2[new_c] += 1

            if cnts_2 == cnts_1:
                return True

        return False

Max Consecutive Ones III (M)

class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:

        # Sliding window
        #   Move end pointer while num is 1 or
        #   num is 0 but num_zeros <= k
        #   If num_zeros > k, move start pointer until 
        #   subarray is valid

        start, end, num_zeros, max_dist = 0, 0, 0, 0  
        
        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while start <= end and end < len(nums):

            if nums[end] == 1:
                end += 1
            else:
                if num_zeros < k:
                    num_zeros += 1
                    end += 1
                else:
                    # If both start and end is at 0,
                    # and there is no zeros to filp,
                    # move until finding the first 1
                    if start == end:
                        end += 1
                        start += 1
                        continue

                    if nums[start] == 0:
                        num_zeros -= 1
                    start += 1
                    
            max_dist = max(max_dist, end-start)

        # max_dist = max(max_dist, end-start)
        return max_dist
