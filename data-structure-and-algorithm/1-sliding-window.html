Longest Substring Without Repeating Characters (75/150/250)

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        
        # "abccdef" ? cdef
        # "aa" ? a

        # Use a set to hold seen chars
        # Move end pointer
        # While new char is in seen, move start pointer
        # Time O(n) n = len(s)
        # Space O(1) fixed number of ASCII chars

        seen = set()
        start, end = 0, 0
        max_len = 0

        while end < len(s):

            while s[end] in seen:
                seen.remove(s[start])
                start += 1

            seen.add(s[end])
            end += 1
            max_len = max(max_len, end-start)

        return max_len

Minimum Window Substring (75/150/250)

class Solution:
    def minWindow(self, s: str, t: str) -> str:
        
        # "ABCDEEFG", "FEEC" ? "CDEEF"

        # Use two dictionary for counting chars in t and window
        # Substring is valid when each char in t is satisfied
        # Move end pointer
        # While substring is valid, move start pointer
        # Time O(n+m) n = len(s), m = len(t)
        # Space O(n+m)

        freq_s, freq_t = defaultdict(int), defaultdict(int)
        num_satisfied = 0
        start, end = 0, 0
        min_len, min_string = math.inf, ""

        for c in t:
            freq_t[c] += 1

        while end < len(s):
            freq_s[s[end]] += 1
            if s[end] in freq_t and freq_s[s[end]] == freq_t[s[end]]:
                num_satisfied += 1
            end += 1

            while len(freq_t) == num_satisfied:
                if end - start < min_len:
                    min_len = end - start
                    min_string = s[start:end]
                if s[start] in freq_t and freq_s[s[start]] == freq_t[s[start]]:
                    num_satisfied -= 1
                freq_s[s[start]] -= 1
                start += 1

        return min_string

Sliding Window Maximum (150/250) (DoorDash)

class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        
        # [1,2,1,0,4,2,6], k = 3 ? [2,2,4,4,6]
         
        # Keep index of max num in at the front of queue
        # While new number > last num in queue, pop last num
        # [1,2,1] -> [2,1,0] -> [4] -> [4,2] -> [6]
        # Time O(n) n = len(nums)
        # Space O(k)

        result = []
        queue = deque()
        for i in range(k):
            while queue and nums[queue[-1]] < nums[i]:
                queue.pop()
            queue.append(i)
        result.append(nums[queue[0]])

        for i in range(k, len(nums)):
            
            while queue and nums[queue[-1]] < nums[i]:
                queue.pop()
            queue.append(i)

            while queue and queue[0] <= i - k:
                queue.popleft()

            result.append(nums[queue[0]])

        return result

Longest Repeating Character Replacement (75/150/250)

class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        
        # "ABBA", k = 2 ? 4
        # "CCBBBAA", k = 2 ? 5

        # Consider count of most frequent char in a window
        # Substring is valid when end - start - most_freq_count <= k
        # Move end pointer
        # While string is not valid, move start pointer
        # Time O(n) n = len(s)
        # Space O(1) 26 chars in alphabet

        most_freq_count = 0
        freq = defaultdict(int)
        start, end = 0, 0 
        max_len = 0

        while end < len(s):

            freq[s[end]] += 1
            most_freq_count = max(most_freq_count, freq[s[end]])
            end += 1

            while end - start - most_freq_count > k:
                freq[s[start]] -= 1
                start += 1
            
            max_len = max(max_len, end-start)

        return max_len

Permutation in String (150/250)

class Solution:
    def checkInclusion(self, s1: str, s2: str) -> bool:
        
        # Create counter for s1
        # Create running counter while iterating s2
        # Check if frequency matches 
        # Time O(nm) n = len(s1), m = len(s2)
        # Space O(m) 

        n, m = len(s1), len(s2)
        freq_1 = Counter(s1)

        freq_2 = defaultdict(int)
        for c in s2[:n]:
            if c in freq_1:
                freq_2[c] += 1

        if freq_1 == freq_2:
            return True

        for i in range(n, m):
            new, old = s2[i], s2[i-n] 
            if new in freq_1:
                freq_2[new] += 1
            if old in freq_1:
                freq_2[old] -= 1
                if freq_2[old] == 0:
                    del freq_2[old]

            if freq_1 == freq_2:
                return True
            
        return False

Max Consecutive Ones III

class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:

        # Sliding window
        #   Move end pointer while num is 1 or
        #   num is 0 but num_zeros <= k
        #   If num_zeros > k, move start pointer until 
        #   subarray is valid

        start, end, num_zeros, max_dist = 0, 0, 0, 0  
        
        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while start <= end and end < len(nums):

            if nums[end] == 1:
                end += 1
            else:
                if num_zeros < k:
                    num_zeros += 1
                    end += 1
                else:
                    # If both start and end is at 0,
                    # and there is no zeros to filp,
                    # move until finding the first 1
                    if start == end:
                        end += 1
                        start += 1
                        continue

                    if nums[start] == 0:
                        num_zeros -= 1
                    start += 1
                    
            max_dist = max(max_dist, end-start)

        # max_dist = max(max_dist, end-start)
        return max_dist

Longest Substring with At Most Two Distinct Characters (Google)

class Solution:
    def lengthOfLongestSubstringTwoDistinct(self, s: str) -> int:

        # Sldiing window
        # Move end pointer while meeting the condition
        # If condition is not met, move start pointer until condition is met
        # Use dictionary where key is char and value is their occurances
        # If length of dictionary > 2, condition is not valid
        # Time O(n) n = len(s)
        # Space O(1) cchars dictioanry cannot contain more than 2 keys

        max_dist, start, end = 0, 0, 0
        chars = defaultdict(int)

        while start <= end and end < len(s):
            chars[s[end]] += 1
            end += 1
            while len(chars) > 2:
                chars[s[start]] -= 1
                if chars[s[start]] == 0:
                    del chars[s[start]]
                start += 1
                
            max_dist = max(max_dist, end-start)

        return max_dist
