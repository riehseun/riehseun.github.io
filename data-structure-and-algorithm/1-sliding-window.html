Longest Substring Without Repeating Characters (75/150)

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:

        # "abccdef" ? 4 (cdef)

        # Interate i and j from the beginning
        # j increments when substring has no repeating char
        # i increments when substring has repeating char
        # Use set to remember chars that have been seen
        # Time O(n) n = len(s)
        # Space O(n)

        i, j, n = 0, 0, len(s)
        seen, max_len = set(), 0

        while i <= j and j < n:
            if s[j] not in seen:
                seen.add(s[j])
                j += 1
            else:
                seen.remove(s[i])
                i += 1
            max_len = max(max_len, j-i)

        return max_len

Minimum Window Substring (75/150)

class Solution:
    def minWindow(self, s: str, t: str) -> str:

        # s = "ABCDE" t = "BCE" ? "BCDE"

        # Iterate i and j from beginning
        # Increment j when not all chars in t are in substring
        # Increment i when all chars in t are in substring
        # For each char in t, compare its count in t vs count in substring
        # Time O(n+m) n = len(s), m = len(t)
        # Space O(n+m) = O(52) only upper and lower english chars

        i, j, n, m = 0, 0, len(s), len(t)
        freq_s, freq_t = defaultdict(int), defaultdict(int)
        num_chars_satisfied = 0
        min_len, min_substring = math.inf, ""
        
        for c in t:
            freq_t[c] += 1

        while i <= j and j < n:

            if num_chars_satisfied < len(freq_t):
                freq_s[s[j]] += 1
                if s[j] in freq_t and freq_s[s[j]] == freq_t[s[j]]:
                    num_chars_satisfied += 1
                j += 1
            else:
                freq_s[s[i]] -= 1
                if s[i] in freq_t and freq_s[s[i]] < freq_t[s[i]]:
                    num_chars_satisfied -= 1
                i += 1
            if j - i < min_len \
                and num_chars_satisfied == len(freq_t):
                min_len = min(min_len, j-i)
                min_substring = s[i:j]
                
        # Move i pointer forwards after j reached n
        while num_chars_satisfied == len(freq_t):
            freq_s[s[i]] -= 1
            if s[i] in freq_t and freq_s[s[i]] < freq_t[s[i]]:
                num_chars_satisfied -= 1
            i += 1
            if j - i < min_len \
                and num_chars_satisfied == len(freq_t):
                min_len = min(min_len, j-i)
                min_substring = s[i:j]

        return min_substring

Longest Repeating Character Replacement (75/150)

class Solution:
    def characterReplacement(self, s: str, k: int) -> int:

        # "ABCC", k = 1 ? 3 (ACCC)
        
        # Iterate i and j from start
        # Increment j when difference <= k
        # Increment i when difference > k
        # Use dictionary to store char count
        #   Substring is valid if total_count - max_count <= k
        # Time O(n) n = len(s)
        # Space O(n) = O(1) only 26 upper case chars 

        i, j, n = 0, 0, len(s)
        total_count, max_count, freq = 0, 0, defaultdict(int)
        max_len = 0

        while i <= j and j < n:

            if total_count - max_count <= k:
                total_count += 1
                freq[s[j]] += 1
                max_count = max(max_count, freq[s[j]])
                j += 1
            else:
                total_count -= 1
                freq[s[i]] -= 1
                i += 1
            if total_count - max_count <= k:
                max_len = max(max_len, j-i+1)
            
        return max_len - 1

Permutation in String (150)

class Solution:
    def checkInclusion(self, s1: str, s2: str) -> bool:

        # Create frequency map for s1
        # Slide s2 where window size is len(s1)
        #   If new char exists in s1 decrement count
        #   If old char exists in s1 increment count
        # Keep track of number of mismatches

        n, m = len(s1), len(s2)
        counts, num_mismatch = defaultdict(int), 0

        # Time O(n)
        #   n = len(s1)
        # Space O(1)
        #   There are only 26 chars in alphabet
        for c in s1:
            counts[c] += 1
            
        for c in s2[:n]:
            if c in counts:
                counts[c] -= 1
        
        for k,v in counts.items():
            num_mismatch += abs(v)

        if num_mismatch == 0:
            return True

        # Time O(m)
        #   m = len(s2)
        # Space O(1)
        #   There are only 26 chars in alphabet
        for i in range(n, m):
            new_c = s2[i]
            old_c = s2[i-n]

            if new_c in counts:
                if counts[new_c] > 0:
                    num_mismatch -= 1
                else:
                    num_mismatch += 1
                counts[new_c] -= 1

            if old_c in counts:
                if counts[old_c] < 0:
                    num_mismatch -= 1
                else:
                    num_mismatch += 1
                counts[old_c] += 1

            if num_mismatch == 0:
                return True

        return False

Max Consecutive Ones III (M)

class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:

        # Sliding window
        #   Move end pointer while num is 1 or
        #   num is 0 but num_zeros <= k
        #   If num_zeros > k, move start pointer until 
        #   subarray is valid

        start, end, num_zeros, max_dist = 0, 0, 0, 0  
        
        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while start <= end and end < len(nums):

            if nums[end] == 1:
                end += 1
            else:
                if num_zeros < k:
                    num_zeros += 1
                    end += 1
                else:
                    # If both start and end is at 0,
                    # and there is no zeros to filp,
                    # move until finding the first 1
                    if start == end:
                        end += 1
                        start += 1
                        continue

                    if nums[start] == 0:
                        num_zeros -= 1
                    start += 1
                    
            max_dist = max(max_dist, end-start)

        # max_dist = max(max_dist, end-start)
        return max_dist
