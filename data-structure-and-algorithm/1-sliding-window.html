Longest Substring Without Repeating Characters (75/150)

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:

        # "abccdef" ? 4 (cdef)

        # Interate i and j from the beginning
        # j increments when substring has no repeating char
        # i increments when substring has repeating char
        # Use set to remember chars that have been seen
        # Time O(n) n = len(s)
        # Space O(n)

        i, j, n = 0, 0, len(s)
        seen, max_len = set(), 0

        while i <= j and j < n:
            if s[j] not in seen:
                seen.add(s[j])
                j += 1
            else:
                seen.remove(s[i])
                i += 1
            max_len = max(max_len, j-i)

        return max_len

Minimum Window Substring (75/150)

class Solution:
    def minWindow(self, s: str, t: str) -> str:

        # Sliding window
        # Increse end pointer until substring is valid
        # Then, increase start pointer while substring is valid
        # How to check validity?
        #   Keep frequency map of chars for t
        #   Increment count of new char in window
        #   Decrement count of old char in window
        #   Remember number of mismatches

        start, end, n = 0, 0, len(s) 
        min_length, min_string = n, ""
        counts, num_mismatch = defaultdict(int), 0
        
        for c in t:
            counts[c] += 1
        num_mismatch = len(counts)

        # Time O(n)
        #   n = len(s)
        # Space O(m)
        #   m = len(t)
        while start <= end and end < n:
    
            if num_mismatch != 0:
                if s[end] in counts:
                    if counts[s[end]] == 1:
                        num_mismatch -= 1 
                    counts[s[end]] -= 1
                end += 1

            while num_mismatch == 0:
                substr = s[start:end]
                if len(substr) <= min_length:
                    min_length = len(substr)
                    min_string = substr
                if s[start] in counts:
                    if counts[s[start]] == 0:
                        num_mismatch += 1
                    counts[s[start]] += 1
                start += 1
            
        return min_string

Longest Repeating Character Replacement (75/150)

class Solution:
    def characterReplacement(self, s: str, k: int) -> int:

        # Find substring that is only differ by k
        # Use hash table to count frequency of elements
        # Substring is valid if
        #   total_count - max_count <= k 

        start, end, max_length = 0, 0, 1
        counts, max_cnt, total_cnt = defaultdict(int), 0, 0

        # Time O(n)
        #   n = len(s)
        # Space O(n)
        while start <= end and end < len(s):
            
            if total_cnt - max_cnt <= k:
                counts[s[end]] += 1
                total_cnt += 1
                if counts[s[end]] > max_cnt:
                    max_cnt = counts[s[end]]
                if total_cnt - max_cnt <= k:
                    max_length = max(max_length, end-start+1)
                end += 1
            else:
                counts[s[start]] -= 1
                total_cnt -= 1
                start += 1

        return max_length

Permutation in String (150)

class Solution:
    def checkInclusion(self, s1: str, s2: str) -> bool:

        # Create frequency map for s1
        # Slide s2 where window size is len(s1)
        #   If new char exists in s1 decrement count
        #   If old char exists in s1 increment count
        # Keep track of number of mismatches

        n, m = len(s1), len(s2)
        counts, num_mismatch = defaultdict(int), 0

        # Time O(n)
        #   n = len(s1)
        # Space O(1)
        #   There are only 26 chars in alphabet
        for c in s1:
            counts[c] += 1
            
        for c in s2[:n]:
            if c in counts:
                counts[c] -= 1
        
        for k,v in counts.items():
            num_mismatch += abs(v)

        if num_mismatch == 0:
            return True

        # Time O(m)
        #   m = len(s2)
        # Space O(1)
        #   There are only 26 chars in alphabet
        for i in range(n, m):
            new_c = s2[i]
            old_c = s2[i-n]

            if new_c in counts:
                if counts[new_c] > 0:
                    num_mismatch -= 1
                else:
                    num_mismatch += 1
                counts[new_c] -= 1

            if old_c in counts:
                if counts[old_c] < 0:
                    num_mismatch -= 1
                else:
                    num_mismatch += 1
                counts[old_c] += 1

            if num_mismatch == 0:
                return True

        return False

Max Consecutive Ones III (M)

class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:

        # Sliding window
        #   Move end pointer while num is 1 or
        #   num is 0 but num_zeros <= k
        #   If num_zeros > k, move start pointer until 
        #   subarray is valid

        start, end, num_zeros, max_dist = 0, 0, 0, 0  
        
        # Time O(n)
        #   n = len(nums)
        # Space O(1)
        while start <= end and end < len(nums):

            if nums[end] == 1:
                end += 1
            else:
                if num_zeros < k:
                    num_zeros += 1
                    end += 1
                else:
                    # If both start and end is at 0,
                    # and there is no zeros to filp,
                    # move until finding the first 1
                    if start == end:
                        end += 1
                        start += 1
                        continue

                    if nums[start] == 0:
                        num_zeros -= 1
                    start += 1
                    
            max_dist = max(max_dist, end-start)

        # max_dist = max(max_dist, end-start)
        return max_dist
