Kth Largest Element in an Array (150/250)

class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:

        # Use min_heap of size k
        # If new num > root of min_heap, replace
        # Time O(nlogk) n = len(nums)
        # Space O(k) 

        # Quickselect
        # Store mapping between frequency and number
        # Retrieve top k frequency by 
        #   Parition freq array by random pivot
        #   Nums greater than pivot comes left
        #   Nums less than pivot go to right
        # If k < p, search on right
        # If k > p, search on left
        # If k == p, return
        # Time O(n) 
        # Space O(n) for calculating frequency, O(1) for quickselect

        min_heap = []

        for num in nums:
            if len(min_heap) < k:
                heappush(min_heap, num)
            else:
                if num > min_heap[0]:
                    heappop(min_heap)
                    heappush(min_heap, num)

        return min_heap[0]

Find Median from Data Stream (75/150/250)

class MedianFinder:

    # [1,2,3] ? 2
    # [1,2] ? 1.5

    # Use two heaps
    # Max heap to store lower half (root can be median)
    # Min heap to store upper half (root can be median)
    # Balance heaps so that len(min_heap) >= len(max_heap) by 1
    # Time O(logn) n = number of elements in stream
    # Space O(n) 

    def __init__(self):
        self.min_heap = []
        self.max_heap = []

    def addNum(self, num: int) -> None:
        if len(self.min_heap) == len(self.max_heap):
            heappush(self.min_heap, num)
        else:
            heappush(self.max_heap, -num)
        while self.min_heap \
            and self.max_heap \
            and self.min_heap[0] < -self.max_heap[0]: 
            a = heappop(self.min_heap)
            b = heappop(self.max_heap)
            heappush(self.min_heap, -b)
            heappush(self.max_heap, -a)

    def findMedian(self) -> float:
        if len(self.min_heap) == len(self.max_heap):
            return (self.min_heap[0]-self.max_heap[0]) / 2
        else:
            return self.min_heap[0]

Top K Frequent Elements (75/150/250)

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:

        # Count occurance of each num
        # Quickselect based on the occurance
        # Partition array such that 
        #   Pick pivot index randomly
        #   Numbers higher than pivot value comes left
        #   Numbers lower than pivot value comes right
        # If pivot index < k, kth should be on right
        # If pivot index > k, kth should be on left 

        # Time O(n)
        #   n = len(nums)
        # Space O(n)
        #   Storing frequencies and keys
        def partition(left, right, array):
            p = random.randint(left, right)
            
            for i in range(left, right+1):
                if array[i][0] < array[p][0] and i < p:
                    array[i], array[p] = array[p], array[i] 
                if array[i][0] > array[p][0] and i > p:
                    array[i], array[p] = array[p], array[i] 
                p = i

            return p

        def quickselect(left, right, array, k):
            
            p = partition(left, right, array)

            if p < k - 1:
                return quickselect(p+1, right, array, k)
            elif p > k - 1:
                return quickselect(left, p-1, array, k)
            else:
                return array[:p+1]
        
        count = defaultdict(int)

        for num in nums:
            count[num] += 1

        array = []
        for key, val in count.items():
            array.append([val, key])

        ret_val = quickselect(0, len(array)-1, array, k)

        result = []
        for count, num in ret_val:
            result.append(num)

        return result

Design Twitter (150/250)

class Twitter:

    # postTweet
    #   Dictionary tweetId:userId
    #   Time O(1)
    #   Space O(n) n = number of tweets
    # getNewsFeed
    #   Use count for tracking time
    #   Insert [count, tweetId] into max_heap 
    #   Retrive onces that matches userId or dic[followerId]
    #   Time O(nlogn)
    #   Space O(n)
    # follow/unfollow
    #   Dictionary followerId:followeeId  
    #   Time O(1)
    #   Space O(n)

    def __init__(self):
        self.tweets = defaultdict(int)
        self.follows = defaultdict(set)
        self.max_heap = []
        self.time = 0
        
    def postTweet(self, userId: int, tweetId: int) -> None:
        self.tweets[tweetId] = userId
        heappush(self.max_heap, [-self.time, tweetId])
        self.time += 1

    def getNewsFeed(self, userId: int) -> List[int]:
        result, count, temp = [], 0, []
        while self.max_heap:
            t, tweet_id = heappop(self.max_heap)
            temp.append([t, tweet_id])
            if self.tweets[tweet_id] == userId \
                or self.tweets[tweet_id] in self.follows[userId]:
                result.append(tweet_id)
                count += 1
            if count == 10:
                break
        while temp:
            t, tweet_id = temp.pop()
            heappush(self.max_heap, [t, tweet_id])
        return result
        
    def follow(self, followerId: int, followeeId: int) -> None:
        self.follows[followerId].add(followeeId)

    def unfollow(self, followerId: int, followeeId: int) -> None:
        if followeeId in self.follows[followerId]:
            self.follows[followerId].remove(followeeId)

Task Scheduler (150/250)

class Solution:
    def leastInterval(self, tasks: List[str], n: int) -> int:
        
        # [A,A,B,B], n = 2 ? 5 (A -> B -> idle -> A -> B)

        # Need to handle most frequent task first
        # Use heap to keep track of most frequent task
        # Remember the last position of each task
        # Insert idle appropriately
        # Instead of reducing frequency by 1 and immediately putting it
        # back to heap, store it in temp array (this is to avoid using
        # same tasks in a cycle)
        # Time O(nlogn) n = len(tasks)
        # Space O(n)

        last, freq, max_heap = defaultdict(int), defaultdict(int), [] 
        for t in tasks:
            freq[t] += 1
        
        for key, val in freq.items():
            heappush(max_heap, [-val, key])
        
        i = 0
        while max_heap:
            cycle, temp = n + 1, []
            while cycle > 0 and max_heap:
                val, key = heappop(max_heap)
                if key in last:
                    for _ in range(n-i+last[key]+1):
                        i += 1
                last[key] = i
                i += 1
                cycle -= 1
                if val != -1:
                    temp.append([val+1, key])
            while temp:
                val, key = temp.pop()
                heappush(max_heap, [val, key])

        return i

class KthLargest:

    # Create a min heap
    # When the size of heap > k, pop root element
    # Time O(logk)
    # Space O(k)

    def __init__(self, k: int, nums: List[int]):
        self.min_heap = []
        self.k = k
        for num in nums:
            heappush(self.min_heap, num)
        
    def add(self, val: int) -> int:
        heappush(self.min_heap, val)
        while len(self.min_heap) > self.k:
            heappop(self.min_heap)
        return self.min_heap[0]

Max Stack (H)

class MaxStack:

    # Push, pop, top is handled by python list in O(1)
    # Use heap to store max value and its index in the list
    # Handle pop
    #   Remember popped elements from stack or heap
    #   Remove popped elements when top/pop/peekMax/popMax  

    # Time O(logn)
    # Space O(n)
    def __init__(self):
        self.count = 0
        self.stack = []
        self.max_heap = []
        self.deleted = set() 
        
    def push(self, x: int) -> None:
        self.stack.append((x, self.count))
        heappush(self.max_heap, (-x, -self.count))
        self.count += 1

    def pop(self) -> int:
        while self.stack and self.stack[-1][1] in self.deleted:
            self.stack.pop()
        num, index = self.stack.pop()
        self.deleted.add(index)
        return num

    def top(self) -> int:
        while self.stack and self.stack[-1][1] in self.deleted:
            self.stack.pop()
        return self.stack[-1][0]

    def peekMax(self) -> int:
        while self.max_heap and -self.max_heap[0][1] in self.deleted:
            heappop(self.max_heap)
        return -self.max_heap[0][0]

    def popMax(self) -> int:
        while self.max_heap and -self.max_heap[0][1] in self.deleted:
            heappop(self.max_heap)
        num, index = heappop(self.max_heap)
        self.deleted.add(-index)
        return -num

Last Stone Weight (150/250)

cclass Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        
        # Use max heap
        # Time O(n) n = len(stones)
        # Space O(n)

        max_heap = []

        for stone in stones:
            heappush(max_heap, -stone)

        while len(max_heap) > 1:
            x = -heappop(max_heap)
            y = -heappop(max_heap)
            if x < y:
                heappush(max_heap, -y+x)
            elif x > y:
                heappush(max_heap, -x+y)

        if max_heap:
            return -max_heap[0]
        else:
            return 0

Stock Price Fluctuation (M)

class StockPrice:

    # We only need latest (or greatest) timestamp
    # Use max heap to keep tuple of timestamp
    # Use dictionary to keep latest price per timestamp
    # How to find min and max?
    #   Use min heap and max heap for prices
    #   Insert incoming (timestamp, price) to both heaps
    #   When getting min/max, if (price, timestamp) matches
    #   the dictionary, return it. Otherwise, pop ot
      
    def __init__(self):
        self.max_heap_timestamp = []
        self.prices = defaultdict(int)
        self.max_heap_price = []
        self.min_heap_price = []
        
    # Time O(logn)
    #   n = number of update operation
    # Space O(n)
    def update(self, timestamp: int, price: int) -> None:
        heappush(self.max_heap_timestamp, -timestamp)
        self.prices[timestamp] = price
        heappush(self.max_heap_price, (-price, timestamp))
        heappush(self.min_heap_price, (price, timestamp))
        
    # Time O(1)
    def current(self) -> int:
        timestamp = self.max_heap_timestamp[0]
        return self.prices[-timestamp]

    def maximum(self) -> int:
        price, timestamp = self.max_heap_price[0]
        # If price is not outdated
        if -price == self.prices[timestamp]:
            return -price
        else:
            heappop(self.max_heap_price)
            return self.maximum() 

    def minimum(self) -> int:
        price, timestamp = self.min_heap_price[0]
        # If price is not outdated
        if price == self.prices[timestamp]:
            return price
        else:
            heappop(self.min_heap_price)
            return self.minimum()
