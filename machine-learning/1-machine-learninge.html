Gradient Descent

class Solution:
    def get_minimizer(self, iterations: int, learning_rate: float, init: int) -> float:
        # Objective function: f(x) = x^2
        # Derivative:         f'(x) = 2x
        # Update rule:        x = x - learning_rate * f'(x)
        # Round final answer to 5 decimal places

        x = init
        for e in range(iterations):
            grad = 2 * x
            x = x - learning_rate * grad 

        return round(x, 5)

Sigmoid & ReLU

import numpy as np
from numpy.typing import NDArray


class Solution:
    
    def sigmoid(self, z: NDArray[np.float64]) -> NDArray[np.float64]:
        # z is a 1D NumPy array
        # Formula: 1 / (1 + e^(-z))
        # return np.round(your_answer, 5)
        return np.round(1/(1+np.exp(-z)), 5)

    def relu(self, z: NDArray[np.float64]) -> NDArray[np.float64]:
        # z is a 1D NumPy array
        # Formula: max(0, z) element-wise
        return np.maximum(0, z)

Softmax

import numpy as np
from numpy.typing import NDArray


class Solution:

    def softmax(self, z: NDArray[np.float64]) -> NDArray[np.float64]:
        # z is a 1D NumPy array of logits
        # Hint: subtract max(z) for numerical stability before computing exp
        # return np.round(your_answer, 4)
        val = np.exp(z-np.max(z)) / np.sum(np.exp(z-np.max(z)))
        return np.round(val, 4)

Cross-Entropy Loss

import numpy as np
from numpy.typing import NDArray


class Solution:

    def binary_cross_entropy(self, y_true: NDArray[np.float64], y_pred: NDArray[np.float64]) -> float:
        # y_true: true labels (0 or 1)
        # y_pred: predicted probabilities
        # Hint: add a small epsilon (1e-7) to y_pred to avoid log(0)
        # return round(your_answer, 4)
        n = y_true.shape[0]
        y_pred = y_pred + 1e-7
        bce_loss = -np.sum(y_true*np.log(y_pred)+(1-y_true)*np.log(1-y_pred)) / n
        return np.round(bce_loss, 4)

    def categorical_cross_entropy(self, y_true: NDArray[np.float64], y_pred: NDArray[np.float64]) -> float:
        # y_true: one-hot encoded true labels (shape: n_samples x n_classes)
        # y_pred: predicted probabilities (shape: n_samples x n_classes)
        # Hint: add a small epsilon (1e-7) to y_pred to avoid log(0)
        # return round(your_answer, 4)
        n = y_true.shape[0]
        y_pred = y_pred + 1e-7
        cce_loss = -np.sum(y_true*np.log(y_pred)) / n
        return np.round(cce_loss, 4)

Linear Regression (Forward)

import numpy as np
from numpy.typing import NDArray


class Solution:

    def get_model_prediction(self, X: NDArray[np.float64], weights: NDArray[np.float64]) -> NDArray[np.float64]:
        # X is (n, m), weights is (m,) -> return (n,) predictions
        # Round to 5 decimal places
        pred = np.dot(X, weights)
        return np.round(pred, 5)
        
    def get_error(self, model_prediction: NDArray[np.float64], ground_truth: NDArray[np.float64]) -> float:
        # Compute mean squared error between predictions and ground truth
        # Round to 5 decimal places
        n = model_prediction.shape[0]
        mse = np.sum((model_prediction-ground_truth)**2) / n 
        return np.round(mse, 5)

Linear Regression (Training)

import numpy as np
from numpy.typing import NDArray


class Solution:
    def get_derivative(self, model_prediction: NDArray[np.float64], ground_truth: NDArray[np.float64], N: int, X: NDArray[np.float64], desired_weight: int) -> float:
        # note that N is just len(X)
        return -2 * np.dot(ground_truth - model_prediction, X[:, desired_weight]) / N

    def get_model_prediction(self, X: NDArray[np.float64], weights: NDArray[np.float64]) -> NDArray[np.float64]:
        return np.squeeze(np.matmul(X, weights))

    learning_rate = 0.01

    def train_model(
        self,
        X: NDArray[np.float64],
        Y: NDArray[np.float64],
        num_iterations: int,
        initial_weights: NDArray[np.float64]
    ) -> NDArray[np.float64]:
        # For each iteration:
        #   1. Compute predictions with get_model_prediction(X, weights)
        #   2. For each weight index j, compute gradient with get_derivative()
        #   3. Update: weights[j] -= learning_rate * gradient
        # Return np.round(final_weights, 5)
        weights = initial_weights.copy()
        n = X.shape[0]
        m = X.shape[1]

        for e in range(num_iterations):
            pred = self.get_model_prediction(X, weights)
            grad = np.zeros(m)
            for j in range(m):
                grad[j] = self.get_derivative(pred, Y, n, X, j)
            weights -= self.learning_rate * grad 
        
        return np.round(weights, 5)

Single Neuron

import numpy as np
from numpy.typing import NDArray


class Solution:
    def forward(self, x: NDArray[np.float64], w: NDArray[np.float64], b: float, activation: str) -> float:
        # x: 1D input array
        # w: 1D weight array (same length as x)
        # b: scalar bias
        # activation: "sigmoid" or "relu"
        #
        # Pre-activation: z = dot(x, w) + b
        # Sigmoid: σ(z) = 1 / (1 + exp(-z))
        # ReLU: max(0, z)
        # return round(your_answer, 5)
        z = np.dot(x, w) + b
        if activation == "relu":
            z = max(0.0, z)
        if activation == "sigmoid":
            z = 1 / (1 + np.exp(-z))
        return round(z, 5)

Multi-class Classification (Softmax Regression)

  • Generalization of logistic regression
  • Number of units in the output layer is \( n^{L} = C \) the number of classes
  • If \( C = 2 \), applying softmax reduces to logistic regression

Activation in the output layer

  • In the output layer \( z^{[L]} = w^{[L]}a^{[L-1]} + b^{[l]} \)
  • The activation in the last layer for softmax is \( a^{[L]} = \dfrac{e^{z^{[L]}}}{\sum_{i=1}^{C}t_{i}} \) where \( t = e^{z^{[L]}} \)

Loss function

  • \( L(\hat{y},y) = -\displaystyle\sum_{j=1}^{C} y_{j}log\hat{y}_{j} \)

Gradient descent

  • \( \textbf{dZ} = \textbf{A} - \textbf{y} \)

K-nearest Neighbor

# Define k - pick odd number to avoid ties
# Compute distance between a new point to all other points
# Pick top k points that are closest
# Amongst top k points, count number of points in each category
# Assign new points to top category

import torch

def find_knn_indices(X_train, X_test_point, k):
    # Calculate Euclidean distance: sqrt(sum((train - test_point)^2))
    distances = torch.sum((X_train - X_test_point) ** 2, axis=1).sqrt()
    
    # Sort distances to find the k smallest
    _, indices = torch.sort(distances)
    
    # Return the indices of the k nearest neighbors
    return indices[:k]

# Example usage:
train_data = torch.randn(20, 3) # 20 points, 3 dimensions
query_point = torch.randn(1, 3) # 1 query point
k_neighbors = 3

nearest_indices = find_knn_indices(train_data, query_point, k_neighbors)

print(f"Query point: {query_point}")
print(f"Indices of {k_neighbors} nearest neighbors in train_data: {nearest_indices}")
print(f"The actual neighbor points are:\n{train_data[nearest_indices]}")

K-means

# Pick k
# Randomly pick k points and make them center of k clusters
# Assign each point to one of k clusters based on distance to centroid
# Adjust centroid of each cluster by taking mean of all points within the cluster
# Repeat until cluster assignment no longer changes

import torch
import numpy as np
import matplotlib.pyplot as plt
from kmeans_pytorch import kmeans, kmeans_predict

# --- 1. Setup and Data Generation ---
# Set random seed for reproducibility
np.random.seed(123)
torch.manual_seed(123)

# Set device to GPU if available, otherwise CPU
if torch.cuda.is_available():
    device = torch.device('cuda:0')
else:
    device = torch.device('cpu')
print(f"Running k-means on: {device}")

# Generate some random data points
data_size, dims, num_clusters = 1000, 2, 3
x_numpy = np.random.randn(data_size, dims) / 6
# Add offsets to create distinct clusters
x_numpy[:300] += np.array([-0.5, -0.5])
x_numpy[300:600] += np.array([0.5, 0.5])
x_numpy[600:] += np.array([-0.5, 0.5])

# Convert numpy array to PyTorch tensor and move to the correct device
x = torch.from_numpy(x_numpy).float().to(device)

# --- 2. Perform K-Means Clustering ---
# The kmeans function returns the cluster IDs for each data point and the final cluster centers
cluster_ids_x, cluster_centers = kmeans(
    X=x,
    num_clusters=num_clusters,
    distance='euclidean',
    device=device
)

# --- 3. (Optional) Predict new data points ---
# Generate some new data for prediction
y_numpy = np.random.randn(100, dims) / 6
y = torch.from_numpy(y_numpy).float().to(device)

# Predict cluster IDs for the new data
cluster_ids_y = kmeans_predict(
    y,
    cluster_centers,
    'euclidean',
    device=device
)

print(cluster_ids_y)
