Jupyter in docker

# Login to dockerhub.
docker login

# Pull jupyter notebook.
docker pull jupyter/tensorflow-notebook

# Run notebook.
docker run -it --rm -p 8888:8888 jupyter/tensorflow-notebook

# Run notebook by mapping local dir to container dir.
docker run -it --rm -p 8888:8888 -v /home/user/work:/home/jovyan/work jupyter/tensorflow-notebook
docker run -it --rm -p 8888:8888 -v /home/osboxes/machine-learning:/home/jovyan/work jupyter/tensorflow-notebook
docker run -it --rm -p 8888:8888 -v /home/osboxes/algorithm:/home/jovyan/work jupyter/tensorflow-notebook
docker run -it --rm -p 8888:8888 -v /home/osboxes:/home/jovyan/work jupyter/tensorflow-notebook

# Pull jupyter notebook. (GPU environment)
docker pull cschranz/gpu-jupyter

# Run notebook by mapping local dir to container dir. (GPU environment)
docker run --gpus all -d -it -p 8888:8888 -v /home/user/work:/home/jovyan/work -e GRANT_SUDO=yes -e JUPYTER_ENABLE_LAB=yes --user root docker pull cschranz/gpu-jupyter

# Access the notebook from the browser at localhost:8888

Pip useful commands

# Install specific version.
pip install <package_nam>==<package_version>

# Install packages in requirements.txt.
pip install --requirement requirements.txt

# Options
# --no-deps: Don't install package dependencies.
# -U / --upgrade: Upgrade all packages to the newest version.
# -I / --ignore-installed: Don't re-install existing packages.
# --compile: Compile python source files into bytecode.
Reference: pip-install - Man Page
