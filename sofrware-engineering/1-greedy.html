Jump Game II (M) (Liked)

class Solution:
    def jump(self, nums: List[int]) -> int:

        # Time O(n)
        # Space O(1)
        # Compute farthest jump in each interation and update current jump to farthest jump when needed.

        farthest_jump = 0  # Farthest index we can reach.
        current_jump = 0  # End index of current jump.
        num_jumps = 0

        for i in range(len(nums)-1):
            farthest_jump = max(farthest_jump, nums[i]+i)
            if i == current_jump:
                current_jump = farthest_jump
                num_jumps += 1

        return num_jumps

Jump game (M) (Interview/Liked)

class Solution:
    def canJump(self, nums: List[int]) -> bool:

        # Time O(n)
        # Space O(1)

        # Loop backwards and update target index.
        # Target number is initially at last index.

        target_index = len(nums)-1

        for i in range(len(nums)-2, -1, -1):
            if nums[i] + i >= target_index:
                target_index = i

        if target_index == 0:
            return True

        return False

Gas stations (M) (Interview)

class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:

        # Time: O(n)
        # Space: O(1)

        # If sum(gas) >= sum(cost), there is enough gas to make a trip from at least one index.
        # Solution is guaranteed to be unique from the problem.

        # If this condition is not met, there is at least one index
        # where the trip can be started with enough gas.

        if sum(gas) < sum(cost):
            return -1

        n = len(gas)
        accumulated_gas = 0
        starting_index = 0

        for i in range(n):
            accumulated_gas += (gas[i]-cost[i])
            if accumulated_gas < 0:
                accumulated_gas = 0
                starting_index = i + 1

        return starting_index

Boats to save people (M)

class Solution:
    def numRescueBoats(self, people: List[int], limit: int) -> int:

        # Time: O(nlogn)
        # Space: O(1)

        # Sort the array.
        # If the sum of lightest and heaviest is less than or equal to limit, increment lightest and decrement heaviest.
        # If not, only decrement heaviest.

        people.sort()
        lightest = 0
        heaviest = len(people) - 1
        num_boats = 0

        while lightest <= heaviest:
            if people[lightest] + people[heaviest] <= limit:
                lightest += 1
                heaviest -= 1
                num_boats += 1
            else:
                heaviest -= 1
                num_boats += 1

        return num_boats

Two city scheduling (M)

class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:

        # Time O(nlogn)
        # Space O(n)

        # First half of sorted array will be the ones that would save the most cost by travelling to A rather than B.

        array = []

        for cost in costs:
            array.append([cost[0]-cost[1], cost[0], cost[1]])

        array.sort(key = lambda x: x[0])

        n = int(len(costs)/2)

        total_cost = 0
        for i in range(n):
            total_cost += array[i][1]
        for i in range(n,2*n):
            total_cost += array[i][2]

        return total_cost
