First bad version

First implement binary search, then modify

def first_bad_version(n):

    return search(n, 1, n, 0)

# "is_bad_version" is given.
def search(n, low, high, count):

    # Time: O(logn)
    # Space: O(1)

    if low > high:
        return low, count

    mid = low + (high-low) / 2
    mid = int(mid)
    version = is_bad_version(mid)
    if version:
        count += 1
        return search(n, low, mid-1, count)
    else:
        return search(n, mid+1, high, count)

Find K closest elements

Use binary search to find the closest element

def find_closest_elements(nums, k, num):

    # Time: O(logn+k)
    # Space: O(1)

    index = binary_search(nums, 0, len(nums)-1, num)

    left = index - 1
    right = index

    while right - left - 1 < k:
        # check for out of bounds
        if left == -1:
            right += 1
            continue

        if right == len(nums) \
            or abs(nums[left] - num) <= abs(nums[right] - num):
            left -= 1
        else:
            right += 1

    return nums[left+1:right]

def binary_search(nums, low, high, target):

    if low > high:
        return low

    mid = low + (high - low) / 2
    mid = int(mid)

    if target > nums[mid]:
        return binary_search(nums, mid+1, high, target)
    elif target < nums[mid]:
        return binary_search(nums, low, mid-1, target)
    else:
        return mid

Single element in a sorted array

Use binary search and update left or right in each recursive call

def single_non_duplicate(nums):

    return search(nums, 0, len(nums)-1)

def search(nums, left, right):

    # Time: O(logn)
    # Space: O(1)

    if left >= right:
        return nums[left]

    mid = left + (right-left) // 2
    if mid % 2 != 0:
        mid -= 1

    if nums[mid] == nums[mid+1]:
        left = mid + 2
    else:
        right = mid

    return search(nums, left, right)

Search in Rotated Sorted Array II

First implement binary search, then modify

def search(arr, t):

    return search_helper(arr, 0, len(arr)-1, t)

def search_helper(arr, low, high, t):

    # Time: O(logn)
    # Space: O(1)

    if low > high:
        return False

    mid = low + (high-low) // 2

    if arr[mid] == t:
        return True

    if arr[low] <= arr[mid]:
        if arr[low] <= t and t < arr[mid]:
            return search_helper(arr, low, mid-1, t)
        else:
            return search_helper(arr, mid+1, high, t)
    else:
        if arr[mid] < t and t <= arr[high]:
            return search_helper(arr, mid+1, high, t)
        else:
            return search_helper(arr, low, mid-1, t)
