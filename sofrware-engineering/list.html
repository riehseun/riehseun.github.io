Two numbers adding up to k

  • Time: O(nlogn)
def find_sum(lst, k):
    lst.sort()

    i = 0
    j = len(lst) - 1

    while i < j:
        if lst[i] + lst[j] > k:
            j -= 1
        elif lst[i] + lst[j] < k:
            i += 1
        else:
            return lst[i], lst[j]
