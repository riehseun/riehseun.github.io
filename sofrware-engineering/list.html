Merge Two Sorted Lists

  • Time: O(n+m)
def merge_lists(lst1, lst2):
    merged = []
    s1 = 0
    s2 = 0

    while s1 < len(lst1) and s2 < len(lst2):
        if lst1[s1] < lst2[s2]:
            merged.append(lst1[s1])
            s1 += 1
        else:
            merged.append(lst2[s2])
            s2 += 1

    while s1 < len(lst1):
        merged.append(lst1[s1])
        s1 += 1

    while s2 < len(lst2):
        merged.append(lst2[s2])
        s2 += 1

    return merged

Two numbers adding up to k

  • Time: O(nlogn)
def find_sum(lst, k):
    lst.sort()

    i = 0
    j = len(lst) - 1

    while i < j:
        if lst[i] + lst[j] > k:
            j -= 1
        elif lst[i] + lst[j] < k:
            i += 1
        else:
            return lst[i], lst[j]

List of Products of all Elements

  • Time: O(n)
def find_product(lst):

    total_product = 1
    num_zeros = 0
    for v in lst:
        if v != 0:
            total_product *= v
        else:
            num_zeros += 1

    for i in range(len(lst)):
        if num_zeros == 0:
            lst[i] = int(total_product/lst[i])
        elif num_zeros == 1:
            if lst[i] == 0:
                lst[i] = total_product
            else:
                lst[i] = 0
        else:
            lst[i] = 0

    return lst

First Non-Repeating Integer in a list

  • Time: O(n)
  • Space: O(n)
from collections import defaultdict

def find_first_unique(lst):

    my_dict = defaultdict(set)
    for i, num in enumerate(lst):
        my_dict[num].add(i)

    for num in lst:
        if len(my_dict[num]) == 1:
            return num

    return None
