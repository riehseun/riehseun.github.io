Number of islands (M) (Interview/Liked)

from collections import defaultdict, deque

class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:

        # Construct a graph.
        # Count number of connnected component using BFS.

        # Connected when a node is 1 and adjacent is also 1.

        graph = defaultdict(list)
        nodes = set()

        n = len(grid)
        m = len(grid[0])
        for i in range(len(grid)):
            for j, num in enumerate(grid[i]):
                if num == "1":
                    if (i,j) not in nodes:
                        nodes.add((i,j))
                    # Check right
                    if j < m-1 and grid[i][j+1] == "1":
                        graph[(i,j)].append((i, j+1))
                        graph[(i,j+1)].append((i,j))
                    # Check left
                    if j > 0 and grid[i][j-1] == "1":
                        graph[(i,j)].append((i, j-1))
                        graph[(i,j-1)].append((i,j))
                    # Check up
                    if i > 0 and grid[i-1][j] == "1":
                        graph[(i,j)].append((i-1, j))
                        graph[(i-1,j)].append((i,j))
                    # Check down
                    if i < n-1 and grid[i+1][j] == "1":
                        graph[(i,j)].append((i+1, j))
                        graph[(i+1,j)].append((i,j))

        explored = set()
        count = 0
        for (i,j) in nodes:
            if (i,j) not in explored:
                self.bfs(graph, explored, (i,j))
                count += 1
        return count

    def bfs(self, graph, explored, start):
        queue = deque()
        queue.append(start)

        while queue:
            node = queue.popleft()
            print(graph[node])
            for child_node in graph[node]:
                if child_node not in explored:
                    explored.add(child_node)
                    queue.append(child_node)

Redundant connection (M)

class Solution:
    def findRedundantConnection(self, edges: List[List[int]]) -> List[int]:

        # Union-find.
        # Consider an edge. If two nodes of the edge are already in the same component,
        # that edge is redundant.

        # Key represents each node and value represents the leader of group
        # that the node belongs to.

        # Initially all node have its own value as its leader.
        group = {}
        for edge in edges:
            if edge[0] not in group:
                group[edge[0]] = edge[0]
            if edge[1] not in group:
                group[edge[1]] = edge[1]

        for edge in edges:

            leader_first = self.find(group, edge[0])
            leader_second = self.find(group, edge[1])

            # If both nodes in edge already belong to the same leader,
            # then the edge is redundant.
            if leader_first == leader_second:
                return edge

            # Let the smallest node be the leader in that group.
            if leader_first < leader_second:
                self.union(group, edge[1], leader_first)
            else:
                self.union(group, edge[0], leader_second)

        return []

    def find(self, group, node):
        return group[node]

    def union(self, group, node, new_leader):

        # Save the previous leader.
        previous_leader = group[group[node]]

        # The node's leader also gets the new leader.
        group[group[node]] = new_leader

        # All nodes that had the previous leader as leader must now have
        # the new leader as leader.
        for k,v in group.items():
            if v == previous_leader:
                group[k] = new_leader
