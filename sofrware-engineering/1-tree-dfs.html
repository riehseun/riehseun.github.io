Binary tree inorder traversal (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def inorderTraversal(self, root: Optional[TreeNode]) -> List[int]:

        # Traverse tree
        # Return the order as left, root, right

        inorder = []
        self.helper(root, inorder)
        return inorder

    # Time O(n) where n is number of nodes
    # Space O(h) where h = n if worst case, h = log(n) if best case
    def helper(self, root, inorder):

        if root:
            self.helper(root.left, inorder)
            inorder.append(root.val)
            self.helper(root.right, inorder)

        return []

Validate binary search tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:

        # Convert to inorder traversal
        # If find non-increasing sequence, return False

        inorder = []
        self.helper(root, inorder)

        for i in range(len(inorder)-1):
            if inorder[i] >= inorder[i+1]:
                return False

        return True

    # Time O(n) where n is number of nodes
    # Space O(h) where h = n if worst case, h = log(n) if best case
    def helper(self, root, inorder):

        if root:
            self.helper(root.left, inorder)
            inorder.append(root.val)
            self.helper(root.right, inorder)

        return []

Recover binary search tree (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """

        # Do inorder traversal
        # During inorder traversal, find elements that need to be swapped

        self.previous = TreeNode(-math.inf)
        self.first = None
        self.second = None

        # Time O(n) where n is the number of nodes in tress
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if root:
                helper(root.left)

                if self.previous.val >= root.val:
                    if not self.first:
                        self.first = self.previous
                    self.second = root
                self.previous = root

                helper(root.right)

        helper(root)
        self.first.val, self.second.val = self.second.val, self.first.val

Balanced binary tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isBalanced(self, root: Optional[TreeNode]) -> bool:

        # Return 0 at the leaf nodes
        # If height difference at any layer is greater than 1, return False
        # If height of left or right subtree return -1, return False
        # Return max height plus 1 in each layer

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if not root:
                return 0

            left = helper(root.left)
            right = helper(root.right)

            if abs(left-right) > 1:
                return -1

            if left == -1 or right == -1:
                return -1

            return max(left, right) + 1

        retval = helper(root)

        if retval == -1:
            return False
        return True

Path sum II (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:

        # Save the result when the running sum equals to target

        self.paths = []

        # Time O(n) where n is number of nodes
        # Space O(n) for each node, we are constructing its path
        def helper(root, path, running_sum, target_sum):

            if root:

                if running_sum + root.val == target_sum and not root.left and not root.right:
                    self.paths.append(path+[root.val])

                helper(root.left, path+[root.val], running_sum+root.val, target_sum)
                helper(root.right, path+[root.val], running_sum+root.val, target_sum)

            return self.paths

        return helper(root, [], 0, targetSum)

Flatten binary tree to linked list (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """

        # Do post-order traversing (right, root, left)
        # Set root.left to None
        # Set root.right to previous root

        self.prev = None

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if root:
                print(root.val)
                helper(root.right)
                helper(root.left)

                root.left = None
                root.right = self.prev
                self.prev = root

        helper(root)

Binary tree maximum path sum (H) (Interview/Liked)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:

        # If node is None, return 0
        # Consider each node. Max sum at each node is the sum of
        #   node value itself
        #   max sum of left subtree
        #   max sum of right subtree
        # Also, in each node, the path can only take either left or right

        self.max_sums = []

        # Time O(n) where n is number of nodes
        # Space O(n) for each node, we are constructing its max sum
        def helper(root):

            if not root:
                return 0

            max_sum_left = max(helper(root.left), 0)
            max_sum_right = max(helper(root.right), 0)

            max_sum = root.val + max_sum_left + max_sum_right
            self.max_sums.append(max_sum)

            max_path = root.val + max(max_sum_left, max_sum_right)
            return max_path

        helper(root)

        return max(self.max_sums)

Sum root to leaf numbers (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sumNumbers(self, root: Optional[TreeNode]) -> int:

        # Find all path from root to all leaf nodes

        self.paths = []

        # Time O(n) where n is number of nodes
        # Space O(n) for each node, we are constructing its path
        def helper(root, path):

            if root:

                # If leaf node
                if not root.left and not root.right:
                    self.paths.append(path+[root.val])

                print(root.val)
                helper(root.left, path+[root.val])
                helper(root.right, path+[root.val])

            return self.paths

        final_sum = 0
        result = helper(root, [])
        for path in result:
            string = ""
            for node in path:
                string += str(node)
            final_sum += int(string)

#     def __init__(self, val=0, left=None, right=None):

class Solution:
    def preorderTraversal(self, root: Optional[TreeNode]) -> List[int]:

        self.preorder = []

        # Time O(n) where n is number of nodes
        # Space O(h) where h = n if worst case, h = log(n) if best case
        def helper(root):

            if root:

                self.preorder.append(root.val)
                helper(root.left)
                helper(root.right)

        return self.preorder

        def helper(root):
