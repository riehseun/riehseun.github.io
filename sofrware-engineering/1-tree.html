Construct binary tree from preorder and inorder traversal (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreexNode]:

        # Preorder - root, left, right
        # Inorder - left, root, right
        # The very first number of preorder is always the root
        # Numbers left of this root from inorder forms left subtree
        # Numbers right of this root from inorder forms left subtree

        # Time O(n) computation takes place on each node of tree

        if inorder:

            root_node = preorder.pop(0)
            root_index = inorder.index(root_node)
            root = TreeNode(root_node)
            root.left = self.buildTree(preorder[:len(inorder[:root_index])], inorder[:root_index])
            root.right = self.buildTree(preorder[len(inorder[:root_index]):], inorder[root_index+1:])

            return root

Construct binary tree from inorder and postorder traversal (M)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:

        # Postorder - left, right, root
        # Inorder - left, root, right
        # The very last number of preorder is always the root
        # Numbers left of this root from inorder forms left subtree
        # Numbers right of this root from inorder forms left subtree

        # Time O(n) computation takes place on each node of tree

        if inorder and postorder:

            root_node = postorder.pop()
            root_index = inorder.index(root_node)
            root = TreeNode(root_node)
            root.left = self.buildTree(inorder[:root_index], postorder[:len(inorder[:root_index])])
            root.right = self.buildTree(inorder[root_index+1:], postorder[len(inorder[:root_index]):])

            return root

Convert Sorted Array to Binary Search Tree (E)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sortedArrayToBST(self, nums: List[int]) -> Optional[TreeNode]:

        # The middle element is always the root
        # Everything on the left side constitute left subtree
        # Everything on the right side constitute right subtree

        # Time O(n) where n is len(nums)
        if nums:
            low = 0
            high = len(nums)-1
            mid = (low+high) // 2

            root = TreeNode(nums[mid])
            root.left = self.sortedArrayToBST(nums[:mid])
            root.right = self.sortedArrayToBST(nums[mid+1:])

            return root

Convert sorted list to binary search tree (M)

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:

        # The middle element is always the root
        # Everything on the left side constitute left subtree
        # Everything on the right side constitute right subtree

        nums = []

        while head:
            nums.append(head.val)
            head = head.next

        return self.helper(nums)

    def helper(self, nums):

        # Time O(n) where n is len of the linked list
        if nums:
            low = 0
            high = len(nums)-1
            mid = (low+high) // 2

            root = TreeNode(nums[mid])
            root.left = self.helper(nums[:mid])
            root.right = self.helper(nums[mid+1:])

            return root
