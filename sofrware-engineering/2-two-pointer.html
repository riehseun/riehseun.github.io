Three sum (M) (Interview/Liked)

class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:

        # Time: O(n^2)
        # Space: O(1)
        # n = length of nums

        # Sort the array.
        # Iterate the array.
        # For each number, set low as numbers less than the number and set high as numbers greater than the number.
        # Check different values of low and high to see if target can be met.

        triplets = []

        if len(nums) < 3:
            return triplets

        nums.sort()  # O(nlogn)

        for i, num in enumerate(nums):
            # If three sum with the same previous number is checked, skip.
            if i > 0 and num == nums[i - 1]:
                continue

            start = i + 1
            end = len(nums) - 1

            while start < end:
                # If the sum is less than zero, need to increment the
                # start index to increase the sum.
                if nums[i] + nums[start] + nums[end] < 0:
                    start += 1
                # If the sum is greater than zero, need to decrement the
                # end index to decrease the sum.
                elif nums[i] + nums[start] + nums[end] > 0:
                    end -= 1
                else:
                    triplets.append([nums[i],nums[start],nums[end]])

                    # We can increment either start or end.
                    if nums[start] <= nums[end]:
                        start += 1
                    else:
                        end -= 1

        unique_triplets = []
        for triplet in triplets:
            if triplet not in unique_triplets:
                unique_triplets.append(triplet)

        return unique_triplets

class Solution:
    def isPalindrome(self, s: str) -> bool:

        # Time: O(n)
        # Space: O(1)
        # n = length of s

        s = re.sub(r'[^a-zA-Z0-9]', '', s)
        s = s.lower()

        start = 0
        end = len(s)-1

        while start <= end:
            if s[start] != s[end]:
                return False
            start += 1
            end -= 1

        return True

Reverse words

1. Swap each words.

  • Preprocess the sentence to remove extra whitespaces.
  • Convert the sentence into an array of words.
  • Swap words at each end while moving start and end pointer towards the middle.
def reverse_words(sentence):

    # Time: O(n) where n is length of sentence.
    # Space: O(1)

    # Convert multiple whitespaces to a single whitespace.
    # Split O(n), join O(n)
    sentence = " ".join(sentence.split())

    sentence_array = sentence.split(" ")

    start = 0
    end = len(sentence_array) - 1

    while start < end:
        temp = sentence_array[start]
        sentence_array[start] = sentence_array[end]
        sentence_array[end] = temp
        start += 1
        end -= 1

    return " ".join(sentence_array)

Valid palindrome II

1. Compare chars at each end and move pointers towards the middle.

  • Allow 1 mismatch to happen. If more than 1 mismatch, return False.
  • When mismatch happens, either increment start pointer or decrement end pointer such that two chars at the end matches. If cannot find match, return False.
def is_palindrome(s):

    # Time: O(n)
    # Space: O(1)

    start = 0
    end = len(s) - 1
    num_mismatch = 0

    while start < end:
        if s[start] == s[end]:
            start += 1
            end -= 1
        else:
            if num_mismatch == 0:
                num_mismatch += 1
                if s[start+1] == s[end]:
                    start += 1
                elif s[start] == s[end-1]:
                    end -= 1
                else:
                    return False
            else:
                return False

    return True

  • String with two chars ("ab")
