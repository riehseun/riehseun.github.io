Find first and last position of element in sorted array (M) (Interview/Liked)

class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        return self.binary_search(nums, 0, len(nums)-1, target)

    def binary_search(self, nums, low, high, target):

        if low > high:
            return [-1, -1]

        mid = low + (high - low) / 2
        mid = int(mid)

        if target > nums[mid]:
            return self.binary_search(nums, mid+1, high, target)
        elif target < nums[mid]:
            return self.binary_search(nums, low, mid-1, target)
        else:
            # return mid
            start = mid
            while nums[mid] == nums[start]:
                start -= 1
                if start < 0:
                    break
            start += 1

            end = mid
            while nums[mid] == nums[end]:
                end += 1
                if end >= len(nums):
                    break
            end -= 1

            return [start, end]

Search in rotated sorted array II (M)

class Solution:
    def search(self, nums: List[int], target: int) -> bool:
        return self.search_helper(nums, 0, len(nums)-1, target)

    def search_helper(self, arr, low, high, t):

        # Time: O(logn)
        # Space: O(1)

        if low > high:
            return False

        mid = low + (high-low) // 2

        if arr[mid] == t:
            return True

        if arr[low] <= arr[mid]:
            if arr[low] <= t and t < arr[mid]:
                return self.search_helper(arr, low, mid-1, t)
            else:
                return self.search_helper(arr, mid+1, high, t)
        else:
            if arr[mid] < t and t <= arr[high]:
                return self.search_helper(arr, mid+1, high, t)
            else:
                return self.search_helper(arr, low, mid-1, t)

Find peak element (M) (Interview)

class Solution:
    def findPeakElement(self, nums: List[int]) -> int:

        # Binary search trying to find local max.
        return self.helper(0, len(nums)-1, nums)

    def helper(self, low, high, nums):

        # mid = low + (high-low) // 2
        mid1 = int((low+high)/2)
        mid2 = mid1 + 1

        if low == high:
            return low

        if nums[mid1] > nums[mid2]:
            return self.helper(low, mid1, nums)
        else:
            return self.helper(mid2, high, nums)
