Linked list

  • Access: O(n)
  • Insert (at head): O(1)
  • Delete (at head): O(1)
  • Insert (at tail): O(n)
  • Delete (at tail): O(n)

Search

def search(lst, value):

    head = lst.get_head()

    while head.next:
        if head.data == value:
            return True
        head = head.next

    return False

Insert at head

insert_at_head(lst, value):

    temp_node = Node(value)
    temp_node.next = lst.get_head()
    lst.head_node = temp_node
    return

Insert at tail

def insert_at_tail(lst, value):

    node = Node(value)
    head = lst.get_head()

    # If linked list is empty
    if head is None:
        lst.head_node = node
        return

    while head.next:
        head = head.next

    head.next = node
    return

def delete_by_value(lst, value):
    if lst.is_empty():
        return

    curr = lst.get_head()
    if curr.data == value:
        lst.head_node = curr.next
        if curr.next is not None and curr.next.prev is not None:
            curr.next.prev = None

    while curr:
        if curr.data == value:
            if curr.next:
                prev = curr.prev
                nex = curr.next
                prev.next = nex
                nex.prev = prev
            else:
                curr.prev.next = None
            break
        curr = curr.next

    return

Find length

def length(lst):
    curr = lst.get_head()
    length = 0

    while curr:
        length += 1
        curr = curr.next
    return length

Find middle node

def find_mid(lst):
    if lst.is_empty():
        return None

    node = lst.get_head()
    mid = 0

    if lst.length() % 2 == 0:
        mid = lst.length() // 2
    else:
        mid = lst.length() // 2 + 1

    for i in range(mid-1):
        node = node.next

    return node

Union

def union(list1, list2):

    if list1.is_empty():
        return list2
    if list2.is_empty():
        return list1

    start = list1.get_head()

    # Traverse the first list till the tail.
    while start.next:
        start = start.next

    start.next = list2.get_head()
    list1.remove_duplicates()  # Assume this function is provided.
    return list1
