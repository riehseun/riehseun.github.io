Reverse linked list

1. Use 3 pointers: pre, cur, nex

  • Time: O(n)
  • Space: O(1)
def reverse(head):

    pre = None
    cur = head
    nex = head.next

    while cur:
        cur.next = pre
        pre = cur
        cur = nex
        if nex:
            nex = nex.next

    return pre
