Subset

Generate binary numbers.

  • If bit is 1, include the number. If bit is 0, exclude the number.
  • Time: O(n * 2**n)
  • Space: O(2**n)
def find_all_subsets(v):

    result = []
    result.append([])

    num_subset = 2 ** len(v)

    for i in range(1, num_subset):
        binary = bin(i)
        binary = binary.replace("0b","")
        padding = "0" * (len(v)-len(binary))
        binary = padding + binary
        subset = []

        for j, bit in enumerate(binary):
            if bit == "1":
                subset.append(v[j])
        result.append(subset)

    return result

Permutation

Swap the chars in the string to compute permutation

  • Time: O(n!)
  • Space: O(n)
def permute_word(word):
    result = []
    permute(result, word, 0, len(word)-1)
    return result

def permute(result, string, start, end):
    if start == end:
        result.append(string)
        return
    else:
        for i in range(start, end+1):
            # Swap two chars in the string.
            swap_index = list(string)
            swap_index[i], swap_index[start] = swap_index[start], swap_index[i]
            string = ''.join(swap_index)
            permute(result, string, start+1, end)

Find K-sum subsets

  • Time: O(n * 2**n)
  • Space: O(2**n)
def get_k_sum_subsets(set_of_integers, target_sum):
    all_subsets = find_all_subsets(set_of_integers)

    result = []
    for subset in all_subsets:
        if sum(subset) == target_sum:
            result.append(subset)

    return result
