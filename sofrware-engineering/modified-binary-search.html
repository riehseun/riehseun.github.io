def binary_search(nums, low, high, target):

    if low > high:
        return -1

    mid = low + (high - low) / 2
    mid = int(mid)

    # If regular binary search.
    # if target > nums[mid]:
    #     return binary_search(nums, mid+1, high, target)
    # elif target < nums[mid]:
    #     return binary_search(nums, low, mid-1, target)
    # else:
    #     return mid

    if nums[mid] == target:
        return mid

    if nums[low] <= nums[mid]:
        if nums[low] <= target and target < nums[mid]:
            return binary_search(nums, low, mid-1, target)
        else:
            return binary_search(nums, mid+1, high, target)
    else:
        if nums[mid] < target and target <= nums[high]:
            return binary_search(nums, mid+1, high, target)
        else:
            return binary_search(nums, low, mid-1, target)

# "is_bad_version" is given.
def search(n, low, high, count):

    if low > high:
        return low, count

    mid = low + (high-low) / 2
    mid = int(mid)
    version = is_bad_version(mid)
    if version:
        count += 1
        return search(n, low, mid-1, count)
    else:
        return search(n, mid+1, high, count)
