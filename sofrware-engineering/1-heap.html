Convert max heap to min heap

from heapq import heappush, heappop, heapify

def convertMax(max_heap):

    min_heap = []

    while max_heap:
        heappush(min_heap, (heappop(max_heap)))

    return min_heap

Find K smallest elements in a list

from heapq import heappush, heappop, heapify

def find_K_smallest(lst, k):

    result = []
    heapify(lst)

    for _ in range(k):
        result.append(heappop(lst))

    return result

Find K largest elements in a list

def findKLargest(lst, k):

    result = []
    negative_list = [-1*i for i in lst]
    heapify(negative_list)

    for _ in range(k):
        result.append(-heappop(negative_list))

    return result
