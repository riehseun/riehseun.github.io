Longest substring without repeating characters (M) (Interview/Liked)

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:

        # Time: O(n)
        # Space: O(n)
        # n = length of input string

        # Move the end pointer until seeing repeating chars.
        # Move the start pointer until substring has no repeating chars.
        # Keep the max diff between end and start pointer.
        # Scan finished when both start and end pointers reach the end of string.

        start = 0
        max_length = 0
        char_map = {}  # Store chracter and its index.

        for i,c in enumerate(s):

            # If duplicate is found, compute the length using start and the i pointer.
            # The second condition ensures that we are only looking at the substring
            # after the start index.
            if c in char_map and char_map[c] >= start:
                max_length = max(max_length, i-start)
                # Move the start pointer to plus 1 of the index where the duplicate is.
                start = char_map[c] + 1
            char_map[c] = i

        max_length = max(max_length, len(s)-start)

        return max_length

Substring with concatenation of all words (H)

class Solution:
    def findSubstring(self, s: str, words: List[str]) -> List[int]:
        result = []

        # For this problem, all words have the same length.
        word_length = len(words[0])
        substring_length = word_length * len(words)
        words_converted_to_dict = {}
        for word in words:
            if word in words_converted_to_dict:
                words_converted_to_dict[word] += 1
            else:
                words_converted_to_dict[word] = 0

        # Example: "goodgoodbestword"
        #          ["word","good","best","word"]

        # Run sliding window of substring through s.
        # Time: O(kn) where k = n / word_legnth.
        # Space: O(k) where k = n / word_legnth.
        for i in range(len(s)-substring_length+1):
            # print(i)
            # print(s[i:i+substring_length])

            # Check s[i:substring_length] can be constructed by words.
            # If so, append i to result.

            string_splitted_into_words = {}
            string = s[i:i+substring_length]
            j = 0
            while j < len(string):
                word = string[j:j+word_length]
                # Increment the occurance by 1.
                if word in string_splitted_into_words:
                    string_splitted_into_words[word] += 1
                # Initialize the occurance to 0.
                else:
                    string_splitted_into_words[word] = 0
                j += word_length

            if words_converted_to_dict == string_splitted_into_words:
                result.append(i)

        return result

Mimimum window substring (H) (Interview/Liked)

class Solution:
    def minWindow(self, s: str, t: str) -> str:

        # Time: O(len(s))
        # Space: O(len(t))

        # Prepare start and end pointers and two hash tables for s and t.
        # Compute occurance of each character in t and store it in a hash table.
        # Iterate s by incrementing the end pointer until the substring contains all chars in t. Remember this substring.
        # This substring may not be the "minimum" window substring. Now increment the start pointer.
        #   If the substring contains all chars in t, keep incrementing the start pointer. (as long as start <= end)
        #   Otherwise, increment the end pointer.

        # In count_s, we will store the count of only the characters
        # that exist in t.
        count_s = {}
        count_t = {}

        min_length = len(s) + 1
        min_substring = ""

        for c in t:
            if c not in count_t:
                count_t[c] = 0
            count_t[c] += 1

        start = 0
        end = 0

        while start <= end and end <= len(s):

            is_valid_substring = True
            for k,v in count_t.items():
                if k not in count_s \
                    or (k in count_s and count_s[k] < count_t[k]):
                    is_valid_substring = False

            if is_valid_substring:

                if end - start < min_length:
                    min_length = end - start
                    min_substring = s[start:end]

                # Whenever "start" pointer moves forward, must remove
                # the previous chracter from count_s set.
                # This block must come before incrementing the start pointer!
                if s[start] in count_t:
                    if count_s[s[start]] > 1:
                        count_s[s[start]] -= 1
                    else:
                        del count_s[s[start]]

                start += 1

            else:
                # If covered the last character in s,
                # increment the start pointer until it reaches the end.
                if end == len(s):
                    start += 1
                else:
                    # Update count_s dictionary each time "end" pointer moves.
                    # This block must come before incrementing the end pointer!
                    if s[end] in count_t:
                        if s[end] not in count_s:
                            count_s[s[end]] = 0
                        count_s[s[end]] += 1

                    end += 1

        return min_substring

Repeated DNA subsequence (M)

class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        # Time: O(k*len(s))
        # Space: O(len(s)-k) where number of substring to store is len(s)-k

        # Use hash table to store occurances of all substrings with length k.

        k = 10
        substrings = set()
        result = set()

        for i in range(0, len(s)-k+1):
            substring = s[i:i+k]
            # To save set space, use hash of substring
            # Instead of the actual substring.
            if hash(substring) not in substrings:
                substrings.add(hash(substring))
            # Substring is repeated.
            else:
                result.add(substring)

        return result

Minimum size subarray sum (M)

class Solution:
    def minSubArrayLen(self, target: int, nums: List[int]) -> int:

        # Time O(n)
        # Space O(1)
        # n = length of nums

        if sum(nums) < target:
            return 0

        min_length = len(nums)

        start = 0
        end = 0

        # Move end pointer until the sum is >= target.
        # Then, move start pointer until the sum is < target.
        # Continue until both start and end finish.

        while start <= end and end < len(nums):

            if sum(nums[start:end+1]) >= target:
                min_length = min(min_length, end-start+1)
                start += 1
            else:
                # If end pointer reached the end of string.
                if end == len(nums) - 1:
                    start += 1
                else:
                    end += 1

        return min_length

Sliding window maximumw (H) (Interview/Liked)

from collections import deque

class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:

        # Time: O(n) where n is list length.
        # Space: O(k)
        # n = length of nums

        # Keep the maximum number at the front of the queue. (We can do this by popping out all smaller numbers than the current number)
        # First, compute the maximum number in the first window.
        # Then, everytime a window moves.
        #   Pop out all smaller numbers than the current number.
        #   Remove previous number, which is now outside of the window, from the queue.
        #   Append the first item in the queue to the result.

        result = []
        queue = deque()

        # Compute the maximum in the first window.
        for i in range(k):

            # Pop out all smaller items than the current item
            # by removing last items in the queue.
            # Then the first itme in the queue will always
            # contains the maximum item.
            while queue and nums[queue[-1]] < nums[i]:
                queue.pop()
            queue.append(i)

        result.append(nums[queue[0]])

        # Move the window
        for i in range(k, len(nums)):

            while queue and nums[queue[-1]] < nums[i]:
                queue.pop()
            queue.append(i)

            # Remove all indexes in the queue that is before
            # and outside than the current window.
            while queue and queue[0] <= (i - k):
                queue.popleft()

            result.append(nums[queue[0]])

        return result

maxSlidingWindow([1,2], 3)  # Window size is greater than the list length.
maxSlidingWindow([1,2], 0)  # Window size is zero.
maxSlidingWindow([], 0)  # Window size and list length are zero.

Longest substring with at least K repeating characters (M) (Interview)

from collections import defaultdict

class Solution:
    def longestSubstring(self, s: str, k: int) -> int:

        # Time O(nlogn)
        # n = length of s

        # Find characters whose occurance is less than k.
        # Find substrings that does not include those chars and meet the
        # criteria of at least k repeating chars.
        # Keep track of the length of those good substrings.

        length_of_good_substring = {}

        self.helper(s, k, length_of_good_substring)

        max_len = 0
        for key, val in length_of_good_substring.items():
            max_len = max(max_len, val)

        return max_len

    def helper(self, s, k, length_of_good_substring):

        # Time O(n)
        # Space O(n)
        bad_chars = self.get_bad_chars(s, k)

        # The entire string is a good string.
        if not bad_chars:
            length_of_good_substring[s] = len(s)
            return

        start = 0

        for i in range(len(s)):
            # Check previous good string whenever hitting a bad char.
            if s[i] in bad_chars:
                if s[start:i] not in length_of_good_substring:
                    self.helper(s[start:i], k, length_of_good_substring)

                start = i + 1

        # Check previous good string at the end.
        self.helper(s[start:len(s)], k, length_of_good_substring)

    def get_bad_chars(self, s, k):

        char = defaultdict(int)

        for i in s:
            char[i] += 1

        bad_chars = set()

        for key,val in char.items():
            if val < k:
                bad_chars.add(key)

        return bad_chars

Find all anagrams in a string (M) (Liked)

lass Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        result = []

        # Keep track of occurance of each char in p.
        count_p = {}
        for char in p:
            if char not in count_p:
                count_p[char] = 1
            else:
                count_p[char] += 1

        # Keep track of occurance of each char in the very beginning
        # substring of s whose length equals to p.
        count_s = {}
        for char in s[0:len(p)]:
            if char not in count_s:
                count_s[char] = 1
            else:
                count_s[char] += 1

        start = 0
        end = len(p)

        # Time O(len(s)-len(p))
        # Space O(len(p))
        while end <= len(s):

            if count_p == count_s:
                result.append(start)

            if end == len(s):
                break

            char_to_remove = s[start]
            char_to_add = s[end]
            start += 1
            end += 1

            # Adjust "count_s" hash table
            if count_s[char_to_remove] == 1:
                del count_s[char_to_remove]
            else:
                count_s[char_to_remove] -= 1

            if char_to_add in count_s:
                count_s[char_to_add] += 1
            else:
                count_s[char_to_add] = 1

        return result

import math

def min_window(str1, str2):

    # Time: O(n^2) where n is length of str1.
    # Space: O(n) where n is length of str1.

    char = {}

    for i in range(len(str1)):
        if str1[i] not in char:
            char[str1[i]] = []
        char[str1[i]].append(i)

    candidate_strings = []
    first_match = str2[0]
    all_other_matches = str2[1:]

    if first_match not in char:
        return ""

    # Must interate all indices of first chracter match.
    # Construct shortest valid string starting at each first character match.
    for index in char[first_match]:
        index_array = []
        index_array.append(index)

        # Most possible number of checks will not exceed the number
        # of characters in str1.
        for c in all_other_matches:
            if c not in char:
                break

            for index in char[c]:
                # Append the smallest index possible.
                if index > index_array[-1]:
                    index_array.append(index)
                    break

        if len(index_array) >= len(str2):
            candidate_strings.append(str1[index_array[0]:index_array[-1]+1])

    min_length = math.inf
    min_string = ""
    for string in candidate_strings:
        if len(string) < min_length:
            min_length = len(string)
            min_string = string

    return min_string

import math

def min_window(str1, str2):

    # Time: O(n) where n is length of str1.
    # Space: O(1) no extra space needed.

    result = ""
    min_distance = len(str1) + 1

    index1 = 0
    index2 = 0

    while index1 < len(str1):

        if str1[index1] == str2[index2]:
            index2 += 1

            # At this point, str1 contains all chracters of str2.
            # index1 is at the last matching character.
            # index2 is at the last matching character plus 1.
            if index2 == len(str2):
                # Bring the index2 to the last index of str2.
                index2 -= 1
                # Idea: the last chracter in this range is guaranteeed to
                #       be the part of the solution in this range.

                start = index1
                end = index1

                while index2 >= 0:
                    if str1[start] == str2[index2]:
                        index2 -= 1
                    start -= 1

                # When index2 is less than 0, then we covered all characters in
                # str2.
                # Reset the position of index2 to 0. (At this point, it would be -1)
                index2 = 0

                if end - start < min_distance:
                    min_distance = end - start
                    result = str1[start+1:end+1]

        index1 += 1

    return result

Best time to buy and sell stock

Iterate once. Remember the minimum price and maximum profit seen so far.

  • Always remember the minimum price and maximum profit seen so far.
  • While iterating the array, the value is either lower or higher than the minimum seen so far.
    • If lower than the minimum, update the minimum seen so far.
    • If greater than the minumum, update the maximum profit by (price - minimum seen so far)
def max_profit(stock_prices):

    # Time O(len(stock_prices)).
    # Space O(1).

    min_price = max(stock_prices)
    max_profit = 0

    for price in stock_prices:

        if price < min_price:
            min_price = price
        else:
            max_profit = max(max_profit, price-min_price)

    return max_profit
