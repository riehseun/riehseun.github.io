Binary tree level order traversal (M) (Interview/Liked)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:

        # Do BFS

        self.nodes = defaultdict(list)

        # Time O(n) where n is number of nodes
        # Space O(n) where n is number of nodes
        def bfs(root):

            explored = set()
            explored.add(root)

            queue = deque()
            queue.append((root,0))

            while queue:
                root, depth = queue.popleft()
                self.nodes[depth].append(root.val)
                if root.left and root.left not in explored:
                    explored.add(root.left)
                    queue.append((root.left,depth+1))
                if root.right and root.right not in explored:
                    explored.add(root.right)
                    queue.append((root.right,depth+1))

        if not root:
            return []

        bfs(root)

        ret_val = []
        for k,v in self.nodes.items():
            ret_val.append(v)

        return ret_val

Binary tree zigzag level order traversal (M) (Interview)

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def zigzagLevelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:

        # Do BFS

        self.nodes = defaultdict(list)

        # Time O(n) where n is number of nodes
        # Space O(n) where n is number of nodes
        def bfs(root):

            explored = set()
            explored.add(root)

            queue = deque()
            queue.append((root,0))

            while queue:
                root, depth = queue.popleft()
                self.nodes[depth].append(root.val)
                if root.left and root.left not in explored:
                    explored.add(root.left)
                    queue.append((root.left,depth+1))
                if root.right and root.right not in explored:
                    explored.add(root.right)
                    queue.append((root.right,depth+1))

        if not root:
            return []

        bfs(root)

        ret_val = []
        for k,v in self.nodes.items():
            if k % 2 == 0:
                ret_val.append(v)
            else:
                v.reverse()
                ret_val.append(v)

        return ret_val

    def levelOrderBottom(self, root: Optional[TreeNode]) -> List[List[int]]:

        # Do BFS

        self.nodes = defaultdict(list)
